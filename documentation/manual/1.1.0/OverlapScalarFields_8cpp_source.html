306 void
307 OverlapScalarFields::plotData(std::string dir, std::string name, bool flag, int counter, mimmo::MimmoObject * geo){
308 
309  if(m_originals.count(geo) == 0) return;
310  dvector1D field;
311  bool flagCalc = (m_results.count(geo) > 0);
312 
313 
314  dvecarr3E points = geo->getVertexCoords();
315  ivector2D connectivity;
316  bitpit::VTKElementType cellType = geo->desumeElement();
317 
318  if (geo->getType() != 3){
319  connectivity = geo->getCompactConnectivity();
320  }
321  else{
322  int np = points.size();
323  connectivity.resize(np);
324  for (int i=0; i<np; i++){
325  connectivity[i].resize(1);
326  connectivity[i][0] = i;
327  }
328  }
329  bitpit::VTKUnstructuredGrid output(dir,name,cellType);
330  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
331  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
332  output.setDimensions(connectivity.size(), points.size());
333 
334  if(flagCalc) field = m_results[geo];
335  else field.resize(points.size(), 0.0);
336 
337  output.addData("field", bitpit::VTKFieldType::SCALAR, bitpit::VTKLocation::POINT, field);
338 
339  std::vector<long> ids(points.size());
340  long ID;
341  for (auto vertex : geo->getVertices()){
342  ID = vertex.getId();
343  ids[geo->getMapDataInv(ID)] = ID;
344  }
345 
346  output.addData("ID", bitpit::VTKFieldType::SCALAR, bitpit::VTKLocation::POINT, ids);
347 
348  output.setCounter(counter);
349  output.setCodex(bitpit::VTKFormat::APPENDED);
350  if(!flag) output.setCodex(bitpit::VTKFormat::ASCII);
351 
352  output.write();
353 };

376 void
377 OverlapScalarFields::execute(){
378 
379  if(m_originals.empty()) return;
380  m_results.clear();
381  int size, listsize;
382  dvector1D temp;
383  dvector1D results;
384  for(auto & obj : m_originals){
385 
386  size = obj.first->getPatch()->getVertexCount();
387  //careful resizing
388  results.resize(size, 0.0);
389  for(auto &vec : obj.second) vec->resize(size,0.0);
390 
391  listsize = obj.second.size();
392  temp.resize(listsize);
393 
394  for(int i=0; i<size; ++i){
395  for(int j=0; j<listsize; j++) temp[j] = (*(obj.second[j]))[i];
396  results[i] = overlapFields(temp);
397  }
398 
399  m_results[obj.first] = results;
400  }
401 }

419 //DEVELOPERS REMIND if more overlap methods are added refer to this method to implement them
420 double
421 OverlapScalarFields::overlapFields(dvector1D & locField){
422  int size = locField.size();
423  if(size < 1) return 0.0;
424 
425  double value = locField[0];
426  if(size ==1 )return value;
427 
428  switch(m_overlapCriterium){
429  case OverlapMethod::MAX :
430  for(auto && loc : locField){
431  value = std::fmax(loc,value);
432  }
433  break;
434  case OverlapMethod::MIN :
435  for(auto && loc : locField){
436  value = std::fmin(loc,value);
437  }
438  break;
439  case OverlapMethod::AVERAGE :
440  value = 0.0;
441  for(auto && loc : locField){
442  value += loc/double(size);
443  }
444  break;
445  case OverlapMethod::SUM :
446  value = 0.0;
447  for(auto && loc : locField){
448  value += loc;
449  }
450  break;
451  default : //never been reached
452  break;
453  }
454 
455  return value;
456 };
