122 void
123 ProjectCloud::execute(){
124 
125  if(getGeometry() == NULL || getGeometry()->isEmpty()) return;
126 
127  if(!getGeometry()->isBvTreeBuilt()) getGeometry()->buildBvTree();
128 
129  //project points on surface.
130  int counter = 0;
131  m_proj.resize(m_points.size());
132  for(auto &val : m_points){
133  m_proj[counter]= bvTreeUtils::projectPoint(&val, getGeometry()->getBvTree());
134  ++counter;
135  }
136  return;
137 };

144 ProjectCloud::plotOptionalResults(){
145  bitpit::VTKFormat codex = bitpit::VTKFormat::APPENDED;
146 
147  int size = m_proj.size();
148  ivector1D conn(size);
149  for(int i=0; i<size; i++){
150  conn[i] = i;
151  }
152  std::string dir = "./";
153  std::string file = m_name + "_" + std::to_string(getClassCounter());
154 
155  bitpit::VTKUnstructuredGrid vtk(dir, file, bitpit::VTKElementType::VERTEX);
156  vtk.setGeomData( bitpit::VTKUnstructuredField::POINTS, m_proj) ;
157  vtk.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, conn) ;
158  vtk.setDimensions(size, size);
159  vtk.setCodex(codex);
160 
161  vtk.write();
162 }
