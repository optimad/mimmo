172 void
173 RotationGeometry::execute(){
174 
175  if (getGeometry() == NULL) return;
176 
177  int nV = m_geometry->getNVertex();
178  m_displ.resize(nV);
179  m_filter.resize(nV, 1.0);
180 
181  //compute coefficients and constant vectors of rodriguez formula
182  double a = cos(m_alpha);
183  darray3E b = (1 - cos(m_alpha)) * m_direction;
184  double c = sin(m_alpha);
185 
186  darray3E point, rotated;
187  long ID;
188  int idx;
189  liimap mapID = m_geometry->getMapDataInv();
190 
191  for (auto vertex : m_geometry->getVertices()){
192  point = vertex.getCoords();
193  ID = vertex.getId();
194  idx = mapID[ID];
195 
196  point -= m_origin;
197  //rodrigues formula
198  rotated = a * point +
199  b * dotProduct(m_direction, point) +
200  c * crossProduct(m_direction, point);
201 
202  rotated += m_origin;
203  point += m_origin;
204 
205  m_displ[idx] = (rotated-point)*m_filter[idx];
206 
207  }
208

215 RotationGeometry::apply(){
216 
217  if (getGeometry() == NULL) return;
218  dvecarr3E vertex = getGeometry()->getVertexCoords();
219  long nv = getGeometry()->getNVertex();
220  nv = long(std::min(int(nv), int(m_displ.size())));
221  livector1D & idmap = getGeometry()->getMapData();
222  for (long i=0; i<nv; i++){
223  vertex[i] += m_displ[i];
224  getGeometry()->modifyVertex(vertex[i], idmap[i]);
225  }
