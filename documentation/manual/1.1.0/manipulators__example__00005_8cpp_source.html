49 void test00005() {
50 
51  /* Creation of mimmo containers.
52  * Input and output MimmoGeometry are instantiated
53  * as two different objects (no loop in chain are permitted).
54  */
55  MimmoGeometry * mimmo0 = new MimmoGeometry();
56  mimmo0->setIOMode(IOMode::CONVERT);
57  mimmo0->setReadDir("geodata");
58  mimmo0->setReadFileType(FileType::STL);
59  mimmo0->setReadFilename("sphere2");
60  mimmo0->setWriteDir(".");
61  mimmo0->setWriteFileType(FileType::STL);
62  mimmo0->setWriteFilename("manipulators_output_00005.0000");
63 
64  MimmoGeometry * mimmo1 = new MimmoGeometry();
65  mimmo1->setIOMode(IOMode::WRITE);
66  mimmo1->setWriteDir(".");
67  mimmo1->setWriteFileType(FileType::STL);
68  mimmo1->setWriteFilename("manipulators_output_00005.0001");
69 
70  /* Creation of a random distribution of 10 points with coordinates between [-0.5, 0.5]
71  */
72  int np = 10;
73  dvecarr3E rbfNodes(10);
74  time_t Time = time(NULL);
75  srand(Time);
76  for (int i=0; i<np; i++){
77  for (int j=0; j<3; j++)
78  rbfNodes[i][j] = 1.0*( (double) (rand()) / RAND_MAX ) - 0.5;
79  }
80 
81  /* Creation of a projection block aimed to project
82  * the point cloud previously defined over the input geometry.
83  * The rbf control nodes will be defined on the surface of the
84  * geometry as output of this block.
85  *
86  */
87  ProjectCloud* proj = new ProjectCloud();
88  proj->setCoords(rbfNodes);
89 
90  /* Instantiation of a MRBF object with a distribution of 10 random control nodes projected
91  * ont he input surface.
92  * Plot Optional results during execution active for MRBF block.
93  */
94  MRBF* mrbf = new MRBF();
95  mrbf->setMode(MRBFSol::NONE);
96  mrbf->setSupportRadius(0.25);
97  mrbf->setPlotInExecution(true);
98 
99 
100  /* Creation of a set of displacements of the control nodes of the radial basis functions.
101  * Use a radial displacements from a center point placed in axes origin.
102  */
103  dvecarr3E displ(np, darray3E{0.0, 0.0, 0.0});
104  darray3E center({0.0, 0.0, 0.0});
105  for (int i=0; i<np; i++){
106  displ[i] = rbfNodes[i] - center;
107  displ[i] /= 2.0*norm2(displ[i]);
108  }
109 
110 
111  /* Set Generic input block with the
112  * displacements defined above.
113  */
114  GenericInput* input = new GenericInput();
115  input->setInput(displ);
116 
117  /* Set Generic output block to write the
118  * nodes defined above.
119  */
120  GenericOutput * outputn = new GenericOutput();
121  outputn->setFilename("manipulators_output_00005n.csv");
122  outputn->setCSV(true);
123 
124  /* Set Generic output block to write the
125  * displacements defined above.
126  */
127  GenericOutput * outputd = new GenericOutput();
128  outputd->setFilename("manipulators_output_00005d.csv");
129  outputd->setCSV(true);
130 
131  /* Create applier block.
132  * It applies the deformation displacements to the original input geometry.
133  */
134  Apply* applier = new Apply();
135 
136  /* Setup pin connections.
137  */
138  addPin(mimmo0, mrbf, PortType::M_GEOM, PortType::M_GEOM);
139  addPin(mimmo0, proj, PortType::M_GEOM, PortType::M_GEOM);
140  addPin(mimmo0, applier, PortType::M_GEOM, PortType::M_GEOM);
141  addPin(proj, mrbf, PortType::M_COORDS, PortType::M_COORDS);
142  addPin(proj, outputn, PortType::M_COORDS, PortType::M_COORDS);
143  addPin(input, mrbf, PortType::M_DISPLS, PortType::M_DISPLS);
144  addPin(input, outputd, PortType::M_DISPLS, PortType::M_DISPLS);
145  addPin(mrbf, applier, PortType::M_GDISPLS, PortType::M_GDISPLS);
146  addPin(applier, mimmo1, PortType::M_GEOM, PortType::M_GEOM);
147 
148  /* Setup execution chain.
149  * The object can be insert in the chain in random order.
150  * The chain object recover the correct order of execution from
151  * the pin connections.
152  */
153  Chain ch0;
154  ch0.addObject(input);
155  ch0.addObject(outputn);
156  ch0.addObject(outputd);
157  ch0.addObject(mimmo0);
158  ch0.addObject(proj);
159  ch0.addObject(applier);
160  ch0.addObject(mrbf);
161  ch0.addObject(mimmo1);
162 
163  /* Execution of chain.
164  * Use debug flag true to print out the execution steps.
165  */
166  ch0.exec(true);
167 
168  /* Clean up & exit;
169  */
170  delete mrbf;
171  delete proj;
172  delete applier;
173  delete input;
174  delete outputn;
175  delete outputd;
176  delete mimmo0;
177  delete mimmo1;
178 
179  proj = NULL;
180  mrbf = NULL;
181  applier = NULL;
182  input = NULL;
183  outputn = NULL;
184  outputd = NULL;
185  mimmo0 = NULL;
186  mimmo1 = NULL;
187 
188  return;
189 }
