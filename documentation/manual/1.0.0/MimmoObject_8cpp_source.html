80 MimmoObject::MimmoObject(int type, dvecarr3E & vertex, ivector2D * connectivity){
81  m_type = max(1,type);
82  m_internalPatch = true;
83  const int id = 0;
84  if (m_type == 2){
85  m_patch = new VolUnstructured(id, 3);
86  dynamic_cast<VolUnstructured*> (m_patch)->setExpert(true);
87  }else{
88  m_patch = new SurfUnstructured(id);
89  dynamic_cast<SurfUnstructured*> (m_patch)->setExpert(true);
90  }
91 
92  bitpit::ElementInfo::Type eltype = bitpit::ElementInfo::UNDEFINED;
93  int sizeVert, sizeCell;
94 
95  sizeVert = vertex.size();
96  m_patch->reserveVertices(sizeVert);
97 
98  for(auto & vv : vertex) addVertex(vv);
99 
100  if(connectivity == NULL) m_type = 3;
101  if (m_type != 3){
102  int sizeConn = (*connectivity)[0].size();
103  sizeCell = connectivity->size();
104 
105  switch(m_type){
106  case 1:
107  if(sizeConn == 3) eltype = bitpit::ElementInfo::TRIANGLE;
108  if(sizeConn == 4) eltype = bitpit::ElementInfo::QUAD;
109  break;
110  case 2:
111  if(sizeConn == 4) eltype = bitpit::ElementInfo::TETRA;
112  if(sizeConn == 8) eltype = bitpit::ElementInfo::HEXAHEDRON;
113  break;
114  case 4:
115  if(sizeConn == 2) eltype = bitpit::ElementInfo::LINE;
116  break;
117  default:
118  // never been reached
119  break;
120  }
121 
122  if(eltype != bitpit::ElementInfo::UNDEFINED){
123 
124  m_patch->reserveCells(sizeCell);
125 
126  for(auto & cc : *connectivity){
127 
128  livector1D temp(cc.size());
129  int counter=0;
130 
131  for(auto && val : cc) {
132  temp[counter] = val;
133  ++counter;
134  }
135 
136  addConnectedCell(temp, eltype);
137  }
138 
139  m_pidsType.insert(0);
140 
141  }else{
142  std::cout<<"Not supported connectivity found for MimmoObject"<<std::endl;
143  std::cout<<"Proceeding as Point Cloud geometry"<<std::endl;
144  }
145  }
146  m_bvTree.setPatch(m_patch);
147  m_bvTreeBuilt = false;
148  m_kdTreeBuilt = false;
149  m_bvTreeSupported = (m_type != 3);
150  m_retrackBvTree = false;
151  m_retrackKdTree = false;
152 };

711 bool
712 MimmoObject::setCells(const bitpit::PiercedVector<Cell> & cells){
713 
714  if (cells.empty() || !m_bvTreeSupported) return false;
715 
716  m_mapCell.clear();
717  m_mapCellInv.clear();
718  m_pidsType.clear();
719 
720  getPatch()->resetCells();
721 
722  int sizeCell = cells.size();
723  getPatch()->reserveCells(sizeCell);
724 
725  long idc;
726  int nVert;
727  short pid;
728  bitpit::ElementInfo::Type eltype;
729  bool checkTot = true;
730  livector1D connectivity;
731 
732  for (const auto & cell : cells){
733  // get ID
734  idc = cell.getId();
735  //check on element type
736  eltype = cell.getType();
737  //check info PID
738  pid = (short)cell.getPID();
739  nVert = cell.getVertexCount();
740  connectivity.resize(nVert);
741  auto conn = cell.getConnect();
742  for(int i=0; i<nVert; ++i){
743  connectivity[i] = conn[i];
744  }
745  checkTot = checkTot && addConnectedCell(connectivity, eltype, pid, idc);
746  }
747 
748  return checkTot;
749 };

766 bool
767 MimmoObject::addConnectedCell(const livector1D & conn, bitpit::ElementInfo::Type type, long idtag){
768 
769  if (isEmpty() || conn.empty() || !m_bvTreeSupported) return false;
770  if(idtag != bitpit::Cell::NULL_ID && m_patch->getCells().exists(idtag)) return false;
771 
772  int sizeElement = checkCellType(type);
773  if(sizeElement < 0) return false;
774 
775  bitpit::PatchKernel::CellIterator it;
776 
777  livector1D conn_dum = conn;
778  conn_dum.resize(sizeElement, 0);
779 
780  long checkedID;
781  if(idtag == bitpit::Cell::NULL_ID){
782  it = m_patch->addCell(type, true, conn_dum);
783  checkedID = it->getId();
784  }else{
785  it = m_patch->addCell(type, true,conn_dum, idtag);
786  checkedID = idtag;
787  }
788 
789  m_pidsType.insert(0);
790 
791  //create inverse map of cells
792  m_mapCell.push_back(checkedID);
793  m_mapCellInv[checkedID] = m_mapCell.size()-1;
794  m_bvTreeBuilt = false;
795  return true;
