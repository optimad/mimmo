286 livector1D
287 SelectionByMapping::extractSelection(){
288 
289  if(!(getGeometry()->isBvTreeBuilt())) getGeometry()->buildBvTree();
290  std::set<long> cellList;
291 
292  for (auto && file : m_geolist){
293  livector1D list = getProximity(file);
294  cellList.insert(list.begin(), list.end());
295  }
296 
297  for (auto mimmo : m_mimmolist){
298  livector1D list = getProximity(mimmo);
299  cellList.insert(list.begin(), list.end());
300  }
301 
302 /* check if dual */
303  livector1D result;
304  if(m_dual){
305  livector1D totID = getGeometry()->getMapCell();
306  result.resize(totID.size() - cellList.size());
307  if(result.size() == 0) return result;
308 
309  std::sort(totID.begin(), totID.end());
310  int counter = 0;
311  auto tot_it = totID.begin();
312  auto cand_it = cellList.begin();
313  while(tot_it != totID.end()){
314  long val = *tot_it;
315  if (cand_it == cellList.end() || val != *cand_it) {
316  result[counter] = val;
317  ++counter;
318  } else {
319  ++cand_it;
320  }
321  ++tot_it;
322  }
323  }else{
324  result.insert(result.end(), cellList.begin(), cellList.end());
325  }
326  return result;
327 };

335 SelectionByMapping::getProximity(std::pair<std::string, int> val){
336 
337  svector1D info = extractInfo(val.first);
338 
339  MimmoGeometry * geo = new MimmoGeometry();
340  geo->setRead(true);
341  geo->setWrite(false);
342  geo->setReadDir(info[0]);
343  geo->setReadFilename(info[1]);
344  geo->setReadFileType(val.second);
345  geo->setBuildBvTree(true);
346  geo->execute();
347 
348  if(geo->getGeometry()->getNVertex() == 0 || geo->getGeometry()->getNCells() == 0 ){
349  std::cout<<"failed to read geometry in SelectionByMapping::getProximity"<<std::endl;
350  return livector1D();
351  }
352  livector1D result = mimmo::bvTreeUtils::selectByPatch(geo->getGeometry()->getBvTree(), getGeometry()->getBvTree(), m_tolerance);
353  delete geo;
354  geo=NULL;
355 
356  return result;

410 
411  BITPIT_UNUSED(name);
412 
413  //checking topology
414  if(slotXML.hasOption("Topology")){

421  }
422  if(m_topo != temp) return;
423  }
424 
425  if(slotXML.hasOption("Priority")){
426  std::string input = slotXML.get("Priority");
427  int value =0;
428  if(!input.empty()){
429  std::stringstream ss(bitpit::utils::trim(input));

432  setPriority(value);

434 
435  //start absorbing

437  std::string input = slotXML.get("Dual");

444  setDual(value);

446

449  input = bitpit::utils::trim(input);

451  if(!input.empty()){
452  std::stringstream ss(input);
453  ss>>temp;
454  }
455  setTolerance(temp);
456 
457  }
458 
459  std::unordered_map<std::string, int> mapp;
460  if(slotXML.hasSection("Files")){
461 
462  const bitpit::Config::Section & filesXML = slotXML.getSection("Files");
463 
464  for(auto & subfile : filesXML.getSections()){
465  std::string path;

545  }
546 
547  if(isPlotInExecution()){
548  slotXML.set("PlotInExecution", std::to_string(1));
549  }
550 
551  if(m_outputPlot != "."){
552  slotXML.set("OutputPlot", m_outputPlot);
