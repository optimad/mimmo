686 darray3E
687 FFDLattice::nurbsEvaluator(darray3E & pointOr){
688 
689  darray3E point = transfToLocal(pointOr);
690  darray3E scaling = getShape()->getScaling();
691 
692  ivector1D knotInterval(3,0);
693  dvector2D BSbasis(3);
694  dvector1D valH(4,0), temp1(4,0),temp2(4,0), zeros(4,0);
695  int uind, vind, wind, index;
696 
697  dvecarr3E* displ = getDisplacements();
698 
699  int i0 = m_mapdeg[0];
700  int i1 = m_mapdeg[1];
701  int i2 = m_mapdeg[2];
702 
703  iarray3E mappedIndex;
704 
705  for(int i=0; i<3; i++){
706  knotInterval[i] = getKnotInterval(point[i],i);
707  BSbasis[i] = basisITS0(knotInterval[i], i, point[i]);
708  }
709 
710  uind = knotInterval[i0] - m_deg[i0];
711  vind = knotInterval[i1] - m_deg[i1];
712  wind = knotInterval[i2] - m_deg[i2];
713 
714  for(int i=0; i<=m_deg[i0]; ++i){
715 
716  mappedIndex[0] = uind+i;
717  temp1 = zeros;
718 
719  for(int j=0; j<=m_deg[i1]; ++j){
720 
721  mappedIndex[1] = vind+j;
722  temp2 = zeros;
723 
724  for(int k=0; k<=m_deg[i2]; ++k){
725 
726  mappedIndex[2] = wind+k;
727  index = accessMapNodes(mappedIndex[0],mappedIndex[1],mappedIndex[2]);
728 
729  for(int intv=0; intv<3; ++intv){
730  temp2[intv] += BSbasis[i2][k]*m_weights[m_intMapDOF[index]]*(*displ)[m_intMapDOF[index]][intv];
731  }
732  temp2[3] += BSbasis[i2][k]*m_weights[index];
733  }
734  for(int intv=0; intv<4; ++intv){
735  temp1[intv] += BSbasis[i1][j]*temp2[intv];
736  }
737 
738  }
739  for(int intv=0; intv<4; ++intv){
740  valH[intv] += BSbasis[i0][i]*temp1[intv];
741  }
742  }
743 
744  darray3E outres;
745  if(isDisplGlobal()){
746  for(int i=0; i<3; ++i){
747  outres[i] = valH[i]/valH[3];
748  }
749 
750  }else{
751  //summing scaled displ in local ref frame;
752  for(int i=0; i<3; ++i){
753  point[i] += valH[i]/(valH[3]*scaling[i]);
754  }
755 
756  //get final displ in global ref frame:
757  outres = transfToGlobal(point) - pointOr;
758  }
759  return(outres);
760 
761 };
