387 void GenericDispls::read(){
388 
389  std::ifstream reading;
390  std::string source = m_dir+"/"+ m_filename;
391  reading.open(source.c_str());
392  if(reading.is_open()){
393 
394  m_displ.clear();
395  m_labels.clear();
396 
397  std::string keyword, line;
398  long label;
399  darray3E dtrial;
400 
401  do{
402  line.clear();
403  keyword.clear();
404  std::getline(reading,line);
405  line = bitpit::utils::trim(line);
406  std::stringstream ss(line);
407  ss>>keyword;
408  keyword = bitpit::utils::trim(keyword);
409  if(keyword == "$DISPL") {
410 
411  ss>>label;
412  m_labels.push_back(label);
413 
414  dtrial.fill(0.0);
415  ss>>dtrial[0]>>dtrial[1]>>dtrial[2];
416  m_displ.push_back(dtrial);
417  }
418 
419  }while(!reading.eof());
420 
421  m_nDispl = m_displ.size();
422 
423  }else{
424  std::cout<<"Error of "<<m_name<<" : cannot open "<<m_filename<< " requested. Exiting... "<<std::endl;
425  exit(1);
426  }
427 
428  reading.close();
429 };

434 void GenericDispls::write(){
435 
436  //assessing data;
437  int sizelabels = m_labels.size();
438  long maxlabel = 0;
439  for(auto & val: m_labels){
440  maxlabel = std::max(maxlabel,val);
441  }
442 
443  int sizedispl = m_displ.size();
444 
445  m_labels.resize(m_displ.size(), -1);
446  if(sizelabels < sizedispl){
447  for(int i = sizelabels; i<sizedispl; ++i){
448  m_labels[i] = maxlabel+1;
449  ++maxlabel;
450  }
451  }
452 
453  std::string filename;
454  if(m_template){
455  filename = "TEMPLATE_"+m_filename;
456  }else{
457  filename = m_filename;
458  }
459 
460  std::ofstream writing;
461  std::string source = m_dir+"/"+ m_filename;
462  writing.open(source.c_str());
463  std::string keyT1 = "{", keyT2 = "}";
464  if(writing.is_open()){
465 
466  int counter = 0;
467  for(auto & dd : m_displ){
468  if(m_template){
469  std::string str1 = keyT1+"x"+std::to_string(m_labels[counter])+keyT2;
470  std::string str2 = keyT1+"y"+std::to_string(m_labels[counter])+keyT2;
471  std::string str3 = keyT1+"z"+std::to_string(m_labels[counter])+keyT2;
472 
473  writing<<"$DISPL"<<'\t'<<m_labels[counter]<<'\t'<<str1<<'\t'<<str2<<'\t'<<str3<<std::endl;
474  }else{
475  writing<<"$DISPL"<<'\t'<<m_labels[counter]<<'\t'<<dd[0]<<'\t'<<dd[1]<<'\t'<<dd[2]<<std::endl;
476  }
477  ++counter;
478  }
479 
480  }else{
481  std::cout<<"Error of "<<m_name<<" : cannot open "<<m_filename<< " requested. Exiting... "<<std::endl;
482  exit(1);
483  }
484 
485  writing.close();
