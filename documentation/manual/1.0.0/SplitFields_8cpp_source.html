192  BITPIT_UNUSED(name);

196  //checking topology

202  std::stringstream ss(input);
203  ss>>temptop;
204  }
205  if(m_topo != temptop) return;
206  }
207 
208  if(slotXML.hasOption("Priority")){
209  input = slotXML.get("Priority");
210  int value =0;
211  if(!input.empty()){
212  std::stringstream ss(bitpit::utils::trim(input));
213  ss>>value;
214  }
215  setPriority(value);
216  };
217 
218  if(slotXML.hasOption("PlotInExecution")){
219  std::string input = slotXML.get("PlotInExecution");
220  input = bitpit::utils::trim(input);
221  bool value = false;
222  if(!input.empty()){
223  std::stringstream ss(input);
224  ss >> value;
225  }
226  setPlotInExecution(value);
227  }
228 
229  if(slotXML.hasOption("OutputPlot")){
230  std::string input = slotXML.get("OutputPlot");
231  input = bitpit::utils::trim(input);
232  std::string temp = ".";
233  if(!input.empty()) setOutputPlot(input);
234  else setOutputPlot(temp);
235  }
