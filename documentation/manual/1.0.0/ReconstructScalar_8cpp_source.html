243 void
244 ReconstructScalar::plotData(std::string dir, std::string name, bool flag, int counter){
245 
246  if(getGeometry() == NULL || getGeometry()->isEmpty()) return;
247  dvecarr3E points = getGeometry()->getVertexCoords();
248  ivector2D connectivity;
249  bitpit::VTKElementType cellType = getGeometry()->desumeElement();
250 
251  if (getGeometry()->getType() != 3){
252  connectivity = getGeometry()->getCompactConnectivity();
253  }
254  else{
255  int np = points.size();
256  connectivity.resize(np);
257  for (int i=0; i<np; i++){
258  connectivity[i].resize(1);
259  connectivity[i][0] = i;
260  }
261  }
262  bitpit::VTKUnstructuredGrid output(dir,name,cellType);
263  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
264  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
265  output.setDimensions(connectivity.size(), points.size());
266 
267  output.addData("field", bitpit::VTKFieldType::SCALAR, bitpit::VTKLocation::POINT, m_result);
268 
269  std::vector<long> ids(points.size());
270  long ID;
271  for (auto vertex : getGeometry()->getVertices()){
272  ID = vertex.getId();
273  ids[getGeometry()->getMapDataInv(ID)] = ID;
274  }
275 
276  output.addData("ID", bitpit::VTKFieldType::SCALAR, bitpit::VTKLocation::POINT, ids);
277 
278  output.setCounter(counter);
279  output.setCodex(bitpit::VTKFormat::APPENDED);
280  if(!flag) output.setCodex(bitpit::VTKFormat::ASCII);
281 
282  output.write();
283 };

289 void
290 ReconstructScalar::execute(){
291 
292  if(getGeometry() == NULL) return;
293 
294  liimap & vMotherMap = getGeometry()->getMapDataInv();
295 
296  m_result.resize(getGeometry()->getPatch()->getVertexCount(), 0.0);
297 
298  if(m_subpatch.empty()) return;
299 
300  std::unordered_map<long, dvector1D> map = checkOverlapping();
301 
302  for(auto && obj : map){
303  m_result[vMotherMap[obj.first]] = overlapFields(obj.second);
304  obj.second.clear();
305  }
306 
307 }

324 //DEVELOPERS REMIND if more overlap methods are added refer to this method to implement them
325 double
326 ReconstructScalar::overlapFields(dvector1D & locField){
327  int size = locField.size();
328  if(size < 1) return 0.0;
329 
330  double value = locField[0];
331  if(size ==1 )return value;
332 
333  switch(m_overlapCriterium){
334  case OverlapMethod::MAX :
335  for(auto && loc : locField){
336  value = std::fmax(loc,value);
337  }
338  break;
339  case OverlapMethod::MIN :
340  for(auto && loc : locField){
341  value = std::fmin(loc,value);
342  }
343  break;
344  case OverlapMethod::AVERAGE :
345  value = 0.0;
346  for(auto && loc : locField){
347  value += loc/double(size);
348  }
349  break;
350  case OverlapMethod::SUM :
351  value = 0.0;
352  for(auto && loc : locField){
353  value += loc;
354  }
355  break;
356  default : //never been reached
357  break;
358  }
359 
360  return value;
361 };

369 std::unordered_map<long, dvector1D>
370 ReconstructScalar::checkOverlapping(){
371 
372  std::unordered_map<long, dvector1D> result;
373  int counter;
374 
375  for(auto && pairInd : m_subpatch){
376 
377  livector1D & vMap = pairInd.first->getMapData();
378  counter = 0;
379  dvector1D field = *(pairInd.second);
380  for(int i=0; i<(int)field.size(); ++i){
381  result[vMap[counter]].push_back(field[i]);
382  ++counter;
383  }
384  }
385 
386  return(result);
387 };

418  BITPIT_UNUSED(name);

430 
431  if(slotXML.hasOption("OverlapCriterium")){

442 
443

445  std::string input = slotXML.get("PlotInExecution");

450  ss >> value;
451  }
452  setPlotInExecution(value);
453  }
454 
455  if(slotXML.hasOption("OutputPlot")){
456  std::string input = slotXML.get("OutputPlot");
