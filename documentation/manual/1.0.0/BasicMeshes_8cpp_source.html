630  void UStructMesh::setShape(ShapeType type){
631  //create internal shape using unique_ptr member.
632  // unlink external shape eventually
633  if(m_shape){
634 
635  m_origin_temp = m_shape.get()->getOrigin();
636  m_span_temp = m_shape.get()->getSpan();
637  m_inflimits_temp = m_shape.get()->getInfLimits();
638  m_refsystem_temp = m_shape.get()->getRefSystem();
639  m_setorigin = true;
640  m_setspan = true;
641  m_setInfLimits = true;
642  m_setRefSys = true;
643 
644  m_shape.reset(nullptr);
645  }
646 
647  darray3E origin{{0,0,0}};
648  dmatrix33E spanMat;
649  spanMat[0].fill(1); spanMat[2].fill(1); spanMat[2].fill(1);
650  spanMat[1][1] = spanMat[2][1] = 2*M_PI;
651  spanMat[2][2] = M_PI;
652 
653  if(m_setorigin){origin = m_origin_temp; m_setorigin= false;}
654  if(m_setspan){
655  for(int i=0; i<3; ++i) spanMat[i] = m_span_temp;
656  m_setspan = false;
657  }
658 
659  switch(type){
660  case ShapeType::CYLINDER :
661  m_shape = std::unique_ptr<BasicShape>(new Cylinder(origin,spanMat[1]));
662  break;
663  case ShapeType::SPHERE :
664  m_shape = std::unique_ptr<BasicShape>(new Sphere(origin, spanMat[2]));
665  break;
666  default://CUBE
667  m_shape = std::unique_ptr<BasicShape>(new Cube(origin, spanMat[0]));
668  break;
669  }
670 
671  if(m_setInfLimits){
672  m_shape.get()->setInfLimits(m_inflimits_temp[0],0);
673  m_shape.get()->setInfLimits(m_inflimits_temp[1],1);
674  m_shape.get()->setInfLimits(m_inflimits_temp[2],2);
675  m_setInfLimits = false;
676  }
677 
678  if(m_setRefSys){
679  m_shape.get()->setRefSystem(m_refsystem_temp);
680  m_setRefSys = false;
681  }
682 
683  m_isBuild = false;
684 }
