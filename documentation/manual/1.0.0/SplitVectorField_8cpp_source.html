131 SplitVectorField::plotOptionalResults(){
132  if(isEmpty()) return;
133  if(m_originals.size() != m_result.size()) return;
134 
135  bitpit::VTKLocation loc = bitpit::VTKLocation::POINT;
136  if(m_mapVertDivision.empty()) loc = bitpit::VTKLocation::CELL;
137 
138 
139  int counter=0;
140  for(auto & geo : m_originals){
141 
142  bitpit::VTKElementType cellType = geo->desumeElement();
143  dvecarr3E points = geo->getVertexCoords();
144 
145  if (cellType == bitpit::VTKElementType::UNDEFINED) continue;
146 
147  if(cellType != bitpit::VTKElementType::VERTEX){
148  ivector2D connectivity = geo->getCompactConnectivity();
149  bitpit::VTKUnstructuredGrid output(".",m_name+std::to_string(getClassCounter()),cellType);
150  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
151  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
152  output.setDimensions(connectivity.size(), points.size());
153  output.addData("field", bitpit::VTKFieldType::VECTOR, loc, m_result[counter]);
154  output.setCounter(counter);
155  output.setCodex(bitpit::VTKFormat::APPENDED);
156  output.write();
157  }else{
158  int size = points.size();
159  ivector1D connectivity(size);
160  for(int i=0; i<size; ++i) connectivity[i]=i;
161  bitpit::VTKUnstructuredGrid output(".",m_name+std::to_string(getClassCounter()), cellType);
162  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
163  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
164  output.setDimensions(connectivity.size(), points.size());
165  output.addData("field", bitpit::VTKFieldType::VECTOR, loc, m_result[counter]);
166  output.setCounter(counter);
167  output.setCodex(bitpit::VTKFormat::APPENDED);
168  output.write();
169  }
170 
171  ++counter;
172  }
173 }

179 bool
180 SplitVectorField::split(){
181  if(m_mapCellDivision.empty() && m_mapVertDivision.empty()) return false;
182  if(isEmpty()) return false;
183 
184  //check original field;
185  bool loc = m_mapVertDivision.empty(); //true by cells, false by vert.
186 
187  int nGeo = m_originals.size();
188  m_result.resize(nGeo);
189 
190  std::unordered_map<long, std::pair<int, long>> * mapp;
191  livector1D * mapIdTarget;
192  std::vector< std::map<long, int> * > invIdLoc(nGeo);
193 
194  if(loc) {
195  m_field.resize(getGeometry()->getNCells(), {{0.0,0.0,0.0}});
196  mapp = &m_mapCellDivision;
197  mapIdTarget = &(getGeometry()->getMapCell());
198  for(int i=0; i<nGeo; ++i){
199  invIdLoc[i] = &(m_originals[i]->getMapCellInv());
200  }
201  }
202  else {
203  m_field.resize(getGeometry()->getNVertex(), {{0.0,0.0,0.0}});
204  mapp = &m_mapVertDivision;
205  mapIdTarget = &(getGeometry()->getMapData());
206  for(int i=0; i<nGeo; ++i){
207  invIdLoc[i] = &(m_originals[i]->getMapDataInv());
208  }
209  }
210 
211  if(m_field.size() != mapp->size()) return false;
212 
213  //allocate memory for results;
214  for(int i=0; i<nGeo; ++i){
215 
216  if(loc) m_result[i].resize(m_originals[i]->getNCells(),{{0.0,0.0,0.0}});
217  else m_result[i].resize(m_originals[i]->getNVertex(),{{0.0,0.0,0.0}});
218  }
219 
220  int counter = 0;
221  long IDtarget;
222  std::pair<int,long> douple;
223  int locali;
224  for(auto & val: m_field){
225 
226  IDtarget = (*mapIdTarget)[counter];
227  if(mapp->count(IDtarget)> 0 ){
228  douple = (*mapp)[IDtarget];
229  if(douple.first > nGeo) continue;
230  if((invIdLoc[douple.first])->count(douple.second) > 0){
231  locali = (*invIdLoc[douple.first])[douple.second];
232  m_result[douple.first][locali] = val;
233  }
234  }
235  ++counter;
236  }
237 
238  return true;
239 }
