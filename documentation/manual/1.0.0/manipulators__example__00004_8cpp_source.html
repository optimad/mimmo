47 void test00004() {
48 
49  /* Creation of mimmo containers.
50  * Input and output MimmoGeometry are instantiated
51  * as two different objects (no loop in chain are permitted).
52  */
53  MimmoGeometry * mimmo0 = new MimmoGeometry();
54  mimmo0->setRead(true);
55  mimmo0->setReadDir("geodata");
56  mimmo0->setReadFileType(FileType::STL);
57  mimmo0->setReadFilename("sphere2");
58  mimmo0->setWrite(true);
59  mimmo0->setWriteDir(".");
60  mimmo0->setWriteFileType(FileType::STL);
61  mimmo0->setWriteFilename("manipulators_output_00004.0000");
62 
63  MimmoGeometry * mimmo1 = new MimmoGeometry();
64  mimmo1->setRead(false);
65  mimmo1->setWrite(true);
66  mimmo1->setWriteDir(".");
67  mimmo1->setWriteFileType(FileType::STL);
68  mimmo1->setWriteFilename("manipulators_output_00004.0001");
69 
70  /* Instantiation of a FFDobject with spherical shape.
71  * Setup of origin and span (radius and span angles) of sphere.
72  * Plot Optional results during execution active for FFD block.
73  */
74  FFDLattice* lattice = new FFDLattice();
75  darray3E origin = {0.0, 0.0,0.0};
76  darray3E span;
77  span[0]= 3.01;
78  span[1]= 2*M_PI;
79  span[2]= M_PI;
80 
81  /* Set number of nodes of the mesh (dim) and degree of nurbs functions (deg).
82  */
83  iarray3E dim, deg;
84  dim[0] = 30;
85  dim[1] = 30;
86  dim[2] = 30;
87  deg[0] = 2;
88  deg[1] = 2;
89  deg[2] = 2;
90 
91  lattice->setLattice(origin,span,ShapeType::SPHERE,dim, deg);
92 
93  /* Change reference system to work in local spherical coordinates.
94  * Set coordinates as CLAMPED (continuity in origins of angles).
95  */
96  lattice->setRefSystem(2, darray3E{0,1,0});
97  lattice->setCoordType(CoordType::CLAMPED, 2);
98  lattice->setPlotInExecution(true);
99 
100  /* Build mesh of lattice outside the execution chain
101  * to use it during setup the displacements.
102  */
103  lattice->build();
104 
105  /* Creation of Generic input block and set it with the
106  * displacements of the control nodes of the lattice.
107  * Use random values to set the displacements of the control nodes at a longitude
108  * angle smaller than PI and expansion on radius direction for nodes with longitude
109  * angle greater than PI.
110  */
111  int ndeg = lattice->getNNodes();
112  dvecarr3E displ(ndeg, darray3E{0,0,0});
113  time_t Time = time(NULL);
114  srand(Time);
115  for (int i=0; i<ndeg; i++){
116  int l1,l2,l3;
117  int index = lattice->accessGridFromDOF(i);
118  lattice->accessPointIndex(index,l1,l2,l3);
119  if(l1 > 0 && lattice->getLocalPoint(l1,l2,l3)[1] < M_PI){
120  displ[i][0] = 1.0*( (double) (rand()) / RAND_MAX );
121  }
122  if( (l1 > 0 && lattice->getLocalPoint(l1,l2,l3)[1] >= M_PI)
123  || lattice->getLocalPoint(l1,l2,l3)[1] == 0){
124  displ[i][0] = 1.25;
125  }
126 
127  }
128 
129  /* Set Generic input block with the
130  * displacements defined above.
131  */
132  GenericInput* input = new GenericInput();
133  input->setInput(displ);
134 
135  /* Set Generic output block to write the
136  * displacements defined above.
137  */
138  GenericOutput * output = new GenericOutput();
139  output->setFilename("manipulators_output_00004.csv");
140  output->setCSV(true);
141 
142  /* Create applier block.
143  * It applies the deformation displacements to the original input geometry.
144  */
145  Apply* applier = new Apply();
146 
147  /* Setup pin connections.
148  */
149  addPin(mimmo0, lattice, PortType::M_GEOM, PortType::M_GEOM);
150  addPin(mimmo0, applier, PortType::M_GEOM, PortType::M_GEOM);
151  addPin(input, lattice, PortType::M_DISPLS, PortType::M_DISPLS);
152  addPin(input, output, PortType::M_DISPLS, PortType::M_DISPLS);
153  addPin(lattice, applier, PortType::M_GDISPLS, PortType::M_GDISPLS);
154  addPin(applier, mimmo1, PortType::M_GEOM, PortType::M_GEOM);
155 
156  /* Setup execution chain.
157  * The object can be insert in the chain in random order.
158  * The chain object recover the correct order of execution from
159  * the pin connections.
160  */
161  Chain ch0;
162  ch0.addObject(input);
163  ch0.addObject(output);
164  ch0.addObject(applier);
165  ch0.addObject(lattice);
166  ch0.addObject(mimmo1);
167  ch0.addObject(mimmo0);
168 
169  /* Execution of chain.
170  * Use debug flag true to print out the execution steps.
171  */
172  ch0.exec(true);
173 
174  /* Clean up & exit;
175  */
176  delete lattice;
177  delete applier;
178  delete input;
179  delete output;
180  delete mimmo0;
181  delete mimmo1;
182 
183  lattice = NULL;
184  applier = NULL;
185  input = NULL;
186  output = NULL;
187  mimmo0 = NULL;
188  mimmo1 = NULL;
189 
190  return;
191 }
