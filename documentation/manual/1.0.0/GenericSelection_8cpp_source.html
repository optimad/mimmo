158 livector1D
159 GenericSelection::constrainedBoundary(){
160 
161  if(getGeometry()->isEmpty() || getPatch()->isEmpty()) return livector1D(0);
162  livector1D sonV, fatherV;
163 
164  sonV = getPatch()->extractBoundaryVertexID();
165  bitpit::PiercedVector<bitpit::Cell> & existentCells = getPatch()->getCells();
166 
167  /*Create dual subpatch.*/
168  std::unique_ptr<MimmoObject> temp(new MimmoObject(m_topo));
169  bitpit::PatchKernel * tri = getGeometry()->getPatch();
170 
171  bitpit::PiercedVector<bitpit::Vertex> & mapV = temp->getPatch()->getVertices();
172 
173  livector1D TT;
174 
175  long idV,idC;
176  int sizeCC;
177  int counter=0;
178  bitpit::ElementInfo::Type eltype;
179 
180  for(auto && cell : tri->getCells()){
181 
182  idC = cell.getId();
183  if (!(existentCells.exists(idC))){
184  eltype = cell.getType();
185  sizeCC = cell.getVertexCount();
186  TT.resize(sizeCC);
187 
188  for(int i=0; i<sizeCC; ++i){
189  idV = cell.getVertex(i);
190  TT[i] = idV;
191  if(!mapV.exists(idV)) temp->addVertex(tri->getVertexCoords(idV),idV);
192  }
193  temp->addConnectedCell(TT,eltype,idC);
194  TT.clear();
195  counter++;
196  }
197  }
198 
199  if(counter == 0) return sonV;
200 
201  livector1D dualpV = temp->extractBoundaryVertexID();
202  if(dualpV.empty()) return sonV;
203 
204  std::sort(sonV.begin(), sonV.end());
205  std::sort(dualpV.begin(), dualpV.end());
206 
207  livector1D::iterator itS, itD;
208  livector1D matching(sonV.size());
209 
210  counter = 0;
211  itS = sonV.begin();
212  itD = dualpV.begin();
213 
214  while (itS != sonV.end()) {
215  if (itD != dualpV.end()){
216  long valS = *itS;
217  long valD = *itD;
218 
219  if(valD == valS){
220  matching[counter] = valS;
221  ++counter;
222  ++itS;
223  ++itD;
224  }else if(valD > valS){
225  ++itS;
226  }else{
227  ++itD;
228  }
229  }else{
230  itS = sonV.end();
231  }
232  }
233  matching.resize(counter);
234  return matching;
235 };

242 void
243 GenericSelection::execute(){
244 
245  if(getGeometry()->isEmpty()) return;
246 
247  m_subpatch.reset(nullptr);
248 
249  livector1D extracted = extractSelection();
250 
251  if(extracted.empty()) return;
252 
253  /*Create subpatch.*/
254  std::unique_ptr<MimmoObject> temp(new MimmoObject(m_topo));
255  bitpit::PatchKernel * tri = getGeometry()->getPatch();
256 
257  bitpit::PiercedVector<bitpit::Vertex> & mapV = temp->getPatch()->getVertices();
258 
259  livector1D TT;
260 
261  long idV;
262  int sizeCC;
263  bitpit::ElementInfo::Type eltype;
264  short PID;
265  if (m_topo != 3){
266  for(auto && idCell : extracted){
267 
268  bitpit::Cell & cell = tri->getCell(idCell);
269  eltype = cell.getType();
270  sizeCC = cell.getVertexCount();
271  PID = (short)cell.getPID();
272  TT.resize(sizeCC);
273 
274  for(int i=0; i<sizeCC; ++i){
275  idV = cell.getVertex(i);
276  TT[i] = idV;
277 
278  if(!mapV.exists(idV)) temp->addVertex(tri->getVertexCoords(idV),idV);
279  }
280 
281  temp->addConnectedCell(TT,eltype, PID, idCell);
282  TT.clear();
283  }
284  }
285  else{
286  for(auto && idV : extracted){
287  temp->addVertex(tri->getVertexCoords(idV),idV);
288  }
289  }
290 
291 
292  m_subpatch = std::move(temp);
293  tri = NULL;
294 };

301 GenericSelection::plotOptionalResults(){
302  if(getPatch() == NULL) return;
303  if(getPatch()->isEmpty()) return;
304 
305  dvecarr3E points = getPatch()->getVertexCoords();
306  ivector2D connectivity;
307  bitpit::VTKElementType cellType;
308 
309  std::string dir = m_outputPlot;
310  std::string name = m_name + "_Patch";
311 
312 
313  if (getPatch()->getType() != 3){
314  connectivity = getPatch()->getCompactConnectivity();
315  }
316  else{
317  int np = points.size();
318  connectivity.resize(np);
319  for (int i=0; i<np; i++){
320  connectivity[i].resize(1);
321  connectivity[i][0] = i;
322 
323  }
324  }
325  cellType = getPatch()->desumeElement();
326 
327 
328  bitpit::VTKUnstructuredGrid output(dir,name,cellType);
329  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
330  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
331  output.setDimensions(connectivity.size(), points.size());
332 
333  auto pids = getPatch()->getCompactPID();
334  if(pids.size() > 0) output.addData("PID", bitpit::VTKFieldType::SCALAR, bitpit::VTKLocation::CELL, pids);
335 
336  output.setCounter(getClassCounter());
337  output.setCodex(bitpit::VTKFormat::APPENDED);
338 
339  output.write();
340 }
