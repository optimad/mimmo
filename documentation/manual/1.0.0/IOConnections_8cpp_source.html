102 IOConnections_MIMMO::absorbConnections(const bitpit::Config & slotXML, bool debug){
103 
104  if(slotXML.getSectionCount() == 0) {
105  if(debug) std::cout<<"IOConnections::absorbConnections does not found any connection to read in current XML slot."<<std::endl;
106  return;
107  }
108 
109  for( auto & sect : slotXML.getSections()){
110 
111  std::string snd_str;
112  std::string rcv_str;
113  std::string sndP_str;
114  std::string rcvP_str;
115 
116  if(sect.second->hasOption("sender")) snd_str = sect.second->get("sender");
117  if(sect.second->hasOption("receiver")) rcv_str = sect.second->get("receiver");
118  if(sect.second->hasOption("senderPort")) sndP_str = sect.second->get("senderPort");
119  if(sect.second->hasOption("receiverPort")) rcvP_str = sect.second->get("receiverPort");
120 
121  snd_str = bitpit::utils::trim(snd_str);
122  rcv_str = bitpit::utils::trim(rcv_str);
123  sndP_str = bitpit::utils::trim(sndP_str);
124  rcvP_str = bitpit::utils::trim(rcvP_str);
125 
126 
127  auto itSend = m_mapConn.find(snd_str);
128  auto itRece = m_mapConn.find(rcv_str);
129  auto itSPort = m_mapPorts.find(sndP_str);
130  auto itRPort = m_mapPorts.find(rcvP_str);
131 
132  if(itSend == m_mapConn.end() || itRece == m_mapConn.end() || itSPort == m_mapPorts.end() || itRPort == m_mapPorts.end()){
133  if(debug) std::cout<<"---------------------------------------------"<<std::endl;
134  if(debug) std::cout<<"sender: "<<snd_str<<std::endl;
135  if(debug) std::cout<<"receiver: "<<rcv_str<<std::endl;
136  if(debug) std::cout<<"senderPort: "<<sndP_str<<std::endl;
137  if(debug) std::cout<<"receiverPort: "<<rcvP_str<<std::endl;
138  if(debug) std::cout<<"---------------------------------------------"<<std::endl;
139  if(debug) std::cout<<""<<std::endl;
140  if(debug) std::cout<<"IOConnections::absorbConnections does not found any sender/receiver and/or port compatible with API."<<std::endl;
141  if(debug) std::cout<<"Please control connection data entries in your XML file"<<std::endl;
142  if(debug) std::cout<<""<<std::endl;
143  continue;
144  }
145 
146  bool check = pin::addPin(itSend->second, itRece->second, itSPort->second, itRPort->second);
147 
148  if(debug && !check){
149  std::cout<<"---------------------------------------------"<<std::endl;
150  std::cout<<"sender: "<<snd_str<<std::endl;
151  std::cout<<"receiver: "<<rcv_str<<std::endl;
152  std::cout<<"senderPort: "<<sndP_str<<std::endl;
153  std::cout<<"receiverPort: "<<rcvP_str<<std::endl;
154  std::cout<<"---------------------------------------------"<<std::endl;
155  std::cout<<""<<std::endl;
156  std::cout<<"IOConnections::absorbConnections failed creating connection."<<std::endl;
157  std::cout<<""<<std::endl;
158  }
159  if(debug && check) std::cout<<"IOConnections::absorbConnections successfully created connection."<<std::endl;
160  }
161 
162  return;
163 };

175 void
176 IOConnections_MIMMO::flushConnections(bitpit::Config & slotXML, bool debug ){
177 
178  int counterGlob = 0;
179  for( auto object : m_mapConn){
180 
181  std::string sender = m_invMapConn[object.second]; //recover sender name
182  auto mapPOut = (object.second)->getPortsOut(); //get its slot of output port
183 
184 
185  for(auto & pOut : mapPOut ){
186 
187  std::string senderPort = m_invMapPorts[pOut.first]; //gets the Port Type of sender
188 
189  std::vector<BaseManipulation * > links = (pOut.second)->getLink(); //get all possible links to this specific port
190  std::vector<short int> lids = (pOut.second)->getPortLink();
191 
192  int counter = 0;
193  for( auto & val : links){
194  std::string receiver = m_invMapConn[val];
195  std::string receiverPort = m_invMapPorts[lids[counter]];
196 
197  std::string title = "conn" + std::to_string(counter+counterGlob);
198  bitpit::Config::Section & conn = slotXML.addSection(title);
199  conn.set("sender", sender);
200  conn.set("senderPort", senderPort);
201  conn.set("receiver", receiver);
202  conn.set("receiverPort", receiverPort);
203 
204  ++counter;
205  }//loop on possible links on the same port
206  counterGlob += counter;
207  }//loop on all the port of the single object
208  }//loop on all instantiated object.
209 
210  if(debug){
211  std::cout<<"IOConnections::flushConnections wrote "<< slotXML.getSectionCount()<< " active connections."<<std::endl;
212  }
213  return;
214 };
