315 ivector1D
316 MRBF::checkDuplicatedNodes(double tol){
317  ivector1D marked;
318  int sizeEff = getTotalNodesCount();
319  if( sizeEff == 0 ) return marked;
320 
321  bvector1D check(sizeEff, false);
322 
323  for(int i=0; i<sizeEff; ++i){
324  for(int j=i+1; j<sizeEff; ++j){
325  double dist = norm2(m_node[j] - m_node[i]);
326  if(!check[j] && dist <= tol){
327  marked.push_back(i);
328  check[j] = true;
329  }
330  }
331  }
332  return(marked);
333 }

469 void
470 MRBF::execute(){
471 
472  MimmoObject * container = getGeometry();
473  if(container->isEmpty() ) return;
474 
475  int size = 0;
476  int sizeF = getDataCount();
477 
478  for (int i=0; i<sizeF; i++){
479 
480  if(m_solver == MRBFSol::NONE) size = m_weight[i].size();
481  else size = m_value[i].size();
482 
483  if(size != getTotalNodesCount()){
484  std::cout << "mimmo : WARNING : " << getName() << " has displacements of " << i << " field with size (" << size << ") that does not fit number of RBF nodes ("<< getTotalNodesCount() << ")" << std::endl;
485  fitDataToNodes(i);
486  }
487  }
488 
489  double bboxDiag;
490  {
491  darray3E pmin, pmax;
492  container->getPatch()->getBoundingBox(pmin, pmax);
493  bboxDiag= norm2(pmax - pmin);
494  }
495 
496  //Checking supportRadius.
497  double distance = 0.0;
498  if(m_SRRatio <=0.0){ //get maximum weight/value displ and assign support radius a 3 times this value.
499 
500  for(int i=0; i<size; ++i){
501 
502  dvector1D data(sizeF);
503 
504  for(int j=0; j<sizeF; ++j){
505  if(m_solver == MRBFSol::NONE) data[j] = m_weight[j][i];
506  else data[j] = m_value[j][i];
507  }
508 
509  distance = std::max(distance, norm2(data));
510  }
511 
512  distance *=3.0;
513 
514  }else{
515  if (m_supRIsValue){
516  distance = m_SRRatio;
517  }
518  else{
519  distance = m_SRRatio * bboxDiag;
520  }
521  }
522 
523  //TODO remove it (I can't use a support radius 0....why?)
524  if(distance <=1.E-18){ //checkSupportRadius if too small, set it to the semidiagonal value of the geometry AABB
525  distance = 0.5*bboxDiag;
526  }
527 
528  const double radius = distance;
529  RBF::setSupportRadius(radius);
530 
531  if (m_solver == MRBFSol::WHOLE) solve();
532  if (m_solver == MRBFSol::GREEDY) greedy(m_tol);
533 
534  int nv = container->getNVertex();
535  dvecarr3E vertex = container->getVertexCoords();
536 
537  m_displ.resize(nv, darray3E{0,0,0});
538  dvector1D displ;
539  for(int i=0; i<nv; ++i){
540  displ = RBF::evalRBF(vertex[i]);
541  for (int j=0; j<3; j++) m_displ[i][j] = displ[j];
542  }
543 
544  //if m_filter is active;
545  if(m_bfilter){
546  m_filter.resize(nv,1.0);
547  int counter = 0;
548  for (auto && vec : m_displ){
549  vec = vec * m_filter[counter];
550  ++counter;
551  }
552  }

581 void
582 MRBF::plotCloud(std::string directory, std::string filename, int counterFile, bool binary, bool deformed){
583 
584  int nnodes = getTotalNodesCount();
585  nnodes = min(nnodes, int(m_displ.size()));
586  dvecarr3E* nodes_ = getNodes();
587  dvecarr3E nodes(nnodes);
588  dvecarr3E data(nnodes);
589  for(int i=0; i<nnodes; ++i){
590  for(int j=0; j<3; ++j){
591  if(m_solver == MRBFSol::NONE) data[i][j] = m_weight[j][i];
592  else data[i][j] = m_value[j][i];
593  }
594  }
595  if(deformed){
596  for(int i=0; i<nnodes; ++i){
597  nodes[i] = (*nodes_)[i] + data[i];
598  }
599  }else{
600  for(int i=0; i<nnodes; ++i){
601  nodes[i] = (*nodes_)[i];
602  }
603  }
604 
605  bitpit::VTKFormat codex = bitpit::VTKFormat::ASCII;
606  if(binary){codex=bitpit::VTKFormat::APPENDED;}
607 
608  ivector1D conn(nnodes);
609  {
610  int counter = 0;
611  for(auto & val: conn){
612  val = counter;
613  ++counter;
614  }
615  }
616  bitpit::VTKUnstructuredGrid vtk(directory, filename, bitpit::VTKElementType::VERTEX);
617  vtk.setGeomData( bitpit::VTKUnstructuredField::POINTS, nodes) ;
618  vtk.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, conn) ;
619  vtk.setDimensions(conn.size(), nnodes);
620  vtk.setCodex(codex);
621  if(counterFile>=0){vtk.setCounter(counterFile);}
622  vtk.write();
623 };

646 
647  if(slotXML.hasOption("RBFShape")){
648  input = slotXML.get("RBFShape");

664  std::stringstream ss(bitpit::utils::trim(input));
