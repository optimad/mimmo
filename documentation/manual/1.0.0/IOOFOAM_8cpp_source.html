306 bool
307 IOOFOAM::read(){
308 
309 
310  //Read OpenFOAM Points
311  {
312 
313  std::ifstream infile(m_rdirV+"/"+m_rfilenameV);
314  bool check = infile.good();
315  if (!check){
316  m_stopat = SHRT_MAX;
317  return false;
318  }
319  infile.close();
320 
321  dvecarr3E Ipoints;
322  readOFP(m_rdirV, m_rfilenameV, Ipoints);
323 
324  //Reverse info in your grids.
325  std::unique_ptr<MimmoObject> patchVol(new MimmoObject(3));
326 
327  int sizeV = Ipoints.size();
328  patchVol->getPatch()->reserveVertices(sizeV);
329 
330  // Stock vertices in volume grid
331  for(auto & vv : Ipoints) patchVol->addVertex(vv);
332 
333  //release the points cloud mesh
334  m_volmesh = std::move(patchVol);
335 
336  }
337 
338  //compute kdtree for points cloud
339  m_volmesh->buildKdTree();
340 
341 
342  //read vtk patches and assign ordered PID starting from 0
343  //TODO input PID for each patch has to be implemented.
344 
345  std::unique_ptr<MimmoObject> patchBnd(new MimmoObject(1));
346 
347  short nPID = m_rfilenameS.size();
348  bool check;
349  for (short iPID = 0; iPID < nPID; iPID++){
350  check = readVTK(m_rdirS[iPID], m_rfilenameS[iPID], iPID, patchBnd.get());
351  if (!check) return check;
352  }
353 
354  if (m_normalize && m_maxf > 0.0){
355  for (auto vertex : patchBnd->getVertices()){
356  m_field[patchBnd->getMapDataInv(vertex.getId())] /= m_maxf;
357  m_field[patchBnd->getMapDataInv(vertex.getId())] *= m_scaling;
358  }
359  }
360  else{
361  for (auto vertex : patchBnd->getVertices()){
362  m_field[patchBnd->getMapDataInv(vertex.getId())] *= m_scaling;
363  }
364  }
365 
366  //release the surface mesh
367  m_surfmesh = std::move(patchBnd);
368 
369  return true;
370 
371 }

409 void IOOFOAM::readOFP(string& inputDir, string& pointsName, dvecarr3E& points){
410 
411  ifstream is(inputDir +"/"+pointsName);
412 
413  points.clear();
414  int ip = 0;
415  int np;
416  darray3E point;
417  string sread;
418  char par;
419 
420  for (int i=0; i<18; i++){
421  getline(is,sread);
422  }
423  is >> np;
424  getline(is,sread);
425  getline(is,sread);
426 
427  points.resize(np);
428  while(!is.eof() && ip<np){
429  is.get(par);
430  for (int i=0; i<3; i++) is >> point[i];
431  is.get(par);
432  getline(is,sread);
433  points[ip] = point;
434  ip++;
435  }
436  is.close();
437  return;

522 bool IOOFOAM::readVTK(string& inputDir, string& surfaceName, short PID, MimmoObject* patchBnd){
523 
524 
525  std::ifstream infile(inputDir+"/"+surfaceName+".vtk");
526  bool check = infile.good();
527  if (!check) return false;
528  infile.close();
529 
530 
531  string inputFilename = inputDir+"/"+surfaceName+".vtk";
532  //int np = 0;
533  int nt = 0;
534  darray3E point;
535  bitpit::Vertex vertex;
536 
537  //mapper for connectivity
538  map<vtkIdType, long> mapID;
539  map<vtkIdType, long> mapid;
540 
541  // Get all data from the file
542  vtkSmartPointer<vtkGenericDataObjectReader> reader =
543  vtkSmartPointer<vtkGenericDataObjectReader>::New();
544  reader->SetFileName(inputFilename.c_str());
545  reader->Update();
546 
547  // All of the standard data types can be checked and obtained like this:
548  if(reader->IsFilePolyData())
549  {
550  vtkPolyData* output = reader->GetPolyDataOutput();
551 
552  vtkSmartPointer<vtkTriangleFilter> tri= vtkSmartPointer<vtkTriangleFilter>::New();
553  tri->SetInputData(output);
554  tri->Update();
555 
556  vtkPolyData* output2 = tri->GetOutput();
557 
558  vtkSmartPointer<vtkPoints> points = output2->GetPoints();
559  vtkCellArray *cells = output2->GetPolys();
560  //vtkCellData *cdata = output2->GetCellData();
561  vtkPointData *pdata = output2->GetPointData();
562 
563 
564  bitpit::KdTree<3, bitpit::Vertex, long> * kdtree = m_volmesh->getKdTree();
565  long ID;
566 
567  double point_[3], tol = 1.0e-08;
568  livector1D ids;
569  livector1D noids;
570  bool check;
571 
572  for (vtkIdType id=0; id<points->GetNumberOfPoints(); id++ ){
573  points->GetPoint(id, point_);
574  for (int i=0; i<3; i++) point[i] = point_[i];
575  vertex.setCoords(point);
576  check = false;
577  while(!check){
578  ids.clear();
579  kdtree->hNeighbors(&vertex, tol, &ids, &noids);
580  if (ids.size() == 0){
581  tol *= 1.5;
582  }
583  else if (ids.size() > 1){
584  tol /= 1.5;
585  }
586  else{
587  ID = ids[0];
588  check = true;
589  }
590  }
591 
592  if (!(patchBnd->getVertices().exists(ID))){
593  patchBnd->addVertex(point, ID);
594  }
595  mapID[id] = ID;
596  mapid[id] = patchBnd->getMapDataInv(ID);
597  }
598 
599  vtkIdType* conn_;
600  vtkIdType npts;
601  nt = cells->GetNumberOfCells();
602  livector1D connectivity(3);
603  bitpit::ElementInfo::Type eltype = bitpit::ElementInfo::TRIANGLE;
604  cells->InitTraversal();
605  for (vtkIdType id=0; id<nt; id++ ){
606  cells->GetNextCell(npts, conn_);
607  for (int i=0; i<npts; i++) connectivity[i] = mapID[conn_[i]];
608  patchBnd->addConnectedCell(connectivity, eltype, PID);
609  }
610 
611  patchBnd->cleanGeometry();
612 
613 
614  //IOOFOAM reads only a scalar field on each patch if present (if not add 0 values)
615  //If m_normalize is active the data are normalized with the maximum value on the patches
616  vtkDataArray* data = pdata->GetArray(0);
617  m_field.resize(patchBnd->getNVertex());
618  if (data != NULL){
619  for (vtkIdType id=0; id<points->GetNumberOfPoints(); id++){
620  m_field[mapid[id]] = (data->GetComponent(id,0));
621  m_maxf = max(m_maxf, abs(m_field[mapid[id]]));
622  }
623  }
624  else{
625  for (vtkIdType id=0; id<points->GetNumberOfPoints(); id++){
626  m_field[mapid[id]] = 0.0;
627  }
628  }
629  }
630  else{
631  std::cout << "mimmo : ERROR : polydata not found in : "<< inputFilename << std::endl;
632  m_stopat = PID;
633  return false;
634  }
635 
636  return true;
637 }
