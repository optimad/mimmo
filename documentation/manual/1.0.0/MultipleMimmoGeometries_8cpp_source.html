224 MultipleMimmoGeometries::getObjMAP(){
225 
226  std::unordered_map<std::string, std::pair<int, MimmoObject*> > objMap;
227 
228  std::string fullpath;

233  info = m_winfo;

236  int totVal = info.size();

238  if(m_isInternal) totGeo = m_intgeo.size();

240  totVal = std::min(totVal,totGeo);
241 
242  for(int i=0; i<totVal; ++i){
243 
244  std::string tag = "vtu";
245  switch(info[i].ftype){
246  case 0: tag="stl"; break;
247  case 5: tag="nas"; break;
248  case 6: tag=""; break;
249  default: break;
250  }
251 
252  fullpath = info[i].fdir+"/"+info[i].fname+"."+tag;
253 
254  if(m_isInternal) objMap[fullpath] = std::make_pair(info[i].ftype, m_intgeo[i].get());
255  else objMap[fullpath] = std::make_pair(info[i].ftype, m_extgeo[i]);
256  }
257  return objMap;

597 bool
598 MultipleMimmoGeometries::write(){
599  if (m_extgeo.empty() || m_winfo.empty()) return false;
600 
601  int totGeo = m_extgeo.size();
602  int totInfo = m_winfo.size();
603 
604  if(totGeo != totInfo) {
605  std::cout<<"WARNING:Not enough output info or some invalid geoemetries found writing in class "<<m_name<<std::endl;
606  }
607 
608  totGeo = std::min(totGeo, totInfo);
609 
610  for(int k=0; k<totGeo; ++k){
611 
612  std::unique_ptr<MimmoGeometry> writer(new MimmoGeometry());
613  writer->setWrite(true);
614  writer->setWriteDir(m_winfo[k].fdir);
615  writer->setWriteFilename(m_winfo[k].fname);
616  writer->setWriteFileType(m_winfo[k].ftype);
617  writer->setCodex(m_codex);
618  writer->setFormatNAS(m_wformat);
619  writer->setGeometry(m_extgeo[k]);
620  writer->execute();
621  }
622  return true;
623 };

629 bool
630 MultipleMimmoGeometries::read(){
631  if(!m_isInternal || m_rinfo.empty()) return false;
632 
633  setGeometry();
634 
635  //absorbing Geometries.
636  int counter = 0;
637  m_intgeo.resize(m_rinfo.size());
638  for(auto & data: m_rinfo){
639  std::unique_ptr<MimmoGeometry> geo(new MimmoGeometry());
640  std::unique_ptr<MimmoObject> subData(new MimmoObject());
641  geo->setRead(true);
642  geo->setReadDir(data.fdir);
643  geo->setReadFilename(data.fname);
644  geo->setReadFileType(data.ftype);
645  geo->execute();
646 
647  subData->setHARDCopy(geo->getGeometry());
648  m_intgeo[counter] = std::move(subData);
649  counter++;
650  }
651  if(counter == 0) return false;
652  m_intgeo.resize(counter);
653 
654  for(auto & val: m_intgeo){
655  val->cleanGeometry();
656  if(m_buildBvTree) val->buildBvTree();
657  if(m_buildKdTree) val->buildKdTree();
658  }
659  return true;
660 };
