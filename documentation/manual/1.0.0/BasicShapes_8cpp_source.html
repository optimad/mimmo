125

143 void BasicShape::setRefSystem(int label, darray3E axis){
144 
145  if(label <0 || label >2 ) return;
146 
147  m_sdr[label] = axis/norm2(axis);
148  dvecarr3E point_mat(3,darray3E{0,0,0});
149  point_mat[0][0] = point_mat[1][1]= point_mat[2][2]=1.0;
150 
151  int next_label = (label + 1)%3;
152  int fin_label = (label + 2)%3;
153 
154  double pj = dotProduct(point_mat[next_label], m_sdr[label]);
155  m_sdr[next_label] = point_mat[next_label] - pj*m_sdr[label];
156  m_sdr[next_label] = m_sdr[next_label]/norm2(m_sdr[next_label]);
157 
158  m_sdr[fin_label] = crossProduct(m_sdr[label],m_sdr[next_label]);
159  m_sdr[fin_label] = m_sdr[fin_label]/norm2(m_sdr[fin_label]);
160 }

568 bool BasicShape::containShapeAABBox(darray3E bMin,darray3E bMax){
569  bool check = true;
570  dvecarr3E points(8);
571  points[0] = bMin;
572  points[1] = bMax;
573  points[2] = bMax; points[2][2] = bMin[2];
574  points[3] = bMin; points[3][2] = bMax[2];

617 void BasicShape::searchKdTreeMatches(bitpit::KdTree<3,bitpit::Vertex,long> & tree, int indexKdNode, int level, livector1D & result, int &counter ){
618 
619  //check indexKdNode admissible.
620  if(indexKdNode <0 || indexKdNode > tree.n_nodes) return;
621 
622  //1st step get data
623  bitpit::KdNode<bitpit::Vertex, long> & target = tree.nodes[indexKdNode];
624  //check target point is in the shape
625  if (bitpit::CGElem::intersectPointBox(target.object_->getCoords(), m_bbox[0], m_bbox[1], 3)){
626  if(isPointIncluded(target.object_->getCoords())){
627  result[counter] = target.label;
628  ++counter;
629  }
630  searchKdTreeMatches(tree, target.lchild_, level+1, result, counter);
631  searchKdTreeMatches(tree, target.rchild_, level+1, result, counter);
632  }else{
633 
634  uint32_t check = intersectShapePlane(level%3, target.object_->getCoords());
635  bool right = (check > 0);
636  bool left = (check%2 == 0);
637  if (left && target.lchild_ >= 0){
638  searchKdTreeMatches(tree, target.lchild_, level+1, result, counter);
639  }
640  if (right && target.rchild_ >= 0){
641  searchKdTreeMatches(tree, target.rchild_, level+1, result, counter);
642  }
643  }
644 
645  return;
646 };

658 void BasicShape::searchBvTreeMatches(mimmo::BvTree & tree, bitpit::PatchKernel * geo, int indexBvNode, livector1D & result, int &counter ){
659 
660  //check indexBvNode admissible.
661  if(indexBvNode <0 || indexBvNode > tree.m_nnodes) return;
662 
663  //1st step get data and control if it's leaf or if shape contain bounding box
664  if(containShapeAABBox(tree.m_nodes[indexBvNode].m_minPoint, tree.m_nodes[indexBvNode].m_maxPoint)){
665 
666  for(int i = tree.m_nodes[indexBvNode].m_element[0]; i<tree.m_nodes[indexBvNode].m_element[1]; ++i){
667  result[counter] = tree.m_elements[i].m_label;
668  ++counter;
669  }
670  return;
671  };
672 
673  if(tree.m_nodes[indexBvNode].m_leaf){
674  for(int i = tree.m_nodes[indexBvNode].m_element[0]; i<tree.m_nodes[indexBvNode].m_element[1]; ++i){
675  if(isSimplexIncluded(geo, tree.m_elements[i].m_label)){
676  result[counter] = tree.m_elements[i].m_label;
677  ++counter;
678  }
679  }
680  return;
681  }
682 
683  if(tree.m_nodes[indexBvNode].m_lchild >= 0) {
684  if( intersectShapeAABBox(tree.m_nodes[tree.m_nodes[indexBvNode].m_lchild].m_minPoint, tree.m_nodes[tree.m_nodes[indexBvNode].m_lchild].m_maxPoint) )
685  searchBvTreeMatches(tree, geo, tree.m_nodes[indexBvNode].m_lchild, result, counter);
686  }
687 
688  if(tree.m_nodes[indexBvNode].m_rchild >= 0) {
689  if( intersectShapeAABBox(tree.m_nodes[tree.m_nodes[indexBvNode].m_rchild].m_minPoint, tree.m_nodes[tree.m_nodes[indexBvNode].m_rchild].m_maxPoint) )
690  searchBvTreeMatches(tree, geo, tree.m_nodes[indexBvNode].m_rchild, result,counter);
691  }
692 
693  return;
694 };

827 darray3E Cube::toWorldCoord(darray3E point){
828 
829  darray3E work, work2;
830  //unscale your local point
831  for(int i =0; i<3; ++i){
832  work[i] = point[i]*m_scaling[i];
833  }
834 
835  //return to local xyz system
836  // -> cube, doing nothing
837 
838  //unapply change to local sdr transformation
839  linearalgebra::matmul(work, m_sdr, work2);
840 
841  //unapply origin translation
842  work = work2 + m_origin;
843  return(work);
844 };

852 darray3E Cube::toLocalCoord(darray3E point){
853 
854  darray3E work, work2;
855 
856  //unapply origin translation
857  work = point - m_origin;
858 
859  //apply change to local sdr transformation
860  dmatrix33E transp = linearalgebra::transpose(m_sdr);
861  linearalgebra::matmul(work, transp, work2);
862 
863  //get to proper local system
864  // -> cube, doing nothing
865 
866  //scale your local point
867  for(int i =0; i<3; ++i){
868  work[i] = work2[i]/m_scaling[i];
869  }
870 
871  return(work);
872
