174 void
175 CGNSPidExtractor::execute(){
176 
177  livector1D extracted;
178 
179  for(auto & val: m_targetpid){
180  livector1D temp= getGeometry()->extractPIDCells(val);
181  extracted.insert(extracted.end(), temp.begin(), temp.end());
182  }
183  std::unique_ptr<MimmoObject> patchTemp(new MimmoObject(1));
184 
185 
186  MimmoObject * mother = getGeometry();
187 
188  if(extracted.empty()){
189  patchTemp->setHARDCopy(getGeometry());
190 
191  }else{
192 
193  for(auto &val: extracted){
194 
195  livector1D conn = mother->getCellConnectivity(val);
196  bitpit::ElementInfo::Type eletype = mother->getPatch()->getCell(val).getType();
197  patchTemp->addConnectedCell(conn, eletype, val);
198  }
199 
200  {
201  darray3E temp;
202  livector2D tempConn = patchTemp->getConnectivity();
203  std::map<long,long> ordIndex;
204 
205  for(auto & val2: tempConn){
206  for (auto val22 : val2){
207  ordIndex[val22] = val22;
208  }
209  }
210 
211  for(auto & val3: ordIndex){
212  temp = mother->getVertexCoords(val3.second);
213  patchTemp->addVertex(temp,val3.second);
214  }
215  }
216 
217  }
218 
219  //check if patch is forced to be triangulated
220  if(m_force){
221 
222  long maxID, newID;
223 
224  auto orderedCellID = patchTemp->getCells().getIds(true);
225  maxID = orderedCellID[(int)orderedCellID.size()-1];
226  newID = maxID+1;
227  bitpit::ElementInfo::Type eletype, eletri = bitpit::ElementInfo::Type::TRIANGLE;
228 
229  for(auto &idcell : orderedCellID){
230 
231  livector1D conn = patchTemp->getCellConnectivity(idcell);
232  eletype = mother->getPatch()->getCell(idcell).getType();
233 
234  if(eletype == bitpit::ElementInfo::Type::QUAD){
235  //create new triangle connectivity
236  livector1D conn1(3), conn2(3);
237  conn1[0] = conn[0];
238  conn1[1] = conn[1];
239  conn1[2] = conn[2];
240 
241  conn2[0] = conn[0];
242  conn2[1] = conn[2];
243  conn2[2] = conn[3];
244 
245  patchTemp->getPatch()->deleteCell(idcell);
246  patchTemp->addConnectedCell(conn1, eletri, idcell);
247  patchTemp->addConnectedCell(conn2, eletri, newID);
248  ++newID;
249  }
250  }
251  }
252 
253  m_patch = std::move(patchTemp);
254  mother=NULL;
255 };

275 
276  BITPIT_UNUSED(name);
277 
278  std::string input;
279 
280  if(slotXML.hasOption("Priority")){

288  };
289 
290  std::vector<short> temp;
291  if(slotXML.hasOption("nPID")){
292  input = slotXML.get("nPID");
293  int value = 0;
294  if(!input.empty()){
295  std::stringstream ss(bitpit::utils::trim(input));
296  ss >> value;
297  value = std::max(0, value);
298  }
299  temp.resize(value, -1);
300  };
301 
302  if(slotXML.hasOption("PID") && !temp.empty()){
303  input = slotXML.get("PID");
304  if(!input.empty()){
305  std::stringstream ss(bitpit::utils::trim(input));
306  for(auto & val: temp){
307  ss >> val;
308  }
309  }
310  setPID(temp);
311  };
312 
313  if(slotXML.hasOption("ForcedToTriangulate")){
314  input = slotXML.get("ForcedToTriangulate");
315  input = bitpit::utils::trim(input);
316  bool value = false;
317  if(!input.empty()){
318  std::stringstream ss(input);
319  ss >> value;
320  }
321  setForcedToTriangulate(value);
322  };
323 
324  if(slotXML.hasOption("PlotInExecution")){
325  std::string input = slotXML.get("PlotInExecution");
326  input = bitpit::utils::trim(input);
327  bool value = false;
328  if(!input.empty()){
329  std::stringstream ss(input);
330  ss >> value;
331  }
332  setPlotInExecution(value);
333  }
334 
335  if(slotXML.hasOption("OutputPlot")){
336  std::string input = slotXML.get("OutputPlot");
337  input = bitpit::utils::trim(input);
338  std::string temp = ".";
339  if(!input.empty()) setOutputPlot(input);
