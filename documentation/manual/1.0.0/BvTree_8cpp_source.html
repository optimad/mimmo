239 void BvTree::buildTree()
240 {
241  if (m_patch_ == NULL) return;
242 
243  if (m_nelements == 0 || m_nodes.size() == 0) setup();
244  if (m_nelements == 0) return;
245 
246  int iel = 0;
247  long id;
248 
249  //fill ids and centroid
250  for ( auto & cell : m_patch_->getCells() )
251  {
252  id = cell.getId();
253  m_elements[iel].m_label = id;
254  m_elements[iel].m_centroid = m_patch_->evalCellCentroid(id);
255  iel++;
256  }
257 
258  //node root
259  BvNode node;
260  node.m_element[0] = 0;
261  node.m_element[1] = m_nelements;
262  node.m_nrange = m_nelements;
263  m_nodes[m_nnodes] = node;
264  if ( m_nelements == 1 )
265  {
266  m_nodes[m_nnodes].m_leaf = true;
267  m_nleaf++;
268  m_nnodes++;
269  }
270  else
271  {
272  m_nnodes++;
273  fillTree(0);
274  }
275  decreaseStack();

282 void BvTree::fillTree(int iparent)
283 {
284 
285  int j;
286 
287  //compute Bounding Box of node
288  computeBoundingBox(iparent);
289 
290 
291  if ( m_nodes[iparent].m_nrange > m_maxsize )
292  {
293 
294  //split parent by mean of centroid in node
295  darray3E meanC = computeMeanPoint(m_nodes[iparent].m_element[0], m_nodes[iparent].m_element[1]);
296 
297  //split by plane normal to maximum dimension of the centroid box
298  darray3E minPC{{1.0e+18, 1.0e+18, 1.0e+18}};
299  darray3E maxPC{{-1.0e+18, -1.0e+18, -1.0e+18}};
300  int ic, ics = m_nodes[iparent].m_element[0], ice = m_nodes[iparent].m_element[1];
301  for ( ic=ics; ic<ice; ++ic )
302  {
303  for ( j=0; j<m_dim; ++j )
304  {
305  minPC[j] = std::min(minPC[j], m_elements[ic].m_centroid[j]);
306  maxPC[j] = std::max(maxPC[j], m_elements[ic].m_centroid[j]);
307  }
308  }
309 
310  int dir = 0;
311  double maxRange = maxPC[dir] - minPC[dir];
312  for ( int idim=1; idim<m_dim; ++idim )
313  {
314  if ((maxPC[idim] - minPC[idim]) > maxRange)
315  {
316  dir = idim;
317  maxRange = maxPC[dir] - minPC[dir];
318  }
319  }
320 
321  //Sort elements by coordinate dir.
322  std::vector<BvElement>::iterator itstart = m_elements.begin()+m_nodes[iparent].m_element[0];
323  std::vector<BvElement>::iterator itend = m_elements.begin()+(m_nodes[iparent].m_element[1]);
324 
325  //Old algorithm
326  // sort( itstart, itend, compareElements(dir) );
327  // //find first element with coords[dir] > of meanC[dir]
328  // int firstRight = findFirstGreater(iparent, meanC, dir);
329 
330 
331  int firstRight = pseudoSort( itstart, itend, meanC, dir );
332 
333  //insert lchild
334  if ( firstRight > m_nodes[iparent].m_element[0] )
335  {
336  m_nodes[iparent].m_lchild = m_nnodes;
337  if ( (int)m_nodes.size() <= m_nnodes+1) increaseStack();
338  BvNode node;
339  node.m_element[0] = m_nodes[iparent].m_element[0];
340  node.m_element[1] = firstRight;
341  node.m_nrange = node.m_element[1] - node.m_element[0];
342  m_nodes[m_nnodes] = node;
343  if ( node.m_nrange == 1 )
344  {
345  m_nodes[m_nnodes].m_leaf = true;
346  computeBoundingBox(m_nnodes);
347  m_nnodes++;
348  m_nleaf++;
349  }
350  else
351  {
352  m_nnodes++;
353  fillTree(m_nnodes-1);
354  }
355  }
356 
357  //insert rchild
358  if ( firstRight < m_nodes[iparent].m_element[1] )
359  {
360  m_nodes[iparent].m_rchild = m_nnodes;
361  if ( (int)m_nodes.size() <= m_nnodes+1) increaseStack();
362  BvNode node;
363  node.m_element[0] = firstRight;
364  node.m_element[1] = m_nodes[iparent].m_element[1];
365  node.m_nrange = node.m_element[1] - node.m_element[0];
366  m_nodes[m_nnodes] = node;
367  if (node.m_nrange == 1){
368  m_nodes[m_nnodes].m_leaf = true;
369  computeBoundingBox(m_nnodes);
370  m_nnodes++;
371  m_nleaf++;
372  }
373  else{
374  m_nnodes++;
375  fillTree(m_nnodes-1);
376  }
377  }
378 
379  }
380  else
381  {
382  //insert this node as leaf
383  m_nodes[iparent].m_leaf = true;
384  m_nleaf++;
385 
386  }
387

397 darray3E BvTree::computeMeanPoint(int istart, int iend)
398 {
399 
400  long id;
401  darray3E meanC({{0.0, 0.0, 0.0}});
402  for ( int i=istart; i<iend; ++i )
403  {
404  id = m_elements[i].m_label;

406  }
407  meanC /= double(iend-istart);

419 int BvTree::pseudoSort(std::vector<BvElement>::iterator itstart,
420  std::vector<BvElement>::iterator itend, darray3E meanC, int dir){
421 
422  double thres = meanC[dir];
423 
424  itend--;
425  while(itstart != itend){
426 
427  while (itstart->m_centroid[dir] <= thres && itstart != itend){
428  itstart++;
429  }
430  while (itend->m_centroid[dir] > thres && itend != itstart){
431  itend--;
432  }
433 
434  if (itstart != itend){
435  std::iter_swap(itstart, itend);
436  }
437  }
438 
439  return (std::max(0, int(distance(m_elements.begin(), itend))));
440 }

453 int BvTree::findFirstGreater(int inode, darray3E meanC, int dir)
454 {
455  double mean = meanC[dir];
456  int guess = (m_nodes[inode].m_element[1] - m_nodes[inode].m_element[0])/2;
457  int step = guess;
458 
459  while( step > 0 )
460  {
461  step /= 2;
462  guess += step * int(sign(mean - m_elements[guess].m_centroid[dir]));
463  guess = std::max(m_nodes[inode].m_element[0], guess);
464  guess = std::min(m_nodes[inode].m_element[1]-1, guess);
465  }
466 
467  while ( mean < m_elements[guess].m_centroid[dir] && guess > m_nodes[inode].m_element[0] )
468  {
469  guess--;
470  }
471  while ( mean > m_elements[guess].m_centroid[dir] && guess < m_nodes[inode].m_element[1]-1 )
472  {
473  guess++;
474  }
475  return guess;
476 }

487 void BvTree::computeBoundingBox(int inode)
488 {
489  BvNode &node(m_nodes[inode]);
490 
491  int is = node.m_element[0];
492  int ie = node.m_element[1];
493 
494  int i, j, iV, nV;
495 
496  node.m_minPoint = {{1.0e+18, 1.0e+18, 1.0e+18}};
497  node.m_maxPoint = {{-1.0e+18, -1.0e+18, -1.0e+18}};
498 
499  bitpit::Cell cell;
500  darray3E coords;
501 
502  for ( i=is; i<ie; i++ )
503  {
504  cell = m_patch_->getCell(m_elements[i].m_label);
505  nV = cell.getVertexCount();
506  for ( iV=0; iV<nV; ++iV )
507  {
508  coords = m_patch_->getVertexCoords(cell.getVertex(iV));
509  for ( j=0; j<m_dim; ++j )
510  {
511  node.m_minPoint[j] = std::min(node.m_minPoint[j], coords[j]);
512  node.m_maxPoint[j] = std::max(node.m_maxPoint[j], coords[j]);
513  }
514  }
515  }
516  node.m_minPoint -= m_tol;
517  node.m_maxPoint += m_tol;
518 }

552 bool BvTree::SphereBoundingBox(darray3E *P_, BvNode *node_, double r)
553 {
554 
555  darray3E closestP;
556  for ( int i=0; i<m_dim; i++ )
557  {
558  closestP[i] = std::min(std::max((*P_)[i], node_->m_minPoint[i]), node_->m_maxPoint[i]);
559  }
560  double dist = dotProduct(closestP-(*P_), closestP-(*P_));
561 
562  return(dist < r*r);
563 }
