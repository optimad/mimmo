165 void
166 TwistGeometry::execute(){
167 
168  if (getGeometry() == NULL) return;
169 
170  int nV = m_geometry->getNVertex();
171  m_displ.resize(nV);
172  m_filter.resize(nV, 1.0);
173 
174 
175  darray3E point, rotated;
176  long ID;
177  int idx;
178  liimap mapID = m_geometry->getMapDataInv();
179  darray3E projected;
180  double distance;
181  //double sign;
182  double rot;
183 
184  for (auto vertex : m_geometry->getVertices()){
185  point = vertex.getCoords();
186  ID = vertex.getId();
187  idx = mapID[ID];
188 
189  //signed distance from origin
190  distance = dotProduct((point-m_origin),m_direction);
191 
192  //compute coefficients and constant vectors of rodriguez formula
193  rot = std::min(m_alpha, (std::abs(distance)/m_distance)*m_alpha);
194  if (distance < 0) rot = -rot;
195  double a = cos(rot);
196  darray3E b = (1 - cos(rot)) * m_direction;
197  double c = sin(rot);
198 
199  //project point on axis (local origin)
200  projected = distance*m_direction;
201 
202 
203  point -= projected;
204  //rodrigues formula
205  rotated = a * point +
206  b * dotProduct(m_direction, point) +
207  c * crossProduct(m_direction, point);
208 
209  rotated += projected;
210  point += projected;
211 
212  m_displ[idx] = (rotated-point)*m_filter[idx];
213 
214  }
