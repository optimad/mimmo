156 void
157 RotationGeometry::execute(){
158 
159  if (getGeometry() == NULL) return;
160 
161  int nV = m_geometry->getNVertex();
162  m_displ.resize(nV);
163  m_filter.resize(nV, 1.0);
164 
165  //compute coefficients and constant vectors of rodriguez formula
166  double a = cos(m_alpha);
167  darray3E b = (1 - cos(m_alpha)) * m_direction;
168  double c = sin(m_alpha);
169 
170  darray3E point, rotated;
171  long ID;
172  int idx;
173  liimap mapID = m_geometry->getMapDataInv();
174 
175  for (auto vertex : m_geometry->getVertices()){
176  point = vertex.getCoords();
177  ID = vertex.getId();
178  idx = mapID[ID];
179 
180  point -= m_origin;
181  //rodrigues formula
182  rotated = a * point +
183  b * dotProduct(m_direction, point) +
184  c * crossProduct(m_direction, point);
185 
186  rotated += m_origin;
187  point += m_origin;
188 
189  m_displ[idx] = (rotated-point)*m_filter[idx];
190 
191  }
192 
193 };
