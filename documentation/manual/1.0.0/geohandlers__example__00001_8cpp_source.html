50 void test00001() {
51 
52  /* Creation of mimmo containers.
53  * Input and output MimmoGeometry are instantiated
54  * as two different objects (no loop in chain are permitted).
55  */
56  MimmoGeometry * mimmo0 = new MimmoGeometry();
57  mimmo0->setRead(true);
58  mimmo0->setReadDir("geodata");
59  mimmo0->setReadFileType(FileType::STL);
60  mimmo0->setReadFilename("sphere2");
61  mimmo0->setWrite(true);
62  mimmo0->setWriteDir("./");
63  mimmo0->setWriteFileType(FileType::STL);
64  mimmo0->setWriteFilename("geohandlers_output_00001.0000");
65 
66  MimmoGeometry * mimmo1 = new MimmoGeometry();
67  mimmo1->setRead(false);
68  mimmo1->setWrite(true);
69  mimmo1->setWriteDir(".");
70  mimmo1->setWriteFileType(FileType::STL);
71  mimmo1->setWriteFilename("geohandlers_output_00001.0001");
72 
73  MimmoGeometry * mimmo2 = new MimmoGeometry();
74  mimmo2->setRead(false);
75  mimmo2->setWrite(true);
76  mimmo2->setWriteDir(".");
77  mimmo2->setWriteFileType(FileType::STL);
78  mimmo2->setWriteFilename("geohandlers_output_00001.0002");
79 
80  /* Instantiation of a Selection By Box block.
81  * Setup of span and origin of cube.
82  */
83  SelectionByBox * boxSel = new SelectionByBox();
84  boxSel->setOrigin({{-0.5,-0.5,0.2}});
85  boxSel->setSpan(0.6,0.6,0.6);
86 
87  /* Instantiation of a Selection By Sphere block.
88  * Setup of span and origin of sphere.
89  */
90  SelectionBySphere * sphSel = new SelectionBySphere();
91  sphSel->setOrigin({{-0.5, 0.5,0.2}});
92  sphSel->setSpan(0.34, 2*M_PI, M_PI);
93 
94  /* Setup pin connections.
95  */
96  pin::addPin(mimmo0, boxSel, PortType::M_GEOM, PortType::M_GEOM);
97  pin::addPin(mimmo0, sphSel, PortType::M_GEOM, PortType::M_GEOM);
98  pin::addPin(boxSel, mimmo1, PortType::M_GEOM, PortType::M_GEOM);
99  pin::addPin(sphSel, mimmo2, PortType::M_GEOM, PortType::M_GEOM);
100 
101  /* Setup execution chain.
102  */
103  Chain ch0;
104  ch0.addObject(mimmo0);
105  ch0.addObject(boxSel);
106  ch0.addObject(sphSel);
107  ch0.addObject(mimmo1);
108  ch0.addObject(mimmo2);
109 
110  /* Execution of chain.
111  * Use debug flag true to to print out the execution steps.
112  */
113  ch0.exec(true);
114 
115  /* Clean up & exit;
116  */
117  delete boxSel;
118  delete sphSel;
119  delete mimmo0;
120  delete mimmo1;
121  delete mimmo2;
122 
123  boxSel = NULL;
124  sphSel = NULL;
125  mimmo0 = NULL;
126  mimmo1 = NULL;
127  mimmo2 = NULL;
128 
129  return;
130 }
