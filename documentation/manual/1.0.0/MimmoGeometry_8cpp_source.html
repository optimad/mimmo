572  }

576  //Export Triangulation Surface VTU
577  {
578  dvecarr3E points = getGeometry()->getVertexCoords();
579  ivector2D connectivity = getGeometry()->getCompactConnectivity();
580  shivector1D pids = getGeometry()->getCompactPID();
581  bitpit::VTKUnstructuredGrid vtk(m_winfo.fdir, m_winfo.fname, bitpit::VTKElementType::TRIANGLE);
582  vtk.setGeomData( bitpit::VTKUnstructuredField::POINTS, points) ;
583  vtk.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity) ;
584  vtk.setDimensions(connectivity.size(), points.size());
585  if(pids.size() > 0) vtk.addData("PID", bitpit::VTKFieldType::SCALAR, bitpit::VTKLocation::CELL, pids);
586  if(!m_codex) vtk.setCodex(bitpit::VTKFormat::ASCII);
587  else vtk.setCodex(bitpit::VTKFormat::APPENDED);
588  vtk.write() ;
589  return true;
590  }
591  break;

725 bool
726 MimmoGeometry::read(){
727 
728  if(!m_isInternal) return false;
729 
730  switch(FileType::_from_integral(m_rinfo.ftype)){
731 
732  //Import STL
733  case FileType::STL :
734  {
735  setGeometry(1);
736  string name;
737 
738  {
739  std::ifstream infile(m_rinfo.fdir+"/"+m_rinfo.fname+".stl");
740  bool check = infile.good();
741  name = m_rinfo.fdir+"/"+m_rinfo.fname+".stl";
742  if (!check){
743  infile.open(m_rinfo.fdir+m_rinfo.fname+".STL");
744  check = infile.good();
745  name = m_rinfo.fdir+"/"+m_rinfo.fname+".STL";
746  if (!check) return false;
747  }
748  }
749 
750  ifstream in(name);
751  string ss, sstype;
752  getline(in,ss);
753  stringstream ins;
754  ins << ss;
755  ins >> sstype;
756  bool binary = true;
757  if (sstype == "solid" || sstype == "SOLID") binary = false;
758  in.close();
759 
760  dynamic_cast<SurfUnstructured*>(getGeometry()->getPatch())->importSTL(name, binary);
761  getGeometry()->cleanGeometry();
762 
763  //count PID if multi-solid
764  auto & map = getGeometry()->getPIDTypeList();
765  for(auto & cell : getGeometry()->getCells() ){
766  map.insert(cell.getPID());
767  }
768  }
769  break;
770 
771  case FileType::STVTU :
772  //Import Triangulation Surface VTU
773  {
774  std::ifstream infile(m_rinfo.fdir+"/"+m_rinfo.fname+".vtu");
775  bool check = infile.good();
776  if (!check) return false;
777 
778  dvecarr3E Ipoints ;
779  ivector2D Iconnectivity ;
780  shivector1D pids;
781 
782  bitpit::VTKUnstructuredGrid vtk(m_rinfo.fdir, m_rinfo.fname, bitpit::VTKElementType::TRIANGLE);
783  vtk.setGeomData( bitpit::VTKUnstructuredField::POINTS, Ipoints) ;
784  vtk.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, Iconnectivity) ;
785  vtk.addData("PID", pids);
786 
787  vtk.read() ;
788 
789  bitpit::ElementInfo::Type eltype = bitpit::ElementInfo::TRIANGLE;
790 
791  setGeometry(1);
792 
793  int sizeV, sizeC, sizeP;
794  sizeV = Ipoints.size();
795  sizeC = Iconnectivity.size();
796  sizeP = pids.size();
797  m_intgeo->getPatch()->reserveVertices(sizeV);
798  m_intgeo->getPatch()->reserveCells(sizeC);
799 
800  for(auto & vv : Ipoints) m_intgeo->addVertex(vv);
801 
802  int ccell = 0;
803  for(auto & cc : Iconnectivity) {
804  livector1D temp(cc.size());
805  int counter = 0;
806  for(auto && val : cc){
807  temp[counter] = val;
808  ++counter;
809  }
810  if(sizeP == 0){
811  m_intgeo->addConnectedCell(temp, eltype);
812  }else{
813  m_intgeo->addConnectedCell(temp, eltype, pids[ccell]);
814  }
815  ccell++;
816  }
817 
818  m_intgeo->cleanGeometry();
819  }
820  break;
821 
822  case FileType::SQVTU :
823  //Import Quadrilateral Surface VTU
824  {
825 
826  std::ifstream infile(m_rinfo.fdir+"/"+m_rinfo.fname+".vtu");
827  bool check = infile.good();
828  if (!check) return false;
829 
830  dvecarr3E Ipoints ;
831  ivector2D Iconnectivity ;
832  shivector1D pids;
833 
834  bitpit::VTKUnstructuredGrid vtk(m_rinfo.fdir, m_rinfo.fname, bitpit::VTKElementType::QUAD);
835  vtk.setGeomData( bitpit::VTKUnstructuredField::POINTS, Ipoints) ;
836  vtk.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, Iconnectivity) ;
837  vtk.addData("PID", pids);
838  vtk.read() ;
839 
840  bitpit::ElementInfo::Type eltype = bitpit::ElementInfo::QUAD;
841 
842  setGeometry(1);
843 
844  int sizeV, sizeC, sizeP;
845  sizeV = Ipoints.size();
