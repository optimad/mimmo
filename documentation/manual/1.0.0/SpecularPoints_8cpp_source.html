270 void
271 SpecularPoints::execute(){
272 
273  /* If an implicit definition is not present it has to be computed
274  * by using origin and normal.
275  */
276  if (!m_implicit) setPlane(m_origin, m_normal);
277 
278  darray3E norm;
279  double offset;
280  for(int i=0; i<3; ++i) norm[i] = m_plane[i];
281  offset = m_plane[3];
282  double normPlane = norm2(norm);
283  if(normPlane < 1.E-18 || m_points.empty()) return;
284  norm /= normPlane;
285  bool project =!(getGeometry() == NULL || getGeometry()->isEmpty());
286 
287 
288  //choosing margin for plane offset
289  double margin = 1.e-12;
290  if(project && getGeometry()->getType() == 1){
291  double aTot = 0.0;
292  int cellSize = getGeometry()->getNCells();
293  bitpit::SurfaceKernel * tri = static_cast<bitpit::SurfaceKernel * >(getGeometry()->getPatch());
294  for(auto &cell: tri->getCells()){
295  aTot += tri->evalCellArea(cell.getId());
296  }
297 
298  if(aTot > 0.0) {margin = 1.2*pow(aTot/((double)cellSize),0.5);}
299 
300  }else if(project && getGeometry()->getType() == 2){
301  double vTot = 0.0;
302  int cellSize = getGeometry()->getNCells();
303  bitpit::VolumeKernel * tetra = static_cast<bitpit::VolumeKernel * >(getGeometry()->getPatch());
304  for(auto &cell: tetra->getCells()){
305  vTot += tetra->evalCellVolume(cell.getId());
306  }
307 
308  if(vTot > 0.0) {margin = 1.2*pow(vTot/((double)cellSize),0.5);}
309  }
310 
311 
312  double sig = (1.0 - 2.0*((int)m_insideout));
313  double distance;
314 
315  //mirroring.
316  m_scalar.resize(m_points.size());
317  m_vector.resize(m_points.size());
318 
319  int counterProj = m_points.size();
320  int counterData = 0;
321 
322  m_proj = m_points;
323  m_scalarMirrored = m_scalar;
324  m_vectorMirrored = m_vector;
325 
326  m_proj.resize(2*counterProj);
327  m_scalarMirrored.resize(2*counterProj);
328  m_vectorMirrored.resize(2*counterProj);
329 
330  for(auto &val: m_points){
331  distance = sig*(dotProduct(norm, val) + offset);
332  if(distance > margin || m_force){
333  m_proj[counterProj] = val - 2.0*distance*sig*norm;
334  m_scalarMirrored[counterProj] = m_scalar[counterData];
335  m_vectorMirrored[counterProj] = m_vector[counterData] -2.0*dotProduct(m_vector[counterData], sig*norm)*sig*norm;
336  counterProj++;
337  }
338  counterData++;
339  }
340 
341  m_proj.resize(counterProj);
342  m_scalarMirrored.resize(counterProj);
343  m_vectorMirrored.resize(counterProj);
344 
345  if(project){
346  if(!getGeometry()->isBvTreeBuilt()) getGeometry()->buildBvTree();
347 
348  //project points on surface.
349  int counter = 0;
350  for(auto &val : m_proj){
351  m_proj[counter]= bvTreeUtils::projectPoint(&val, getGeometry()->getBvTree());
352  ++counter;
353  }
354  }
355 };

466 SpecularPoints::flushSectionXML(bitpit::Config::Section & slotXML, std::string name){
467 
468  BITPIT_UNUSED(name);
469 
470  slotXML.set("ClassName", m_name);
471  slotXML.set("Priority", std::to_string(getPriority()));
472 
473  int value = m_force;
474  slotXML.set("Force", std::to_string(value));
475 
476  value = m_insideout;
477  slotXML.set("InsideOut", std::to_string(value));
478 
479 
480  {
481  darray4E org = getPlane();
482  darray3E normal;
483  darray3E point = {{0.0,0.0,0.0}};
484  int imax = -1;
485  double dum = 0.0;
486  for(int i=0; i<3; ++i) {
487  normal[i] =org[i];
488  if(abs(normal[i]) > dum) {
489  imax = i;
490  dum = abs(normal[i]);
491  }
492  }
493  if(imax != -1) point[imax] = -1.0*org[3]/normal[imax];
494 
495  std::stringstream ss1, ss2;
496  ss1<<std::scientific<<point[0]<<'\t'<<point[1]<<'\t'<<point[2];
497  ss2<<std::scientific<<normal[0]<<'\t'<<normal[1]<<'\t'<<normal[2];
498 
499  bitpit::Config::Section & planeXML = slotXML.addSection("Plane");
500  planeXML.set("Point",ss1.str());
501  planeXML.set("Normal",ss2.str());
502  }
503 
504  if(isPlotInExecution()){
505  slotXML.set("PlotInExecution", std::to_string(1));
506  }
507 
508  if(m_outputPlot != "."){
509  slotXML.set("OutputPlot", m_outputPlot);
510  }
