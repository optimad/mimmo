142 void BasicShape::setRefSystem(darray3E axis0, darray3E axis1, darray3E axis2){
143 
144  double a = norm2(axis0);
145  double b = norm2(axis1);
146  double c = norm2(axis2);
147 
148  bool check = ( a>0.0 && b >0.0 && c > 0.0);
149  if (! check){
150  assert(false && "one or more 0-vector passed as arguments in BasicShape::SetRefSystem method");
151  return;
152  }
153 
154  axis0 /= a;
155  axis1 /= b;
156  axis2 /= c;
157 
158  double tol = 1.0e-12;
159  check = std::abs(dotProduct(axis0,axis1)) + std::abs(dotProduct(axis1,axis2)) + std::abs(dotProduct(axis0,axis2));
160  if(check > tol) {
161  assert(false && "not orthogonal axes passed as arguments in BasicShape::SetRefSystem method");
162  return;
163  }
164 
165  m_sdr[0] = axis0;
166  m_sdr[1] = axis1;
167  m_sdr[2] = axis2;
168 }

175 void BasicShape::setRefSystem(int label, darray3E axis){
176 
177  if(label <0 || label >2 ){
178  assert(false && "no correct label passed as argument in BasicShape::setRefSystem");
179  return;
180  }
181 
182  double a = norm2(axis);
183  if (!(a > 0.0) ){
184  assert(false && "0-vector passed as axis argument in BasicShape::SetRefSystem method");
185  return;
186  }
187 
188  m_sdr[label] = axis/a;
189  dvecarr3E point_mat(3,darray3E{0,0,0});
190  point_mat[0][0] = point_mat[1][1]= point_mat[2][2]=1.0;
191 
192  int next_label = (label + 1)%3;
193  int fin_label = (label + 2)%3;
194 
195  double pj = dotProduct(point_mat[next_label], m_sdr[label]);
196  m_sdr[next_label] = point_mat[next_label] - pj*m_sdr[label];
197  m_sdr[next_label] = m_sdr[next_label]/norm2(m_sdr[next_label]);
198 
199  m_sdr[fin_label] = crossProduct(m_sdr[label],m_sdr[next_label]);
200  m_sdr[fin_label] = m_sdr[fin_label]/norm2(m_sdr[fin_label]);
201 }

647 void BasicShape::searchKdTreeMatches(bitpit::KdTree<3,bitpit::Vertex,long> & tree, livector1D & result ){
648 
649  //1st step get data
650  std::vector<int> candidates;
651  std::vector< std::pair<int, int> > nodeStackLoc; //cointains touple with kdNode id and level label.
652 
653  nodeStackLoc.push_back(std::make_pair(0, 0));
654  while(!nodeStackLoc.empty()){
655  std::pair<int, int> toupleId = nodeStackLoc.back();
656  bitpit::KdNode<bitpit::Vertex, long> & target = tree.nodes[toupleId.first];
657  nodeStackLoc.pop_back();
658 
659  //add children to the stack, if node has any of them.
660  uint32_t check = intersectShapePlane((toupleId.second)%3, target.object_->getCoords());
661  if((check%2 == 0) && (target.lchild_ >= 0)){
662  nodeStackLoc.push_back(std::make_pair(target.lchild_, (toupleId.second) + 1));
663  }
664  if((check > 0) && (target.rchild_ >= 0)){
665  nodeStackLoc.push_back(std::make_pair(target.rchild_, (toupleId.second) + 1));
666  }
667 
668  //push node as candidate if is in the raw bounding box of the shape
669  if (bitpit::CGElem::intersectPointBox(target.object_->getCoords(), m_bbox[0], m_bbox[1], 3)){
670  candidates.push_back(toupleId.first);
671  }
672  }
673 
674  result.clear();
675  result.reserve(candidates.size());
676  for (const auto & idCand : candidates){
677  bitpit::KdNode<bitpit::Vertex, long> & target = tree.nodes[idCand];
678  if(isPointIncluded(target.object_->getCoords())){
679  result.push_back(target.label);
680  }
681  }
682 };

692 void BasicShape::searchBvTreeMatches(bitpit::PatchSkdTree & tree, bitpit::PatchKernel * geo, livector1D & result){
693 
694  std::size_t rootId = 0;
695 
696  std::vector<std::size_t> toBeCandidates;
697  std::vector<long> sureCells;
698  sureCells.reserve((geo->getCellCount()));
699 
700  std::vector<size_t> nodeStack;
701  nodeStack.push_back(rootId);
702  while(!nodeStack.empty()){
703  std::size_t nodeId = nodeStack.back();
704  const SkdNode & node = tree.getNode(nodeId);
705  nodeStack.pop_back();
706 
707  //first step: check if node AABB is completely inside the shape.
708  //if it is, save all its cells and continue.
709  if(containShapeAABBox(node.getBoxMin(), node.getBoxMax())){
710  std::vector<long> cellids = node.getCells();
711  sureCells.insert(sureCells.end(),cellids.begin(), cellids.end());
712  continue;
713  };
714 
715  //second step: if the current node AABB does not intersect or does not completely contains the Shape, then thrown it away and continue.
716  if(!intersectShapeAABBox(node.getBoxMin(), node.getBoxMax()) ){
717  continue;
718  }
719 
720  //now the only option is to visit node's children. If node is not leaf, add children to the stack,
721  //otherwise add current node as a possible candidate in shape inclusion.
722  bool isLeaf = true;
723  for (int i = SkdNode::CHILD_BEGIN; i != SkdNode::CHILD_END; ++i) {
724  SkdNode::ChildLocation childLocation = static_cast<SkdNode::ChildLocation>(i);
725  std::size_t childId = node.getChildId(childLocation);
726  if (childId != SkdNode::NULL_ID) {
727  isLeaf = false;
728  nodeStack.push_back(childId);
729  }
730  }
731 
732  if (isLeaf) {
733  toBeCandidates.push_back(nodeId);
734  }
735  }
736 
737  result.clear();
738  result.reserve(toBeCandidates.size() + sureCells.size());
739  for (const auto & idCand : toBeCandidates){
740  const SkdNode &node = tree.getNode(idCand);
741  std::vector<long> cellids = node.getCells();
742  for(const auto & id : cellids){
743  if(isSimplexIncluded(geo, id)){
744  result.push_back(id);
745  }
746  }
747  }
748  result.insert(result.end(), sureCells.begin(), sureCells.end());
749 };

814 darray3E Cube::toWorldCoord(darray3E point){
815 
816  darray3E work, work2;
817  //unscale your local point
818  for(int i =0; i<3; ++i){
819  work[i] = point[i]*m_scaling[i];
820  }
821 
822  //return to local xyz system
823  // -> cube, doing nothing
824 
825  //unapply change to local sdr transformation
826  linearalgebra::matmul(work, m_sdr, work2);
827 
828  //unapply origin translation
829  work = work2 + m_origin;
830  return(work);
831 };

839 darray3E Cube::toLocalCoord(darray3E point){
840 
841  darray3E work, work2;
842 
843  //unapply origin translation
844  work = point - m_origin;
845 
846  //apply change to local sdr transformation
847  dmatrix33E transp = linearalgebra::transpose(m_sdr);

851  // -> cube, doing nothing
852 
853  //scale your local point
854  for(int i =0; i<3; ++i){
855  work[i] = work2[i]/m_scaling[i];
856  }
857

934 void Cube::getTempBBox(){
935 
936  m_bbox.clear();
937  m_bbox.resize(2);
938  m_bbox[0].fill(1.e18);
939  m_bbox[1].fill(-1.e18);
940 
941  dvecarr3E locals(8,darray3E{{0.0,0.0,0.0}});
942  darray3E temp;
943  locals[1].fill(1.0);
944  locals[2][0]= locals[2][1]=1.0;
945  locals[3][2]=1.0;
946  locals[4][0]= 1.0;
947  locals[5][1]= locals[5][2] = 1.0;
948  locals[6][1] = 1.0;
949  locals[7][0] = locals[7][2] = 1.0;
950 
951  for(auto &vv : locals){
