163 void
164 SwitchVectorField::plotOptionalResults(){
165 
166  if (m_result.size() == 0 || getGeometry() == NULL) return;
167  if (getGeometry()->isEmpty()) return;
168 
169  bitpit::VTKLocation loc = bitpit::VTKLocation::UNDEFINED;
170  switch(m_loc){
171  case MPVLocation::POINT :
172  loc = bitpit::VTKLocation::POINT;
173  break;
174  case MPVLocation::CELL :
175  loc = bitpit::VTKLocation::CELL;
176  break;
177  default:
178  (*m_log)<<"Warning: Undefined Reference Location in plotOptionalResults of "<<m_name<<std::endl;
179  (*m_log)<<"Interface locations are not supported in VTU writing." <<std::endl;
180  break;
181  }
182 
183  if(loc == bitpit::VTKLocation::UNDEFINED) return;
184 
185  //check size of field and adjust missing values to zero for writing purposes only.
186  dmpvecarr3E field_supp = m_result;
187  if(!field_supp.completeMissingData({{0.0,0.0,0.0}})) return;
188  dvecarr3E field = field_supp.getDataAsVector();
189 
190  if(getGeometry()->getType() != 3){
191  getGeometry()->getPatch()->getVTK().addData("field", bitpit::VTKFieldType::VECTOR, loc, field);
192  getGeometry()->getPatch()->write(m_outputPlot+"/"+m_name+std::to_string(getId()));
193  getGeometry()->getPatch()->getVTK().removeData("field");
194  }else{
195  if(loc == bitpit::VTKLocation::CELL){
196  (*m_log)<<"Warning: Attempt writing Cell data field on cloud point in plotOptionalResults of "<<m_name<<std::endl;
197  return;
198  }
199  liimap mapDataInv;
200  dvecarr3E points = getGeometry()->getVertexCoords(&mapDataInv);
201  int size = points.size();
202  ivector2D connectivity(size, ivector1D(1));
203  for(int i=0; i<size; ++i) connectivity[i][0]=i;
204  bitpit::VTKUnstructuredGrid output(m_outputPlot,m_name+std::to_string(getId()), bitpit::VTKElementType::VERTEX);
205  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
206  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
207  output.setDimensions(connectivity.size(), points.size());
208  output.addData("field", bitpit::VTKFieldType::VECTOR, loc, field);
209  output.setCodex(bitpit::VTKFormat::APPENDED);
210  output.write();
211  }

219 bool
220 SwitchVectorField::mswitch(){
221 
222  if (getGeometry() == NULL) return false;
223 
224  m_result.clear();
225 
226  //Extract by link to geometry
227  for (const auto & field : m_fields){
228  if (field.getGeometry() == getGeometry()){
229  m_result = field;
230  //geometry and location are copied from field.
231  return true;
232  }
233  }
234 
235  //Extract by geometric mapping if active and if no positive match is found.
236  if (m_mapping && getGeometry()->getType() != 3){
237 
238  m_result.setGeometry(getGeometry());
239  m_result.setDataLocation(m_loc);
240 
241  ExtractVectorField * ef = new ExtractVectorField();
242  ef->setGeometry(getGeometry());
243  ef->setMode(ExtractMode::MAPPING);
244  ef->setTolerance(m_tol);
245 
246  //create map for overlapping ids purpose;
247  std::unordered_map<long, int> idRepetition;
248 
249  for (const auto & field : m_fields){
250  ef->setField(field);
251  bool check = ef->extract();
252  if(!check) continue;
253 
254  auto temp = ef->getExtractedField();
255  dmpvecarr3E::iterator itB;
256  auto itE = temp.end();
257  long id;
258  for(itB = temp.begin(); itB != itE; ++itB){
259  id = itB.getId();
260  if(!m_result.exists(id)){
261  m_result.insert(id, *itB);
262  }else{
263  m_result[id] += *itB;
264  if(idRepetition.count(id)>0){
265  ++idRepetition[id];
266  }else{
267  idRepetition[id] = 2;
268  }
269  }
270  }
271  }
272 
273  //resolve overlapping ids by averaging correspondent value;
274 
275  for(auto &itval : idRepetition){
276  m_result[itval.first] /= double(itval.second);
277  }
278 
279  delete ef;
280  }
281 
282  if (m_result.isEmpty()) return false;
283  return true;
284 }
