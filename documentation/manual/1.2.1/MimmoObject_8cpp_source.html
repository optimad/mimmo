144 MimmoObject::MimmoObject(int type){

146  m_log = &bitpit::log::cout(MIMMO_LOG_FILE);
147  m_type = max(type,1);
148  if (m_type > 4){

150  throw std::runtime_error ("MimmoObject : unrecognized mesh type in class construction");

152  switch(m_type){
153  case 1:
154  m_patch = std::move(std::unique_ptr<PatchKernel>(new MimmoSurfUnstructured(2)));
155  m_skdTree = std::move(std::unique_ptr<PatchSkdTree>(new bitpit::SurfaceSkdTree(dynamic_cast<SurfaceKernel*>(m_patch.get()))));

161  m_kdTree = std::move(std::unique_ptr<KdTree<3,bitpit::Vertex,long> >(new KdTree<3,bitpit::Vertex, long>()));
162  break;
163  case 3:
164  m_patch = std::move(std::unique_ptr<PatchKernel>(new MimmoPointCloud()));
165  m_kdTree = std::move(std::unique_ptr<KdTree<3,bitpit::Vertex,long> >(new KdTree<3,bitpit::Vertex, long>()));

580 livector2D
581 MimmoObject::getCompactConnectivity(liimap & mapDataInv){
582 
583  livector2D connecti(getNCells());
584  int np, counter =0;
585 
586  for(auto const & cell : getCells()){
587  np = cell.getConnectSize();
588  const long * conn_ = cell.getConnect();
589  connecti[counter].resize(np);
590  bitpit::ElementType eltype = cell.getType();
591 
592  if(eltype == bitpit::ElementType::POLYGON){
593  connecti[counter][0] = conn_[0];
594  for (int i=1; i<np; ++i){
595  connecti[counter][i] = mapDataInv[conn_[i]];
596  }
597  }else if(eltype == bitpit::ElementType::POLYHEDRON){
598  connecti[counter][0] = conn_[0];
599  for(int nF = 0; nF < conn_[0]-1; ++nF){
600  int facePos = cell.getFaceStreamPosition(nF);
601  int beginVertexPos = facePos + 1;
602  int endVertexPos = facePos + 1 + conn_[facePos];
603  connecti[counter][facePos] = conn_[facePos];
604  for (int i=beginVertexPos; i<endVertexPos; ++i){
605  connecti[counter][i] = mapDataInv[conn_[i]];
606  }
607  }
608  }else{
609  for (int i=0; i<np; ++i){
610  connecti[counter][i] = mapDataInv[conn_[i]];
611  }
612  }
613  ++counter;
614  }
615  return connecti;
616 }
