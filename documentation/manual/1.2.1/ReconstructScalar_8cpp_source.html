190 ReconstructScalar::removeData(MimmoObject * patch){
191  //in m_subpatch remove progressively all pierced vector elements in last position
192  //which links towards a geometry of type patch.
193  while(m_subpatch.back().getGeometry() == patch){
194  m_subpatch.pop_back();
195  }
196 
197  //start searching from the begin on element linking patch
198  std::vector<dmpvector1D>::iterator it = m_subpatch.begin();
199 
200  while(it != m_subpatch.end()){
201  if (it->getGeometry() == patch){
202  *it = m_subpatch.back();
203  m_subpatch.pop_back();
204  //be always sure the last element is clean and does not link
205  // the patch
206  while(m_subpatch.back().getGeometry() == patch){
207  m_subpatch.pop_back();
208  }
209  }
210  ++it; //forward the iterator.
211  }

243  if(getGeometry() == NULL) return;

247  bitpit::VTKLocation loc = bitpit::VTKLocation::POINT;

252  dvector1D field = m_result.getDataAsVector();
253 
254  if (getGeometry()->getType() != 3){

262  dvecarr3E points = getGeometry()->getVertexCoords(&mapData);
263  ivector2D connectivity;

271  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
272  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
273  output.setDimensions(connectivity.size(), points.size());
274  output.addData("scalarfield", bitpit::VTKFieldType::SCALAR, loc, field);
275  output.setDataCodex(bitpit::VTKFormat::APPENDED);

336 void
337 ReconstructScalar::execute(){
338 
339  if(getGeometry() == NULL){
340  throw std::runtime_error(m_name + "NULL pointer to linked geometry found");
341  }
342 
343  if(getGeometry()->isEmpty()){
344  throw std::runtime_error(m_name + "empty linked geometry found");
345  }
346  //Overlap fields
347  m_result.clear();
348  m_result.setGeometry(getGeometry());
349  m_result.setDataLocation(m_loc);
350 
351  m_subresults.clear();
352 
353  std::unordered_set<long> idsTarget;
354  {
355  livector1D ids = idsGeoDataLocation(getGeometry());
356  idsTarget.insert(ids.begin(), ids.end());
357  }
358 
359  bitpit::PiercedVector<int> counter;
360  for (int i=0; i<getNData(); i++){
361  dmpvector1D* pv = &m_subpatch[i];
362  livector1D ids = idsGeoDataLocation(pv->getGeometry());
363  for (auto ID: ids){
364  if(idsTarget.count(ID)==0) continue;
365 
366  if (!m_result.exists(ID)){
367  m_result.insert(ID, (*pv)[ID]);
368  counter.insert(ID, 1);
369  }
370  else{
371  overlapFields(ID, (*pv)[ID]);
372  counter[ID]++;
373  }
374  }
375  }
376 
377  if (m_result.isEmpty()){
378  (*m_log)<<"Error in "<<m_name<<". Resulting reconstructed field is empty.This is could be caused by unrelated fields linked geometry and target geometry"<<std::endl;
379  throw std::runtime_error(m_name + "empty field reconstructed in class execution.");
380  }
381 
382  if (m_overlapCriterium == OverlapMethod::AVERAGE){
383  long int ID;
384  auto itend = m_result.end();
385  for (auto it=m_result.begin(); it!=itend; ++it){
386  ID = it.getId();
387  (*it) = (*it) / double(counter[ID]);
388  }
389  }
390 
391  //Update field on whole geometry
392  m_result.completeMissingData(0.0);
393 
394  //Create subresults
395  m_subresults.resize(getNData());
396  for (int i=0; i<getNData(); i++){
397  dmpvector1D* pv = &m_subpatch[i];
398  m_subresults[i].setGeometry(pv->getGeometry());
399  m_subresults[i].setDataLocation(m_loc);
400  livector1D ids = idsGeoDataLocation(pv->getGeometry());
401  for (auto ID : ids){
402  if(idsTarget.count(ID)>0){
403  m_subresults[i].insert(ID, m_result[ID]);
404  }
405  }
406  }
407 }

429 void
430 ReconstructScalar::overlapFields(long int ID, double & locField){
431 
432  switch(m_overlapCriterium){
433  case OverlapMethod::MAX :
434  {
435  double actual = m_result[ID];
436  double dummy = locField;
437 
438  if (actual < dummy)
439  m_result[ID] = locField;
440  }
441  break;
442  case OverlapMethod::MIN :
443  {
444  double actual = m_result[ID];
445  double dummy = locField;
446 
447  if (actual > dummy)
448  m_result[ID] = locField;
449  }
450  break;
451  case OverlapMethod::AVERAGE :
452  m_result[ID] += locField;
453  break;
454  case OverlapMethod::SUM :
455  m_result[ID] += locField;
456  break;
457  default : //never been reached
458  break;
459  }

488 ReconstructScalar::absorbSectionXML(const bitpit::Config::Section & slotXML, std::string name){
489 
490  BITPIT_UNUSED(name);
491 
492  if(slotXML.hasOption("DataLocation")){
493  std::string input = slotXML.get("DataLocation");
494  input = bitpit::utils::string::trim(input);
495  int temp = -1;
496  if(!input.empty()){
497  std::stringstream ss(input);
498  ss>>temp;
499  }
500  if(int(m_loc) != temp){
501  (*m_log)<<"Error absorbing DataLocation in "<<m_name<<". Class and read locations mismatch"<<std::endl;
502  if (temp == 0) (*m_log)<<"XML DataLocation in "<<m_name<<" is set to 0-UNDEFINED"<<std::endl;
503  if (temp == 3) (*m_log)<<"XML DataLocation in "<<m_name<<" is set to 3-INTERFACE, not supported for now."<<std::endl;
504  throw std::runtime_error (m_name + " : xml absorbing failed.");
505  }
506  }
507  //start absorbing
508  BaseManipulation::absorbSectionXML(slotXML, name);
509 
510  if(slotXML.hasOption("OverlapCriterium")){
511  std::string input = slotXML.get("OverlapCriterium");
512  input = bitpit::utils::string::trim(input);
513  int value = 1;
514  if(!input.empty()){
515  std::stringstream ss(input);
516  ss >> value;
517  value = std::min(std::max(1, value),4);
518  }
519  setOverlapCriterium(value);
520  }
