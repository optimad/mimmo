311 template<typename T>
312 MimmoPiercedVector< T >*
313 GenericInputMPVData::_getResult(){
314  MimmoPiercedVector< T > data;
315  if (getGeometry() == NULL) return NULL;
316 
317  int n_loc;
318  long nSize = 0, readNSize;
319  long id;
320  T data_T;
321 
322  if(m_csv) m_binary = false;
323  data.setGeometry(getGeometry());
324 
325  std::fstream file;
326  file.open(m_dir+"/"+m_filename, std::fstream::in);
327  if (file.is_open()){
328  if (m_binary){
329  bitpit::genericIO::absorbBINARY(file, n_loc);
330  bitpit::genericIO::absorbBINARY(file, nSize);
331  data.reserve(nSize);
332  for(int i=0; i<nSize; ++i){
333  bitpit::genericIO::absorbBINARY(file, id);
334  bitpit::genericIO::absorbBINARY(file, data_T);
335  data.insert(id,data_T);
336  }
337  data.setDataLocation(n_loc);
338  }
339  else if (m_csv){
340  inputCSVStream::ifstreamcsv(file, data);
341  }else{
342  bitpit::genericIO::absorbASCII(file, n_loc);
343  bitpit::genericIO::absorbASCII(file, readNSize);
344  nSize = std::max(nSize,readNSize);
345  data.reserve(nSize);
346  for(int i=0; i<nSize; ++i){
347  bitpit::genericIO::absorbASCII(file, id);
348  bitpit::genericIO::absorbASCII(file, data_T);
349  data.insert(id,data_T);
350  }
351  data.setDataLocation(n_loc);
352  }
353  file.close();
354  }else{
355  (*m_log) << "file not open --> exit" << std::endl;
356  throw std::runtime_error (m_name + " : cannot open " + m_filename + " requested");
357  }
358 
359  //mandatory check:if field is id-uncoherent with current geometry, does not set anything
360  if(data.checkDataIdsCoherence()){
361  _setResult(data);
362  }else{
363  throw std::runtime_error (m_name + " : read data are not coherent with linked geometry");
364  }
365 
366  return static_cast<IODataT<MimmoPiercedVector< T > >*>(m_result.get())->getData();
367 }
