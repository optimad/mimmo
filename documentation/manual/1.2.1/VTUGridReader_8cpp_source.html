199 void VTUGridStreamer::decodeRawData(bitpit::PatchKernel & patch)
200 {
201 
202  //time to check reading result.
203  std::size_t nVertices, nCells;
204  nVertices = points.size();
205  nCells = offsets.size();
206  if(nVertices == 0){
207  throw std::runtime_error("Error VTUGridStreamer : no point coordinates detected while reading *.vtu file.");
208  }
209  if(nCells == 0 || types.empty() || connectivitylist.size() < nCells ){
210  throw std::runtime_error("Error VTUGridStreamer : no valid connectivity/offsets/types info detected while reading *.vtu file.");
211  }
212 
213  patch.reserveVertices(nVertices);
214  patch.reserveCells(nCells);
215 
216  //reading mesh nodes and store it in vertices.
217  //check labels if any;
218  bool checkPointsID;
219  {
220  std::unordered_set<long> checkSet(pointsID.begin(), pointsID.end());
221  checkPointsID = ( checkSet.size() == nVertices);
222  }
223  //insert points and recover local/global map of vertices;
224  std::unordered_map<int, long> mapVert;
225  int counter = 0;
226  long idV=0;
227  for(const auto & p : points){
228  idV = bitpit::Vertex::NULL_ID;
229  if(checkPointsID) idV = pointsID[counter];
230  bitpit::PatchKernel::VertexIterator it = patch.addVertex(p, idV);
231  mapVert[counter] = (*it).getId();
232  ++counter;
233  }
234 
235  //reading mesh connectivity by offsets and store it in cells.
236  //check cell labels if any;
237  bool checkCellsID, checkFaceOffset, checkPID;
238  {
239  std::unordered_set<long> checkSet(cellsID.begin(), cellsID.end());
240  checkCellsID = ( checkSet.size() == nCells);
241  checkFaceOffset = ( (faceoffsets.size() == nCells) && (faces.size()>= nCells) );
242  checkPID = (pids.size() == nCells);
243  }
244  //insert points;
245  counter = 0;
246  long idC=0;
247  int posCellBegin = 0, posFaceBegin=0;
248  bitpit::ElementType eltype;
249  short PID;
250  livector1D conn;
251  std::size_t connSize;
252  for(const auto & off : offsets){
253  idC = bitpit::Cell::NULL_ID;
254  PID = 0;
255  if(checkCellsID) {idC= cellsID[counter];}
256  if(checkPID) {PID = pids[counter];}
257  eltype = types[counter];
258 
259  if(eltype == bitpit::ElementType::POLYHEDRON){
260  if(!checkFaceOffset){
261  throw std::runtime_error("Error VTUGridStreamer : trying to acquire POLYHEDRON info without faces and faceoffsets data");
262  }
263  connSize = faceoffsets[counter] - posFaceBegin;
264  conn.resize(connSize);
265  int loc = 0;
266  for(int i=posFaceBegin; i<faceoffsets[counter]; ++i){
267  conn[loc] = faces[i];
268  ++loc;
269  }
270  //remap vertices: conn is now written face by face with local vertex indices
271  int posfbegin = 1, posfend; //begin from 1- value. 0 value of conn contains the total number fo faces
272  while(posfbegin < int(connSize)){
273  posfend = posfbegin +conn[posfbegin] + 1;
274  for(int i=posfbegin+1; i<posfend; ++i){
275  conn[i] = mapVert[conn[i]];
276  }
277  posfbegin = posfend;
278  }
279  }else if(eltype == bitpit::ElementType::POLYGON){
280  connSize = off - posCellBegin;
281  conn.resize(connSize +1);
282  conn[0] = connSize;
283  int loc =1;
284  for(int i=posCellBegin; i<off; ++i){
285  conn[loc] = mapVert[connectivitylist[i]];
286  ++loc;
287  }
288  }else{
289  connSize = off - posCellBegin;
290  conn.resize(connSize);
291  int loc =0;
292  for(int i=posCellBegin; i<off; ++i){
293  conn[loc] = mapVert[connectivitylist[i]];
294  ++loc;
295  }
296  }
297 
298  bitpit::PatchKernel::CellIterator it = patch.addCell(eltype, true, conn, idC);
299  (*it).setPID(PID);
300 
301  posCellBegin = off;
302  if(checkFaceOffset) posFaceBegin = faceoffsets[counter];
303  ++counter;
304  }
305 }

424 void VTUGridReader::readMetaInformation(){
425  std::fstream str;
426  std::string line, temp;
427 
428  std::fstream::pos_type position;
429 
430  str.open( m_fh.getPath( ), std::ios::in ) ;
431 
432  getline( str, line);
433  while( ! bitpit::utils::string::keywordInString( line, "<VTKFile")){
434  getline(str, line);
435  }
436 
437  if( bitpit::utils::string::getAfterKeyword( line, "header_type", '\"', temp) ){
438  VTKUnstructuredGrid::setHeaderType( temp) ;
439  }
440 
441  while( ! bitpit::utils::string::keywordInString( line, "<Piece")){
442  getline(str, line);
443  }
444 
445  bitpit::utils::string::getAfterKeyword( line, "NumberOfPoints", '\"', temp) ;
446  bitpit::utils::string::convertString( temp, m_points );
447 
448  bitpit::utils::string::getAfterKeyword( line, "NumberOfCells", '\"', temp) ;
449  bitpit::utils::string::convertString( temp, m_cells );
450 
451 
452  position = str.tellg() ;
453  VTKUnstructuredGrid::readDataHeader( str ) ;
454 
455 
456  for( auto &field : m_geometry ){
457  str.seekg( position) ;
458  if( ! readDataArray( str, field ) ) {
459  #if ENABLE_DEBUG
460  log::cout() << field.getName() << " DataArray not found" << std::endl ;
461  #endif
462  field.disable();
463  }
464  }
465 
466  str.close() ;
467  setDimensions( m_cells, m_points, readConnectivityEntries(), readFaceStreamEntries() ) ;
468 }

474 uint64_t VTUGridReader::readFaceStreamEntries( ){
475  uint64_t nface(0) ;
476  std::fstream str ;
477  std::fstream::pos_type position_appended;
478  std::string line;
479  char c_ ;
480  uint32_t nbytes32 ;
481  uint64_t nbytes64 ;
482 
483 
484  // Geometry id of the facestream
485  int facestream_gid = VTKUnstructuredGrid::_findFieldIndex("faces", m_geometry);
486  if(facestream_gid < 0 ) return nface;
487  if(!m_geometry[facestream_gid].isEnabled()) return nface;
488 
489  if( m_geometry[facestream_gid].getCodification() == VTKFormat::APPENDED ){
490  str.open( m_fh.getPath( ), std::ios::in ) ;
491  //Go to the initial position of the appended section
492  while( getline(str, line) && (! bitpit::utils::string::keywordInString( line, "<AppendedData")) ){}
493  str >> c_;
494  while( c_ != '_') str >> c_;
495  position_appended = str.tellg();
496  str.close();
497  str.clear();
498  }
499 
500  //Open in binary for read
501  if( m_geometry[facestream_gid].getCodification() == VTKFormat::APPENDED ){
502  str.open( m_fh.getPath( ), std::ios::in | std::ios::binary);
503  str.seekg( position_appended) ;
504  str.seekg( m_geometry[facestream_gid].getOffset(), std::ios::cur) ;
505 
506  int dataSize = VTKTypes::sizeOfType( m_geometry[facestream_gid].getDataType() ) ;
507  assert(dataSize > 0) ;
508 
509  if( m_headerType== "UInt32") {
510  genericIO::absorbBINARY( str, nbytes32 ) ;
511  nface = nbytes32 / dataSize ;
512  }
513 
514  if( m_headerType== "UInt64") {
515  genericIO::absorbBINARY( str, nbytes64 ) ;
516  nface = nbytes64 / dataSize ;
517  }
518 
519  str.close();
520  }
521 
522  //Read geometry ASCII
523  if( m_geometry[facestream_gid].getCodification() == VTKFormat::ASCII ){
524  str.open( m_fh.getPath( ), std::ios::in );
525  str.seekg( m_geometry[facestream_gid].getPosition() ) ;
526 
527  std::string line ;
528  std::vector<uint64_t> temp;
529 
530  nface = 0 ;
531 
532  getline( str, line) ;
533  while( ! bitpit::utils::string::keywordInString(line,"/DataArray") ) {
534  temp.clear() ;
535  bitpit::utils::string::convertString( line, temp) ;
536  nface += temp.size() ;
537  getline( str, line) ;
538  }
539 
540  str.close();
541  }
542 
543 
544  return nface ;
545 }
