313 IOCGNS::execute(){
314  bool check = true;
315  if (m_read) {
316  check = read();
317  }
318  if (!check){
319  (*m_log) << "error: file corrupted or not found : "<< m_rfilename << std::endl;
320  (*m_log) << " " << std::endl;
321  throw std::runtime_error ("file corrupted or not found : " + m_rfilename);
322  }
323  if (m_write) {
324  check = write();
325  }
326  if (!check){
327  (*m_log) << "error: write not done : geometry not linked " << std::endl;
328  (*m_log) << " " << std::endl;
329  throw std::runtime_error ("write not done : geometry not linked ");
330  }
331 }

336 bool
337 IOCGNS::read(){
338 
339  std::string file = m_rdir+"/"+m_rfilename+".cgns";
340  std::string error_string = "read CGNS grid: " + file;
341 
342  long nVertices;
343  // long nCells, nBoundVertices;
344  int nCoords; //, nConnSections;
345  std::array< std::vector<double>,3 > coords;
346  std::vector<CG_ElementType_t> orderedConns;
347  std::unordered_map<int, ivector1D> conns;
348  std::unordered_map<int, ivector1D > bcLists;
349 
350  //Open cgns file
351  int indexfile;
352  if(cg_open(file.c_str(), CG_MODE_READ, &indexfile) != CG_OK){
353  return false;
354  }
355 
356  //Read number of bases
357  int nbases;
358  if(cg_nbases(indexfile, &nbases) != CG_OK){
359  return false;
360  }
361  if(nbases > 1){
362  (*m_log) << m_name << " more than one base found in grid file -> only the first zone will be read" << std::endl;
363  }
364 
365  //Read name of basis and physical dimension
366  char basename[33];
367  int physdim, celldim;
368  if(cg_base_read(indexfile, 1, basename, &celldim, &physdim) != CG_OK){
369  return false;
370  }
371  if(celldim != 3 || physdim !=3){
372  //Only volume mesh supported
373  //TODO NO SURFACE MESH (ONLY SURFACE WITHOUT VOLUME)?
374  return false;
375  }
376 
377  //Read number of zone in basis, check if only one.
378  int nzones;
379  if(cg_nzones(indexfile, 1, &nzones)!=CG_OK){
380  return false;
381  }
382  if(nzones > 1){
383  //Only single volume zone supported for now
384  return false;
385  }
386 
387  //Read type mesh of zone
388  CG_ZoneType_t zoneType;
389  int index_dim;
390  if(cg_zone_type(indexfile,1,1, &zoneType)!= CG_OK){
391  return false;
392  }
393  if(cg_index_dim(indexfile,1,1, &index_dim)!= CG_OK){
394  return false;
395  }
396  if(zoneType != CG_ZoneType_t::CG_Unstructured || index_dim != 1){
397  //Only unstructured mesh supported for now (index_dim == 1 for unstructured)
398  return false;
399  }
400 
401  //Read size of zone (n nodes, n cells, n boundary nodes)
402  std::vector<cgsize_t> sizeG(3);
403  char zonename[33];
404  if(cg_zone_read(indexfile,1,1, zonename, sizeG.data()) != CG_OK ){
405  return false;
406  }
407 
408 
409  nVertices = sizeG[0];
410  // nCells = sizeG[1];
411  // nBoundVertices = sizeG[2];
412 
413  //Read Vertices.
414  if(cg_ncoords(indexfile,1,1, &nCoords)!= CG_OK){
415  return false;
416  }
417  for(int i = 1; i <= nCoords; ++i){
418  cgsize_t startIndex = 1;
419  cgsize_t finishIndex = sizeG[0];
420  CG_DataType_t datatype;
421  char name[33];
422  coords[i-1].resize(nVertices);
423  if(cg_coord_info(indexfile,1,1,i, &datatype, name)!=CG_OK){
424  return false;
425  }
426  if(cg_coord_read(indexfile,1,1,name, datatype, &startIndex, &finishIndex, coords[i-1].data() )!=CG_OK){
427  return false;
428  }
429  }
430 
431  //Read connectivities.
432  //They are read starting from 1, fortran style. When matching up w/ coords
433  //vector positions remember to diminish conn value of 1.
434  //Read number of sections of connectivity structure
435  int nSections;
436  if(cg_nsections(indexfile,1,1, &nSections)!= CG_OK){
437  return false;
438  }
439 
440  for(int sec = 1; sec <= nSections; ++sec){
441 
442  //Read section sec
443  char elementname[33];
444  CG_ElementType_t type;
445  cgsize_t eBeg, eEnd, enBdry;
446  int parent_flag;
447  std::vector<cgsize_t> connlocal;
448  cgsize_t size;
449 
450  //Read elements name, type and range
451  if(cg_section_read(indexfile,1,1, sec, elementname, &type, &eBeg, &eEnd,&enBdry, & parent_flag)!= CG_OK){
452  return false;
453  }
454 
455  //Read size of connectivity data
456  if(cg_ElementDataSize(indexfile,1,1, sec,&size)!= CG_OK){
457  return false;
458  }
459  //cgsize_t nelements = eEnd-eBeg + 1;
460 
461 
462  connlocal.resize((size_t) size);
463  int *ptr = NULL;
464  if(cg_elements_read(indexfile,1,1,sec, connlocal.data(),ptr) !=CG_OK){
465  return false;
466  }
467 
468  conns[sec].resize(connlocal.size());
469  int count=0;
470  for(const auto &val: connlocal){
471  conns[sec][count] = (int)val;
472  ++count;
473  }
474 
475  orderedConns.push_back(type);
476  }
477 
478  //explore superficial boundary conditions definition, for boundary surface extraction.
479  int nBcs;
480  if(cg_nbocos(indexfile,1,1, &nBcs)!= CG_OK){
481  return false;
482  }
483 
484  //up to now i'm not able to encoding all the possible variants of bc node structure.
485  // i rely on the element list . Need to study better the cgns docs.
486  // TODO ????
487  for(int bc=1; bc<=nBcs; ++bc){
488 
489  char name[33];
490  CG_BCType_t bocotype;
491  CG_PointSetType_t ptset_type;
492  std::vector<cgsize_t> nBCElements(2);
493  int normalIndex;
494  cgsize_t normalListSize;
495  CG_DataType_t normalDataType;
496  int ndataset;
497  CG_GridLocation_t location;
498 
499  if(cg_boco_gridlocation_read(indexfile,1,1,bc, &location) != CG_OK){
500  return false;
501  }
502 
503  if(cg_boco_info(indexfile,1,1,bc, name, &bocotype, &ptset_type, nBCElements.data(),
504  &normalIndex, &normalListSize, &normalDataType, &ndataset) != CG_OK){
505  return false;
506  }
507 
508  if(ptset_type == CG_PointSetType_t::CG_ElementList || ptset_type == CG_PointSetType_t::CG_ElementRange){
509 
510  cgsize_t dim = nBCElements[0];
511 
512  std::vector<cgsize_t> localbc;
513 
514  if(ptset_type == CG_PointSetType_t::CG_ElementList){
515  localbc.resize((size_t) dim);
516  int *ptr = NULL;
517 
518  if(cg_boco_read(indexfile,1,1,bc, localbc.data(), ptr )!= CG_OK){
519  return false;
520  }
521 
522  }
523  else{
524 
525  std::vector<cgsize_t> rangeidx(2);
526  int *ptr = NULL;
527 
528  if(cg_boco_read(indexfile,1,1,bc, rangeidx.data(), ptr )!= CG_OK){
529  return false;
530  }
531 
532  dim = rangeidx[1] - rangeidx[0] + 1;
533  localbc.reserve((size_t) dim);
534 
535  for (auto idx = rangeidx[0]; idx <= rangeidx[1]; idx++){
536  localbc.push_back(idx);
537  }
538 
539  }
540 
541  bcLists[bc-1].resize(localbc.size());
542  int count=0;
543  for(const auto &val: localbc){
544  // WHY INT AND NOT LONG INT (IT IS A BITPIT PID) ??
545  bcLists[bc-1][count] = (int)val;
546  ++count;
547  }
548 
549  }
550 
551  m_storedBC->mcg_pidtobc[bc] = bocotype;
552  m_storedBC->mcg_pidtoname[bc] = name;
553 
554  }
555 
556  m_storedBC->mcg_pidtobc[0] = CG_BCTypeNull;
557  m_storedBC->mcg_pidtoname[0] = "undefined";
558 
559  //Finish reading CGNS file
560  cg_close(indexfile);
561 
562  //Reverse info in your grids.
563  std::unique_ptr<MimmoObject> patchVol(new MimmoObject(2));
564  std::unique_ptr<MimmoObject> patchBnd(new MimmoObject(1));
565  long id;
566  darray3E temp;
567 
568  // Stock vertices in volume grid
569  for(int i=0; i<nVertices; ++i){
570 
571  for(int j=0; j<3; ++j) temp[j] = coords[j][i];
572  id = i+1; //not C indexing, labeling coherent with connectivity.
573  patchVol->addVertex(temp,id);
574  }
575 
576  //debug only;
577  //TODO ????
578 
579  id = 1;
580  int idsection = 0;
581  //Unpack 3D elements connectivities and store it in volume grid. Label same species elements w PID.
582  for(const auto & val: orderedConns){
583 
584  idsection++;
585 
586  livector1D lConn;
587  bitpit::ElementType btype;
588  short PID = 0;
589  int size = conns[idsection].size();
590 
591  switch(val){
592  case CGNS_ENUMV(MIXED):
593 
594  unpack3DElementsMixedConns(patchVol.get(), patchBnd.get(), conns[idsection], id);
595  break;
596 
597  case CGNS_ENUMV(TETRA_4):
598  case CGNS_ENUMV(TETRA_10):
599 
600  btype = bitpit::ElementType::TETRA;
601  lConn.resize(4);
602  for(int i=0; i<size; i+=4){
603  for(int j=0; j<4; ++j){
604  lConn[j] = conns[idsection][i+j];
605  }
606  patchVol->addConnectedCell(lConn, btype, id );
607  patchVol->setPIDCell(id,PID);
608  id++;
609  }
610  break;
611 
612  case CGNS_ENUMV(PYRA_5):
613  case CGNS_ENUMV(PYRA_14):
614 
615  btype = bitpit::ElementType::PYRAMID;
616  lConn.resize(5);
617  for(int i=0; i<size; i+=5){
618  for(int j=0; j<5; ++j){
619  lConn[j] = conns[idsection][i+j];
620  }
621  patchVol->addConnectedCell(lConn, btype, id );
622  patchVol->setPIDCell(id,PID);
623  id++;
624  }
625  break;
626 
627  case CGNS_ENUMV(PENTA_6):
628  case CGNS_ENUMV(PENTA_15):
629  case CGNS_ENUMV(PENTA_18):
630 
631  btype = bitpit::ElementType::WEDGE;
632  lConn.resize(6);
633  for(int i=0; i<size; i+=6){
634  for(int j=0; j<6; ++j){
635  lConn[j] = conns[idsection][i+j];
636  }
637  patchVol->addConnectedCell(lConn, btype, id );
638  patchVol->setPIDCell(id,PID);
639  id++;
640  }
641 
642  break;
643 
644  case CGNS_ENUMV(HEXA_8):
645  case CGNS_ENUMV(HEXA_20):
646  case CGNS_ENUMV(HEXA_27):
647 
648  btype = bitpit::ElementType::HEXAHEDRON;
649  lConn.resize(8);
650  for(int i=0; i<size; i+=8){
651  for(int j=0; j<8; ++j){
652  lConn[j] = conns[idsection][i+j];
653  }
654  patchVol->addConnectedCell(lConn, btype, id );
655  patchVol->setPIDCell(id,PID);
656  id++;
657  }
658  break;
659 
660  case CGNS_ENUMV(TRI_3):
661  case CGNS_ENUMV(TRI_6):
662 
663  btype = bitpit::ElementType::TRIANGLE;
664  lConn.resize(3);
665 
666  for(int i=0; i<size; i+=3){
667  for(int j=0; j<3; ++j){
668  lConn[j] = conns[idsection][i+j];
669  }
670  patchBnd->addConnectedCell(lConn, btype, id );
671  patchBnd->setPIDCell(id,PID);
672  id++;
673  }
674  break;
675 
676  case CGNS_ENUMV(QUAD_4):
677  case CGNS_ENUMV(QUAD_8):
678  case CGNS_ENUMV(QUAD_9):
679 
680  btype = bitpit::ElementType::QUAD;
681  lConn.resize(4);
682  for(int i=0; i<size; i+=4){
683  for(int j=0; j<4; ++j){
684  lConn[j] = conns[idsection][i+j];
685  }
686  patchBnd->addConnectedCell(lConn, btype, id );
687  patchBnd->setPIDCell(id,PID);
688  id++;
689  }
690  break;
691 
692  default:
693  //do nothing
694  break;
695  }
696  }
697 
698  //divide patchBnd in subpatch if any bc is present.
699  if(bcLists.size() > 0) {
700  for(const auto & sel: bcLists){
701  if(sel.second.size() > 0){
702  short PID = sel.first + 1;
703  for(const auto &ind : sel.second){
704  id = ind;
705  patchBnd->setPIDCell(id,PID);
706  }
707  }
708  }//end for
709 
710  }
711 
712 
713  //adding vstand alone vertices to boundary patches and release all structures
714  {
715  //TODO this is not the best way to get this. In case of polygonal meshes it does not work.
716  //Anyway CGNS does not support polygons for now. So be it.
717  livector2D tempConn;
718  tempConn = patchBnd->getConnectivity();
719  std::set<long> ordIndex;
720 
721  for(const auto & val: tempConn){
722  ordIndex.insert(val.begin(), val.end());
723  }
724 
725  for(const auto & val: ordIndex){
726  temp = patchVol->getVertexCoords(val);
727  patchBnd->addVertex(temp,val);
728  }
729  }
730 
731  //release the meshes
732  m_volmesh = std::move(patchVol);
733  m_surfmesh = std::move(patchBnd);
734 
735  return true;
736 }
