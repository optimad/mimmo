33 void example00001() {
34 
35  /* Create IO_CGNS object to import input file. */
36  IOCGNS * cgnsI = new IOCGNS();
37  cgnsI->setRead(true);
38  cgnsI->setReadDir("geodata");
39  cgnsI->setReadFilename("grid");
40 
41  /* Create IO_CGNS object to export output file. */
42  IOCGNS * cgnsO = new IOCGNS();
43  cgnsO->setRead(false);
44  cgnsO->setWriteDir(".");
45  cgnsO->setWriteFilename("iocgns_output_00001");
46 
47  /* Create CGNS PID extractor object to test input file.
48  * Extraction of PID = 1,2,3 (All boundaries in input file).
49  */
50  CGNSPidExtractor * cgnsExtr = new CGNSPidExtractor();
51  cgnsExtr->setPID({1, 2, 3});
52  cgnsExtr->setForcedToTriangulate(false);
53  cgnsExtr->setPlotInExecution(true);
54 
55  /* Create CGNS PID extractor object to test input file.
56  * Extraction of PID = 1,2 (Wing wall and symmetry plane boundaries
57  * in input file).
58  */
59  CGNSPidExtractor * cgnsExtr2 = new CGNSPidExtractor();
60  cgnsExtr2->setPID({1, 2});
61  cgnsExtr2->setForcedToTriangulate(false);
62  cgnsExtr2->setPlotInExecution(true);
63 
64  /* Instantiation of a Selection By Box block.
65  * Setup of span and origin of cube.
66  */
67  SelectionByBox * boxSel = new SelectionByBox();
68  boxSel->setOrigin({{700., 800., 0.}});
69  boxSel->setSpan(1500.,1600.,100.);
70  boxSel->setPlotInExecution(true);
71 
72  /* Creation of rotation block.
73  */
74  RotationGeometry* rotation = new RotationGeometry();
75  rotation->setDirection(darray3E{0.,1.,0.});
76  rotation->setRotation((M_PI/6));
77 
78  /* Create reconstruct vector block and set to reconstruct over the whole
79  * input surface geometry the displacements field.
80  */
81  ReconstructVector* recon = new ReconstructVector();
82 
83  /* Create propagate vector block and set to propagate over the whole
84  * input volume geometry the displacements field.
85  */
86  PropagateVectorField* prop = new PropagateVectorField();
87  prop->setWeightConstant(1.0);
88 // prop->setSmoothingSteps(50);
89  prop->setPlotInExecution(true);
90  prop->setConvergence(true);
91  prop->setTolerance(1.0e-03);
92  prop->setDumpingFactor(1.0);
93  prop->setDumpingRadius(3000);
94  prop->setSolver(true);
95 
96  /* Create propagate vector block and set to propagate over the whole
97  * input volume geometry the displacements field.
98  */
99  ExtractVectorField* extrF = new ExtractVectorField();
100  extrF->setMode(1);
101  extrF->setPlotInExecution(true);
102 
103  /* Create applier block.
104  * It applies the deformation displacements
105  * to the selected input volume geometry.
106  */
107  Apply* applier = new Apply();
108 
109  /* Create applier block.
110  * It applies the deformation displacements
111  * to the selected input surface geometry.
112  */
113  Apply* applierS = new Apply();
114 
115  /* Create PINs. */
116  addPin(cgnsI, cgnsExtr, M_GEOM2, M_GEOM);
117  addPin(cgnsExtr2, boxSel, M_GEOM, M_GEOM);
118  addPin(boxSel, rotation, M_GEOM, M_GEOM);
119  addPin(rotation, recon, M_GDISPLS, M_VECTORFIELD);
120  addPin(cgnsI, cgnsExtr2, M_GEOM2, M_GEOM);
121  addPin(cgnsExtr2, recon, M_GEOM, M_GEOM);
122  addPin(cgnsI, prop, M_GEOM, M_GEOM);
123  addPin(cgnsExtr2, prop, M_GEOM, M_GEOM2);
124  addPin(recon, prop, M_VECTORFIELD, M_GDISPLS);
125  addPin(prop, applier, M_GDISPLS, M_GDISPLS);
126  addPin(cgnsI, applier, M_GEOM, M_GEOM);
127 
128  addPin(cgnsExtr, extrF, M_GEOM, M_GEOM);
129  addPin(prop, extrF, M_GDISPLS, M_VECTORFIELD);
130  addPin(extrF, applierS, M_VECTORFIELD, M_GDISPLS);
131  addPin(cgnsExtr, applierS, M_GEOM, M_GEOM);
132 
133  addPin(applier, cgnsO, M_GEOM, M_GEOM);
134  addPin(applierS, cgnsO, M_GEOM, M_GEOM2);
135  addPin(cgnsI, cgnsO, M_BCCGNS, M_BCCGNS);
136 
137  /* Create and execute chain. */
138  Chain ch0;
139  ch0.addObject(cgnsI);
140  ch0.addObject(cgnsExtr);
141  ch0.addObject(cgnsExtr2);
142  ch0.addObject(boxSel);
143  ch0.addObject(rotation);
144  ch0.addObject(recon);
145  ch0.addObject(prop);
146  ch0.addObject(extrF);
147  ch0.addObject(applier);
148  ch0.addObject(applierS);
149  ch0.addObject(cgnsO);
150  ch0.exec(true);
151 
152  /* Destroy objects. */
153  delete cgnsI;
154  delete cgnsExtr;
155  delete cgnsExtr2;
156  delete cgnsO;
157  delete boxSel;
158  delete rotation;
159  delete recon;
160  delete prop;
161  delete extrF;
162  delete applier;
163  delete applierS;
164 
165  return;
166 }
