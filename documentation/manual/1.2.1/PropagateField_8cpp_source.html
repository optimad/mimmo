289 void
290 PropagateField::computeConnectivity(){
291 
292  //DOESN'T WORK FOR POINTS CLOUD OR CURVES
293 
294  if (m_geometry == NULL) return;
295 
296  bitpit::PatchKernel * patch_ = getGeometry()->getPatch();
297 
298  int type = getGeometry()->getType();
299 
300  m_conn.resize(m_np);
301  m_weights.resize(m_np);
302 
303  //map for edges already visited
304  std::map<std::pair<long, long>, bool> visitedge;
305 
306  //compute connectivity between points
307  for (auto & cell : patch_->getCells()){
308  int edgecount;
309  if (type == 1 || patch_->getDimension() == 2){
310  edgecount = cell.getFaceCount();
311  }
312  else{
313  edgecount = cell.getEdgeCount();
314  }
315  for (int ie=0; ie<edgecount; ie++){
316  bitpit::ConstProxyVector<long> econn;
317  if (type == 1 || patch_->getDimension() == 2){
318  econn = cell.getFaceConnect(ie);
319  }
320  else{
321  econn = cell.getEdgeConnect(ie);
322  }
323  if (!visitedge[std::pair<long, long>(econn[0],econn[1])]){
324  visitedge[std::pair<long, long>(econn[0],econn[1])] = true;
325  visitedge[std::pair<long, long>(econn[1],econn[0])] = true;
326  if (!m_conn.exists(econn[0])){
327  m_conn.insert(econn[0], livector1D(1, econn[1]));
328  }
329  else{
330  m_conn[econn[0]].push_back(econn[1]);
331  }
332  if (!m_conn.exists(econn[1])){
333  m_conn.insert(econn[1], livector1D(1, econn[0]));
334  }
335  else{
336  m_conn[econn[1]].push_back(econn[0]);
337  }
338  }
339  }
340  }

346 void
347 PropagateField::computeWeights(){
348 
349  if (m_geometry == NULL) return;
350 
351  bitpit::PatchKernel * patch_ = getGeometry()->getPatch();
352 
353  int nsize;
354  livector1D ids;
355  double dist;
356  double sumdist;
357 
358  m_weights.clear();
359  darray3E point;
360  dvector1D lweights;
361  for (long ID : m_conn.getIds()){
362  ids = m_conn[ID];
363  point = patch_->getVertex(ID).getCoords();
364  nsize = ids.size();
365  lweights.resize(nsize);
366  sumdist = 0.0;
367  for (int j=0; j<nsize; j++){
368  dist = norm2(point-patch_->getVertex(ids[j]).getCoords());
369  lweights[j] = 1/(std::pow(dist, m_gamma))*m_dumping[ids[j]];
370  sumdist += lweights[j];
371  }
372  lweights /= sumdist;
373  m_weights.insert(ID, lweights);
374  }
375 }

497 void
498 PropagateVectorField::computeDumpingFunction(){
499 
500  if (m_geometry == NULL ) return;
501 
502  if (m_bc.getGeometry() != m_bsurface){
503  throw std::runtime_error (m_name + " : boundary conditions not linked to boundary surface");
504  }
505  if (m_bc.getDataLocation() != mimmo::MPVLocation::POINT){
506  throw std::runtime_error (m_name + " : boundary conditions not defined on points");
507  }
508 
509  bitpit::PatchKernel * patch_ = getGeometry()->getPatch();
510 
511  double dist;
512  long ID;
513 
514  /*Maxdist should be the maximum distance between boundaries with zero values and
515  * boundaries with values different from zero.
516  */
517  //TODO compute it
518  double maxd = m_radius;
519 
520  m_dumping.clear();
521  darray3E point;
522  double val;
523  if (m_dumpingFactor <= 1.0e-12){
524  for (auto const & vertex : patch_->getVertices()){
525  ID = vertex.getId();
526  m_dumping.insert(ID, 1.0);
527  }
528  }
529  else{
530 
531  int dim = m_bsurface->getPatch()->getDimension();
532 
533  std::unique_ptr<MimmoObject> activeBoundary;
534  if (dim == 2){
535  activeBoundary = std::unique_ptr<MimmoObject>(new MimmoObject(1));
536  }
537  else if (dim == 1){
538  activeBoundary = std::unique_ptr<MimmoObject>(new MimmoObject(4));
539  }
540 
541  for (auto const & vertex : m_bsurface->getVertices()){
542  ID = vertex.getId();
543  if (norm2(m_bc[ID]) >= 1.0e-12){
544  activeBoundary->addVertex(vertex, ID);
545  }
546  }
547  for (auto const & cell : m_bsurface->getCells()){
548  ID = cell.getId();
549  bitpit::ConstProxyVector<long> vconn = cell.getVertexIds();
550  bool toinsert = true;
551  for (const auto & iV : vconn){
552  if (norm2(m_bc[iV]) < 1.0e-12){
553  toinsert = false;
554  break;
555  }
556  }
557  if (toinsert){
558  auto conn = m_bsurface->getCellConnectivity(ID);
559  activeBoundary->addConnectedCell(conn, cell.getType());
560  }
561  }
562  activeBoundary->buildAdjacencies();
563  activeBoundary->buildSkdTree();
564  bitpit::SurfaceSkdTree* tree = static_cast<bitpit::SurfaceSkdTree*>(activeBoundary->getSkdTree());
565 
566  for (auto const & vertex : patch_->getVertices()){
567  ID = vertex.getId();
568  point = vertex.getCoords();
569  dist = max(1.0e-08, tree->evalPointDistance(point));
570  val = std::max(1.0, std::pow((maxd/dist), m_dumpingFactor));
571  m_dumping.insert(ID, val);
572  }
573  }
574 }

580 void
581 PropagateVectorField::solveSmoothing(int nstep){
582 
583  int nsize;
584  livector1D ids;
585  livector1D noids;
586  long ID;
587 
588  {
589 
590  m_field.clear();
591  double maxval = 0.0;
592  for (auto vertex : getGeometry()->getVertices()){
593  ID = vertex.getId();
594  if (m_isbp[ID]){
595  m_field.insert(ID, m_bc[ID]);
596  maxval = std::max(maxval, norm2(m_bc[ID]));
597  }
598  else{
599  m_field.insert(ID, darray3E({0.0, 0.0, 0.0}));
600  }
601  }
602 
603  (*m_log)<< m_name <<" starts field propagation."<<std::endl;
604  for (int istep = 0; istep < nstep; istep++){
605 
606  if (!m_convergence) (*m_log)<<m_name << " smoothing step : " << istep+1 << " / " << nstep <<std::endl;
607 
608  double maxdiff;
609  maxdiff = 0.0;
610  for (auto vertex : getGeometry()->getVertices()){
611 
612  ID = vertex.getId();
613  darray3E value = m_field[ID];
614 
615  if (!m_isbp[ID]){
616 
617  m_field[ID] = darray3E({0.0, 0.0, 0.0});
618  nsize = m_conn[ID].size();
619  ids = m_conn[ID];
620  for (int j=0; j<nsize; j++){
621  m_field[ID] += m_field[ids[j]]*m_weights[ID][j];
622  }
623  }
624  if (m_convergence)
625  maxdiff = std::max(maxdiff, norm2(value - m_field[ID])/maxval);
626  }//end for vertex
627 
628  if (m_convergence) (*m_log)<< m_name<<" residual : " << maxdiff <<std::endl;
629 
630  //convergence
631  if (m_convergence){
632  if (maxdiff <= m_tol)
633  istep = nstep;
634  else{
635  nstep = istep+2;
636  }
637  }
638 
639  }// end step
640  (*m_log)<< m_name<<" ends field propagation."<<std::endl;
641 
642  m_field.setDataLocation(MPVLocation::POINT);
643  m_field.setGeometry(getGeometry());
644 
645  }
646 
647  m_conn.clear();
648  m_weights.clear();
649 
650 }

943 PropagateScalarField::computeDumpingFunction(){
944 
945  if (m_geometry == NULL ) return;
946 
947  if (m_bc.getGeometry() != m_bsurface){
948  throw std::runtime_error (m_name + " : boundary conditions not linked to boundary surface");
949  }
950  if (m_bc.getDataLocation() != mimmo::MPVLocation::POINT){
951  throw std::runtime_error (m_name + " : boundary conditions not defined on points");
952  }
953 
954  bitpit::PatchKernel * patch_ = getGeometry()->getPatch();
955 
956  double dist;
957  long ID;
958 
959  /*Maxdist should be the maximum distance between boundaries with zero values and
960  * boundaries with values different from zero.
961  */
962  //TODO compute it
963  double maxd = m_radius;
964 
965  m_dumping.clear();
966  darray3E point;
967  double val;
968  if (m_dumpingFactor <= 1.0e-12){
969  for (auto const & vertex : patch_->getVertices()){
970  ID = vertex.getId();
971  m_dumping.insert(ID, 1.0);
972  }
973  }
974  else{
975 
976  int dim = m_bsurface->getPatch()->getDimension();
977 
978  std::unique_ptr<MimmoObject> activeBoundary;
979  if (dim == 2){
980  activeBoundary = std::unique_ptr<MimmoObject>(new MimmoObject(1));
981  }
982  else if (dim == 1){
983  activeBoundary = std::unique_ptr<MimmoObject>(new MimmoObject(4));
984  }
985 
986  for (auto const & vertex : m_bsurface->getVertices()){
987  ID = vertex.getId();
988  if (std::abs(m_bc[ID]) >= 1.0e-12){
989  activeBoundary->addVertex(vertex, ID);
990  }
991  }
992  for (auto const & cell : m_bsurface->getCells()){
993  ID = cell.getId();
994  bitpit::ConstProxyVector<long> vconn = cell.getVertexIds();
995  bool toinsert = true;
996  for (const auto & iV : vconn){
997  if (std::abs(m_bc[iV]) < 1.0e-12){
998  toinsert = false;
999  break;
1000  }
1001  }
1002  if (toinsert){
1003  auto conn = m_bsurface->getCellConnectivity(ID);
1004  activeBoundary->addConnectedCell(conn, cell.getType());
1005  }
1006  }
1007  activeBoundary->buildAdjacencies();
1008  activeBoundary->buildSkdTree();
1009  bitpit::SurfaceSkdTree* tree = static_cast<bitpit::SurfaceSkdTree*>(activeBoundary->getSkdTree());
1010 
1011  for (auto const & vertex : patch_->getVertices()){
1012  ID = vertex.getId();
1013  point = vertex.getCoords();
1014  dist = max(1.0e-08, tree->evalPointDistance(point));
1015  val = std::max(1.0, std::pow((maxd/dist), m_dumpingFactor));
1016  m_dumping.insert(ID, val);
1017  }
1018  }
1019 }

1026 PropagateScalarField::solveSmoothing(int nstep){
1027 
1028  int nsize;
1029  livector1D ids;
1030  long ID;
1031 
1032  {
1033  m_field.clear();
1034  double maxval = 0.0;
1035  for (auto vertex : getGeometry()->getVertices()){
1036  ID = vertex.getId();
1037  if (m_isbp[ID]){
1038  m_field.insert(ID, m_bc[ID]);
1039  maxval = std::max(maxval, std::abs(m_bc[ID]));
1040  }
1041  else{
1042  m_field.insert(ID, 0.0);
1043  }
1044  }
1045 
1046  (*m_log)<< m_name <<" starts field propagation."<<std::endl;
1047  for (int istep = 0; istep < nstep; istep++){
1048 
1049  if (!m_convergence) (*m_log)<<"Smoothing step : " << istep+1 << " / " << nstep <<std::endl;
1050 
1051  double maxdiff;
1052  maxdiff = 0.0;
1053  for (auto vertex : getGeometry()->getVertices()){
1054 
1055  ID = vertex.getId();
1056  double value = m_field[ID];
1057 
1058  if (!m_isbp[ID]){
1059 
1060  m_field[ID] = 0.0;
1061  nsize = m_conn[ID].size();
1062  ids = m_conn[ID];
1063  for (int j=0; j<nsize; j++){
1064  m_field[ID] += m_field[ids[j]]*m_weights[ID][j];
1065  }
1066  }
1067  if (m_convergence)
1068  maxdiff = std::max(maxdiff, std::abs(value - m_field[ID])/maxval);
1069  }//end for vertex
1070 
1071  if (m_convergence) (*m_log)<< m_name<<" residual : " << maxdiff <<std::endl;
1072 
1073  //convergence
1074  if (m_convergence){
1075  if (maxdiff <= m_tol)
1076  istep = nstep;
1077  else{
1078  nstep = istep+2;
1079  }
1080  }
1081 
1082  }// end step
1083  (*m_log)<< m_name<<" ends field propagation."<<std::endl;
1084 
1085  m_field.setDataLocation(MPVLocation::POINT);
1086  m_field.setGeometry(getGeometry());
1087  }
1088 
1089  m_conn.clear();
1090  m_weights.clear();
1091
