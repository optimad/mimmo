286 bool
287 IOVTKScalar::read(){
288 
289  std::ifstream infile(m_rdir+"/"+m_rfilename+".vtk");
290  bool check = infile.good();
291  if (!check) return false;
292  infile.close();
293 
294  if (getGeometry() == NULL){
295  //Local Instantiation of mimmo Object.
296  m_local = true;
297  MimmoObject* mimmo0 = new MimmoObject(1);
298  setGeometry(mimmo0);
299  }
300 
301  string inputFilename = m_rdir+"/"+m_rfilename+".vtk";
302  int np = 0;
303  int nt = 0;
304  darray3E point;
305 
306  // Get all data from the file
307  vtkSmartPointer<vtkGenericDataObjectReader> reader =
308  vtkSmartPointer<vtkGenericDataObjectReader>::New();
309  reader->SetFileName(inputFilename.c_str());
310  reader->Update();
311 
312  // All of the standard data types can be checked and obtained like this:
313  if(reader->IsFilePolyData())
314  {
315  //unlock m_polydata to NULL;
316  if(m_polydataInternal) m_polydata->Delete();
317  m_polydata = NULL;
318  m_polydata = vtkPolyData::New();
319  m_polydata->DeepCopy(reader->GetPolyDataOutput());
320  m_polydataInternal = true;
321  vtkPolyData* output = reader->GetPolyDataOutput();
322 
323  vtkSmartPointer<vtkTriangleFilter> tri= vtkSmartPointer<vtkTriangleFilter>::New();
324  tri->SetInputData(output);
325  tri->Update();
326 
327  vtkPolyData* output2 = tri->GetOutput();
328 
329  vtkSmartPointer<vtkPoints> points = output2->GetPoints();
330  vtkCellArray *cells = output2->GetPolys();
331  //vtkCellData *cdata = output2->GetCellData();
332  vtkPointData *pdata = output2->GetPointData();
333 
334  double point_[3];
335  for (vtkIdType id=0; id<points->GetNumberOfPoints(); id++ ){
336  points->GetPoint(id, point_);
337  for (int i=0; i<3; i++) point[i] = point_[i];
338  getGeometry()->addVertex(point);
339  }
340 
341  vtkIdType* conn_;
342  vtkIdType npts;
343  nt = cells->GetNumberOfCells();
344  livector1D connectivity(3);
345  bitpit::ElementType eltype = bitpit::ElementType::TRIANGLE;
346  cells->InitTraversal();
347  for (vtkIdType id=0; id<nt; id++ ){
348  cells->GetNextCell(npts, conn_);
349  for (int i=0; i<npts; i++) connectivity[i] = conn_[i];
350  getGeometry()->addConnectedCell(connectivity, eltype);
351  }
352 
353  np = getGeometry()->getNVertex();
354  vtkDataArray* data = pdata->GetArray(0);
355  m_field.clear();
356  if (data != NULL){
357  double maxf = 0.0;
358  for (long int i=0; i<np; i++){
359  vtkIdType id = i;
360  m_field.insert(i, data->GetComponent(id,0));
361  maxf = max(maxf, abs(m_field[i]));
362  }
363  if (m_normalize){
364  for (int i=0; i<np; i++){
365  m_field[i] /= maxf;
366  m_field[i] *= m_scaling;
367  }
368  }
369  else{
370  for (int i=0; i<np; i++){
371  m_field[i] *= m_scaling;
372  }
373  }
374  }
375  m_field.setGeometry(getGeometry());
376  m_field.setDataLocation(MPVLocation::POINT);
377  }
378  else{
379  (*m_log) << m_name << " error: polydata not found in : "<< m_rfilename << std::endl;
380  return false;
381  }
382 
383  return true;
384 
385 }

392 bool
393 IOVTKScalar::write(){
394 
395  if (m_polydata == NULL && getGeometry() == NULL) return false;
396 
397  if (m_polydata != NULL){
398  if (getGeometry() != NULL){
399  vtkSmartPointer<vtkPoints> points = m_polydata->GetPoints();
400  double point_[3];
401  darray3E point;
402  int np = points->GetNumberOfPoints();
403  np = min(np, int(getGeometry()->getNVertex()));
404  for (vtkIdType id=0; id<np; id++ ){
405  point = getGeometry()->getVertexCoords(id);
406  for (int i=0; i<3; i++) point_[i] = point[i];
407  points->SetPoint(id, point_);
408  }
409  }
410 
411  string outputFilename = m_wdir+"/"+m_wfilename+".vtk";
412  vtkSmartPointer<vtkPolyDataWriter> writer = vtkSmartPointer<vtkPolyDataWriter>::New();
413  writer->SetFileName(outputFilename.c_str());
414  writer->SetInputData(m_polydata);
415  writer->Write();
416 
417  }
418  else{
419 
420  m_polydata = vtkPolyData::New();
421  m_polydata->Allocate();
422  m_polydataInternal = true;
423 
424  /* Set polydata points. */
425  vtkPoints* points = vtkPoints::New() ;
426  double point_[3];
427  darray3E point;
428  int np = int(getGeometry()->getNVertex());
429  points->SetNumberOfPoints(np);
430  for (vtkIdType id=0; id<np; id++ ){
431  point = getGeometry()->getVertexCoords(id);
432  for (int i=0; i<3; i++) point_[i] = point[i];
433  points->SetPoint(id, point_);
434  }
435  m_polydata->SetPoints(points);
436  points->Delete();
437  points = NULL;
438 
439  /* Set polydata cells. */
440  bitpit::PiercedVector<bitpit::Cell> & cells = getGeometry()->getCells();
441  for (auto & cell : cells){
442  bitpit::ConstProxyVector<long> vList = cell.getVertexIds();
443  std::vector<vtkIdType> c (vList.size(), 0);
444  int countV = 0;
445  for (const auto & iV : vList){
446  c[countV] = vtkIdType(iV);
447  ++countV;
448  }
449  if (vList.size() == 3){
450  m_polydata->InsertNextCell(VTK_TRIANGLE, 3, c.data());
451  }
452  else if (vList.size() == 4 ){
453  m_polydata->InsertNextCell(VTK_QUAD, 4, c.data());
454  }
455  else{
456  m_polydata->InsertNextCell(VTK_POLYGON, vList.size(), c.data());
457  }
458  }
459 
460  /* Set polydata field. */
461  //check if a valid scalarfield is connected
462  bool check = m_field.getGeometry()==getGeometry();
463  check = check && m_field.getDataLocation()==MPVLocation::POINT;
464  check = check && m_field.completeMissingData(0.0);
465  if(check){
466  //get MPV as normal vector sequenced as geometry vertices actual ordering.
467  dvector1D field = m_field.getDataAsVector();
468 
469  vtkSmartPointer<vtkPointData> pdata = m_polydata->GetPointData();
470  vtkDoubleArray* data = vtkDoubleArray::New();
471  data->SetName( "mimmo.field" );
472  data->SetNumberOfComponents( 1 );
473  for (int i=0; i<np; i++){
474  data->InsertNextTuple( &field[i]);
475  }
476  pdata->AddArray(data);
477  data->Delete();
478  data = NULL;
479  }
480 
481  string outputFilename = m_wdir+"/"+m_wfilename+".vtk";
482  vtkSmartPointer<vtkPolyDataWriter> writer = vtkSmartPointer<vtkPolyDataWriter>::New();
483  writer->SetFileName(outputFilename.c_str());
484  writer->SetInputData(m_polydata);
485  writer->Write();
486  }
487  return true;
488 }

509  }
