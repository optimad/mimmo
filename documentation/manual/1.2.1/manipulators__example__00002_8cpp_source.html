91  deg[2] = 2;

102  input->setFilename("manipulators_input_00002.txt");
103

119  cout << " add pin info : " << boolalpha << addPin(mimmo0, applier, M_GEOM, M_GEOM) << endl;

121  cout << " " << endl;
122 
123  /* Setup execution chain.
124  */
125  Chain ch0;
126  ch0.addObject(mimmo0);
127  ch0.addObject(input);
128  ch0.addObject(lattice);
129  ch0.addObject(applier);
