182 void
183 StitchGeometry::execute(){
184  if(m_extgeo.empty()){
185  throw std::runtime_error(m_name + " : no source geometries to stich were found");
186  }
187 
188  std::unique_ptr<MimmoObject> dum(new MimmoObject(m_topo));
189  long nCells = 0;
190  long nVerts = 0;
191 
192  for(auto &obj:m_extgeo){
193  nCells += obj.first->getNCells();
194  nVerts += obj.first->getNVertex();
195  }
196 
197  //reserving memory
198  dum->getPatch()->reserveVertices(nVerts);
199  dum->getPatch()->reserveCells(nCells);
200 
201  long cV = 0, cC = 0;
202 
203  //start filling your stitched object.
204  {
205  //optional vars;
206  long vId, cId;
207  short PID;
208  bitpit::ElementType eltype;
209 
210  for(auto &obj : m_extgeo){
211 
212  std::unordered_map<long,long> mapVloc;
213 
214  //start extracting/reversing vertices of the current obj
215  for(const auto & vv : obj.first->getVertices()){
216  vId = vv.getId();
217  dum->addVertex(obj.first->getVertexCoords(vId), cV);
218  //update map;
219  mapVloc[vId] = cV;
220  cV++;
221  }
222 
223  //start extracting/reversing cells of the current obj
224  for(const auto & cc : obj.first->getCells()){
225  cId = cc.getId();
226  PID = cc.getPID();
227  eltype = cc.getType();
228  //get the local connectivity and update with new vertex numbering;
229  livector1D conn = obj.first->getCellConnectivity(cId);
230  livector1D connloc(conn.size());
231 
232  if(eltype == bitpit::ElementType::POLYGON){
233  std::size_t size = conn.size();
234  connloc[0] = conn[0];
235  for(std::size_t i = 1; i < size; ++i){
236  connloc[i] = mapVloc[conn[i]];
237  }
238 
239  }else if(eltype == bitpit::ElementType::POLYHEDRON){
240  connloc[0] = conn[0];
241  for(int nF = 0; nF < conn[0]-1; ++nF){
242  int facePos = cc.getFaceStreamPosition(nF);
243  int beginVertexPos = facePos + 1;
244  int endVertexPos = facePos + 1 + conn[facePos];
245  connloc[facePos] = conn[facePos];
246  for (int i=beginVertexPos; i<endVertexPos; ++i){
247  connloc[i] = mapVloc[conn[i]];
248  }
249  }
250  }else{
251  int ic = 0;
252  for (const auto & v : conn){
253  connloc[ic] = mapVloc[v];
254  ++ic;
255  }
256  }
257  dum->addConnectedCell(connloc, eltype, PID, cC);
258  //update map;
259  cC++;
260  }
261  }
262  }//scope for optional vars;
263 
264  m_patch = std::move(dum);
265  m_patch->cleanGeometry();
266 }
