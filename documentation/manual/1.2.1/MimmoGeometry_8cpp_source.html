613 bool
614 MimmoGeometry::write(){
615 
616  if (isEmpty()) return false;
617 
618  //adjusting with reference pid;
619  auto locpids = getGeometry()->getCompactPID();
620  for( auto & val: locpids) val+=m_refPID;
621  setPID(locpids);
622  //action completed, reset m_refPID to zero.
623  m_refPID = 0;
624 
625  //writing
626  switch(FileType::_from_integral(m_winfo.ftype)){
627 
628  case FileType::STL :
629  //Export STL
630  {
631  string name = (m_winfo.fdir+"/"+m_winfo.fname+".stl");
632  dynamic_cast<SurfUnstructured*>(getGeometry()->getPatch())->exportSTL(name, m_codex, m_multiSolidSTL, false);
633  return true;
634  }
635  break;
636 
637 
638  case FileType::SURFVTU :
639  case FileType::VOLVTU :
640  case FileType::CURVEVTU:
641  //Export Surface/Volume/3DCurve VTU
642  {
643  if(!m_codex) getGeometry()->getPatch()->getVTK().setCodex(bitpit::VTKFormat::ASCII);
644  else getGeometry()->getPatch()->getVTK().setCodex(bitpit::VTKFormat::APPENDED);
645  getGeometry()->getPatch()->write(m_winfo.fdir+"/"+m_winfo.fname);
646  getGeometry()->getPatch()->getVTK().setCodex(bitpit::VTKFormat::APPENDED);
647  return true;
648  }
649  break;
650 
651  case FileType::NAS :
652  //Export Nastran file
653  {
654  liimap mapDataInv;
655  dvecarr3E points = getGeometry()->getVertexCoords(&mapDataInv);
656  livector1D pointsID = getGeometry()->getVertices().getIds(false);
657  livector2D connectivity = getGeometry()->getCompactConnectivity(mapDataInv);
658  livector1D elementsID = getGeometry()->getCells().getIds(false);
659  NastranInterface nastran;
660  nastran.setWFormat(m_wformat);
661  shivector1D pids = getGeometry()->getCompactPID();
662  std::unordered_set<short> pidsset = getGeometry()->getPIDTypeList();
663  if (pids.size() == connectivity.size()){
664  nastran.write(m_winfo.fdir,m_winfo.fname,points, pointsID, connectivity,elementsID, &pids, &pidsset);
665  }else{
666  nastran.write(m_winfo.fdir,m_winfo.fname,points, pointsID, connectivity, elementsID);
667  }
668  return true;
669  }
670  break;
671 
672  case FileType::OFP : //Export ascii OpenFOAM point cloud
673  {
674 
675  dvecarr3E points = getGeometry()->getVertexCoords();
676  writeOFP(m_winfo.fdir, m_winfo.fname, points);
677  return true;
678  }
679  break;
680 
681  case FileType::PCVTU :
682  //Export Point Cloud VTU
683  {
684  dvecarr3E points = getGeometry()->getVertexCoords();
685  ivector2D connectivity(points.size(), ivector1D(1,0));
686  int counter = 0;
687  for(auto & c:connectivity){
688  c[0] = counter;
689  counter++;
690  }
691  bitpit::VTKUnstructuredGrid vtk(m_winfo.fdir, m_winfo.fname, bitpit::VTKElementType::VERTEX);
692  vtk.setGeomData( bitpit::VTKUnstructuredField::POINTS, points) ;
693  vtk.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity) ;
694  vtk.setDimensions(connectivity.size(), points.size());
695  if(!m_codex) vtk.setCodex(bitpit::VTKFormat::ASCII);
696  else vtk.setCodex(bitpit::VTKFormat::APPENDED);
697  vtk.write() ;
698  return true;
699  }
700  break;
701 
702  case FileType::MIMMO :
703  //Export in mimmo (bitpit) dump format
704  {
705  string name = (m_winfo.fdir+"/"+m_winfo.fname+".geomimmo");
706  std::filebuf buffer;
707  std::ostream out(&buffer);
708  buffer.open(name, std::ios::out);
709  if (buffer.is_open()){
710  getGeometry()->dump(out);
711  buffer.close();
712  }
713  return true;
714  }
715  break;
716 
717  default: //never been reached
718  break;
719  }
720 
721  return false;
722 };

728 bool
729 MimmoGeometry::read(){
730 
731  if(!m_isInternal) return false;
732 
733  switch(FileType::_from_integral(m_rinfo.ftype)){
734 
735  //Import STL
736  case FileType::STL :
737  {
738  setGeometry(1);
739  string name;
740 
741  {
742  std::ifstream infile(m_rinfo.fdir+"/"+m_rinfo.fname+".stl");
743  bool check = infile.good();
744  name = m_rinfo.fdir+"/"+m_rinfo.fname+".stl";
745  if (!check){
746  infile.open(m_rinfo.fdir+m_rinfo.fname+".STL");
747  check = infile.good();
748  name = m_rinfo.fdir+"/"+m_rinfo.fname+".STL";
749  if (!check) return false;
750  }
751  }
752 
753  ifstream in(name);
754  string ss, sstype;
755  getline(in,ss);
756  stringstream ins;
757  ins << ss;
758  ins >> sstype;
759  bool binary = true;
760  if (sstype == "solid" || sstype == "SOLID") binary = false;
761  in.close();
762 
763  dynamic_cast<SurfUnstructured*>(getGeometry()->getPatch())->importSTL(name, binary);
764  getGeometry()->cleanGeometry();
765 
766  //count PID if multi-solid
767  auto & map = getGeometry()->getPIDTypeList();
768  for(const auto & cell : getGeometry()->getCells() ){
769  map.insert(cell.getPID());
770  }
771  }
772  break;
773 
774  case FileType::SURFVTU :
775  //Import Surface VTU meshes
776  {
777  std::ifstream infile(m_rinfo.fdir+"/"+m_rinfo.fname+".vtu");
778  bool check = infile.good();
779  if (!check) return false;
780 
781  setGeometry(1);
782  VTUGridStreamer vtustreamer;
783  VTUGridReader input(m_rinfo.fdir, m_rinfo.fname, vtustreamer, *(getGeometry()->getPatch()));
784  input.read() ;
785 
786  getGeometry()->resyncPID();
787  }
788  break;
789 
790  case FileType::VOLVTU :
791  //Import Volume VTU meshes
792  {
793  std::ifstream infile(m_rinfo.fdir+"/"+m_rinfo.fname+".vtu");
794  bool check = infile.good();
795  if (!check) return false;
796 
797  setGeometry(2);
798  VTUGridStreamer vtustreamer;
799  VTUGridReader input(m_rinfo.fdir, m_rinfo.fname, vtustreamer, *(getGeometry()->getPatch()));
800  input.read() ;
801 
802  getGeometry()->resyncPID();
803 
804  }
805  break;
806 
807  case FileType::NAS :
808  //Import Surface NAS
809  {
810  std::ifstream infile(m_rinfo.fdir+"/"+m_rinfo.fname+".nas");
811  bool check = infile.good();
812  if (!check) return false;
813  infile.close();
814 
815  dvecarr3E Ipoints ;
816  livector2D Iconnectivity ;
817  livector1D pointsID;
818  livector1D cellsID;
819 
820  NastranInterface nastran;
821  nastran.setWFormat(m_wformat);
822 
823  shivector1D pids;
824  nastran.read(m_rinfo.fdir, m_rinfo.fname, Ipoints, pointsID, Iconnectivity, cellsID, pids );
825 
826  bitpit::ElementType eltype;
827 
828  setGeometry(1);
829 
830  int sizeV, sizeC;
831  sizeV = Ipoints.size();
832  sizeC = Iconnectivity.size();
833  m_intgeo->getPatch()->reserveVertices(sizeV);
834  m_intgeo->getPatch()->reserveCells(sizeC);
835 
836  int counter = 0;
837  for(const auto & vv : Ipoints){
838  m_intgeo->addVertex(vv, pointsID[counter]);
839  ++counter;
840  }
841  counter = 0;
842  for(const auto & cc : Iconnectivity) {
843  eltype = bitpit::ElementType::UNDEFINED;
844  std::size_t ccsize = cc.size();
845  if(ccsize == 3) eltype = bitpit::ElementType::TRIANGLE;
846  if(ccsize == 4) eltype = bitpit::ElementType::QUAD;
847  m_intgeo->addConnectedCell(cc, eltype, cellsID[counter]);
848  ++counter;
849  }
850  m_intgeo->setPID(pids);
851  }
852 
853  break;
854 
855  //Import ascii OpenFOAM point cloud
856  case FileType::OFP :
857  {
858 
859  std::ifstream infile(m_rinfo.fdir+"/"+m_rinfo.fname);
860  bool check = infile.good();
861  if (!check) return false;
862  infile.close();
863 
864  dvecarr3E Ipoints;
865  readOFP(m_rinfo.fdir, m_rinfo.fname, Ipoints);
866 
867  setGeometry(3);
868 
869  int sizeV = Ipoints.size();
870  m_intgeo->getPatch()->reserveVertices(sizeV);
871 
872  for(const auto & vv : Ipoints) m_intgeo->addVertex(vv);
873 
874  }
875  break;
876 
877  //Import point cloud from vtu
878  case FileType::PCVTU :
879  {
880 
881  std::ifstream infile(m_rinfo.fdir+"/"+m_rinfo.fname+".vtu");
882  bool check = infile.good();
883  if (!check) return false;
884 
885  setGeometry(3);
886  VTUPointCloudStreamer vtustreamer;
887  VTUGridReader input(m_rinfo.fdir, m_rinfo.fname, vtustreamer, *(getGeometry()->getPatch()));
888  input.read() ;
889 
890  }
891  break;
892 
893  case FileType::CURVEVTU :
894  //Import 3D Curve VTU
895  {
896  std::ifstream infile(m_rinfo.fdir+"/"+m_rinfo.fname+".vtu");
897  bool check = infile.good();
898  if (!check) return false;
899 
900  setGeometry(4);
901  VTUGridStreamer vtustreamer;
902  VTUGridReader input(m_rinfo.fdir, m_rinfo.fname, vtustreamer, *(getGeometry()->getPatch()));
903  input.read() ;
904 
905  getGeometry()->resyncPID();
906  }
907  break;
908 
909  case FileType::MIMMO :
910  //Import in mimmo (bitpit) restore format
911  {
912  string name = (m_rinfo.fdir+"/"+m_rinfo.fname+".geomimmo");
913  std::filebuf buffer;
914  std::istream in(&buffer);
915  buffer.open(name, std::ios::in);
916  if (buffer.is_open()){
917  m_intgeo->restore(in);
918  buffer.close();
919  }
920  return true;
921  }
