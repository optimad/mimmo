352 livector1D
353 SelectionByMapping::extractSelection(){
354 
355  if(!(getGeometry()->isSkdTreeSync())) getGeometry()->buildSkdTree();
356  std::set<long> cellList;
357 
358  for (auto & file : m_geolist){
359  livector1D list = getProximity(file);
360  cellList.insert(list.begin(), list.end());
361  }
362 
363 
364  for (auto & mimmoptr : m_mimmolist){
365  livector1D list = getProximity(mimmoptr);
366  cellList.insert(list.begin(), list.end());
367  }
368 
369 /* check if dual */
370  livector1D result;
371  if(m_dual){
372  livector1D totID = getGeometry()->getCellsIds();
373  result.resize(totID.size() - cellList.size());
374  if(result.size() == 0) return result;
375 
376  std::sort(totID.begin(), totID.end());
377  int counter = 0;
378  auto tot_it = totID.begin();
379  auto cand_it = cellList.begin();
380  while(tot_it != totID.end()){
381  long val = *tot_it;
382  if (cand_it == cellList.end() || val != *cand_it) {
383  result[counter] = val;
384  ++counter;
385  } else {
386  ++cand_it;
387  }
388  ++tot_it;
389  }
390  }else{
391  result.insert(result.end(), cellList.begin(), cellList.end());
392  }
393  return result;
394 };

402 SelectionByMapping::getProximity(std::pair<std::string, int> val){
403 
404  svector1D info = extractInfo(val.first);
405 
406  MimmoGeometry * geo = new MimmoGeometry();
407  geo->setIOMode(IOMode::READ);
408  geo->setDir(info[0]);
409  geo->setFilename(info[1]);
410  geo->setFileType(val.second);
411  geo->setBuildSkdTree(true);
412  geo->execute();
413 
414  if(geo->getGeometry()->getNVertices() == 0 || geo->getGeometry()->getNCells() == 0 || geo->getGeometry()->getType()==3 ){
415  m_log->setPriority(bitpit::log::NORMAL);
416  (*m_log)<< m_name << " failed to read or unsuitable geometry in SelectionByMapping::getProximity"<<std::endl;
417  m_log->setPriority(bitpit::log::DEBUG);
418  return livector1D();
419  }
420  livector1D result = mimmo::skdTreeUtils::selectByPatch(geo->getGeometry()->getSkdTree(), getGeometry()->getSkdTree(), m_tolerance);
421  delete geo;
422  geo=NULL;
423 
424  return result;
425 };

478 SelectionByMapping::absorbSectionXML(const bitpit::Config::Section & slotXML, std::string name){
479 
480  BITPIT_UNUSED(name);
481 
482  //checking topology
483  if(slotXML.hasOption("Topology")){
484  std::string input = slotXML.get("Topology");
485  input = bitpit::utils::string::trim(input);
486  int temp = -1;
487  if(!input.empty()){
488  std::stringstream ss(input);
489  ss>>temp;
490  }
491  if(m_topo != temp) {
492  throw std::runtime_error (m_name + " : xml absorbing failed.");
493  }
494  }
495 
496  BaseManipulation::absorbSectionXML(slotXML, name);
497  //start absorbing
498  if(slotXML.hasOption("Dual")){
499  std::string input = slotXML.get("Dual");
500  input = bitpit::utils::string::trim(input);
501  bool value = false;
502  if(!input.empty()){
503  std::stringstream ss(input);
504  ss >> value;
505  }
506  setDual(value);
507  }
508 
509  if(slotXML.hasOption("Tolerance")){
510  std::string input = slotXML.get("Tolerance");
511  input = bitpit::utils::string::trim(input);
512  double temp = 1.0E-8;
513  if(!input.empty()){
514  std::stringstream ss(input);
515  ss>>temp;
516  }
517  setTolerance(temp);
518 
519  }
520 
521  std::unordered_map<std::string, int> mapp;
522  if(slotXML.hasSection("Files")){
523 
524  const bitpit::Config::Section & filesXML = slotXML.getSection("Files");
525 
526  for(auto & subfile : filesXML.getSections()){
527  std::string path;
528  std::string tag;
529 
530  if(subfile.second->hasOption("fullpath")) {
531  path = subfile.second->get("fullpath");
532  path = bitpit::utils::string::trim(path);
533  }
534  if(subfile.second->hasOption("tag")){
535  tag = subfile.second->get("tag");
536  tag = bitpit::utils::string::trim(tag);
537  //check tag;
538  auto maybe_tag = FileType::_from_string_nothrow(tag.c_str());
539  if(!maybe_tag) tag.clear();
540  else tag = maybe_tag->_to_string();
541  }
542 
543  if(!path.empty() && !tag.empty()){
544  mapp[path] = (int) FileType::_from_string(tag.c_str());
545  }
546  }
547 
548  setFiles(mapp);
549 
550  }
