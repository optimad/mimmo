147 void
148 TranslationGeometry::execute(){
149 
150  if(getGeometry() == NULL){
151  (*m_log)<<m_name + " : NULL pointer to linked geometry found"<<std::endl;
152  throw std::runtime_error(m_name + "NULL pointer to linked geometry found");
153  }
154 
155 
156  if(getGeometry()->isEmpty()){
157  (*m_log)<<m_name + " : empty linked geometry found"<<std::endl;
158  }
159 
160  checkFilter();
161  m_displ.clear();
162  m_displ.clear();
163  m_displ.setDataLocation(mimmo::MPVLocation::POINT);
164  m_displ.reserve(getGeometry()->getNVertices());
165  m_displ.setGeometry(getGeometry());
166 
167 
168  long ID;
169  darray3E value;
170  for (const auto & vertex : m_geometry->getVertices()){
171  ID = vertex.getId();
172  value = m_alpha*m_direction*m_filter[ID];
173  m_displ.insert(ID, value);
174  }
175 };

189 void
190 TranslationGeometry::checkFilter(){
191  bool check = m_filter.getDataLocation() == mimmo::MPVLocation::POINT;
192  check = check && m_filter.completeMissingData(0.0);
193  check = check && m_filter.getGeometry() == getGeometry();
194 
195  if (!check){
196  m_log->setPriority(bitpit::log::Verbosity::DEBUG);
197  (*m_log)<<"Not valid filter found in "<<m_name<<". Proceeding with default unitary field"<<std::endl;
198  m_log->setPriority(bitpit::log::Verbosity::NORMAL);
199 
200  m_filter.clear();
201  m_filter.setGeometry(m_geometry);
202  m_filter.setDataLocation(mimmo::MPVLocation::POINT);
203  m_filter.reserve(getGeometry()->getNVertices());
204  for (const auto & vertex : getGeometry()->getVertices()){
205  m_filter.insert(vertex.getId(), 1.0);
206  }
207  }
208 }
