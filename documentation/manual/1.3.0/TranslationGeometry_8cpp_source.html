147 TranslationGeometry::execute(){
148 
149  if (getGeometry() == NULL){
150  throw std::runtime_error (m_name + " : NULL pointer to linked geometry");
151  }
152  if (getGeometry()->isEmpty()){
153  throw std::runtime_error (m_name + " : empty linked geometry");
154  }
155 
156  checkFilter();
157 
158  m_displ.clear();
159  m_displ.clear();
160  m_displ.setDataLocation(mimmo::MPVLocation::POINT);
161  m_displ.reserve(getGeometry()->getNVertex());
162  m_displ.setGeometry(getGeometry());
163 
164  long ID;
165  darray3E value;
166  for (const auto & vertex : m_geometry->getVertices()){
167  ID = vertex.getId();
168  value = m_alpha*m_direction*m_filter[ID];
169  m_displ.insert(ID, value);

177 TranslationGeometry::apply(){
178 
179  if (getGeometry() == NULL) return;
180  if (getGeometry()->isEmpty() || m_displ.isEmpty()) return;
181  darray3E vertexcoords;
182  long int ID;
183  for (const auto & vertex : m_geometry->getVertices()){
184  vertexcoords = vertex.getCoords();
185  ID = vertex.getId();
186  vertexcoords += m_displ[ID];
187  getGeometry()->modifyVertex(vertexcoords, ID);
188  }

196 void
197 TranslationGeometry::checkFilter(){
198  bool check = m_filter.getDataLocation() == mimmo::MPVLocation::POINT;
199  check = check && m_filter.completeMissingData(0.0);
200  check = check && m_filter.getGeometry() == getGeometry();
201 
202  if (!check){
203  m_log->setPriority(bitpit::log::Verbosity::DEBUG);
204  (*m_log)<<"Not valid filter found in "<<m_name<<". Proceeding with default unitary field"<<std::endl;
205  m_log->setPriority(bitpit::log::Verbosity::NORMAL);
206 
207  m_filter.clear();
208  m_filter.setGeometry(m_geometry);
209  m_filter.setDataLocation(mimmo::MPVLocation::POINT);
210  m_filter.reserve(getGeometry()->getNVertex());
211  for (const auto & vertex : getGeometry()->getVertices()){
212  m_filter.insert(vertex.getId(), 1.0);
213  }
214  }
