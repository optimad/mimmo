229 livector1D
230 SelectionByPID::extractSelection(){
231  livector1D result;
232  std::set<long> extraction;
233 
234  syncPIDList();
235  livector1D pids = getActivePID();
236 
237  for(auto && pid : pids){
238  livector1D local = getGeometry()->extractPIDCells(pid);
239  extraction.insert(local.begin(),local.end());
240  }
241 
242  //check if dual selection is triggered
243 
244  if(m_dual){
245  livector1D totID = getGeometry()->getCellsIds();
246  result.resize(totID.size() - extraction.size());
247  if(result.size() == 0) return result;
248 
249  std::sort(totID.begin(), totID.end());
250  int counter = 0;
251  auto tot_it = totID.begin();
252  auto cand_it = extraction.begin();
253  while(tot_it != totID.end()){
254  long val = *tot_it;
255  if (cand_it == extraction.end() || val != *cand_it) {
256  result[counter] = val;
257  ++counter;
258  } else {
259  ++cand_it;
260  }
261  ++tot_it;
262  }
263  }else{
264  result.insert(result.end(), extraction.begin(), extraction.end());
265  }
266  return result;
267 };

274 void
275 SelectionByPID::syncPIDList(){
276  if(getGeometry() == NULL) return;
277 
278  if(m_setPID.count(-1) == 0){
279 
280  std::unordered_set<long>::iterator itU;
281  livector1D negative;
282  for(itU = m_setPID.begin(); itU != m_setPID.end(); ++itU){
283  long value = *itU;
284  if(m_activePID.count(value) > 0) m_activePID[value] = true;
285  else negative.push_back(value);
286  }
287 
288  for(auto &val: negative) m_setPID.erase(val);
289 
290 
291  }else{
292  m_setPID.clear();
293  for(auto &val: m_activePID ){
294  m_setPID.insert(val.first);
295  val.second = true;
296 
297  }
298  }
299 }
