246 void
247 BendGeometry::execute(){
248 
249  if(getGeometry() == NULL){
250  (*m_log)<<m_name + " : NULL pointer to linked geometry found"<<std::endl;
251  throw std::runtime_error(m_name + "NULL pointer to linked geometry found");
252  }
253 
254 
255  if(getGeometry()->isEmpty()){
256  (*m_log)<<m_name + " : empty linked geometry found"<<std::endl;
257  }
258 
259  m_displ.clear();
260  m_displ.setDataLocation(mimmo::MPVLocation::POINT);
261  m_displ.reserve(getGeometry()->getNVertices());
262  m_displ.setGeometry(getGeometry());
263 
264 
265  //check coherence of degrees and coeffs;
266  for(int i=0; i<3; ++i){
267  for(int j=0; j<3; ++j){
268  m_coeffs[i][j].resize(m_degree[i][j]+1, 0.0);
269  }
270  }
271 
272  checkFilter();
273 
274  long ID;
275  darray3E value;
276  darray3E point, point0;
277  for (const auto & vertex : m_geometry->getVertices()){
278  point = vertex.getCoords();
279  if (m_local){
280  point0 = point;
281  point = toLocalCoord(point);
282  }
283  ID = vertex.getId();
284  value.fill(0.0);
285  for (int j=0; j<3; j++){
286  for (int z=0; z<3; z++){
287  if (m_degree[j][z] > 0){
288  for (int k=0; k<(int)m_degree[j][z]+1; k++){
289  value[j] += pow(point[z],(double)k)*m_coeffs[j][z][k]*m_filter[ID];
290  }
291  }
292  }
293  }
294  if (m_local){
295  point += value;
296  point = toGlobalCoord(point);
297  value = point - point0;
298  }
299  m_displ.insert(ID, value);
300  }
301 };

316 BendGeometry::checkFilter(){
317  bool check = m_filter.getDataLocation() == mimmo::MPVLocation::POINT;
318  check = check && m_filter.completeMissingData(0.0);
319  check = check && m_filter.getGeometry() == getGeometry();
320 
321  if (!check){
322  m_log->setPriority(bitpit::log::Verbosity::DEBUG);
323  (*m_log)<<"Not valid filter found in "<<m_name<<". Proceeding with default unitary field"<<std::endl;
324  m_log->setPriority(bitpit::log::Verbosity::NORMAL);
325 
326  m_filter.clear();
327  m_filter.setGeometry(m_geometry);
328  m_filter.setDataLocation(mimmo::MPVLocation::POINT);
329  m_filter.reserve(getGeometry()->getNVertices());
330  for (const auto & vertex : getGeometry()->getVertices()){
331  m_filter.insert(vertex.getId(), 1.0);
332  }
333  }

355 darray3E BendGeometry::toGlobalCoord(darray3E point){
356 
357  darray3E work, work2;
358  //unscale your local point
359  for(int i =0; i<3; ++i){
360  work[i] = point[i];
361  }
362 
363  //unapply change to local sdr transformation
364  work2 = matmul(work, m_system);
365 
366  //unapply origin translation
367  return (work2 + m_origin);

376 darray3E BendGeometry::matmul(const darray3E & vec, const dmatrix33E & mat){
377 
378  darray3E out;
379  for (std::size_t i = 0; i < 3; i++) {
380  out[i] = 0.0;
381  for (std::size_t j = 0; j <3; j++) {
382  out[i] += vec[j]*mat[j][i];
383  } //next j
384  } //next i
385 
386  return out;

414 BendGeometry::absorbSectionXML(const bitpit::Config::Section & slotXML, std::string name){
415 
416  BITPIT_UNUSED(name);
417 
418  std::string input;
419 
420  BaseManipulation::absorbSectionXML(slotXML, name);
421 
422  if(slotXML.hasSection("DegreesMatrix")){
423  auto & subslot = slotXML.getSection("DegreesMatrix");
424  umatrix33E temp;
425  temp.fill({{0,0,0}});
426 
427  if(subslot.hasOption("xDispl")){
428  input = subslot.get("xDispl");
429  std::stringstream ss(bitpit::utils::string::trim(input));
430  for(int i=0; i<3; ++i) ss>>temp[0][i];
431  }
432 
433  if(subslot.hasOption("yDispl")){
434  input = subslot.get("yDispl");
435  std::stringstream ss(bitpit::utils::string::trim(input));
436  for(int i=0; i<3; ++i) ss>>temp[1][i];
437  }
438 
439  if(subslot.hasOption("zDispl")){
440  input = subslot.get("zDispl");
441  std::stringstream ss(bitpit::utils::string::trim(input));
442  for(int i=0; i<3; ++i) ss>>temp[2][i];
443  }
444 
445  setDegree(temp);
446  };
447 
448  // set Degree already set correctly the matrix of coeffs..
449  if(slotXML.hasSection("PolyCoefficients")){
450  auto & subslot = slotXML.getSection("PolyCoefficients");
451  dmat33Evec &temp = m_coeffs;
452  std::string rootPoly = "Poly";
453  std::string locPoly;
454  int ik,jk;
455  for (int k=0; k<9; ++k){
456  locPoly = rootPoly + std::to_string(k);
457  if(subslot.hasOption(locPoly)){
458  input = subslot.get(locPoly);
459  std::stringstream ss(bitpit::utils::string::trim(input));
460  ik = (int)(k/3);
461  jk = k%3;
462  for(double & val: temp[ik][jk]) ss>>val;
463  }
464  }
465  };
466 
467  if(slotXML.hasOption("Origin")){
468  std::string input = slotXML.get("Origin");
469  input = bitpit::utils::string::trim(input);
470  darray3E temp = {{0.0,0.0,0.0}};
471  if(!input.empty()){
472  std::stringstream ss(input);
473  for(auto &val : temp) ss>>val;
474  }
475  setOrigin(temp);
476  };
477 
478  if(slotXML.hasSection("RefSystem")){
479  const bitpit::Config::Section & rfXML = slotXML.getSection("RefSystem");
480  std::string rootAxis = "axis";
481  std::string axis;
482  dmatrix33E temp;
483  temp[0].fill(0.0); temp[0][0] = 1.0;
484  temp[1].fill(0.0); temp[1][1] = 1.0;
485  temp[2].fill(0.0); temp[2][2] = 1.0;
486  for(int i=0; i<3; ++i){
487  axis = rootAxis + std::to_string(i);
488  std::string input = rfXML.get(axis);
489  input = bitpit::utils::string::trim(input);
490  if(!input.empty()){
491  std::stringstream ss(input);
492  for(auto &val : temp[i]) ss>>val;
493  }
494  }
495  setRefSystem(temp);
496  };
497 };

505 BendGeometry::flushSectionXML(bitpit::Config::Section & slotXML, std::string name){
506 
507  BITPIT_UNUSED(name);
508 
509  BaseManipulation::flushSectionXML(slotXML, name);
510 
511  bitpit::Config::Section & degXML = slotXML.addSection("DegreesMatrix");
512

525 
526  bitpit::Config::Section & polyXML = slotXML.addSection("PolyCoefficients");
527  std::string rootPoly = "Poly";
528  std::string locPoly;
529  for(int i=0; i<3; i++){
530  for(int j =0; j<3; j++){
531  if(m_coeffs[i][j].empty()) continue;
532  locPoly = rootPoly + std::to_string(int(i*3+j));
533  std::stringstream ss;
534  for(auto & loc: m_coeffs[i][j]) ss<<loc<<'\t';
535  polyXML.set(locPoly, ss.str());
536  }
537  }
538 
539  {
540  std::stringstream ss;
541  ss<<std::scientific<<getOrigin()[0]<<'\t'<<getOrigin()[1]<<'\t'<<getOrigin()[2];
542  slotXML.set("Origin", ss.str());
543  }
544 
545  {
546  auto rs = getRefSystem();
547  bitpit::Config::Section & rsXML = slotXML.addSection("RefSystem");
548  std::string rootAxis = "axis";
549  std::string localAxis;
550  int counter=0;
551  for(auto &axis : rs){
552  localAxis = rootAxis+std::to_string(counter);
553  std::stringstream ss;
554  ss<<std::scientific<<axis[0]<<'\t'<<axis[1]<<'\t'<<axis[2];
555  rsXML.set(localAxis, ss.str());
556  ++counter;
