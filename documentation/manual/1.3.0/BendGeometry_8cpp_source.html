235 void
236 BendGeometry::execute(){
237 
238  if (getGeometry() == NULL){
239  throw std::runtime_error (m_name + " : NULL pointer to linked geometry");
240  }
241  if (getGeometry()->isEmpty()){
242  throw std::runtime_error (m_name + " : empty linked geometry");
243  }
244 
245  //check coherence of degrees and coeffs;
246  for(int i=0; i<3; ++i){
247  for(int j=0; j<3; ++j){
248  m_coeffs[i][j].resize(m_degree[i][j]+1, 0.0);
249  }
250  }
251 
252  checkFilter();
253 
254  m_displ.clear();
255  m_displ.setDataLocation(mimmo::MPVLocation::POINT);
256  m_displ.reserve(getGeometry()->getNVertex());
257  m_displ.setGeometry(getGeometry());
258 
259  long ID;
260  darray3E value;
261  darray3E point, point0;
262  for (const auto & vertex : m_geometry->getVertices()){
263  point = vertex.getCoords();
264  if (m_local){
265  point0 = point;
266  point = toLocalCoord(point);
267  }
268  ID = vertex.getId();
269  value.fill(0.0);
270  for (int j=0; j<3; j++){
271  for (int z=0; z<3; z++){
272  if (m_degree[j][z] > 0){
273  for (int k=0; k<(int)m_degree[j][z]+1; k++){
274  value[j] += pow(point[z],(double)k)*m_coeffs[j][z][k]*m_filter[ID];
275  }
276  }
277  }
278  }
279  if (m_local){
280  point += value;
281  point = toGlobalCoord(point);
282  value = point - point0;
283  }
284  m_displ.insert(ID, value);
285  }
286 };

312 BendGeometry::checkFilter(){
313  bool check = m_filter.getDataLocation() == mimmo::MPVLocation::POINT;
314  check = check && m_filter.completeMissingData(0.0);
315  check = check && m_filter.getGeometry() == getGeometry();
316 
317  if (!check){
318  m_log->setPriority(bitpit::log::Verbosity::DEBUG);
319  (*m_log)<<"Not valid filter found in "<<m_name<<". Proceeding with default unitary field"<<std::endl;
320  m_log->setPriority(bitpit::log::Verbosity::NORMAL);
321 
322  m_filter.clear();
323  m_filter.setGeometry(m_geometry);
324  m_filter.setDataLocation(mimmo::MPVLocation::POINT);
325  m_filter.reserve(getGeometry()->getNVertex());
326  for (const auto & vertex : getGeometry()->getVertices()){
327  m_filter.insert(vertex.getId(), 1.0);
328  }
329  }

336 darray3E
337 BendGeometry::toLocalCoord(darray3E point){
338  darray3E work, work2;
339  //unapply origin translation
340  work = point - m_origin;
341  //apply change to local sdr transformation
342  dmatrix33E transp = linearalgebra::transpose(m_system);
343  linearalgebra::matmul(work, transp, work2);
344  return(work2);
345 };

353 darray3E BendGeometry::toGlobalCoord(darray3E point){
354 
355  darray3E work, work2;
356  //unscale your local point
357  for(int i =0; i<3; ++i){
358  work[i] = point[i];
359  }
360 
361  //unapply change to local sdr transformation
362  linearalgebra::matmul(work, m_system, work2);
363 
364  //unapply origin translation
365  work = work2 + m_origin;
366  return(work);
367 };

377 BendGeometry::absorbSectionXML(const bitpit::Config::Section & slotXML, std::string name){
378 
379  BITPIT_UNUSED(name);
380 
381  std::string input;
382 
383  BaseManipulation::absorbSectionXML(slotXML, name);
384 
385  if(slotXML.hasSection("DegreesMatrix")){
386  auto & subslot = slotXML.getSection("DegreesMatrix");
387  umatrix33E temp;
388  temp.fill({{0,0,0}});
389 
390  if(subslot.hasOption("xDispl")){
391  input = subslot.get("xDispl");
392  std::stringstream ss(bitpit::utils::string::trim(input));
393  for(int i=0; i<3; ++i) ss>>temp[0][i];
394  }
395 
396  if(subslot.hasOption("yDispl")){
397  input = subslot.get("yDispl");
398  std::stringstream ss(bitpit::utils::string::trim(input));
399  for(int i=0; i<3; ++i) ss>>temp[1][i];
400  }
401 
402  if(subslot.hasOption("zDispl")){
403  input = subslot.get("zDispl");
404  std::stringstream ss(bitpit::utils::string::trim(input));
405  for(int i=0; i<3; ++i) ss>>temp[2][i];
406  }
407 
408  setDegree(temp);
409  };
410 
411  if(slotXML.hasSection("PolyCoefficients")){
412  auto & subslot = slotXML.getSection("PolyCoefficients");
413  dmat33Evec temp = getCoeffs();
414  std::string rootPoly = "Poly";
415  std::string locPoly;
416  int ik,jk;
417  for (int k=0; k<9; ++k){
418  locPoly = rootPoly + std::to_string(k);
419  if(subslot.hasOption(locPoly)){
420  input = subslot.get(locPoly);
421  std::stringstream ss(bitpit::utils::string::trim(input));
422  ik = (int)(k/3);
423  jk = k%3;
424  for(auto & val: temp[ik][jk]) ss>>val;
425  }
426  }
427  setCoeffs(temp);
428  };
429 
430  if(slotXML.hasOption("Origin")){
431  std::string input = slotXML.get("Origin");
432  input = bitpit::utils::string::trim(input);
433  darray3E temp = {{0.0,0.0,0.0}};
434  if(!input.empty()){
435  std::stringstream ss(input);
436  for(auto &val : temp) ss>>val;
437  }
438  setOrigin(temp);
439  };
440 
441  if(slotXML.hasSection("RefSystem")){
442  const bitpit::Config::Section & rfXML = slotXML.getSection("RefSystem");
443  std::string rootAxis = "axis";
444  std::string axis;
445  dmatrix33E temp;
446  temp[0].fill(0.0); temp[0][0] = 1.0;
447  temp[1].fill(0.0); temp[1][1] = 1.0;
448  temp[2].fill(0.0); temp[2][2] = 1.0;
449  for(int i=0; i<3; ++i){
450  axis = rootAxis + std::to_string(i);
451  std::string input = rfXML.get(axis);
452  input = bitpit::utils::string::trim(input);
453  if(!input.empty()){
454  std::stringstream ss(input);
455  for(auto &val : temp[i]) ss>>val;
456  }
457  }
458  setRefSystem(temp);
459  };
460 };
