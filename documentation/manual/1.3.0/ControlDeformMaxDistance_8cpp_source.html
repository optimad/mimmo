175 void
176 ControlDeformMaxDistance::execute(){
177 
178  MimmoObject * geo = getGeometry();
179  if(geo == NULL){
180  (*m_log)<<m_name + " : NULL pointer to linked geometry found"<<std::endl;
181  throw std::runtime_error(m_name + "NULL pointer to linked geometry found");
182  }
183 
184  if(geo->isEmpty()){
185  (*m_log)<<m_name + " : empty linked geometry found"<<std::endl;
186  return;
187  }
188 
189 
190  bool check = m_defField.getGeometry() == geo;
191  check = check && m_defField.getDataLocation() == MPVLocation::POINT;
192  check = check && m_defField.completeMissingData({{0.0,0.0,0.0}});
193  if(!check){
194  (*m_log)<<"Error in "<<m_name<<": Unsuitable deformation field linked"<<std::endl;
195  throw std::runtime_error (m_name + " : Unsuitable deformation field linked.");
196  }
197 
198  m_violationField.clear();
199 
200  if(!(geo->isSkdTreeSync())) geo->buildSkdTree();
201 
202  dmpvecarr3E points;
203  dmpvector1D normDef;
204  long int ID;
205  for (const auto & v : geo->getVertices()){
206  ID = v.getId();
207  points.insert(ID, geo->getVertexCoords(ID) + m_defField[ID] );
208  normDef.insert(ID, norm2(m_defField[ID]));
209  m_violationField.insert(ID, -1.0e+18);
210  }
211 
212  double dist;
213  double radius ;
214  double rate = 0.05;
215  int kmax = 200;
216  int kiter;
217  bool flag;
218  long int IDC;
219  for(const auto &ID : points.getIds()){
220  dist =1.0E+18;
221  kiter = 0;
222  flag = true;
223  radius = std::fmax(1.0E-8, normDef[ID]);
224  while(flag && kiter < kmax){
225  dist = skdTreeUtils::distance(&points[ID], geo->getSkdTree(), IDC, radius);
226  flag = (dist == 1.0E+18);
227  if(flag) radius *= (1.0+ rate*((double)flag));
228  kiter++;
229  }
230  if(kiter == kmax) dist = m_maxDist - dist;
231  m_violationField[ID] = (dist - m_maxDist);
232  }
233 
234  m_violationField.setGeometry(getGeometry());
235  m_violationField.setDataLocation(MPVLocation::POINT);
236 
237  //write log
238  std::string logname = m_name+std::to_string(getId())+"_violation.log";
239  std::ofstream log;
240  log.open(logname);
241  log<<"mimmo "<<m_name<<" resume file"<<std::endl;
242  log<<std::endl;
243  log<<std::endl;
244  log<<" violation value : " << getViolation() << std::endl;
245  log.close();
246 
247 };

294 void
295 ControlDeformMaxDistance::plotOptionalResults(){
296  if(getGeometry() == NULL) return;
297  if(getGeometry()->isEmpty()) return;
298  if(!m_defField.completeMissingData({{0,0,0}}) || !m_violationField.completeMissingData(0.0)) return;
299 
300  //get deformation on points
301  dvecarr3E deff = m_defField.getDataAsVector();
302  //get violation field
303  dvector1D viol = m_violationField.getDataAsVector();
304 
305  //add deformation field as vector field
306  bitpit::VTKUnstructuredGrid & vtk = getGeometry()->getPatch()->getVTK();
307  vtk.addData("Deformation Field", bitpit::VTKFieldType::VECTOR, bitpit::VTKLocation::POINT, deff);
308  //add violation field as a custom data field of the original geometry
309  vtk.addData("Violation Distance Field", bitpit::VTKFieldType::SCALAR, bitpit::VTKLocation::POINT, viol);
310 
311  vtk.setDirectory(m_outputPlot+"/");
312  vtk.setName(m_name+std::to_string(getId()));
313  getGeometry()->getPatch()->write();
314 
315  //reset geometry VTK to its original setup
316  vtk.removeData("Deformation Field");
317  vtk.removeData("Violation Distance Field");
318 }
