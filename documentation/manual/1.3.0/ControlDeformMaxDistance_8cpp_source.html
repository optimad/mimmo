175 void
176 ControlDeformMaxDistance::execute(){
177 
178  MimmoObject * geo = getGeometry();
179  if (geo == NULL){
180  throw std::runtime_error (m_name + " : NULL pointer to linked geometry");
181  }
182  if (geo->isEmpty() ){
183  throw std::runtime_error (m_name + " : linked geometry is empty or a point cloud mesh");
184  }
185 
186  bool check = m_defField.getGeometry() == geo;
187  check = check && m_defField.getDataLocation() == MPVLocation::POINT;
188  check = check && m_defField.completeMissingData({{0.0,0.0,0.0}});
189  if(!check){
190  (*m_log)<<"Error in "<<m_name<<": Unsuitable deformation field linked"<<std::endl;
191  throw std::runtime_error (m_name + " : Unsuitable deformation field linked.");
192  }
193 
194  m_violationField.clear();
195 
196  if(!(geo->isSkdTreeSync())) geo->buildSkdTree();
197 
198  dmpvecarr3E points;
199  dmpvector1D normDef;
200  long int ID;
201  for (const auto & v : geo->getVertices()){
202  ID = v.getId();
203  points.insert(ID, geo->getVertexCoords(ID) + m_defField[ID] );
204  normDef.insert(ID, norm2(m_defField[ID]));
205  m_violationField.insert(ID, -1.0e+18);
206  }
207 
208  double dist;
209  double radius ;
210  double rate = 0.05;
211  int kmax = 200;
212  int kiter;
213  bool flag;
214  long int IDC;
215  for(const auto &ID : points.getIds()){
216  dist =1.0E+18;
217  kiter = 0;
218  flag = true;
219  radius = std::fmax(1.0E-8, normDef[ID]);
220  while(flag && kiter < kmax){
221  dist = skdTreeUtils::distance(&points[ID], geo->getSkdTree(), IDC, radius);
222  flag = (dist == 1.0E+18);
223  if(flag) radius *= (1.0+ rate*((double)flag));
224  kiter++;
225  }
226  if(kiter == kmax) dist = m_maxDist - dist;
227  m_violationField[ID] = (dist - m_maxDist);
228  }
229 
230  m_violationField.setGeometry(getGeometry());
231  m_violationField.setDataLocation(MPVLocation::POINT);
232 
233  //write log
234  std::string logname = m_name+"_violation";
235  bitpit::Logger* log = &bitpit::log::cout(logname);
236  log->setPriority(bitpit::log::DEBUG);
237  (*log)<<" violation value : " << getViolation() << std::endl;
238 
239 };

286 void
287 ControlDeformMaxDistance::plotOptionalResults(){
288  if(getGeometry() == NULL) return;
289  if(getGeometry()->isEmpty()) return;
290  if(!m_defField.completeMissingData({{0,0,0}}) || !m_violationField.completeMissingData(0.0)) return;
291 
292  //get deformation on points
293  dvecarr3E deff = m_defField.getDataAsVector();
294  //get violation field
295  dvector1D viol = m_violationField.getDataAsVector();
296 
297  //add deformation field as vector field
298  getGeometry()->getPatch()->getVTK().addData("Deformation Field", bitpit::VTKFieldType::VECTOR, bitpit::VTKLocation::POINT, deff);
299  //add violation field as a custom data field of the original geometry
300  getGeometry()->getPatch()->getVTK().addData("Violation Distance Field", bitpit::VTKFieldType::SCALAR, bitpit::VTKLocation::POINT, viol);
301  getGeometry()->getPatch()->write(m_outputPlot+"/"+m_name+std::to_string(getId()));
302 
303  //reset geometry VTK to its original setup
304  getGeometry()->getPatch()->getVTK().removeData("Deformation Field");
305  getGeometry()->getPatch()->getVTK().removeData("Violation Distance Field");
306 }
