41 double distancePointPolygon(const darray3E & p,const dvecarr3E & vertCoords){
42 
43  //calculate barycenter
44  darray3E barycenter = {{0.0,0.0,0.0}};
45  for(const auto & vv : vertCoords){
46  barycenter += vv;
47  }
48  barycenter /= double(vertCoords.size());
49 
50  //split it in subtriangles - fixed barycenter, run 2-by-2 along vertCoords to get the other two vertices
51  double distance(std::numeric_limits<double>::max());
52  std::size_t sizeV = vertCoords.size();
53  for(std::size_t i=0; i<sizeV-1; ++i){
54  distance = std::min(distance, bitpit::CGElem::distancePointTriangle(p, barycenter, vertCoords[i], vertCoords[i+1]));
55  }
56  return distance;
57 }

66 bool isPointInsideSegment(const darray3E & p, const darray3E & V0,const darray3E & V1 ){
67  darray3E Pv = p - V0;
68  darray3E segment = V1 - V0;
69  double normSegment = norm2(segment);
70  if(normSegment > std::numeric_limits<double>::min()) segment /= normSegment;
71  double V = dotProduct(Pv, segment);
72  if(norm2(Pv - V*segment)> 1.E-12) return false;
73  return ( V >=0.0 && V <= normSegment);
74 }

83 bool isPointInsideTriangle(const darray3E & p, const darray3E & V0,const darray3E & V1,const darray3E & V2){
84 
85  darray3E normal = crossProduct(V1 - V0,V2 - V0);
86  double normNormal = norm2(normal);
87  if(normNormal > std::numeric_limits<double>::min()) normal /= normNormal;
88  else return false;
89 
90  if(std::abs(dotProduct((p-V0), normal)) > std::numeric_limits<double>::min()){ return false;}
91 
92  double a1 = bitpit::CGElem::areaTriangle(p,V0,V1)/(0.5*normNormal);
93  double a2 = bitpit::CGElem::areaTriangle(p,V1,V2)/(0.5*normNormal);
94  double a3 = bitpit::CGElem::areaTriangle(p,V2,V0)/(0.5*normNormal);
95  return ( (a1+a2+a3 -1.0) <=1.0E-12 );
96 }
