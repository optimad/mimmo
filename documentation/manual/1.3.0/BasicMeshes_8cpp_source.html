710  void UStructMesh::setShape(ShapeType type){
711  //create internal shape using unique_ptr member.
712  // unlink external shape eventually
713  if(m_shape){
714 
715  m_origin_temp = m_shape.get()->getOrigin();
716  m_span_temp = m_shape.get()->getSpan();
717  m_inflimits_temp = m_shape.get()->getInfLimits();
718  m_refsystem_temp = m_shape.get()->getRefSystem();
719  m_setorigin = true;
720  m_setspan = true;
721  m_setInfLimits = true;
722  m_setRefSys = true;
723 
724  m_shape.reset(nullptr);
725  }
726 
727  darray3E origin{{0,0,0}};
728  dmatrix33E spanMat;
729  spanMat[0].fill(1); spanMat[2].fill(1); spanMat[2].fill(1);
730  spanMat[1][1] = spanMat[2][1] = 2*M_PI;
731  spanMat[2][2] = M_PI;
732 
733  if(m_setorigin){origin = m_origin_temp; m_setorigin= false;}
734  if(m_setspan){
735  for(int i=0; i<3; ++i) spanMat[i] = m_span_temp;
736  m_setspan = false;
737  }
738 
739  switch(type){
740  case ShapeType::WEDGE :
741  m_shape = std::unique_ptr<BasicShape>(new Wedge(origin,spanMat[0]));
742  break;
743  case ShapeType::CYLINDER :
744  m_shape = std::unique_ptr<BasicShape>(new Cylinder(origin,spanMat[1]));
745  break;
746  case ShapeType::SPHERE :
747  m_shape = std::unique_ptr<BasicShape>(new Sphere(origin, spanMat[2]));
748  break;
749  default://CUBE
750  m_shape = std::unique_ptr<BasicShape>(new Cube(origin, spanMat[0]));
751  break;
752  }
753 
754  if(m_setInfLimits){
755  m_shape.get()->setInfLimits(m_inflimits_temp[0],0);
756  m_shape.get()->setInfLimits(m_inflimits_temp[1],1);
757  m_shape.get()->setInfLimits(m_inflimits_temp[2],2);
758  m_setInfLimits = false;
759  }
760 
761  if(m_setRefSys){
762  m_shape.get()->setRefSystem(m_refsystem_temp);
763  m_setRefSys = false;
764  }
765 
766  m_isBuild = false;
767 }
