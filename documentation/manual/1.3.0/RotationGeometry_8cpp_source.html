173 void
174 RotationGeometry::execute(){
175 
176  if(getGeometry() == NULL){
177  (*m_log)<<m_name + " : NULL pointer to linked geometry found"<<std::endl;
178  throw std::runtime_error(m_name + "NULL pointer to linked geometry found");
179  }
180 
181  if(getGeometry()->isEmpty()){
182  (*m_log)<<m_name + " : empty linked geometry found"<<std::endl;
183  }
184 
185  checkFilter();
186 
187  m_displ.clear();
188  m_displ.setDataLocation(mimmo::MPVLocation::POINT);
189  m_displ.reserve(getGeometry()->getNVertices());
190  m_displ.setGeometry(getGeometry());
191 
192 
193  //compute coefficients and constant vectors of rodriguez formula
194  double a = std::cos(m_alpha);
195  darray3E b = (1.0 - std::cos(m_alpha)) * m_direction;
196  double c = std::sin(m_alpha);
197 
198  darray3E point, rotated;
199  long ID;
200  darray3E value;
201  for (const auto & vertex : m_geometry->getVertices()){
202  point = vertex.getCoords();
203  ID = vertex.getId();
204 
205  point -= m_origin;
206  //rodrigues formula
207  rotated = a * point +
208  b * dotProduct(m_direction, point) +
209  c * crossProduct(m_direction, point);
210 
211  rotated += m_origin;
212  point += m_origin;
213 
214  value = (rotated-point)*m_filter[ID];
215  m_displ.insert(ID, value);
216 
217  }
218 };

232 void
233 RotationGeometry::checkFilter(){
234  bool check = m_filter.getDataLocation() == mimmo::MPVLocation::POINT;
235  check = check && m_filter.completeMissingData(0.0);
236  check = check && m_filter.getGeometry() == getGeometry();
237 
238  if (!check){
239  m_log->setPriority(bitpit::log::Verbosity::DEBUG);
240  (*m_log)<<"Not valid filter found in "<<m_name<<". Proceeding with default unitary field"<<std::endl;
241  m_log->setPriority(bitpit::log::Verbosity::NORMAL);
242 
243  m_filter.clear();
244  m_filter.setGeometry(m_geometry);
245  m_filter.setDataLocation(mimmo::MPVLocation::POINT);
246  m_filter.reserve(getGeometry()->getNVertices());
247  for (const auto & vertex : getGeometry()->getVertices()){
248  m_filter.insert(vertex.getId(), 1.0);
249  }
250  }
