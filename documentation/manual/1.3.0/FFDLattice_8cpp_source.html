583  if(container == NULL){
584  (*m_log)<<m_name + " : NULL pointer to linked geometry found"<<std::endl;
585  throw std::runtime_error(m_name + "NULL pointer to linked geometry found");
586  }
587  if(container->isEmpty()){
588  (*m_log)<<m_name + " : empty linked geometry found"<<std::endl;
589  }
590 
591  //reset displacement in a unique vector
592  m_gdispl.clear();
593  m_gdispl.setDataLocation(mimmo::MPVLocation::POINT);

595  m_gdispl.setGeometry(getGeometry());
596

598  if(container->isSkdTreeSupported() && !container->isSkdTreeSync()) container->buildSkdTree();
599  else if(!container->isKdTreeSync()) container->buildKdTree();
600 
601  if(!isBuilt()){
602  build();
603  }
604 
605  livector1D map;
606  dvecarr3E localdef = apply(map);
607 
608  long int ID;
609  darray3E zero;
610  zero.fill(0.0);
611  for (const auto & vertex : container->getVertices()){
612  ID = vertex.getId();
613  m_gdispl.insert(ID, zero);
614  }
615  {
616  int counter = 0;
617  for(const auto &mapp: map){
618  m_gdispl[mapp] = localdef[counter];
619  ++counter;
620  }
621  }

771 darray3E
772 FFDLattice::nurbsEvaluator(darray3E & pointOr){
773 
774  darray3E point = transfToLocal(pointOr);
775  darray3E scaling = getShape()->getScaling();
776 
777  ivector1D knotInterval(3,0);
778  dvector2D BSbasis(3);
779  dvector1D valH(4,0), temp1(4,0),temp2(4,0), zeros(4,0);
780  int uind, vind, wind, index;
781 
782  dvecarr3E* displ = getDisplacements();
783 
784  int i0 = m_mapdeg[0];
785  int i1 = m_mapdeg[1];
786  int i2 = m_mapdeg[2];
787 
788  iarray3E mappedIndex;
789 
790  for(int i=0; i<3; i++){
791  knotInterval[i] = getKnotInterval(point[i],i);
792  BSbasis[i] = basisITS0(knotInterval[i], i, point[i]);
793  }
794 
795  uind = knotInterval[i0] - m_deg[i0];
796  vind = knotInterval[i1] - m_deg[i1];
797  wind = knotInterval[i2] - m_deg[i2];
798 
799  for(int i=0; i<=m_deg[i0]; ++i){
800 
801  mappedIndex[0] = uind+i;
802  temp1 = zeros;
803 
804  for(int j=0; j<=m_deg[i1]; ++j){
805 
806  mappedIndex[1] = vind+j;
807  temp2 = zeros;
808 
809  for(int k=0; k<=m_deg[i2]; ++k){
810 
811  mappedIndex[2] = wind+k;
812  index = accessMapNodes(mappedIndex[0],mappedIndex[1],mappedIndex[2]);
813 
814  for(int intv=0; intv<3; ++intv){
815  temp2[intv] += BSbasis[i2][k]*m_weights[m_intMapDOF[index]]*(*displ)[m_intMapDOF[index]][intv];
816  }
817  temp2[3] += BSbasis[i2][k]*m_weights[index];
818  }
819  for(int intv=0; intv<4; ++intv){
820  temp1[intv] += BSbasis[i1][j]*temp2[intv];
821  }
822 
823  }
824  for(int intv=0; intv<4; ++intv){
825  valH[intv] += BSbasis[i0][i]*temp1[intv];
826  }
827  }
828 
829  darray3E outres;
830  if(isDisplGlobal()){
831  for(int i=0; i<3; ++i){
832  outres[i] = valH[i]/valH[3];
833  }
834 
835  }else{
836  //summing scaled displ in local ref frame;
837  for(int i=0; i<3; ++i){
838  point[i] += valH[i]/(valH[3]*scaling[i]);
839  }
840 
841  //get final displ in global ref frame:
842  outres = transfToGlobal(point) - pointOr;
843  }
844  return(outres);
845 
846 };
