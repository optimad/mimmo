72 IOConnections_MIMMO::absorbConnections(const bitpit::Config & slotXML, bool debug){
73 
74  if(debug)
75  m_log->setPriority(bitpit::log::NORMAL);
76 
77  if(slotXML.getSectionCount() == 0) {
78  (*m_log)<<"IOConnections::absorbConnections does not found any connection to read in current XML slot."<<std::endl;
79  return;
80  }
81  bool checkConnection = true;
82 
83  for( auto & sect : slotXML.getSections()){
84  std::string snd_str;
85  std::string rcv_str;
86  std::string sndP_str;
87  std::string rcvP_str;
88 
89  if(sect.second->hasOption("sender")) snd_str = sect.second->get("sender");
90  if(sect.second->hasOption("receiver")) rcv_str = sect.second->get("receiver");
91  if(sect.second->hasOption("senderPort")) sndP_str = sect.second->get("senderPort");
92  if(sect.second->hasOption("receiverPort")) rcvP_str = sect.second->get("receiverPort");
93 
94  snd_str = bitpit::utils::string::trim(snd_str);
95  rcv_str = bitpit::utils::string::trim(rcv_str);
96  sndP_str = bitpit::utils::string::trim(sndP_str);
97  rcvP_str = bitpit::utils::string::trim(rcvP_str);
98 
99 
100  auto itSend = m_mapConn.find(snd_str);
101  auto itRece = m_mapConn.find(rcv_str);
102  bool checkSP = mimmo::PortManager::instance().containsPort(sndP_str);
103  bool checkRP = mimmo::PortManager::instance().containsPort(rcvP_str);
104 
105  if(itSend == m_mapConn.end() || itRece == m_mapConn.end() || !checkSP || !checkRP ){
106  (*m_log)<<"---------------------------------------------"<<std::endl;
107  (*m_log)<<"sender: "<<snd_str<<std::endl;
108  (*m_log)<<"receiver: "<<rcv_str<<std::endl;
109  (*m_log)<<"senderPort: "<<sndP_str<<std::endl;
110  (*m_log)<<"receiverPort: "<<rcvP_str<<std::endl;
111  (*m_log)<<"---------------------------------------------"<<std::endl;
112  (*m_log)<<""<<std::endl;
113  (*m_log)<<"IOConnections::absorbConnections does not found any sender/receiver and/or port compatible with API."<<std::endl;
114  (*m_log)<<"Please control connection data entries in your XML file"<<std::endl;
115  (*m_log)<<""<<std::endl;
116  continue;
117  }
118 
119  m_log->setPriority(bitpit::log::NORMAL);
120  bool check = pin::addPin(itSend->second, itRece->second, sndP_str, rcvP_str);
121  if(!debug) m_log->setPriority(bitpit::log::DEBUG);
122 
123  if(!check){
124  (*m_log)<<"---------------------------------------------"<<std::endl;
125  (*m_log)<<"sender: "<<snd_str<<std::endl;
126  (*m_log)<<"receiver: "<<rcv_str<<std::endl;
127  (*m_log)<<"senderPort: "<<sndP_str<<std::endl;
128  (*m_log)<<"receiverPort: "<<rcvP_str<<std::endl;
129  (*m_log)<<"---------------------------------------------"<<std::endl;
130  (*m_log)<<""<<std::endl;
131  (*m_log)<<"IOConnections::absorbConnections failed creating connection."<<std::endl;
132  (*m_log)<<""<<std::endl;
133  checkConnection = false;
134  }
135  else{
136  (*m_log)<<"IOConnections::absorbConnections successfully created connection."<<std::endl;
137  }
138  }
139 
140  m_log->setPriority(bitpit::log::DEBUG);
141  if(!checkConnection){
142  throw std::runtime_error ("IOConnections : one or more failed pin connections found.");
143  }
144  return;

158 IOConnections_MIMMO::flushConnections(bitpit::Config & slotXML, bool debug ){
159 
160  int counterGlob = 0;
161  for( auto object : m_mapConn){
162 
163  std::string sender = m_invMapConn[object.second]; //recover sender name
164  auto mapPOut = (object.second)->getPortsOut(); //get its slot of output port
165 
166 
167  for(auto & pOut : mapPOut ){
168 
169  std::string senderPort = pOut.first; //gets the Port Type of sender
170 
171  std::vector<BaseManipulation * > links = (pOut.second)->getLink(); //get all possible links to this specific port
172  std::vector<PortID> lids = (pOut.second)->getPortLink();
173 
174  int counter = 0;
175  for( auto & val : links){
176  std::string receiver = m_invMapConn[val];
177  std::string receiverPort = lids[counter];
178 
179  std::string title = "conn" + std::to_string(counter+counterGlob);
180  bitpit::Config::Section & conn = slotXML.addSection(title);
181  conn.set("sender", sender);
182  conn.set("senderPort", senderPort);
183  conn.set("receiver", receiver);
184  conn.set("receiverPort", receiverPort);
185 
186  ++counter;
187  }//loop on possible links on the same port
188  counterGlob += counter;
189  }//loop on all the port of the single object
190  }//loop on all instantiated object.
191 
192  if(debug)
193  m_log->setPriority(bitpit::log::NORMAL);
194  (*m_log)<<"IOConnections::flushConnections wrote "<< slotXML.getSectionCount()<< " active connections."<<std::endl;
195  m_log->setPriority(bitpit::log::DEBUG);
196  return;
197 };
