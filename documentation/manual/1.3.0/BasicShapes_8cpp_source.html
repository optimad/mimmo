142 void BasicShape::setRefSystem(darray3E axis0, darray3E axis1, darray3E axis2){
143 
144  double a = norm2(axis0);
145  double b = norm2(axis1);
146  double c = norm2(axis2);
147 
148  bool check = ( a>0.0 && b >0.0 && c > 0.0);
149  if (! check){
150  assert(false && "one or more 0-vector passed as arguments in BasicShape::SetRefSystem method");
151  return;
152  }
153 
154  axis0 /= a;
155  axis1 /= b;
156  axis2 /= c;
157 
158  double tol = 1.0e-12;
159  check = std::abs(dotProduct(axis0,axis1)) + std::abs(dotProduct(axis1,axis2)) + std::abs(dotProduct(axis0,axis2));
160  if(check > tol) {
161  assert(false && "not orthogonal axes passed as arguments in BasicShape::SetRefSystem method");
162  return;
163  }
164 
165  m_sdr[0] = axis0;
166  m_sdr[1] = axis1;
167  m_sdr[2] = axis2;
168 }

175 void BasicShape::setRefSystem(int label, darray3E axis){
176 
177  if(label <0 || label >2 ){
178  assert(false && "no correct label passed as argument in BasicShape::setRefSystem");
179  return;
180  }
181 
182  double a = norm2(axis);
183  if (!(a > 0.0) ){
184  assert(false && "0-vector passed as axis argument in BasicShape::SetRefSystem method");
185  return;
186  }
187 
188  m_sdr[label] = axis/a;
189  dvecarr3E point_mat(3,darray3E{0,0,0});
190  point_mat[0][0] = point_mat[1][1]= point_mat[2][2]=1.0;
191 
192  int next_label = (label + 1)%3;
193  int fin_label = (label + 2)%3;
194 
195  double pj = dotProduct(point_mat[next_label], m_sdr[label]);
196  m_sdr[next_label] = point_mat[next_label] - pj*m_sdr[label];
197  m_sdr[next_label] = m_sdr[next_label]/norm2(m_sdr[next_label]);
198 
199  m_sdr[fin_label] = crossProduct(m_sdr[label],m_sdr[next_label]);
200  m_sdr[fin_label] = m_sdr[fin_label]/norm2(m_sdr[fin_label]);
201 }

620 void BasicShape::searchKdTreeMatches(bitpit::KdTree<3,bitpit::Vertex,long> & tree, livector1D & result ){
621 
622  //1st step get data
623  std::vector<int> candidates;
624  std::vector< std::pair<int, int> > nodeStackLoc; //cointains touple with kdNode id and level label.
625 
626  nodeStackLoc.push_back(std::make_pair(0, 0));
627  while(!nodeStackLoc.empty()){
628  std::pair<int, int> toupleId = nodeStackLoc.back();
629  bitpit::KdNode<bitpit::Vertex, long> & target = tree.nodes[toupleId.first];
630  nodeStackLoc.pop_back();
631 
632  //add children to the stack, if node has any of them.
633  uint32_t check = intersectShapePlane((toupleId.second)%3, target.object_->getCoords());
634  if((check%2 == 0) && (target.lchild_ >= 0)){
635  nodeStackLoc.push_back(std::make_pair(target.lchild_, (toupleId.second) + 1));
636  }
637  if((check > 0) && (target.rchild_ >= 0)){
638  nodeStackLoc.push_back(std::make_pair(target.rchild_, (toupleId.second) + 1));
639  }
640 
641  //push node as candidate if is in the raw bounding box of the shape
642  if (bitpit::CGElem::intersectPointBox(target.object_->getCoords(), m_bbox[0], m_bbox[1], 3)){
643  candidates.push_back(toupleId.first);
644  }
645  }
646 
647  result.clear();
648  result.reserve(candidates.size());
649  for (const auto & idCand : candidates){
650  bitpit::KdNode<bitpit::Vertex, long> & target = tree.nodes[idCand];
651  if(isPointIncluded(target.object_->getCoords())){
652  result.push_back(target.label);
653  }
654  }
655 };

665 void BasicShape::searchBvTreeMatches(bitpit::PatchSkdTree & tree, bitpit::PatchKernel * geo, livector1D & result){
666 
667  std::size_t rootId = 0;
668 
669  std::vector<std::size_t> toBeCandidates;
670  std::vector<long> sureCells;
671  sureCells.reserve((geo->getCellCount()));
672 
673  std::vector<size_t> nodeStack;
674  nodeStack.push_back(rootId);
675  while(!nodeStack.empty()){
676  std::size_t nodeId = nodeStack.back();
677  const SkdNode & node = tree.getNode(nodeId);
678  nodeStack.pop_back();
679 
680 
681  //second step: if the current node AABB does not intersect or does not completely contains the Shape, then thrown it away and continue.
682  if(!intersectShapeAABBox(node.getBoxMin(), node.getBoxMax()) ){
683  continue;
684  }
685 
686  //now the only option is to visit node's children. If node is not leaf, add children to the stack,
687  //otherwise add current node as a possible candidate in shape inclusion.
688  bool isLeaf = true;
689  for (int i = SkdNode::CHILD_BEGIN; i != SkdNode::CHILD_END; ++i) {
690  SkdNode::ChildLocation childLocation = static_cast<SkdNode::ChildLocation>(i);
691  std::size_t childId = node.getChildId(childLocation);
692  if (childId != SkdNode::NULL_ID) {
693  isLeaf = false;
694  nodeStack.push_back(childId);
695  }
696  }
697 
698  if (isLeaf) {
699  toBeCandidates.push_back(nodeId);
700  }
701  }
702 
703  result.clear();
704  result.reserve(toBeCandidates.size() + sureCells.size());
705  for (const auto & idCand : toBeCandidates){
706  const SkdNode &node = tree.getNode(idCand);
707  std::vector<long> cellids = node.getCells();
708  for(const auto & id : cellids){
709  if(isSimplexIncluded(geo, id)){
710  result.push_back(id);
711  }
712  }
713  }
714  result.insert(result.end(), sureCells.begin(), sureCells.end());
715 };

780 darray3E Cube::toWorldCoord(const darray3E &point){
781 
782  darray3E work, work2;
783 
784  //unscale your local point
785  for(int i =0; i<3; ++i){
786  work[i] = point[i]*m_scaling[i];
787  }
788 
789  //return to local xyz system
790  // -> cube, doing nothing
791 
792  //unapply change to local sdr transformation
793  linearalgebra::matmul(work, m_sdr, work2);
794  //unapply origin translation
795  return(work2 + m_origin);
796 };

804 darray3E Cube::toLocalCoord(const darray3E &point){
805 
806  darray3E work, work2;
807 
808  //unapply origin translation
809  work = point - m_origin;
810 
811  //apply change to local sdr transformation
812  dmatrix33E transp = linearalgebra::transpose(m_sdr);
813  linearalgebra::matmul(work, transp, work2);
814 
815  //get to proper local system
816  // -> cube, doing nothing
817 
818  //scale your local point
819  for(int i =0; i<3; ++i){
820  work[i] = work2[i]/m_scaling[i];
821  }
822 
823  return(work);
824

933 bool Cube::intersectShapeAABBox(const darray3E &bMin, const darray3E &bMax){
934 
935  dvecarr3E points(8, bMin);
936  points[1][0] = points[2][0] = points[5][0]=points[6][0]= bMax[0];
937  points[2][1] = points[3][1] = points[6][1]=points[7][1]= bMax[1];
938  points[4][2] = points[5][2] = points[6][2]=points[7][2]= bMax[2];
939 
940  for(auto &val:points) val += -1.0*m_origin;
941 
942  for(int j=0; j<3; ++j){
943  double ref1 = -0.5*m_scaling[j];
944  double ref2 = 0.5*m_scaling[j];
945 
946  double tmin, tmax, t;
947  t= dotProduct(points[0],m_sdr[j]);
948  tmax=tmin=t;
949
