164 void
165 SwitchVectorField::plotOptionalResults(){
166 
167  if (m_result.size() == 0 || getGeometry() == NULL) return;
168  if (getGeometry()->isEmpty()) return;
169 
170  bitpit::VTKLocation loc = bitpit::VTKLocation::UNDEFINED;
171  switch(m_loc){
172  case MPVLocation::POINT :
173  loc = bitpit::VTKLocation::POINT;
174  break;
175  case MPVLocation::CELL :
176  loc = bitpit::VTKLocation::CELL;
177  break;
178  default:
179  (*m_log)<<"Warning: Undefined Reference Location in plotOptionalResults of "<<m_name<<std::endl;
180  (*m_log)<<"Interface locations are not supported in VTU writing." <<std::endl;
181  break;
182  }
183 
184  if(loc == bitpit::VTKLocation::UNDEFINED) return;
185 
186  //check size of field and adjust missing values to zero for writing purposes only.
187  dmpvecarr3E field_supp = m_result;
188  if(!field_supp.completeMissingData({{0.0,0.0,0.0}})) return;
189  dvecarr3E field = field_supp.getDataAsVector();
190 
191  if(getGeometry()->getType() != 3){
192  getGeometry()->getPatch()->getVTK().addData("field", bitpit::VTKFieldType::VECTOR, loc, field);
193  getGeometry()->getPatch()->write(m_outputPlot+"/"+m_name+std::to_string(getId()));
194  getGeometry()->getPatch()->getVTK().removeData("field");
195  }else{
196  if(loc == bitpit::VTKLocation::CELL){
197  (*m_log)<<"Warning: Attempt writing Cell data field on cloud point in plotOptionalResults of "<<m_name<<std::endl;
198  return;
199  }
200  liimap mapDataInv;
201  dvecarr3E points = getGeometry()->getVerticesCoords(&mapDataInv);
202  int size = points.size();
203  ivector2D connectivity(size, ivector1D(1));
204  for(int i=0; i<size; ++i) connectivity[i][0]=i;
205  bitpit::VTKUnstructuredGrid output(m_outputPlot,m_name+std::to_string(getId()), bitpit::VTKElementType::VERTEX);
206  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
207  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
208  output.setDimensions(connectivity.size(), points.size());
209  output.addData("field", bitpit::VTKFieldType::VECTOR, loc, field);
210  output.setCodex(bitpit::VTKFormat::APPENDED);
211  output.write();
212  }

220 bool
221 SwitchVectorField::mswitch(){
222 
223  if (getGeometry() == NULL) return true; //false;
224 
225  m_result.clear();
226 
227  //Extract by link to geometry
228  for (const auto & field : m_fields){
229  if (field.getGeometry() == getGeometry()){
230  m_result = field;
231  //geometry and location are copied from field.
232  return true;
233  }
234  }
235 
236  //Extract by geometric mapping if active and if no positive match is found.
237  if (m_mapping && getGeometry()->getType() != 3){
238 
239  m_result.setGeometry(getGeometry());
240  m_result.setDataLocation(m_loc);
241 
242  ExtractVectorField * ef = new ExtractVectorField();
243  ef->setGeometry(getGeometry());
244  ef->setMode(ExtractMode::MAPPING);
245  ef->setTolerance(m_tol);
246 
247  //create map for overlapping ids purpose;
248  std::unordered_map<long, int> idRepetition;
249 
250  for (dmpvecarr3E & field : m_fields){
251  ef->setField(&field);
252  bool check = ef->extract();
253  if(!check) continue;
254 
255  dmpvecarr3E * temp = ef->getExtractedField();
256  dmpvecarr3E::iterator itB;
257  auto itE = temp->end();
258  long id;
259  for(itB = temp->begin(); itB != itE; ++itB){
260  id = itB.getId();
261  if(!m_result.exists(id)){
262  m_result.insert(id, *itB);
263  }else{
264  m_result[id] += *itB;
265  if(idRepetition.count(id)>0){
266  ++idRepetition[id];
267  }else{
268  idRepetition[id] = 2;
269  }
270  }
271  }
272  }
273 
274  //resolve overlapping ids by averaging correspondent value;
275 
276  for(auto &itval : idRepetition){
277  m_result[itval.first] /= double(itval.second);
278  }
279 
280  delete ef;
281  }
282 
283  if (m_result.isEmpty()) return false;
284  return true;
285 }
