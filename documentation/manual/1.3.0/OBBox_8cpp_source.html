286 void
287 OBBox::execute(){
288  if(m_listgeo.empty()){
289  throw std::runtime_error(m_name + " : found empty list of geometries");
290  };
291 
292  darray3E pmin, pmax;
293  pmin.fill(1.e18);
294  pmax.fill(-1.e18);
295  double val;
296  dmatrix33E eye;
297  {
298  int count = 0;
299  for(auto & local: eye) {
300  local.fill(0);
301  local[count] = 1.0;
302  m_axes[count] = local;
303  ++count;
304  }
305  }
306 
307  std::unordered_map<MimmoObject*, int>::iterator itB = m_listgeo.begin();
308  while(itB != m_listgeo.end() && itB->second != 3){
309  itB++;
310  }
311  bool allCloud = (itB != m_listgeo.end());
312 
313  if(!m_forceAABB){
314 
315  dmatrix33E covariance, covContributes;
316  darray3E spectrum;
317  darray3E etaPoint;
318 
319 
320  etaPoint = evaluateMassCenter(getGeometries(), allCloud);
321  assemblyCovContributes(getGeometries(), allCloud, covContributes);
322  covariance = evaluateCovarianceMatrix(covContributes, etaPoint);
323 
324  m_axes = eigenVectors(covariance, spectrum);
325  adjustBasis(m_axes, spectrum);
326  }
327 
328 
329  for(auto ptr : getGeometries()){
330  for(auto & vert: ptr->getVertices()){
331  darray3E coord = vert.getCoords();
332  for(int i=0;i<3; ++i){
333  val = dotProduct(coord, m_axes[i]);
334  pmin[i] = std::fmin(pmin[i], val);
335  pmax[i] = std::fmax(pmax[i], val);
336  }
337  }
338  }
339 
340  dmatrix33E trasp = bitpit::linearalgebra::transpose(m_axes);
341  m_span = pmax - pmin;
342  //check if one of the span goes to 0;
343  double avg_span = 0.0;
344  for(auto & val: m_span) avg_span+=val;
345  avg_span /= 3.0;
346 
347  for(auto &val : m_span) {
348  val = std::fmax(val, 1.E-04*avg_span);
349  }
350 
351  darray3E originLoc = 0.5*(pmin+pmax);
352  for(int i=0; i<3; ++i){
353  m_origin[i] = dotProduct(originLoc, trasp[i]);
354  }
355 
356  double volOBB = m_span[0]*m_span[1]*m_span[2];
357 
358  if(!m_forceAABB){ //check if AABB get a better result
359  pmin.fill(1.e18);
360  pmax.fill(-1.e18);
361 
362  for(auto ptr : getGeometries()){
363  for(auto & vert: ptr->getVertices()){
364  darray3E coord = vert.getCoords();
365  for(int i=0;i<3; ++i){
366  pmin[i] = std::fmin(pmin[i], coord[i]);
367  pmax[i] = std::fmax(pmax[i], coord[i]);
368  }
369  }
370  }
371 
372  darray3E span2 = pmax - pmin;
373  darray3E orig = 0.5*(pmin+pmax);
374  //check if one of the span goes to 0;
375  avg_span = 0.0;
376  for(auto & val: span2) avg_span+=val;
377  avg_span /= 3.0;
378 
379  for(auto &val : span2) {
380  val = std::fmax(val, 1.E-04*avg_span);
381  }
382 
383  double volAAA =span2[0]*span2[1]*span2[2];
384 
385 
386  if(volAAA <= volOBB){
387  for(int i=0; i<3; ++i) m_axes[i] = eye[i];
388  m_span = span2;
389  m_origin = orig;
390  }
391  }

453  dmatrix33E result;
454 
455  for(auto & val: result) val.fill(0);
456 
457  for(int i=0; i<3; ++i){

477 void
478 OBBox::assemblyCovContributes(std::vector<MimmoObject*> list, bool flag, dmatrix33E & covContributes){
479 
480  dmatrix33E loc;
481  for(auto & val:covContributes) val.fill(0.0);
482  int size, counter;
483  dvector1D area;
484  darray3E temp;
485  double areaTot = 0.0;
486 
487  if(flag){
488  for(auto geo: list){
489  for(auto & val:loc) val.fill(0.0);
490  size = geo->getNVertex();
491 
492  //evaluate local contributes to covariance
493  for(auto & vert: geo->getVertices()){
494  temp = vert.getCoords();
495  for(int j=0; j<3; ++j){
496  for(int k=j; k<3; ++k){
497  loc[j][k] += temp[j]*temp[k];
498  }
499  }
500  }
501  //store temporarely in covContributes
502  counter = 0;
503  for(auto & val: covContributes){
504  val += loc[counter];
505  ++counter;
506  }
507 
508  areaTot += double(size);
509  }
510  }else{
511  for(auto geo: list){
512  for(auto & val:loc) val.fill(0.0);
513  size = geo->getNCells();
514  bitpit::SurfUnstructured * tri = static_cast<bitpit::SurfUnstructured * >(geo->getPatch());
515  area.resize(size);
516  counter = 0;
517  for(auto &cell : tri->getCells()){
518  area[counter] = tri->evalCellArea(cell.getId());
519  areaTot +=area[counter];
520  ++counter;
521  }
522 
523  counter = 0;
524  darray3E centroid;
525  for(auto & cell: tri->getCells()){
526  centroid = tri->evalCellCentroid(cell.getId());
527 
528  for(int j=0; j<3; ++j){
529  for(int k=j; k<3; ++k){
530  loc[j][k] += area[counter]*centroid[j]*centroid[k];
531  }
532  }
533  ++counter;
534  }
535 
536  //store temporarely in covContributes
537  counter = 0;
538  for(auto & val: covContributes){
539  val += loc[counter];
540  ++counter;
541  }
542  }
543  }
544 
545  covContributes[1][0] = covContributes[0][1];
546  covContributes[2][0] = covContributes[0][2];
547  covContributes[2][1] = covContributes[1][2];
548 
549  counter = 0;
550  for(auto & val: covContributes){
551  val /= areaTot;
552  ++counter;
553  }
554 };

562 darray3E
563 OBBox::evaluateMassCenter(std::vector<MimmoObject *> list, bool flag){
564 
565  long size = 0;
566  darray3E centermass = {{0,0,0}}, eta;
567  double areaTot = 0.0;
568  int counter;
569 
570  if(flag){
571  for(auto geo : list){
572  size = geo->getNVertex();
573  eta.fill(0.0);
574  //evaluate eta;
575  for(auto & vert: geo->getVertices()){
576  eta += vert.getCoords();
577  }
578  centermass += eta;
579  areaTot += double(size);
580  }
581  }else{
582  for(auto geo : list){
583  size = geo->getNCells();
584  bitpit::SurfUnstructured * tri = static_cast<bitpit::SurfUnstructured * >(geo->getPatch());
585  dvector1D area(size);
586  counter = 0;
587  for(auto &cell : tri->getCells()){
588  area[counter] = tri->evalCellArea(cell.getId());
589  areaTot +=area[counter];
590  ++counter;
591  }
592 
593  //evaluate eta;
594  eta.fill(0.0);
595  counter = 0;
596  for(auto & cell: tri->getCells()){
597  eta += tri->evalCellCentroid(cell.getId())*area[counter];
598  ++counter;
599  }
600  centermass += eta;
601  }
602  }
603 
604  centermass /=areaTot;
605  return centermass;
606 };

614 dmatrix33E
615 OBBox::eigenVectors( dmatrix33E & matrix, darray3E & eigenvalues){
616 
617  dmatrix33E result;
618  double * a = new double [9];
619  double * u = new double [9];
620  double * vt = new double [9];
621  double * s = new double[3];
622  double * superb = new double[3];
623 
624  int k=0;
625  for(int i=0; i<3; i++){
626  for(int j=0; j<3; j++){
627  //transpose
628  a[k] = matrix[j][i];
629  k++;
630  }
631  }
632 
633  LAPACKE_dgesvd( LAPACK_COL_MAJOR, 'N','S', 3, 3, a, 3, s, u, 3, vt, 3, superb);
634 
635  //solution norm
636  for (int i=0; i<9; i++){
637  result[i/3][i%3] = vt[i];
638  }
639  for(int i=0; i<2; ++i) {
640  result[i] /= norm2(result[i]);
641  eigenvalues[i] = s[i];
642  }
643  result[2] = crossProduct(result[0],result[1]);
644  eigenvalues[2] = s[2];
645 
646  delete [] a; a = NULL;
647  delete [] u; u = NULL;
648  delete [] vt; vt = NULL;
649  delete [] s; s = NULL;
650  delete [] superb; superb = NULL;
651 
652  return result;
653 }

664 void
665 OBBox::adjustBasis(dmatrix33E & eigVec, darray3E & eigVal){
666 
667  darray3E diff;
668  double tol = 1.0E-3;
669 
670  diff[0] = std::abs(eigVal[1]-eigVal[0]);
671  diff[1] = std::abs(eigVal[2]-eigVal[1]);
672  diff[2] = std::abs(eigVal[0]-eigVal[2]);
673 
674  if(diff[0]>tol && diff[1]>tol && diff[2]>tol) return;
675  if(diff[0]<=tol && diff[1]<=tol && diff[2]<= tol){
676  int counter=0;
677  for(auto & val : eigVec){
678  val.fill(0.0);
679  val[counter] = 1.0;
680  ++counter;
681  }
682  return;
683  }
684 
685  int guess = 0, third =1, stable=2;
686 
687  if(diff[1] <= tol) {
688  guess = 1;
689  third = 2;
690  stable= 0;
691  }
692 
693  if(diff[2] <= tol) {
694  guess = 2;
695  third = 0;
696  stable= 1;
697  }
698 
699  dmatrix33E axes; //, trasp;
700  int counter=0;
701  for(const auto & val: eigVec){
702  axes[counter] = val;
703  ++counter;
704  }
705 
706  darray3E guessVec = axes[guess];
707  darray3E refVec = axes[stable];
708 
709  int nstage = 15;
710  int niterations = 8;
711 
712  double distance = M_PI/(2.0*(nstage));
713  double volume, theta, val;
714  std::map<double, double> mapval;
715  darray3E pmin, pmax, temp;
716 
717  for(int i=0; i<nstage; ++i){
718 
719  theta = i*distance;
720 
721  axes[guess] = guessVec*std::cos(theta) + std::sin(theta)*crossProduct(refVec, guessVec);
722  axes[third] = crossProduct(axes[stable],axes[guess]);
723 
724  pmin.fill(1.e18);
725  pmax.fill(-1.e18);
726 
727  for(auto geo : getGeometries()){
728  for(auto & vert: geo->getVertices()){
729  darray3E coord = vert.getCoords();
730  for(int i=0;i<3; ++i){
731  val = dotProduct(coord, axes[i]); //trasp[i]);
732  pmin[i] = std::fmin(pmin[i], val);
733  pmax[i] = std::fmax(pmax[i], val);
734  }
735  }
736  }
737 
738  temp = pmax - pmin;
739  volume = temp[0]*temp[1]*temp[2];
740 
741  mapval[volume] = theta;
742  }
743 
744  int it =0;
745  while (it < niterations){
746 
747 
748  distance /= 2.0;
749 
750  double thetadx = mapval.begin()->second + distance;
751  if(thetadx >= M_PI/2.0) thetadx += -1.0*M_PI/2.0;
752 
753  double thetasx = mapval.begin()->second - distance;
754  if(thetasx <= 0.0) thetasx += M_PI/2.0;
755 
756  //evaluate on the right
757  axes[guess] = guessVec*std::cos(thetadx) + std::sin(thetadx)*crossProduct(refVec, guessVec);
758  axes[third] = crossProduct(axes[stable],axes[guess]);
759 
760 
761  //trasp = bitpit::linearalgebra::transpose(axes);
762  pmin.fill(1.e18);
763  pmax.fill(-1.e18);
764  for(auto geo : getGeometries()){
765  for(auto & vert: geo->getVertices()){
766  darray3E coord = vert.getCoords();
767  for(int i=0;i<3; ++i){
768  val = dotProduct(coord, axes[i]); //trasp[i]);
769  pmin[i] = std::fmin(pmin[i], val);
770  pmax[i] = std::fmax(pmax[i], val);
771  }
772  }
773  }
774 
775  temp = pmax - pmin;
776  volume = temp[0]*temp[1]*temp[2];
777 
778  mapval[volume] = thetadx;
779 
780  //evaluate on the left
781  axes[guess] = guessVec*std::cos(thetasx) + std::sin(thetasx)*crossProduct(refVec, guessVec);
782  axes[third] = crossProduct(axes[stable],axes[guess]);
783 
784 
785  pmin.fill(1.e18);
786  pmax.fill(-1.e18);
787  for(auto geo : getGeometries()){
788  for(auto & vert: geo->getVertices()){
789  darray3E coord = vert.getCoords();
