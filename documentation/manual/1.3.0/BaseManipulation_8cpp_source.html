541 void
542 BaseManipulation::exec(){
543 
544  if (!MIMMO_EXPERT){
545  std::map<int, std::vector<PortIn*> > families;
546  std::map<int, std::vector<PortID> > familiesID;
547  for (std::unordered_map<PortID, PortIn*>::iterator i=m_portIn.begin(); i!=m_portIn.end(); i++){
548  if (i->second->isMandatory()){
549  families[i->second->getFamily()].push_back(i->second);
550  familiesID[i->second->getFamily()].push_back(i->first);
551  }
552  }
553  std::map<int, std::vector<PortID> >::iterator itID = familiesID.begin();
554  for (std::map<int, std::vector<PortIn*> >::iterator i=families.begin(); i!=families.end(); i++){
555  if (i->first == 0){
556  int count = 0;
557  for (auto ip : i->second){
558  std::vector<BaseManipulation*> linked = ip->getLink();
559  if (linked.size() == 0){
560  (*m_log) << "error: " << m_name << " mandatory port " << itID->second[count] << " not linked -> exit! " << std::endl;
561  throw std::runtime_error (m_name + " mandatory port " + itID->second[count] + " not linked");
562  }
563  count++;
564  }
565  }
566  else{
567  bool check = false;
568  for (auto ip : i->second){
569  std::vector<BaseManipulation*> linked = ip->getLink();
570  check = check || linked.size();
571  }
572  if (!check){
573  (*m_log) << "error: " << m_name << " none of mandatory ports " << itID->second << " linked -> exit! " << std::endl;
574  std::string ports;
575  for (auto p : itID->second)
576  ports += " " + p + " ";
577  throw std::runtime_error (m_name + " none of mandatory ports " + ports + " linked");
578  }
579  }
580  itID++;
581  }
582  }
583 
584  if (m_active) execute();
585 
586  for (std::unordered_map<PortID, PortOut*>::iterator i=m_portOut.begin(); i!=m_portOut.end(); i++){
587  std::vector<BaseManipulation*> linked = i->second->getLink();
588  if (linked.size() > 0){
589  i->second->exec();
590  }
591  }
592 
593  if(isPlotInExecution()) plotOptionalResults();
594  if(isApply()) apply();
595 }
