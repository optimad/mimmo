41 void test00002() {
42 
43  /* Creation of mimmo containers.
44  * Input and output MimmoGeometry are instantiated
45  * as two different objects (no loop in chain are permitted).
46  */
47  mimmo::MimmoGeometry * mimmo0 = new mimmo::MimmoGeometry();
48  mimmo0->setIOMode(IOMode::READ);
49  mimmo0->setReadDir("geodata");
50  mimmo0->setReadFileType(FileType::STL);
51  mimmo0->setReadFilename("sphere2");
52 
53  mimmo::MimmoGeometry * mimmo1 = new mimmo::MimmoGeometry();
54  mimmo1->setIOMode(IOMode::WRITE);
55  mimmo1->setWriteDir(".");
56  mimmo1->setWriteFileType(FileType::SURFVTU);
57  mimmo1->setWriteFilename("parallel_output_00002.0001");
58 
59  mimmo::MimmoGeometry * mimmo2 = new mimmo::MimmoGeometry();
60  mimmo2->setIOMode(IOMode::WRITE);
61  mimmo2->setWriteDir(".");
62  mimmo2->setWriteFileType(FileType::SURFVTU);
63  mimmo2->setWriteFilename("parallel_output_00002.0002");
64 
65  /* Instantiation of a Partition object with default patition method space filling curve.
66  * Plot Optional results during execution active for Partition block.
67  */
68  mimmo::Partition* partition = new mimmo::Partition();
69  partition->setPlotInExecution(true);
70 
71  /* Instantiation of a FFDobject with spherical shape.
72  * Setup of origin and span (radius and span angles) of sphere.
73  * Plot Optional results during execution active for FFD block.
74  */
75  mimmo::FFDLattice* lattice = new mimmo::FFDLattice();
76  darray3E origin = {0.0, 0.0,0.0};
77  darray3E span;
78  span[0]= 3.01;
79  span[1]= 2*M_PI;
80  span[2]= M_PI;
81 
82  /* Set number of nodes of the mesh (dim) and degree of nurbs functions (deg).
83  */
84  iarray3E dim, deg;
85  dim[0] = 30;
86  dim[1] = 30;
87  dim[2] = 30;
88  deg[0] = 2;
89  deg[1] = 2;
90  deg[2] = 2;
91 
92  lattice->setLattice(origin,span,mimmo::ShapeType::SPHERE,dim, deg);
93 
94  /* Change reference system to work in local spherical coordinates.
95  * Set coordinates as CLAMPED (continuity in origins of angles).
96  */
97  lattice->setRefSystem(2, darray3E{0,1,0});
98  lattice->setCoordType(mimmo::CoordType::CLAMPED, 2);
99  lattice->setPlotInExecution(false);
100 
101  /* Build mesh of lattice outside the execution chain
102  * to use it during setup the displacements.
103  */
104  lattice->build();
105 
106  /* Creation of Generic input block and set it with the
107  * displacements of the control nodes of the lattice.
108  * Use random values to set the displacements of the control nodes at a longitude
109  * angle smaller than PI and expansion on radius direction for nodes with longitude
110  * angle greater than PI.
111  */
112  int ndeg = lattice->getNNodes();
113  dvecarr3E displ(ndeg, darray3E{0,0,0});
114  time_t Time = time(NULL);
115  srand(Time);
116  for (int i=0; i<ndeg; i++){
117  int l1,l2,l3;
118  int index = lattice->accessGridFromDOF(i);
119  lattice->accessPointIndex(index,l1,l2,l3);
120  if(l1 > 0 && lattice->getLocalPoint(l1,l2,l3)[1] < M_PI){
121  displ[i][0] = 1.0*( (double) (rand()) / RAND_MAX );
122  }
123  if( (l1 > 0 && lattice->getLocalPoint(l1,l2,l3)[1] >= M_PI)
124  || lattice->getLocalPoint(l1,l2,l3)[1] == 0){
125  displ[i][0] = 1.25;
126  }
127 
128  }
129 
130  /* Set Generic input block with the
131  * displacements defined above.
132  */
133  mimmo::GenericInput* input = new mimmo::GenericInput();
134  input->setInput(displ);
135 
136  /* Create applier block.
137  * It applies the deformation displacements to the original input geometry.
138  */
139  mimmo::Apply* applier = new mimmo::Apply();
140 
141  /* Setup pin connections.
142  */
143  mimmo::pin::addPin(mimmo0, partition, M_GEOM, M_GEOM);
144  mimmo::pin::addPin(partition, mimmo1, M_GEOM, M_GEOM);
145  mimmo::pin::addPin(mimmo1, lattice, M_GEOM, M_GEOM);
146  mimmo::pin::addPin(mimmo1, applier, M_GEOM, M_GEOM);
147  mimmo::pin::addPin(input, lattice, M_DISPLS, M_DISPLS);
148  mimmo::pin::addPin(lattice, applier, M_GDISPLS, M_GDISPLS);
149  mimmo::pin::addPin(applier, mimmo2, M_GEOM, M_GEOM);
150 
151  /* Setup execution chain.
152  * The object can be insert in the chain in random order.
153  * The chain object recover the correct order of execution from
154  * the pin connections.
155  */
156  mimmo::Chain ch0;
157  ch0.addObject(input);
158  ch0.addObject(partition);
159  ch0.addObject(applier);
160  ch0.addObject(lattice);
161  ch0.addObject(mimmo0);
162  ch0.addObject(mimmo1);
163  ch0.addObject(mimmo2);
164 
165  /* Execution of chain.
166  * Use debug flag false to avoid printing intermediate results of the execution steps.
167  */
168  ch0.exec(true);
169 
170  /* Clean up & exit;
171  */
172  delete lattice;
173  delete applier;
174  delete input;
175  delete mimmo0;
176  delete partition;
177  delete mimmo1;
178  delete mimmo2;
179 
180  lattice = NULL;
181  applier = NULL;
182  input = NULL;
183  mimmo0 = NULL;
184  partition = NULL;
185  mimmo1 = NULL;
186  mimmo2 = NULL;
187 
188  return;
189 }
