151 FVSelectionBySphere::extractSelection(livector1D & bulk, std::unordered_map<short,livector1D> & boundary){
152 
153  livector1D wholebnd;
154  if(m_dual){
155  bulk = excludeGeometry(m_geometry);
156  wholebnd = excludeGeometry(m_bndgeometry);
157  }else{
158  bulk = includeGeometry(m_geometry);
159  wholebnd = includeGeometry(m_bndgeometry);
160  }
161 
162  //initialize boundary
163  boundary.clear();
164  for(const auto & pid: m_bndgeometry->getPIDTypeList()){
165  boundary.insert(std::make_pair(pid, livector1D()));
166  boundary.reserve(wholebnd.size());
167  }
168 
169  //split wholebnd for pid.
170  for(const auto & cid : wholebnd){
171  int pid = m_bndgeometry->getPatch()->getCell(cid).getPID();
172  boundary[short(pid)].push_back(cid);
173  }
174 };
