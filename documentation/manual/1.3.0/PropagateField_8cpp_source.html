721 void PropagateVectorField::initializeSlipSurface(){

723  //If it not exists use the slip boundary patch as slip reference surface
724  if (m_slipreferencesurface == nullptr)
725  m_slipreferencesurface = m_slipsurface;
726 
727 #if MIMMO_ENABLE_MPI
728 
729  if (getTotalProcs() > 1){
730 
731  //MPI version
732  m_originalslipsurface = std::move(std::unique_ptr<MimmoObject>(new MimmoObject(1)));
733 
734  //fill serialized geometry
735  for (bitpit::Vertex &vertex : m_slipreferencesurface->getVertices()){
736  long vertexId = vertex.getId();
737  if (m_slipreferencesurface->isPointInterior(vertexId)){
738  m_originalslipsurface->addVertex(vertex, vertexId);
739  }
740  }
741  for (bitpit::Cell &cell : m_slipreferencesurface->getCells()){
742  if (cell.isInterior()){
743  m_originalslipsurface->addCell(cell, cell.getId());
744  }
745  }
746 
747  //Receive vertices and cells
748  for (int sendRank=0; sendRank<m_nprocs; sendRank++){
749 
750  if (m_rank != sendRank){
751 
752  // Vertex data
753  long vertexBufferSize;

1248  const liimap & maplocals,

1254  rhs.resize(geo->getPatch()->getInternalCount(), 0.0);
1255

1257  (*m_log)<<"Warning in "<<m_name<<". Unable to assign BC to the system. The solver is not yet initialized."<<std::endl;
1258  return;
1259  }
1260 
1261  if (!borderLaplacianStencil || !borderCCGradientStencil) {
1262  (*m_log)<<"Warning in "<<m_name<<". Unable to reach border cells stencils data. Nothing to do."<<std::endl;
1263  return;
1264  }
1265 
1266  //extract all interfaces from border cells.
1267  livector1D interfaceList = geo->getInterfaceFromCellList(borderLaplacianStencil->getIds());
1268 
1269  //copy laplacian stencils in a work mpv .
1270  FVolStencil::MPVDivergenceUPtr lapwork(new FVolStencil::MPVDivergence(*borderLaplacianStencil));
1271 
1272  //correct the original border laplacian stencils applying the Dirichlet/Neumann corrections.
1273  //renumber it and update the laplacian matrix and fill the rhs.
1274  bitpit::StencilVector correction;
1275  bitpit::PiercedVector<bitpit::Interface> & mesh_interfaces = geo->getInterfaces();
1276 
1277  long idOwner;
1278  std::array<double,3> interfaceNormal, interfaceCentroid, ownerCentroid;
1279  double volume, iarea;
1280  for(long idInterface : interfaceList){
1281 
1282  correction.clear(false);
1283 
1284  if(!mesh_interfaces.at(idInterface).isBorder()) continue; //skip non-border interfaces;
1285 
1286  idOwner = mesh_interfaces.at(idInterface).getOwner();
1287  interfaceNormal = geo->evalInterfaceNormal(idInterface);
1288  volume = geo->evalCellVolume(idOwner);
1289  iarea = geo->evalInterfaceArea(idInterface);
1290 
1291  //apply the correction relative to bc @ interface.
1292  switch(m_isbp.at(idInterface)){
1293  case 1: //DIRICHLET
1294  ownerCentroid = geo->evalCellCentroid(idOwner);
1295  interfaceCentroid = geo->evalInterfaceCentroid(idInterface);
1296  //get the correction.
1297  correction = FVolStencil::correctionDirichletBCFaceGradient(m_bc_dir.at(idInterface)[comp],
1298  idOwner,ownerCentroid, interfaceCentroid, interfaceNormal,
1299  dotProduct(interfaceCentroid-ownerCentroid, interfaceNormal),
1300  borderCCGradientStencil->at(idOwner) );
1301  break;
1302  case 2: //SLIP/IMPERMEABILITY
1303 
1304  if(!slipCorrect){ //predictor stage - push neumann condition.
1305  correction = FVolStencil::correctionNeumannBCFaceGradient(0.0, interfaceNormal);
1306  }else{ //corrector stage - push dirichlet condition from m_slip_bc_dir you have fixed.
1307  ownerCentroid = geo->evalCellCentroid(idOwner);
1308  interfaceCentroid = geo->evalInterfaceCentroid(idInterface);
1309  //get the correction.
1310  correction = FVolStencil::correctionDirichletBCFaceGradient(m_slip_bc_dir.at(idInterface)[comp],
1311  idOwner,ownerCentroid, interfaceCentroid, interfaceNormal,
1312  dotProduct(interfaceCentroid-ownerCentroid, interfaceNormal),
1313  borderCCGradientStencil->at(idOwner) );
1314  }
1315  break;
1316  default: //NEUMANN for non zero flux, change the first entry.
1317  correction = FVolStencil::correctionNeumannBCFaceGradient(0.0, interfaceNormal);
1318  break;
1319  }
1320 
1321  //calculate the laplacian correction and push it in work laplacian
1322  lapwork->at(idOwner) += (m_dumping.at(idOwner) * iarea / volume) * dotProduct(correction, interfaceNormal);
1323 
1324  }
1325 
1326  // now its time to update the solver matrix and to extract the rhs contributes.
1327  updateLaplaceSolver(lapwork.get(), maplocals);
1328 
1329  // now get the rhs
1330  for(auto it = lapwork->begin(); it != lapwork->end();++it){
1331  auto index = maplocals.at(it.getId());
1332 #if MIMMO_ENABLE_MPI
1333  //correct index if in parallel
1334  index -= getGeometry()->getPatchInfo()->getCellGlobalCountOffset();
1335 #endif
1336  rhs[index] -= it->getConstant();
1337  }
