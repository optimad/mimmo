172 bool PropagateScalarField::checkBoundariesCoherence(){
173 
174  //initialize m_isbp
175  bitpit::PiercedVector<bitpit::Vertex> & pVtarget = m_geometry->getVertices();
176 
177  //1st step: verify boundary IDs of Dirichlet boundary patch and target are coherent
178  // and fill m_isbp with flag true and mark 1 for Dirichlet condition.
179  long id;
180  for(const auto & vert: m_bsurface->getVertices()){
181  id= vert.getId();
182  if(!pVtarget.exists(id)) {
183  m_isbp.clear();
184  return false;
185  }
186  m_isbp.insert(id, 1);
187  }
188 
189  //2nd step verify coherence of the Dirichlet field with boundary surface
190  if(m_bc_dir.getGeometry() != m_bsurface || !m_bc_dir.completeMissingData({0.0})){
191  m_isbp.clear();
192  return false;
193  }
194 
195  return true;
196 }

233 void
234 PropagateScalarField::execute(){
235 
236  if(getGeometry() == NULL){
237  (*m_log)<<"Error in "<<m_name<<" .No target volume mesh linked"<<std::endl;
238  throw std::runtime_error("Error in PropagateScalarField execute. No target volume mesh linked");
239  }
240 
241  if(m_bsurface == NULL ){
242  (*m_log)<<"Error in "<<m_name<<" .No Dirichlet Boundary patch linked"<<std::endl;
243  throw std::runtime_error("Error in PropagateScalarField execute. No Dirichlet Boundary patch linked");
244  }
245 
246  if(!checkBoundariesCoherence()){
247  (*m_log)<<"Error in "<<m_name<<" .Boundary patches linked are uncoherent with target bulk geometry"
248  "or bc-fields not coherent with boundary patches"<<std::endl;
249  throw std::runtime_error("Error in PropagateScalarField execute. Boundary patches linked are uncoherent"
250  "with target bulk geometry or bc-fields not coherent with boundary patches");
251  }
252 
253  ivector2D stencils;
254  dvector2D weights;
255  dvector1D rhs;
256  liimap dataInv = getGeometry()->getMapDataInv();
257 
258  computeDumpingFunction();
259  computeStencils(dataInv, stencils, weights);
260  correctStencils(dataInv, stencils, weights);
261  computeRHS(m_bc_dir, dataInv, rhs);
262 
263  if (m_laplace){
264  solveLaplace(stencils, weights, rhs, dataInv, m_field);
265  }else{
266  solveSmoothing(m_sstep, stencils, weights, rhs, dataInv, m_field);
267  }
268 }

547 bool PropagateVectorField::checkBoundariesCoherence(){
548 
549  //initialize m_isbp
550  bitpit::PiercedVector<bitpit::Vertex> & pVtarget = m_geometry->getVertices();
551 
552  //1st step: verify boundary IDs of Dirichlet boundary patch and target are coherent
553  // and fill m_isbp with flag true and mark 1 for Dirichlet condition.
554  long id;
555  for(const auto & vert: m_bsurface->getVertices()){
556  id= vert.getId();
557  if(!pVtarget.exists(id)) {
558  m_isbp.clear();
559  return false;
560  }
561  m_isbp.insert(id, 1);
562  }
563 
564  //2nd step verify coherence of the Dirichlet field with boundary surface
565  if(m_bc_dir.getGeometry() != m_bsurface || !m_bc_dir.completeMissingData({{0.0, 0.0,0.0}})){
566  m_isbp.clear();
567  return false;
568  }
569 
570  // verify if optional Neumann condition are set. If not, exit with true condition
571  if(m_slipsurface == NULL) return true; //ok
572 
573  //neumann surface is not null, double step check as m_bsurface
574  //3rd step: verify boundary IDs of Neumann boundary patch and target are coherent
575  // and fill m_isbp with flag true and mark 2 for Neumann condition.
576  for(const auto & vert: m_slipsurface->getVertices()){
577  id= vert.getId();
578  if(!pVtarget.exists(id)) {
579  m_isbp.clear();
580  return false;
581  }
582  if( !(m_isbp.exists(id)) ){
583  m_isbp.insert(id, 2);
584  }
585  }
586 
587 
588  m_vNormals.clear();
589  m_vNormals.reserve(m_slipsurface->getNVertex());
590 
591  bitpit::ConstProxyVector<long> verts;
592  std::size_t size;
593  long idN;
594  //save the vertex Normals using the boundary surface m_slipsurface;
595  //check m_slipratio stuff and correct accordingly the normals.
596 
597  bitpit::SurfaceKernel* skernel = static_cast<bitpit::SurfaceKernel*>(m_slipsurface->getPatch());
598 
599  for(const auto & cell: m_slipsurface->getCells()){
600  verts= cell.getVertexIds();
601  size = verts.size();
602  for(std::size_t i=0; i<size; ++i){
603  idN = verts[i];
604  if(m_isbp[idN]==2 && !m_vNormals.exists(idN)){
605  m_vNormals.insert(idN, skernel->evalVertexNormal(cell.getId(), i));
606 
607  int comp = 0;
608  if(std::abs(m_vNormals[idN][1]) > std::abs(m_vNormals[idN][comp])) comp = 1;
609  if(std::abs(m_vNormals[idN][2]) > std::abs(m_vNormals[idN][comp])) comp = 2;
610 
611  if(m_slipratio * std::abs(m_vNormals[idN][(comp+1)%3]/m_vNormals[idN][comp]) < 1.0) m_vNormals[idN][(comp+1)%3] = 0.0;
612  if(m_slipratio * std::abs(m_vNormals[idN][(comp+2)%3]/m_vNormals[idN][comp]) < 1.0) m_vNormals[idN][(comp+2)%3] = 0.0;
613 
614  m_vNormals[idN] /= norm2(m_vNormals[idN]);
615  }
616  }
617  }
618 
619  //if it is survived, then it's all ok.
620  return true;
621 }

635 void
636 PropagateVectorField::correctStencils(liimap & dataInv, ivector2D &stencils, dvector2D &weights)
637 {
638  int ind;
639  long ID;
640  for(auto it = m_isbp.begin(); it != m_isbp.end(); ++it){
641 
642  ID = it.getId();
643  ind = dataInv[ID];
644 
645  switch(*it){
646  case 1: //Dirichlet boundary type
647  for(int comp=0; comp<3; ++comp){
648  stencils[ind + comp*m_np] = ivector1D(1, ind + comp*m_np);
649  weights[ind + comp*m_np] = dvector1D(1,1.0);
650  }
651  break;
652  case 2: //Slip boundary type
653  {
654  int comp = 0;
655  if(std::abs(m_vNormals[ID][1]) > std::abs(m_vNormals[ID][comp])) comp = 1;
656  if(std::abs(m_vNormals[ID][2]) > std::abs(m_vNormals[ID][comp])) comp = 2;
657 
658  stencils[ind+comp*m_np].resize(3);
659  stencils[ind+comp*m_np] = {{ind, ind+m_np, ind+2*m_np}};
660  weights[ind+comp*m_np].resize(3);
661  for(int i=0; i<3; ++i){
662  weights[ind +comp*m_np][i] = m_vNormals[ID][i]/m_vNormals[ID][comp];
663 // if(std::abs(weights[ind +comp*m_np][i]) < 1.0e-8)
664 // weights[ind +comp*m_np][i] = 0.0;
665  }
666  }
667  break;
668  default:
669  //do nothing
670  break;
671  }
672  }
673 }

812 PropagateVectorField::execute(){
813 
814  if(getGeometry() == NULL){
815  (*m_log)<<"Error in "<<m_name<<" .No target volume mesh linked"<<std::endl;
816  throw std::runtime_error("Error in PropagateVectorField execute. No target volume mesh linked");
817  }
818 
819  if(m_bsurface == NULL ){
820  (*m_log)<<"Error in "<<m_name<<" .No Dirichlet Boundary patch linked"<<std::endl;
821  throw std::runtime_error("Error in PropagateVectorField execute. No Dirichlet Boundary patch linked");
822  }
823 
824  if(!checkBoundariesCoherence()){
825  (*m_log)<<"Error in "<<m_name<<" .Boundary patches linked are uncoherent with target bulk geometry"
826  "or bc-fields not coherent with boundary patches"<<std::endl;
827  throw std::runtime_error("Error in PropagateVectorField execute. Boundary patches linked are uncoherent"
828  "with target bulk geometry or bc-fields not coherent with boundary patches");
829  }
830 
831  ivector2D stencils;
832  dvector2D weights;
833  dvector1D rhs;
834  liimap dataInv = getGeometry()->getMapDataInv();
835 
836  if (m_laplace){
837  bitpit::PiercedVector<bitpit::Vertex> vertices0;
838  if (m_nstep > 1){
839  subdivideBC();
840  vertices0 = getGeometry()->getVertices();
841  }
842
