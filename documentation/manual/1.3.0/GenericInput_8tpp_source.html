426 MimmoPiercedVector< T >*
427 GenericInputMPVData::_getResult(){
428 
429  MimmoPiercedVector< T > data;
430  MimmoObject * refgeo = getGeometry();
431 
432 #if MIMMO_ENABLE_MPI
433  if(m_rank == 0)
434 #endif
435  {
436  int n_loc;
437  long nSize = 0, readNSize;
438  long id;
439  T data_T;
440 
441  if(m_csv) m_binary = false;
442 
443  std::fstream file;
444  file.open(m_dir+"/"+m_filename, std::fstream::in);
445  if (file.is_open()){
446  if (m_binary){
447  bitpit::genericIO::absorbBINARY(file, n_loc);
448  bitpit::genericIO::absorbBINARY(file, nSize);
449  data.reserve(nSize);
450  for(int i=0; i<nSize; ++i){
451  bitpit::genericIO::absorbBINARY(file, id);
452  bitpit::genericIO::absorbBINARY(file, data_T);
453  data.insert(id,data_T);
454  }
455  data.setDataLocation(n_loc);
456  }
457  else if (m_csv){
458  inputCSVStream::ifstreamcsv(file, data);
459  }else{
460  bitpit::genericIO::absorbASCII(file, n_loc);
461  bitpit::genericIO::absorbASCII(file, readNSize);
462  if(file.good()) nSize = long(readNSize);
463  data.reserve(nSize);
464  for(long i=0; i<nSize; ++i){
465  bitpit::genericIO::absorbASCII(file, id);
466  bitpit::genericIO::absorbASCII(file, data_T);
467  data.insert(id,data_T);
468  }
469  data.setDataLocation(n_loc);
470  }
471  file.close();
472  }else{
473  (*m_log) << "file not open --> exit" << std::endl;
474  throw std::runtime_error (m_name + " : cannot open " + m_filename + " requested");
475  }
476  }
477 
478 #if MIMMO_ENABLE_MPI
479  //if there are any other procs send data to them.
480  if(m_nprocs > 1) sendReadDataToAllProcs(data);
481 #endif
482 
483  if(refgeo == nullptr){
484  _setResult(data);
485  }else{
486  data.setGeometry(refgeo);
487  MimmoPiercedVector<T> data2 = data.resizeToCoherentDataIds();
488  _setResult(data2);
489  }
490 
491  return static_cast<IODataT<MimmoPiercedVector< T > >*>(m_result.get())->getData();
492 }
