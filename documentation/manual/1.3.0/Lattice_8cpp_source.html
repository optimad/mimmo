442 int
443 Lattice::reduceDimToDOF(int nx, int ny, int nz, bvector1D & info){
444 
445  int delta = 0;
446  double dval;
447  switch(getShapeType()){
448 
449  case ShapeType::WEDGE :
450  delta = ny*(1 -nz);
451  info.push_back(true);
452  break;
453 
454  case ShapeType::CYLINDER :
455  if(!(getShape()->getLocalOrigin()[0] >0.0 )){
456  info.push_back(false);
457  }else{
458  delta += nz;
459  nx--;
460  info.push_back(true);
461  }
462  if(getCoordType(1) == CoordType::PERIODIC) ny--;
463 
464 
465  info.push_back(getCoordType(1) == CoordType::PERIODIC);
466  break;
467 
468  case ShapeType::SPHERE :
469  if(!(getShape()->getLocalOrigin()[0] >0.0 )){
470  info.push_back(false);
471  }else{
472  delta ++;
473  nx--;
474  info.push_back(true);
475  }
476  if(getCoordType(1) == CoordType::PERIODIC) ny--;
477  dval = getInfLimits()[2];
478  if(dval == 0.0) {
479  nz--;
480  delta += nx;
481  }
482  if((dval + getLocalSpan()[2]) == M_PI){
483  nz--;
484  delta += nx;
485  }
486 
487  info.push_back(getCoordType(1) == CoordType::PERIODIC);
488  info.push_back(dval==0.0);
489  info.push_back((dval + getLocalSpan()[2]) == M_PI);
490  break;
491 
492  default:
493  //doing nothing
494  break;
495  }
496 
497  int result = nx*ny*nz + delta;
498  return(result);
499 };
