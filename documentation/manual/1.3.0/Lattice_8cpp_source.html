291 void Lattice::absorbSectionXML(const bitpit::Config::Section & slotXML, std::string name){
292 
293  BITPIT_UNUSED(name);
294 
295  BaseManipulation::absorbSectionXML(slotXML, name);
296 
297  if(slotXML.hasOption("Shape")){
298  std::string input = slotXML.get("Shape");
299  input = bitpit::utils::string::trim(input);
300 
301  if(input == "CYLINDER"){
302  setShape(ShapeType::CYLINDER);
303  }else if(input =="SPHERE"){
304  setShape(ShapeType::SPHERE);

306  setShape(ShapeType::WEDGE);
307  }else{
308  setShape(ShapeType::CUBE);

314  input = bitpit::utils::string::trim(input);
315  darray3E temp = {{0.0,0.0,0.0}};
316  if(!input.empty()){
317  std::stringstream ss(input);
318  for(auto &val : temp) ss>>val;
319  }
320  setOrigin(temp);
321  };
322 
323  if(slotXML.hasOption("Span")){
324  std::string input = slotXML.get("Span");
325  input = bitpit::utils::string::trim(input);

327  if(!input.empty()){
328  std::stringstream ss(input);
329  for(auto &val : temp) ss>>val;
330  }
331  setSpan(temp);
332  };
333 
334  if(slotXML.hasSection("RefSystem")){
335  const bitpit::Config::Section & rfXML = slotXML.getSection("RefSystem");
336  std::string rootAxis = "axis";
337  std::string axis;
338  dmatrix33E temp;
339  temp[0].fill(0.0); temp[0][0] = 1.0;

343  axis = rootAxis + std::to_string(i);
344  std::string input = rfXML.get(axis);
345  input = bitpit::utils::string::trim(input);
346  if(!input.empty()){
347  std::stringstream ss(input);
348  for(auto &val : temp[i]) ss>>val;
349  }
350  }
351  setRefSystem(temp);
352  };
353 
354  if(slotXML.hasOption("InfLimits")){
355  std::string input = slotXML.get("InfLimits");
356  input = bitpit::utils::string::trim(input);
357  darray3E temp = {{0.0,0.0,0.0}};
358  if(!input.empty()){
359  std::stringstream ss(input);
360  for(auto &val : temp) ss>>val;
361  }
362  setInfLimits(temp);
363  };
364 
365  if(slotXML.hasOption("Dimension")){
366  std::string input = slotXML.get("Dimension");
367  input = bitpit::utils::string::trim(input);
368  iarray3E temp = {{2,2,2}};
369  if(!input.empty()){
370  std::stringstream ss(input);
371  for(auto &val : temp) ss>>val;

458 int
459 Lattice::reduceDimToDOF(int nx, int ny, int nz, bvector1D & info){
460 
461  int delta = 0;
462  double dval;
463  switch(getShapeType()){
464 
465  case ShapeType::WEDGE :
466  delta = ny*(1 -nz);
467  info.push_back(true);
468  break;
469 
470  case ShapeType::CYLINDER :
471  if(!(getShape()->getLocalOrigin()[0] >0.0 )){
472  info.push_back(false);
473  }else{
474  delta += nz;
475  nx--;
476  info.push_back(true);
477  }
478  if(getCoordType(1) == CoordType::PERIODIC) ny--;
479 
480 
481  info.push_back(getCoordType(1) == CoordType::PERIODIC);
482  break;
483 
484  case ShapeType::SPHERE :
485  if(!(getShape()->getLocalOrigin()[0] >0.0 )){
486  info.push_back(false);
487  }else{
488  delta ++;
489  nx--;
490  info.push_back(true);
491  }
492  if(getCoordType(1) == CoordType::PERIODIC) ny--;
493  dval = getInfLimits()[2];
494  if(dval == 0.0) {
495  nz--;
496  delta += nx;
497  }
498  if((dval + getLocalSpan()[2]) == M_PI){
499  nz--;
500  delta += nx;
501  }
502 
503  info.push_back(getCoordType(1) == CoordType::PERIODIC);
504  info.push_back(dval==0.0);
505  info.push_back((dval + getLocalSpan()[2]) == M_PI);
506  break;
507 
508  default:
509  //doing nothing
510  break;
511  }
512 
513  int result = nx*ny*nz + delta;
514  return(result);
515 };

522 void
523 Lattice::resizeMapDof(){
524  //reallocate your displacement node
525  m_intMapDOF.clear();
526  m_intMapDOF.resize((m_nx+1)*(m_ny+1)*(m_nz+1), -1);
527  ivector1D::iterator itMapBegin = m_intMapDOF.begin();
528  ivector1D::iterator itMap = itMapBegin;
529  ivector1D::iterator itMapEnd = m_intMapDOF.end();
530  bvector1D info;
531  m_np = reduceDimToDOF(m_nx+1,m_ny+1,m_nz+1, info);
532 
533  //set m_intMapDOF
534 
535  int target;
536  int index;
537  ivector1D dummy;
538 
539  int i0,i1,i2;
540  switch(getShapeType()){
541 
542  case ShapeType::WEDGE :
543 
544  target=0;
545  while(itMap != itMapEnd){
546 
547  *itMap = target;
548  index = std::distance(itMapBegin, itMap);
549  accessPointIndex(index,i0,i1,i2);
550 
551  if(info[0] && i0 == m_nx){
552  for(int k=0; k<=m_nz;++k){
553  m_intMapDOF[accessPointIndex(i0,i1,k)] = target;
554  }
555  }
556 
557  itMap = find(m_intMapDOF.begin(), itMapEnd,-1);
558  target++;
559  }
560  break;
561 
562  case ShapeType::CYLINDER :
563  target=0;
564  while(itMap != itMapEnd){
565 
566  *itMap = target;
567  index = std::distance(itMapBegin, itMap);
568  accessPointIndex(index,i0,i1,i2);
569 
570  if(info[0] && i0 == 0){
571  for(int k=0; k<=m_ny;++k){
572  m_intMapDOF[accessPointIndex(i0,k,i2)] = target;
573  }
574  }
575  if(info[1] && i1 == 0){
576  m_intMapDOF[accessPointIndex(i0,m_ny,i2)] = target;
577  }
578 
579  itMap = find(m_intMapDOF.begin(), itMapEnd,-1);
580  target++;
581  }
582  break;
583 
584  case ShapeType::SPHERE :
585 
586  target = 0;
587  while(itMap != itMapEnd){
588 
589  *itMap = target;
590  index = std::distance(itMapBegin, itMap);
591  accessPointIndex(index,i0,i1,i2);
592 
593  if(info[0] && i0 == 0){
594  for(int k1=0; k1<=m_ny;++k1){
595  for(int k2=0; k2<=m_nz; ++k2){
596  m_intMapDOF[accessPointIndex(i0,k1,k2)] = target;
597  }
598  }
599  }
600 
601  if(info[1] && i1 == 0){
602  m_intMapDOF[accessPointIndex(i0,m_ny-1,i2)] = target;
603  }
604 
605  if(info[2] && i2 == 0){
606  for(int k1=0; k1<=m_ny; ++k1){
607  m_intMapDOF[accessPointIndex(i0,k1,i2)] = target;
608  }
609  }
610 
611  if(info[3] && i2 == (m_nz)){
612  for(int k1=0; k1<=m_ny;++k1){
613  m_intMapDOF[accessPointIndex(i0,k1,i2)] = target;
614  }
615  }
616 
617  itMap = find(m_intMapDOF.begin(), itMapEnd,-1);
618  target++;
619  }
620  break;
621 
622 
623  case ShapeType::CUBE :
624  target = 0;
625  while(itMap != itMapEnd){
626 
627  *itMap = target;
628  itMap = find(m_intMapDOF.begin(), itMapEnd,-1);
629  target++;
630  }
631  break;
632 
633  default: //doing nothing

635  }//end switch
