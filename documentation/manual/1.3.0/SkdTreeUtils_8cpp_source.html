86 double signedDistance(std::array<double,3> *P_, bitpit::PatchSkdTree *bvtree_, long &id, std::array<double,3> &n, double &r)
87 {
88 
89  double h = 1.E+18;
90 
91  if(!bvtree_ ){
92  throw std::runtime_error("Invalid use of skdTreeUtils::signedDistance method: a void tree is detected.");
93  }
94  const bitpit::SurfUnstructured *spatch = dynamic_cast<const bitpit::SurfUnstructured*>(&(bvtree_->getPatch()));
95  if(!spatch){
96  throw std::runtime_error("Invalid use of skdTreeUtils::signedDistance method: a not surface patch tree is detected.");
97  }
98 
99  static_cast<bitpit::SurfaceSkdTree*>(bvtree_)->findPointClosestCell(*P_, r, &id, &h);
100 
101  //signed distance only for 2D element patches(quads, pixels, triangles or segments)
102  const bitpit::Cell & cell = spatch->getCell(id);
103  bitpit::ConstProxyVector<long> vertIds = cell.getVertexIds();
104  dvecarr3E VS(vertIds.size());
105  int count = 0;
106  for (const auto & iV: vertIds){
107  VS[count] = spatch->getVertexCoords(iV);
108  ++count;
109  }
110 
111  darray3E xP = {{0.0,0.0,0.0}};
112  darray3E normal= {{0.0,0.0,0.0}};
113 
114  if ( vertIds.size() == 3 ){ //TRIANGLE
115  darray3E lambda;
116  h = bitpit::CGElem::distancePointTriangle((*P_), VS[0], VS[1], VS[2],lambda);
117  int count = 0;
118  for(const auto &val: lambda){
119  normal += val * spatch->evalVertexNormal(id,count) ;
120  xP += val * VS[count];
121  ++count;
122  }
123  }else if ( vertIds.size() == 2 ){ //LINE/SEGMENT
124  darray2E lambda;
125  h = bitpit::CGElem::distancePointSegment((*P_), VS[0], VS[1], lambda);
126  int count = 0;
127  for(const auto &val: lambda){
128  normal += val * spatch->evalVertexNormal(id,count) ;
129  xP += val * VS[count];
130  ++count;
131  }
132  }else{ //GENERAL POLYGON
133  std::vector<double> lambda;
134  h = bitpit::CGElem::distancePointPolygon((*P_), VS,lambda);
135  int count = 0;
136  for(const auto &val: lambda){
137  normal += val * spatch->evalVertexNormal(id,count) ;
138  xP += val * VS[count];
139  ++count;
140  }
141  }
142 
143  double s = sign( dotProduct(normal, (*P_) - xP) );
144  if(s == 0.0) s =1.0;
145  h = s * h;
146  //pseudo-normal (direction P and xP closest point on triangle)
147  n = s * ((*P_) - xP);
148  double normX = norm2(n);
149  if(normX < 1.E-15){
150  n = normal/norm2(normal);
151  }else{
152  n /= norm2(n);
153  }
154  return h;
155 
156 }

171  int nnodess = selection->getNodeCount();

180  leafSelection[count] = &(selection->getNode(i));

212 void extractTarget(bitpit::PatchSkdTree *target, const std::vector<const bitpit::SkdNode*> & leafSelection, std::vector<long> &extracted, double tol){
213 
214  if(leafSelection.empty()) return;
215  std::size_t rootId =0;
216 
217  std::vector<std::size_t> candidates;
218  std::vector<std::pair< std::size_t, std::vector<const bitpit::SkdNode*> > > nodeStack;
219 
220  std::vector<const bitpit::SkdNode*> tocheck;
221  for (int i=0; i<(int)leafSelection.size(); i++){
222  if (bitpit::CGElem::intersectBoxBox(leafSelection[i]->getBoxMin()-tol,
223  leafSelection[i]->getBoxMax()+tol,
224  target->getNode(rootId).getBoxMin(),
225  target->getNode(rootId).getBoxMax() ) )
226  {
227  tocheck.push_back(leafSelection[i]);
228  }
229  }
230  nodeStack.push_back(std::make_pair(rootId, tocheck) );
231 
232  while(!nodeStack.empty()){
233 
234  std::pair<std::size_t, std::vector<const bitpit::SkdNode*> > touple = nodeStack.back();
235  const SkdNode & node = target->getNode(touple.first);
236  nodeStack.pop_back();
237 
238 
239  bool isLeaf = true;
240  for (int i = SkdNode::CHILD_BEGIN; i != SkdNode::CHILD_END; ++i) {
241  SkdNode::ChildLocation childLocation = static_cast<SkdNode::ChildLocation>(i);
242  std::size_t childId = node.getChildId(childLocation);
243  if (childId != SkdNode::NULL_ID) {
244  isLeaf = false;
245  tocheck.clear();
246  for (int i=0; i<(int)touple.second.size(); i++){
247  if (bitpit::CGElem::intersectBoxBox(touple.second[i]->getBoxMin()-tol,
248  touple.second[i]->getBoxMax()+tol,
249  target->getNode(childId).getBoxMin(),
250  target->getNode(childId).getBoxMax() ) )
251  {
252  tocheck.push_back(touple.second[i]);
253  }
254  }
255  if(!tocheck.empty()) nodeStack.push_back(std::make_pair(childId, tocheck) );
256  }
257  }
258 
259  if (isLeaf) {
260  candidates.push_back(touple.first);
261  }
262  }
263 
264  std::unordered_set<long> cellExtracted;
265  for(const auto & nodeId: candidates){
266  const SkdNode & node = target->getNode(nodeId);
267  std::vector<long> cellids = node.getCells();
268  cellExtracted.insert(cellids.begin(), cellids.end());
269  }
270 
271  extracted.insert(extracted.end(), cellExtracted.begin(), cellExtracted.end());
272 }

287 darray3E projectPoint( std::array<double,3> *P_, bitpit::PatchSkdTree *bvtree_, double r )
288 {
289 
290  long id;
291  darray3E normal;
292 
293  double dist = 1.0e+18;
294  while (std::abs(dist) >= 1.0e+18){
295  //use method sphere by default
296  dist = signedDistance(P_, bvtree_, id, normal, r);
297  r *= 1.5;
298  }
299 
300  darray3E projP = (*P_) - dist*normal;
301 
302  return (projP);

313 long locatePointOnPatch(const std::array<double, 3> &point, bitpit::PatchSkdTree &tree)
314 {
315  if(!dynamic_cast<const bitpit::SurfUnstructured*>(&(tree.getPatch()))){
316  throw std::runtime_error("Invalid use of skdTreeUtils::locatePointOnPatch method: a non surface patch or void patch was detected.");
317  }
318  // Initialize the cell id
319  long id = Cell::NULL_ID;
320 
321  // Initialize the distance with an estimate
322  //
323  // The real distance will be lesser than or equal to the estimate.
324  std::size_t rootId = 0;
325  const SkdNode &root = tree.getNode(rootId);
326  double distance = root.evalPointMaxDistance(point);
327 
328  // Get a list of candidates nodes
329  //
330  std::vector<std::size_t> m_candidateIds;
331 
332  std::vector<std::size_t> nodeStack;
333  nodeStack.push_back(rootId);
334  while (!nodeStack.empty()) {
335  std::size_t nodeId = nodeStack.back();
336  const SkdNode &node = tree.getNode(nodeId);
337  nodeStack.pop_back();
338 
339  // Do not consider nodes with a minimum distance greater than
340  // the distance estimate
341  double nodeMinDistance = node.evalPointMinDistance(point);
342  if (nodeMinDistance > distance) {
343  continue;
344  }
345 
346  // Update the distance estimate
347  //
348  // The real distance will be lesser than or equal to the
349  // estimate.
350  double nodeMaxDistance = node.evalPointMaxDistance(point);
351  distance = std::min(nodeMaxDistance, distance);
352 
353  // If the node is a leaf add it to the candidates, otherwise
354  // add its children to the stack.
355  bool isLeaf = true;
356  for (int i = SkdNode::CHILD_BEGIN; i != SkdNode::CHILD_END; ++i) {
357  SkdNode::ChildLocation childLocation = static_cast<SkdNode::ChildLocation>(i);
358  std::size_t childId = node.getChildId(childLocation);
359  if (childId != SkdNode::NULL_ID) {
360  isLeaf = false;
361  nodeStack.push_back(childId);
362  }
363  }
364 
365  if (isLeaf) {
366  m_candidateIds.push_back(nodeId);
367  }
368  }
369 
370  // Process the candidates and find which cell contains the point
371  for (std::size_t k = 0; k < m_candidateIds.size(); ++k) {
372 
373  // Evaluate the min distance between all cells in the candidate node
374  std::size_t nodeId = m_candidateIds[k];
375  const SkdNode &node = tree.getNode(nodeId);
376 
377  bitpit::ElementType eltype ;
378  bool checkBelong;
379  for(const auto & cellId : node.getCells() ){
380  eltype = tree.getPatch().getCell(cellId).getType();
381  auto vList = tree.getPatch().getCell(cellId).getVertexIds();
382  dvecarr3E vertCoords(vList.size());
383  int counterList = 0;
384  for(const auto & idV : vList){
385  vertCoords[counterList] = tree.getPatch().getVertexCoords(idV);
386  ++counterList;
387  }
388  switch(eltype){
389  case ElementType::LINE:
390  checkBelong = mimmoCGUtils::isPointInsideSegment(point, vertCoords[0], vertCoords[1]);
391  break;
392  case ElementType::TRIANGLE:
393  checkBelong = mimmoCGUtils::isPointInsideTriangle(point, vertCoords[0], vertCoords[1], vertCoords[2]);
394  break;
395  case ElementType::PIXEL:
396  case ElementType::QUAD:
397  case ElementType::POLYGON:
398  checkBelong = mimmoCGUtils::isPointInsidePolygon(point, vertCoords);
399  break;
400  default:
401  throw std::runtime_error("Not supported cell type");
402  break;
403  }
404  if (checkBelong) {
405  id = cellId;
406  }
407  }
408  }
409  return id;
410 }
