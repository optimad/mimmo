428 void
429 IOCloudPoints::read(){
430 
431  std::unordered_map<long, int> mapP;
432  std::ifstream reading;
433  std::string source = m_dir+"/"+m_filename;
434  reading.open(source.c_str());
435  if(reading.is_open()){
436 
437  m_points.clear();
438  m_labels.clear();
439  m_scalarfield.clear();
440  m_vectorfield.clear();
441 
442  std::string keyword, line;
443  long label;
444  darray3E dtrial;
445  double temp;
446 
447  do{
448  line.clear();
449  keyword.clear();
450  std::getline(reading,line);
451  line = bitpit::utils::string::trim(line);
452  std::stringstream ss(line);
453  ss>>keyword;
454  keyword = bitpit::utils::string::trim(keyword);
455  if(keyword == "$POINT") {
456 
457  ss>>label;
458  m_labels.push_back(label);
459 
460  dtrial.fill(0.0);
461  ss>>dtrial[0]>>dtrial[1]>>dtrial[2];
462  m_points.push_back(dtrial);
463  }
464 
465  }while(!reading.eof());
466 
467  int counter = 0;
468  for(auto &lab :m_labels){
469  mapP[lab] = counter;
470  ++counter;
471  }
472 
473  m_scalarfield.resize(m_points.size(),0.0);
474  m_vectorfield.resize(m_points.size(),{{0.0,0.0,0.0}});
475 
476  reading.clear();
477  reading.seekg(0, std::ios::beg);
478 
479  do{
480  line.clear();
481  keyword.clear();
482  std::getline(reading,line);
483  line = bitpit::utils::string::trim(line);
484  std::stringstream ss(line);
485  ss>>keyword;
486  keyword = bitpit::utils::string::trim(keyword);
487  if(keyword == "$SCALARF") {
488 
489  ss>>label>>temp;
490  m_scalarfield[mapP[label]] = temp;
491  }
492  if(keyword == "$VECTORF") {
493 
494  dtrial.fill(0.0);
495  ss>>label>>dtrial[0]>>dtrial[1]>>dtrial[2];
496  m_vectorfield[mapP[label]] = dtrial;
497  }
498 
499  }while(!reading.eof());
500 
501  }else{
502  (*m_log)<<"error of "<<m_name<<" : cannot open "<<m_filename<< " requested. Exiting... "<<std::endl;
503  throw std::runtime_error (m_name + " : cannot open " + m_filename + " requested. Exiting... ");
504  }
505 
506  reading.close();
507 };

512 void
513 IOCloudPoints::write(){
514 
515  //assessing data;
516  int sizelabels = m_labels.size();
517  long maxlabel = 0;
518  for(auto & val: m_labels){
519  maxlabel = std::max(maxlabel,val);
520  }
521 
522  int sizedispl = m_points.size();
523 
524  m_labels.resize(m_points.size(), -1);
525  if(sizelabels < sizedispl){
526  for(int i = sizelabels; i<sizedispl; ++i){
527  m_labels[i] = maxlabel+1;
528  ++maxlabel;
529  }
530  }
531 
532  m_scalarfield.resize(m_points.size(), 0.0);
533  m_vectorfield.resize(m_points.size(), {{0.0,0.0,0.0}});
534 
535  std::string filename;
536  if(m_template){
537  filename = "TEMPLATE_"+m_filename;
538  }else{
539  filename = m_filename;
540  }
541 
542  std::ofstream writing;
543  std::string source = m_dir+"/"+m_filename;
544  writing.open(source.c_str());
545  std::string keyT1 = "{", keyT2 = "}";
546  if(writing.is_open()){
547 
548  int counter = 0;
549  for(auto & dd : m_points){
550  writing<<"$POINT"<<'\t'<<m_labels[counter]<<'\t'<<dd[0]<<'\t'<<dd[1]<<'\t'<<dd[2]<<std::endl;
551  ++counter;
552  }
553  writing<<""<<std::endl;
554 
555  counter = 0;
556  for(auto & dd : m_scalarfield){
557  if(m_template){
558  std::string str1 = keyT1+"s"+std::to_string(m_labels[counter])+keyT2;
559  writing<<"$SCALARF"<<'\t'<<m_labels[counter]<<'\t'<<str1<<std::endl;
560  }else{
561  writing<<"$SCALARF"<<'\t'<<m_labels[counter]<<'\t'<<dd<<std::endl;
562  }
563  ++counter;
564  }
565  writing<<""<<std::endl;
566 
567  counter = 0;
568  for(auto & dd : m_vectorfield){
569  if(m_template){
570  std::string str1 = keyT1+"x"+std::to_string(m_labels[counter])+keyT2;
571  std::string str2 = keyT1+"y"+std::to_string(m_labels[counter])+keyT2;
572  std::string str3 = keyT1+"z"+std::to_string(m_labels[counter])+keyT2;
573 
574  writing<<"$VECTORF"<<'\t'<<m_labels[counter]<<'\t'<<str1<<'\t'<<str2<<'\t'<<str3<<std::endl;
575  }else{
576  writing<<"$VECTORF"<<'\t'<<m_labels[counter]<<'\t'<<dd[0]<<'\t'<<dd[1]<<'\t'<<dd[2]<<std::endl;
577  }
578  ++counter;
579  }
580  writing<<""<<std::endl;
581  }else{
582  (*m_log)<<"error of "<<m_name<<" : cannot open "<<m_filename<< " requested. Exiting... "<<std::endl;
583  throw std::runtime_error (m_name + " : cannot open " + m_filename + " requested. Exiting... ");
584  }
585 
586  writing.close();
587 };
