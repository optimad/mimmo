56 void OFOAM_manip() {
57 
58  mimmo::IOOFOAM * reader = new mimmo::IOOFOAM(IOOFMode::READ);
59  reader->setDir("geodata/OFOAM");
60 
61  mimmo::FFDLattice * ffd = new mimmo::FFDLattice();
62  ffd->setShape(mimmo::ShapeType::CUBE);
63 
64  darray3E origin = {{0.367, 0., 0.}};
65  darray3E span = {{0.2, 0.1, 0.1}};
66  dvecarr3E displ(12,{{0.0,0.0,0.0}});
67  displ[4][1] = 0.045;
68  displ[5][1] = 0.045;
69  displ[6][1] = -0.045;
70  displ[7][1] = -0.045;
71 
72  ffd->setOrigin(origin);
73  ffd->setSpan(span);
74  ffd->setDimension(ivector1D({{3,2,2}}));
75  ffd->setDegrees(iarray3E({{2,2,2}}));
76  ffd->setDisplacements(displ);
77  ffd->setPlotInExecution(true);
78 
79  mimmo::Apply * applier = new mimmo::Apply();
80 
81  mimmo::IOOFOAM * writer = new mimmo::IOOFOAM(IOOFMode::WRITEPOINTSONLY);
82  writer->setOverwrite(false);
83 
84  mimmo::pin::addPin(reader, ffd, M_GEOMOFOAM, M_GEOM);
85  mimmo::pin::addPin(reader, applier, M_GEOMOFOAM, M_GEOM);
86  mimmo::pin::addPin(ffd, applier, M_GDISPLS, M_GDISPLS);
87  mimmo::pin::addPin(applier, writer, M_GEOM, M_GEOMOFOAM);
88 
89  mimmo::Chain c0;
90  c0.addObject(reader);
91  c0.addObject(ffd);
92  c0.addObject(applier);
93  c0.addObject(writer);
94  c0.exec(true);
95 
96  delete reader;
97  delete ffd;
98  delete applier;
99  delete writer;
100 }

104 void OFOAM_sensi() {
105 
106  mimmo::IOOFOAM * reader = new mimmo::IOOFOAM(IOOFMode::READ);
107  reader->setDir("geodata/OFOAM");
108 
109  mimmo::IOOFOAMScalarField * fieldreader = new mimmo::IOOFOAMScalarField();
110  fieldreader->setDir("geodata/OFOAM");
111  fieldreader->setFieldName("p");
112 
113  mimmo::Apply * applier = new mimmo::Apply();
114  applier->setScaling(0.1);
115 
116  mimmo::MimmoGeometry * writer = new mimmo::MimmoGeometry();
117  writer->setIOMode(IOMode::WRITE);
118  writer->setWriteDir(".");
119  writer->setWriteFileType(FileType::SURFVTU);
120  writer->setWriteFilename("ofoam_sensi_output");
121 
122 
123  mimmo::pin::addPin(reader, fieldreader, M_GEOMOFOAM2, M_GEOMOFOAM2);
124  mimmo::pin::addPin(reader, fieldreader, M_UMAPIDS, M_UMAPIDS);
125  mimmo::pin::addPin(reader, applier, M_GEOMOFOAM2, M_GEOM);
126  mimmo::pin::addPin(fieldreader, applier, M_SCALARFIELD2, M_SCALARFIELD);
127  mimmo::pin::addPin(applier, writer, M_GEOM, M_GEOM);
128 
129 
130  mimmo::Chain c0;
131  c0.addObject(reader);
132  c0.addObject(fieldreader);
133  c0.addObject(applier);
134  c0.addObject(writer);
135  c0.exec(true);
136 
137  delete reader;
138  delete fieldreader;
139  delete applier;
140  delete writer;
141 }
