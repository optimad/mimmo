195 void
196 ReconstructScalar::removeData(MimmoObject * patch){
197  //in m_subpatch remove progressively all pierced vector elements in last position
198  //which links towards a geometry of type patch.
199  while(m_subpatch.back().getGeometry() == patch){
200  m_subpatch.pop_back();
201  }
202 
203  //start searching from the begin on element linking patch
204  std::vector<dmpvector1D>::iterator it = m_subpatch.begin();
205 
206  while(it != m_subpatch.end()){
207  if (it->getGeometry() == patch){
208  *it = m_subpatch.back();
209  m_subpatch.pop_back();
210  //be always sure the last element is clean and does not link
211  // the patch
212  while(m_subpatch.back().getGeometry() == patch){
213  m_subpatch.pop_back();
214  }
215  }
216  ++it; //forward the iterator.
217  }

344 void
345 ReconstructScalar::execute(){
346 
347  if(getGeometry() == NULL){
348  (*m_log)<<m_name + " : NULL pointer to linked geometry found"<<std::endl;
349  throw std::runtime_error("Error in "+m_name + " : NULL pointer to linked geometry found");
350  }
351 
352  if(getGeometry()->isEmpty()){
353  (*m_log)<<m_name + " : empty linked geometry found"<<std::endl;
354  }
355  //Overlap fields
356  m_result.clear();
357  m_result.setGeometry(getGeometry());
358  m_result.setDataLocation(m_loc);
359 
360  m_subresults.clear();
361 
362  std::unordered_set<long> idsTarget;
363  {
364  livector1D ids = idsGeoDataLocation(getGeometry());
365  idsTarget.insert(ids.begin(), ids.end());
366  }
367 
368  bitpit::PiercedVector<int> counter;
369  for (int i=0; i<getNData(); i++){
370  dmpvector1D* pv = &m_subpatch[i];
371  livector1D ids = idsGeoDataLocation(pv->getGeometry());
372  for (auto ID: ids){
373  if(idsTarget.count(ID)==0) continue;
374 
375  if (!m_result.exists(ID)){
376  m_result.insert(ID, (*pv)[ID]);
377  counter.insert(ID, 1);
378  }
379  else{
380  overlapFields(ID, (*pv)[ID]);
381  counter[ID]++;
382  }
383  }
384  }
385 
386  if (m_result.isEmpty()){
387  (*m_log)<<"Error in "<<m_name<<". Resulting reconstructed field is empty.This is could be caused by unrelated fields linked geometry and target geometry"<<std::endl;
388  }
389 
390  if (m_overlapCriterium == OverlapMethod::AVERAGE){
391  long int ID;
392  auto itend = m_result.end();
393  for (auto it=m_result.begin(); it!=itend; ++it){
394  ID = it.getId();
395  (*it) = (*it) / double(counter[ID]);
396  }
397  }
398 
399  //Update field on whole geometry
400  if(!m_result.completeMissingData(0.0) ){
401  (*m_log)<<"Warning in "<<m_name<<". It seems a reconstruct field with values uncoherent with target geometry is generated."<<std::endl;
402  }
403 
404  //Create subresults
405  m_subresults.resize(getNData());
406  for (int i=0; i<getNData(); i++){
407  dmpvector1D* pv = &m_subpatch[i];
408  m_subresults[i].setGeometry(pv->getGeometry());
409  m_subresults[i].setDataLocation(m_loc);
410  livector1D ids = idsGeoDataLocation(pv->getGeometry());
411  for (auto ID : ids){
412  if(idsTarget.count(ID)>0){
413  m_subresults[i].insert(ID, m_result[ID]);
414  }
415  }
416  }
417 }

439 void
440 ReconstructScalar::overlapFields(long int ID, double & locField){
441 
442  switch(m_overlapCriterium){
443  case OverlapMethod::MAX :
444  {
445  double actual = m_result[ID];
446  double dummy = locField;
447 
448  if (actual < dummy)
449  m_result[ID] = locField;
450  }
451  break;
452  case OverlapMethod::MIN :
453  {
454  double actual = m_result[ID];
455  double dummy = locField;
456 
457  if (actual > dummy)
458  m_result[ID] = locField;
459  }
460  break;
461  case OverlapMethod::AVERAGE :
462  m_result[ID] += locField;
463  break;
464  case OverlapMethod::SUM :
465  m_result[ID] += locField;
466  break;
467  default : //never been reached
468  break;
469  }

498 ReconstructScalar::absorbSectionXML(const bitpit::Config::Section & slotXML, std::string name){
499 
500  BITPIT_UNUSED(name);
501 
502  if(slotXML.hasOption("DataLocation")){
503  std::string input = slotXML.get("DataLocation");
504  input = bitpit::utils::string::trim(input);
505  int temp = -1;
506  if(!input.empty()){
507  std::stringstream ss(input);
508  ss>>temp;
509  }
510  if(int(m_loc) != temp){
511  (*m_log)<<"Error absorbing DataLocation in "<<m_name<<". Class and read locations mismatch"<<std::endl;
512  if (temp == 0) (*m_log)<<"XML DataLocation in "<<m_name<<" is set to 0-UNDEFINED"<<std::endl;
513  if (temp == 3) (*m_log)<<"XML DataLocation in "<<m_name<<" is set to 3-INTERFACE, not supported for now."<<std::endl;
514  throw std::runtime_error (m_name + " : xml absorbing failed.");
515  }
516  }
517  //start absorbing
518  BaseManipulation::absorbSectionXML(slotXML, name);
519 
520  if(slotXML.hasOption("OverlapCriterium")){
521  std::string input = slotXML.get("OverlapCriterium");
522  input = bitpit::utils::string::trim(input);
523  int value = 1;
524  if(!input.empty()){
525  std::stringstream ss(input);
526  ss >> value;
527  value = std::min(std::max(1, value),4);
528  }
529  setOverlapCriterium(value);
530  }
