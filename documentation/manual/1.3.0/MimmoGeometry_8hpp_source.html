249  template<class Type>
250  void writeValue (Type& value, std::ofstream& os){
251 
252  std::size_t offset = 5;
253  if (wformat == Long) offset = 13;
254 
255  std::stringstream manip;
256  manip << value;
257  std::string manips = manip.str();
258  std::string mantissa, expon;
259  std::size_t pos = manips.find("E");
260  if (pos < manips.size()){
261  mantissa = manips.substr(0,std::min(offset,pos));
262  expon = manips.substr(pos+1,manips.size());
263  manips = mantissa + expon;
264  }
265  pos = manips.find("e");
266  if (pos < manips.size()){
267  mantissa = manips.substr(0,std::min(offset,pos));
268  expon = manips.substr(pos+1,manips.size());
269  manips = mantissa + expon;
270  }
271  manips = manips.substr(0, offset+3);
272 
273  switch (wformat)
274  {
275  case Short:
276  {
277  os << std::setw(8) << manips;
278  break;
279  }
280  case Long:
281  {
282  os << std::setw(16) << manips;
283  break;
284  }
285  }
286  }
287 };
