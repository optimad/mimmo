226 void
227 ProjectCloud::execute(){
228 
229  if(getGeometry() == NULL){
230  (*m_log)<<m_name + " : NULL pointer to linked geometry found"<<std::endl;
231  throw std::runtime_error(m_name + "NULL pointer to linked geometry found");
232  }
233 
234  if(getGeometry()->isEmpty()){
235  (*m_log)<<m_name + " : empty linked geometry found"<<std::endl;
236  return;
237  }
238  if(getGeometry()->getType() != 1){
239  return;
240  }
241 
242  if(!getGeometry()->isSkdTreeSync()) getGeometry()->buildSkdTree();
243 
244  //project points on surface.
245  int counter = 0;
246  m_proj.resize(m_points.size(), {{0.0,0.0,0.0}});
247  for(darray3E &val : m_points){
248  m_proj[counter]= skdTreeUtils::projectPoint(&val, getGeometry()->getSkdTree());
249  ++counter;
250  }
251  return;
252 };

258 void
259 ProjectCloud::plotOptionalResults(){
260  bitpit::VTKFormat codex = bitpit::VTKFormat::APPENDED;
261 
262  int size = m_proj.size();
263  ivector1D conn(size);
264  for(int i=0; i<size; i++){
265  conn[i] = i;
266  }
267  std::string dir = "./";
268  std::string file = m_name + "_" + std::to_string(getId());
269 
270  livector1D labels = m_labels;
271  labels.resize(m_proj.size());
272 
273  bitpit::VTKUnstructuredGrid vtk(dir, file, bitpit::VTKElementType::VERTEX);
274  vtk.setGeomData( bitpit::VTKUnstructuredField::POINTS, m_proj) ;
275  vtk.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, conn) ;
276  vtk.setDimensions(size, size);
277  vtk.addData("labels", bitpit::VTKFieldType::SCALAR, bitpit::VTKLocation::POINT, labels);
278  vtk.setCodex(codex);
279 
280  vtk.write();
281 }
