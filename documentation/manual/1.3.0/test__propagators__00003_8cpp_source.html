47 std::unique_ptr<mimmo::MimmoObject> createTestVolumeMesh(std::unique_ptr<mimmo::MimmoObject> & boundary){
48 
49  std::array<double,3> center({{0.0,0.0,0.0}});
50  double radiusin(2.0), radiusout(5.0);
51  double azimuthin(0.0), azimuthout(0.5*BITPIT_PI);
52  double heightbottom(-1.0), heighttop(1.0);
53  int nr(6), nt(10), nh(6);
54 
55  double deltar = (radiusout - radiusin)/ double(nr);
56  double deltat = (azimuthout - azimuthin)/ double(nt);
57  double deltah = (heighttop - heightbottom)/ double(nh);
58 
59  std::vector<std::array<double,3> > verts ((nr+1)*(nt+1)*(nh+1));
60 
61  int counter = 0;
62  for(int k=0; k<=nh; ++k){
63  for(int j=0; j<=nt; ++j){
64  for(int i=0; i<=nr; ++i){
65  verts[counter][0] =(radiusin + i*deltar)*std::cos(azimuthin + j*deltat);
66  verts[counter][1] =(radiusin + i*deltar)*std::sin(azimuthin + j*deltat);
67  verts[counter][2] =(heightbottom + k*deltah);
68  ++counter;
69  }
70  }
71  }
72 
73  //create the volume mesh mimmo.
74  std::unique_ptr<mimmo::MimmoObject> mesh = std::unique_ptr<mimmo::MimmoObject>(new mimmo::MimmoObject(2));
75  mesh->getPatch()->reserveVertices((nr+1)*(nt+1)*(nh+1));
76 
77  //pump up the vertices
78  for(const auto & vertex : verts){
79  mesh->addVertex(vertex); //automatic id assigned to vertices.
80  }
81  mesh->getPatch()->reserveCells(nr*nt*nh);
82 
83  //create connectivities for hexa elements
84  std::vector<long> conn(8,0);
85  for(int k=0; k<nh; ++k){
86  for(int j=0; j<nt; ++j){
87  for(int i=0; i<nr; ++i){
88  conn[0] = (nr+1)*(nt+1)*k + (nr+1)*j + i;
89  conn[1] = (nr+1)*(nt+1)*k + (nr+1)*j + i+1;
90  conn[2] = (nr+1)*(nt+1)*k + (nr+1)*(j+1) + i+1;
91  conn[3] = (nr+1)*(nt+1)*k + (nr+1)*(j+1) + i;
92  conn[4] = (nr+1)*(nt+1)*(k+1) + (nr+1)*j + i;
93  conn[5] = (nr+1)*(nt+1)*(k+1) + (nr+1)*j + i+1;
94  conn[6] = (nr+1)*(nt+1)*(k+1) + (nr+1)*(j+1) + i+1;
95  conn[7] = (nr+1)*(nt+1)*(k+1) + (nr+1)*(j+1) + i;
96  mesh->addConnectedCell(conn, bitpit::ElementType::HEXAHEDRON);
97  }
98  }
99  }
100 
101  mesh->buildAdjacencies();
102  mesh->buildInterfaces();
103 
104  //decompose manually the six boundary faces.
105  std::vector<std::vector<long>> bverts(6, std::vector<long>());
106 
107  //vertex at azimuthal limits.
108  for(int k=0; k<=nh; ++k){
109  for(int i=0; i<=nr; ++i){
110  bverts[0].push_back((nr+1)*(nt+1)*k + i);
111  bverts[1].push_back((nr+1)*(nt+1)*k + (nr+1)*nt + i);
112  }
113  }
114 
115  //vertex at quote limits.
116  for(int j=0; j<=nt; ++j){
117  for(int i=0; i<=nr; ++i){
118  bverts[2].push_back((nr+1)*j + i);
119  bverts[3].push_back((nr+1)*(nt+1)*nh + (nr+1)*j + i);
120  }
121  }
122 
123  //vertex at radius limits.
124  for(int k=0; k<=nh; ++k){
125  for(int j=0; j<=nt; ++j){
126  bverts[4].push_back((nr+1)*(nt+1)*k + (nr+1)*j);
127  bverts[5].push_back((nr+1)*(nt+1)*k + (nr+1)*j + nr);
128  }
129  }
130 
131  //put together
132  mimmo::MimmoPiercedVector<long> pidfaces;
133  long pidder = 1;
134  for(livector1D & vert : bverts ){
135  livector1D boundaryfaces = mesh->getInterfaceFromVertexList(vert, true, true);
136  for(long id : boundaryfaces){
137  pidfaces.insert(id, pidder);
138  }
139  ++pidder;
140  }
141  bverts.clear();
142  std::vector<long> boundaryverts = mesh->extractBoundaryVertexID();
143 
144  //create the boundary meshes
145  auto orinterfaces = mesh->getInterfaces();
146  auto orcells = mesh->getCells();
147 
148  boundary = std::unique_ptr<mimmo::MimmoObject>(new mimmo::MimmoObject(1));
149  boundary->getPatch()->reserveVertices(boundaryverts.size());
150  boundary->getPatch()->reserveCells(pidfaces.size());
151 
152  //push in verts
153  for(long id : boundaryverts){
154  boundary->addVertex(mesh->getVertexCoords(id), id);
155  }
156 
157  //push in interfaces ad 2D cells
158  for(auto it=pidfaces.begin(); it!=pidfaces.end(); ++it){
159  bitpit::Interface & ii = orinterfaces.at(it.getId());
160  long * conn = ii.getConnect();
161  std::size_t connsize = ii.getConnectSize();
162 
163  bitpit::ElementType et = orcells.at(ii.getOwner()).getFaceType(ii.getOwnerFace());
164 
165  boundary->addConnectedCell(std::vector<long>(&conn[0], &conn[connsize]), et, *it,it.getId());
166  }
167  boundary->buildAdjacencies();
168  boundary->getPatch()->write("piddedBoundary_test3");
169 
170  return mesh;
171 }

174 std::unique_ptr<mimmo::MimmoObject> pidExtractor(mimmo::MimmoObject* boundary, const livector1D & pids){
175  if(!boundary) return nullptr;
176 
177  livector1D cellsID = boundary->extractPIDCells(pids);
178  livector1D vertID = boundary->getVertexFromCellList(cellsID);
179 
180  std::unique_ptr<mimmo::MimmoObject> extracted(new mimmo::MimmoObject(1));
181  extracted->getPatch()->reserveVertices(vertID.size());
182  extracted->getPatch()->reserveCells(cellsID.size());
183 
184  for(long id: vertID){
185  extracted->addVertex(boundary->getVertexCoords(id), id);
186  }
187  for(long id: cellsID){
188  extracted->addCell(boundary->getCells().at(id), id);
189  }
190  extracted->buildAdjacencies();
191  return extracted;
192 }

196 int test1() {
197 
198  std::unique_ptr<mimmo::MimmoObject> bDirMesh;
199  std::unique_ptr<mimmo::MimmoObject> mesh = createTestVolumeMesh(bDirMesh);
200 
201  std::unique_ptr<mimmo::MimmoObject> bDIR = pidExtractor(bDirMesh.get(), {{1,2}});
202  std::unique_ptr<mimmo::MimmoObject> bSLIP = pidExtractor(bDirMesh.get(), {{5,6}});
203 
204 //TESTING THE VECTOR PROPAGATOR //////
205  // and the scalar field of Dirichlet values on pidded surface 0 and 1 nodes.
206  mimmo::MimmoPiercedVector<std::array<double,3>> bc_surf_3Dfield;
207  bc_surf_3Dfield.setGeometry(bDIR.get());
208  bc_surf_3Dfield.setDataLocation(mimmo::MPVLocation::POINT);
209  bc_surf_3Dfield.reserve(bDIR->getNVertices());
210 
211  std::vector<long> d1 = bDIR->getVertexFromCellList(bDIR->extractPIDCells(1));
212  for(long id : d1){
213  bc_surf_3Dfield.insert(id, {{0.0, 0.0, 0.0}});
214  }
215  //apply a rotation of alpha around z centered in the origin 0,0,0, so that deformation is
216  //x-directed and equal to tan(alpha) * y-coord of the point.
217  //BEWARE IF THE ROTATION ARC IS similar to a linear segment, this deformation works,
218  //for large rotation, where the arc path is very different from a different segment, this
219  // could lead to not properly expected results.
220  double alpha = 9.0*BITPIT_PI/180.0;
221  std::vector<long> d2 = bDIR->getVertexFromCellList(bDIR->extractPIDCells(2));
222  for(long id : d2){
223  darray3E coord = bDIR->getVertexCoords(id);
224  double normc = std::sqrt(coord[0]*coord[0] + coord[1]*coord[1]);
225  bc_surf_3Dfield.insert(id, {{std::sin(alpha)*normc,
226  (std::cos(alpha)-1.0)*normc,
227  0.0}} );
228  }
229 
230  // Now create a PropagateScalarField and solve the laplacian.
231  mimmo::PropagateVectorField * prop3D = new mimmo::PropagateVectorField();
232  prop3D->setName("test00003_PropagateVectorField");
233  prop3D->setGeometry(mesh.get());
234  prop3D->setDirichletBoundarySurface(bDIR.get());
235  prop3D->setDirichletConditions(&bc_surf_3Dfield);
236  prop3D->setSlipBoundarySurface(bSLIP.get());
237  prop3D->setSlipReferenceSurface(bSLIP.get());
238 
239  prop3D->setMethod(mimmo::PropagatorMethod::GRAPHLAPLACE);
240 
241  prop3D->setDumping(true);
242  prop3D->setDumpingType(0);
243  prop3D->setDecayFactor(2.0);
244  prop3D->setDumpingInnerDistance(0.5);
245  prop3D->setDumpingOuterDistance(2.5);
246 
247  prop3D->setSolverMultiStep(3);
248 
249  prop3D->setPlotInExecution(true);
250 
251  prop3D->exec();
252 
253  //deform the mesh and write deformation
254  mimmo::dmpvecarr3E * values3D = prop3D->getPropagatedField();
255  darray3E work;
256  for(auto it=values3D->begin(); it!=values3D->end(); ++it){
257  work = mesh->getVertexCoords(it.getId());
258  mesh->modifyVertex(work + *it, it.getId());
259  }
260 
261  mesh->getPatch()->write("test00003_deformedMesh");
262 
263  bool check = false;
264  long targetNode = (10 +1)*(6+1)*3 + (6+1)*5 + 3;
265 
266  std::cout<<values3D->at(targetNode)<<std::endl;
267 
268 // check = check || (norm2(values3D.at(targetNode)-std::array<double,3>({{0.206202, -0.0621586, 4.61279e-17}})) > 1.0E-5);
269  check = false;
270 
271  delete prop3D;
272  return check;
273 }
