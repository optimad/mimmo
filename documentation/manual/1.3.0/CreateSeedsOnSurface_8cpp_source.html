404 void
405 CreateSeedsOnSurface::solveLSet(bool debug){
406 
407  if(debug) (*m_log)<<m_name<<" : started LevelSet engine"<<std::endl;
408  dvecarr3E initList;
409  m_deads.reserve(m_nPoints);
410 
411  //understand if the class is a pure triangulation or not
412  std::unique_ptr<MimmoObject> objTriangulated;
413  MimmoObject * workgeo;
414  dmpvector1D worksensitivity;
415  if(!checkTriangulation()){
416  objTriangulated = std::move(triangulate());
417  workgeo = objTriangulated.get();
418  worksensitivity = m_sensitivity;
419  }else{
420  workgeo = getGeometry();
421  worksensitivity = m_sensitivity_triangulated;
422  }
423 
424  if(!(workgeo->areAdjacenciesBuilt()) ) workgeo->buildAdjacencies();
425  if(!(workgeo->isSkdTreeSync())) workgeo->buildSkdTree();
426  if(!(workgeo->isKdTreeSync())) workgeo->buildKdTree();
427  bitpit::SurfUnstructured * tri = static_cast<bitpit::SurfUnstructured * >(workgeo->getPatch());
428 
429  double distance = 0.0;
430  for(const auto &cell : tri->getCells()){
431  distance += tri->evalCellArea(cell.getId());
432  }
433  distance /= double(tri->getCellCount());
434  distance = std::pow(distance, 0.5);
435 
436  livector1D neighs, excl;
437  darray3E projSeed = skdTreeUtils::projectPoint(&m_seed, workgeo->getSkdTree());
438  bitpit::Vertex vertSeed(0, projSeed);
439  //find the vertex of the mesh nearest to projSeed.
440  int nSize = 0;
441  while( nSize < 1){
442  workgeo->getKdTree()->hNeighbors(&vertSeed, distance, &neighs, & excl );
443  nSize = neighs.size();
444  distance *= 1.1;
445  }
446 
447  long candidate = neighs[0];
448  double minDist = norm2(projSeed - tri->getVertexCoords(candidate));
449  for(int i=1; i<nSize; ++i){
450  double val = norm2(projSeed - tri->getVertexCoords(neighs[i]));
451  if(val < minDist){
452  candidate = neighs[i];
453  minDist = val;
454  }
455  }
456 
457  m_deads.push_back(candidate);
458  int deadSize = m_deads.size();
459  if(debug) (*m_log)<<m_name<<" : projected seed point"<<std::endl;
460 
461  std::unordered_map<long,long> invConn = getInverseConn(*(workgeo->getPatch()));
462  if(debug) (*m_log)<<m_name<<" : created geometry inverse connectivity"<<std::endl;
463 
464  while(deadSize < m_nPoints){
465 
466  dmpvector1D field;
467  for (const auto & v : tri->getVertices()){
468  field.insert(v.getId(), 1.0E+18);
469  }
470  for(const auto & dd : m_deads) field[dd] = 0.0;
471 
472  solveEikonal(1.0,1.0, *(workgeo->getPatch()), invConn, field);
473 
474  //modulate field with current working sensitivity field
475  auto itSE=worksensitivity.end();
476  for(auto itSX =worksensitivity.begin(); itSX != itSE; ++itSX){
477  field[itSX.getId()] *= *itSX;
478  }
479 
480  double maxField= 0.0;
481  long candMax =0;
482  auto itE=field.end();
483  for(auto itX =field.begin(); itX != itE; ++itX){
484  if(*itX > maxField){
485  maxField = *itX;
486  candMax = itX.getId();
487  }
488  }
489 
490  m_deads.push_back(candMax);
491 
492  deadSize = m_deads.size();
493  if(debug) (*m_log)<<m_name<<" : geodesic distance field for point "<<deadSize-1<<" found"<<std::endl;
494  }
495 
496  //store result in m_points.
497  m_points.reserve(deadSize);
498  for(const auto & val: m_deads){
499  m_points.push_back(tri->getVertexCoords(val));
500  }
501 
502  m_minDist = 1.E18;
503 
504  for(int i=0; i<deadSize; ++i){
505  for(int j=i+1; j<deadSize; ++j){
506  m_minDist = std::fmin(m_minDist,norm2(m_points[i] - m_points[j]));
507  }
508  }
509 
510  m_deads.clear();
511  if(debug) (*m_log)<<m_name<<" : distribution of point successfully found w/ LevelSet engine "<<std::endl;
512 };

520 void
521 CreateSeedsOnSurface::solveGrid(bool debug){
522 
523  if(debug) (*m_log)<<m_name<<" : started CartesianGrid engine"<<std::endl;
524  iarray3E dim;
525 
526  if(!(getGeometry()->isSkdTreeSync())) getGeometry()->buildSkdTree();
527 
528  //get the seed and project it on surface
529  darray3E projSeed = skdTreeUtils::projectPoint(&m_seed, getGeometry()->getSkdTree());
530  if(debug) (*m_log)<<m_name<<" : projected seed point"<<std::endl;
531  if (m_nPoints == 1) {
532  m_points.clear();
533  m_points.push_back(projSeed);
534  return;
535  }
536 
537  m_minDist = norm2(bbox->getSpan());
538  double dx = m_minDist/int(std::pow(double(m_nPoints),0.5)+ 0.5);
539  {
540  //check dimension
541  std::vector<std::pair<double,int> > mapDimension(3);
542  std::vector<std::pair<double,int> >::iterator itm;
543  std::vector<std::pair<double,int> >::reverse_iterator ritm;
544 
545  mapDimension[0] = std::make_pair(bbox->getSpan()[0]/dx, 0);
546  mapDimension[1] = std::make_pair(bbox->getSpan()[1]/dx, 1);
547  mapDimension[2] = std::make_pair(bbox->getSpan()[2]/dx, 2);
548 
549  std::sort(mapDimension.begin(), mapDimension.end());
550 
551  for(itm = mapDimension.begin(); itm !=mapDimension.end(); ++itm){
552  dim[itm->second] = std::max(int(itm->first + 0.5),1);
553  }
554 
555  int cumCells = dim[0]*dim[1]*dim[2];
556  ritm=mapDimension.rbegin();
557  while (cumCells < m_nPoints){
558 
559  dim[ritm->second] += 1;
560  ritm++;
561  if(ritm == mapDimension.rend()) ritm = mapDimension.rbegin();
562 
563  cumCells = dim[0]*dim[1]*dim[2];
564  }
565  }
566 
567  //raise dimension to number of effective nodes for each direction.
568  dim[0] +=1;
569  dim[1] +=1;
570  dim[2] +=1;
571 
572  //create a lattice on it
573  mimmo::Lattice * grid = new Lattice();
574  grid->setShape(mimmo::ShapeType::CUBE);
575  grid->setOrigin(bbox->getOrigin());
576  grid->setSpan(bbox->getSpan());
577  grid->setRefSystem(bbox->getAxes());
578  grid->setDimension(dim);
579  grid->execute();
580  //grid->plotGrid("./", "lattice",0,false);
581 
582  if(debug) (*m_log)<<m_name<<" : build volume cartesian grid wrapping 3D surface"<<std::endl;
583  //find narrow band cells and extracting their centroids
584  dvecarr3E centroids = grid->getGlobalCellCentroids();
585 
586  dvecarr3E initList;
587  initList.reserve(centroids.size());
588 
589  double distR = 0.5*norm2(grid->getSpacing());
590  double dist, dummy;
591  long id;
592  darray3E normal;
593  for(auto & p : centroids){
594  dummy=distR;
595  dist = mimmo::skdTreeUtils::signedDistance(&p,getGeometry()->getSkdTree(),id, normal, dummy);
596  if(std::abs(dist) < distR){
597  initList.push_back(p-dist*normal);
598  }
599  normal.fill(0.0);
600  }
601  if(debug) (*m_log)<<m_name<<" : found grid cell centers in the narrow band of 3D surface and projected them on it "<<std::endl;
602 
603 
604  //rearrange the list, putting the most nearest point to the seed on top
605  // and decimate points up to desired value.
606  int initListSize = initList.size();
607  if( initListSize > m_nPoints){
608 
609  dvecarr3E secondList;
610  secondList.reserve(initList.size());
611  {
612  dvecarr3E::iterator it, itMaxNorm = initList.begin();
613  double normPoint, minValue = 1.E18;
614 
615  for(it= initList.begin(); it !=initList.end(); ++it){
616  normPoint= norm2(*it - m_seed);
617  if(normPoint < minValue){
618  minValue = normPoint;
619  itMaxNorm= it;
620  }
621  }
622 
623  darray3E temp = *itMaxNorm;
624  initList.erase(itMaxNorm);
625  secondList.push_back(temp);
626  secondList.insert(secondList.end(),initList.begin(), initList.end());
627  initList.clear();
628  }
629 
630  initList = decimatePoints(secondList);
631  if(debug) (*m_log)<<m_name<<" : candidates decimated "<<std::endl;
632  }
633  //store result in m_points.
634  m_points = initList;
635  m_nPoints = (int)m_points.size();
636  if(debug) (*m_log)<<m_name<<" : distribution of point successfully found w/ CartesianGrid engine "<<std::endl;
637  delete grid; grid = NULL;
638 };

646 void
647 CreateSeedsOnSurface::solveRandom(bool debug){
648 
649  if(debug) (*m_log)<<m_name<<" : started Random engine"<<std::endl;
650  dvecarr3E initList(getNPoints());
651 
652  if(!(getGeometry()->isSkdTreeSync())) getGeometry()->buildSkdTree();
653 
654  //get the seed and project it on surface
655  initList[0] = skdTreeUtils::projectPoint(&m_seed, getGeometry()->getSkdTree());
656  if(debug) (*m_log)<<m_name<<" : projected seed point"<<std::endl;
657  if (m_nPoints == 1) {
658  m_points.clear();
659  m_points = initList;
660  return;
661  }
662 
663  m_minDist = norm2(bbox->getSpan()) / 2.0;
664 
665  //fill the oriented bounding box of the figure randomly with max of 100 and 5*m_nPoints
666  dvecarr3E tentative;
667  {
668  darray3E span = bbox->getSpan();
669  dmatrix33E axes = bbox->getAxes();
670  darray3E minP = bbox->getOrigin();
671  for(int i=0; i<3; ++i) {minP += - 0.5*span[i]*axes[i];}
672 
673 
674  if (m_randomFixed <0 ){
675  m_randomFixed = (unsigned int)time(NULL);
676  }
677  srand(static_cast<unsigned int>(m_randomFixed));
678 
679  int nTent = std::max(100,5*m_nPoints);
680  tentative.resize(nTent+1);
681  tentative[0] = initList[0];
682 
683  for(int i = 0; i<nTent; ++i){
684  tentative[i+1] = minP;
685  for(int j=0; j<3; ++j){
686 
687  double valrand = (std::rand()%100)/99.0;
688  tentative[i+1] += valrand * span[j]*axes[j];
689  }
690  }
691 
692  //project tentative points on surface.
693  for(int i = 0; i<nTent; ++i){
694  minP = skdTreeUtils::projectPoint(&tentative[i+1], getGeometry()->getSkdTree());
695  tentative[i+1] = minP;
696  }
697  }
698 
699 
700  if(debug) (*m_log)<<m_name<<" : found random points"<<std::endl;
701  //decimate points up to desired value
702  initList = decimatePoints(tentative);
703  if(debug) (*m_log)<<m_name<<" : decimated random points"<<std::endl;
704  //store result in m_points.
705  m_points = initList;
706  m_nPoints = (int)m_points.size();
707  if(debug) (*m_log)<<m_name<<" : distribution of point successfully found w/ Random engine "<<std::endl;
708 };

754 dvecarr3E
755 CreateSeedsOnSurface::decimatePoints(dvecarr3E & list){
756 
757  dvecarr3E result(m_nPoints);
758  int listS = list.size();
759 
760  //reference all coordinate list to the seed candidate 0;
761  // modulate the coordinate of each point with its respective sensitivity.
762  dvecarr3E listRefer(listS, {{0.0,0.0,0.0}});
763  for(int j=0; j<listS; ++j){
764  listRefer[j] = (list[j] - list[0])*interpolateSensitivity(list[j]);
765  }
766 
767  bitpit::KdTree<3,darray3E,long> kdT;
768  //build a kdtree of points
769  kdT.nodes.resize(listS + kdT.MAXSTK);
770  long label=0;
771  for(auto & val : listRefer ){
772  kdT.insert(&val, label);
773  ++label;
774  }
775 
776  long candidate = 0; //starting seed;
777 
778  //use kdTree to search points within a radius of m_minDist from the seed.

870 double
871 CreateSeedsOnSurface::updateEikonal(double g, double s, long tVert,long tCell, bitpit::PatchKernel &tri, std::unordered_map<long int, short int> &flag, dmpvector1D & field){
872 
873  BITPIT_UNUSED(s);
874 
875  livector1D oneRing;
876  long I, U, V, W;
877 
878  int k, m;
879 
880  int select = -1;
881  double value(1.0e+18);
882  double dVU, dVW, dWU, dVP;
883  std::array<double,3> eVU, eVW, eWU, eVP, P;
884 
885 
886  double a, b, c, A, B, C, K, discr ;
887  double phi_U, phi_W, phi_P;
888  int discrType;
889 
890  double xi1, xi2;
891  double tempVal1, tempVal2;
892 
893  //get current field value of the node;
894  value = std::abs(field[tVert]);
895 
896  V = tVert;
897 
898  {
899  // find 1-Ring cells around target node
900  bitpit::Cell & targetCell = tri.getCell(tCell);
901  int locVert = targetCell.findVertex(tVert);
902  if(locVert == bitpit::Vertex::NULL_ID) return value;
903  oneRing = tri.findCellVertexOneRing(tCell, locVert);
904  }
905 
906  // Loop over cells in the 1-Ring --------------------------------------------------- //
907  for (auto && oneIndex : oneRing) {
908 
909  // Cell data get id of vertex composing triangular cell
910  I = oneIndex;
911  bitpit::Cell & cellI = tri.getCell(I);
912  long * connCellI = cellI.getConnect();
913  k = cellI.findVertex(V);
914  k = (k + 1) % cellI.getVertexCount();
915  U = connCellI[k];
916  m = (k + 1) % cellI.getVertexCount();
917  W = connCellI[m];
918 
919  // discriminate case, according to flag vector of deads, alives and far-aways
920  if ((flag[U] == 0) && (flag[W] == 0)) {
921  select = 2;
922  }
923  else {
924  if ((flag[U] == 0) || (flag[W] == 0)) {
925  select = 1;
926  if (flag[W] == 0) {
927  U = W;
928  }
929  }
930  else {
931  select = 0;
932  }
933  }
934 
935  //Compute solution to the 2D Eikonal equation
936  switch (select){
937 
938  case 1 : // with 1 dead node
939  eVU = tri.getVertexCoords(V) - tri.getVertexCoords(U);
940  dVU = norm2(eVU);
941  value = std::min(value, std::abs(field[U]) + g*dVU);
942  break;
943 
944  case 2 : // with 2 dead nodes
945 
946  eVW = tri.getVertexCoords(V) - tri.getVertexCoords(W);
947  dVW = norm2(eVW);
948  eVW = eVW/dVW;
949  eVU = tri.getVertexCoords(V) - tri.getVertexCoords(U);
950  dVU = norm2(eVU);
951  eVU = eVU/dVU;
952  eWU = tri.getVertexCoords(W) - tri.getVertexCoords(U);
953  dWU = norm2(eWU);
954  eWU = eWU/dWU;
955 
956  // Coeffs -------------------------------------------------------------------- //
957  phi_U = std::abs(field[U]);
958  phi_W = std::abs(field[W]);
959  K = phi_W - phi_U;
960  a = pow(dWU, 2);
961  b = -dWU*dVU*dotProduct(eWU, eVU);
962  c = pow(dVU, 2);
963  A = a*(pow(K, 2) - a);
964  B = b*(pow(K, 2) - a);
965  C = (pow(K, 2)*c - pow(b, 2));
966  discr = pow(B, 2) - A*C;
967 
968  // Find optimal solution ----------------------------------------------------- //
969  discrType = (discr < -1.0e-12) + 2*(std::abs(A) > 1.0e-12) +3*((std::abs(A) < 1.0e-12) && (std::abs(A) >= 0.0)) -1;
970 
971  switch(discrType){
972  case 1: //2 distinct solutions
973  discr = std::abs(discr);
974 
975  xi1 = (-B - sqrt(discr))/A;
976  xi2 = (-B + sqrt(discr))/A;
977 
978  // Restriction of solutions onto [0, 1]
979  xi1 = std::min(1.0, std::max(0.0, xi1));
980  xi2 = std::min(1.0, std::max(0.0, xi2));
981 
982  // Solution #1
983  P = (1.0 - xi1) * tri.getVertexCoords(U) + xi1 * tri.getVertexCoords(W);
984  eVP = tri.getVertexCoords(V) - P;
985  dVP = norm2(eVP);
986  eVP = eVP/dVP;
987  phi_P = (1.0 - xi1) * phi_U + xi1 * phi_W;
988  tempVal1 = phi_P + g * dVP;
989 
990  // Solution #2
991  P = (1.0 - xi2) * tri.getVertexCoords(U) + xi2 * tri.getVertexCoords(W);
992  eVP = tri.getVertexCoords(V) - P;
993  dVP = norm2(eVP);
994  eVP = eVP/dVP;
995  phi_P = (1.0 - xi2) * phi_U + xi2 * phi_W;
996  tempVal2 = phi_P + g * dVP;
997 
998  break;
999 
1000  case 2: // coincident solutions
1001  discr = std::abs(discr);
1002 
1003  // Solution #1
1004  P = tri.getVertexCoords(U);
1005  eVP = tri.getVertexCoords(V) - P;
1006  dVP = norm2(eVP);
1007  eVP = eVP/dVP;
1008  phi_P = phi_U;
1009  tempVal1 = phi_P + g*dVP;
1010 
1011  // Solution #2
1012  P = tri.getVertexCoords(W);
1013  eVP = tri.getVertexCoords(V) - P;
1014  dVP = norm2(eVP);
1015  eVP = eVP/dVP;
1016  phi_P = phi_W;
1017  tempVal2 = phi_P + g*dVP;
1018 
1019  break;
1020 
1021  default: //no real solutions indeed
1022  tempVal1 = value;
1023  tempVal2 = value;
1024  break;
1025  }//end on switch discrType
1026 
1027  // Update solution for case 2:
1028  value = std::min(value, std::min(tempVal1, tempVal2));
1029  break;
1030 
1031  default: //doing really nothing. No dead nodes to hang out.
1032  break;
1033  }//end switch select
1034 
1035  } //loop on oneRing
1036 
1037  return(value);
1038 };
