543 template<std::size_t NCOMP>
544 void PropagateField<NCOMP>::initializeDumpingSurface(){
545 
546  m_originalDumpingSurface = nullptr;
547 
548  if (!m_dumpingActive || m_decayFactor <= 1.E-12) {
549  return;
550  };
551 
552  MimmoObject * dumptarget= m_dsurface;
553  if(m_dsurface == nullptr) dumptarget = m_bsurface;
554 
555 #if MIMMO_ENABLE_MPI
556 
557  if (getTotalProcs() > 1){
558 
559  //MPI version
560  m_originalDumpingSurface = std::move(std::unique_ptr<MimmoObject>(new MimmoObject(1)));
561 
562  //fill serialized geometry
563  for (bitpit::Vertex &vertex : dumptarget->getVertices()){
564  long vertexId = vertex.getId();
565  if (dumptarget->isPointInterior(vertexId)){
566  m_originalDumpingSurface->addVertex(vertex, vertexId);
567  }
568  }
569  for (bitpit::Cell &cell : dumptarget->getCells()){
570  if (cell.isInterior()){
571  m_originalDumpingSurface->addCell(cell, cell.getId());
572  }
573  }
574 
575  //Receive vertices and cells
576  for (int sendRank=0; sendRank<m_nprocs; sendRank++){
577 
578  if (m_rank != sendRank){
579 
580  // Vertex data
581  long vertexBufferSize;
582  MPI_Recv(&vertexBufferSize, 1, MPI_LONG, sendRank, 100, m_communicator, MPI_STATUS_IGNORE);
583 
584  bitpit::IBinaryStream vertexBuffer(vertexBufferSize);
585  MPI_Recv(vertexBuffer.data(), vertexBuffer.getSize(), MPI_CHAR, sendRank, 110, m_communicator, MPI_STATUS_IGNORE);
586 
587  // Cell data
588  long cellBufferSize;
589  MPI_Recv(&cellBufferSize, 1, MPI_LONG, sendRank, 200, m_communicator, MPI_STATUS_IGNORE);
590 
591  bitpit::IBinaryStream cellBuffer(cellBufferSize);
592  MPI_Recv(cellBuffer.data(), cellBuffer.getSize(), MPI_CHAR, sendRank, 210, m_communicator, MPI_STATUS_IGNORE);
593 
594  // There are no duplicate in the received vertices, but some of them may
595  // be already a local vertex of a interface cell.
596  //TODO GENERALIZE IT
597  //NOTE! THE COHINCIDENT VERTICES ARE SUPPOSED TO HAVE THE SAME ID!!!!
598  long nRecvVertices;
599  vertexBuffer >> nRecvVertices;
600  m_originalDumpingSurface->getPatch()->reserveVertices(m_originalDumpingSurface->getPatch()->getVertexCount() + nRecvVertices);
601 
602  // Do not add the vertices with Id already in serialized geometry
603  for (long i = 0; i < nRecvVertices; ++i) {
604  bitpit::Vertex vertex;
605  vertexBuffer >> vertex;
606  long vertexId = vertex.getId();
607 
608  if (!m_originalDumpingSurface->getVertices().exists(vertexId)){
609  m_originalDumpingSurface->addVertex(vertex, vertexId);
610  }
611  }
612 
613  //Receive and add all Cells
614  long nReceivedCells;
615  cellBuffer >> nReceivedCells;
616  m_originalDumpingSurface->getPatch()->reserveCells(m_originalDumpingSurface->getPatch()->getCellCount() + nReceivedCells);
617 
618  for (long i = 0; i < nReceivedCells; ++i) {
619  // Cell data
620  bitpit::Cell cell;
621  cellBuffer >> cell;
622 
623  long cellId = cell.getId();
624 
625  // Add cell
626  m_originalDumpingSurface->addCell(cell, cellId);
627 
628  }
629  }
630  else{
631 
632  //Send local vertices and local cells to ranks
633 
634  //
635  // Send vertex data
636  //
637  bitpit::OBinaryStream vertexBuffer;
638  long vertexBufferSize = 0;
639  long nVerticesToCommunicate = 0;
640 
641  // Fill buffer with vertex data
642  vertexBufferSize += sizeof(long);
643  for (long vertexId : dumptarget->getVertices().getIds()){
644  if (dumptarget->isPointInterior(vertexId)){
645  vertexBufferSize += dumptarget->getVertices()[vertexId].getBinarySize();
646  nVerticesToCommunicate++;
647  }
648  }
649  vertexBuffer.setSize(vertexBufferSize);
650 
651  vertexBuffer << nVerticesToCommunicate;
652  for (long vertexId : dumptarget->getVertices().getIds()){
653  if (dumptarget->isPointInterior(vertexId)){
654  vertexBuffer << dumptarget->getVertices()[vertexId];
655  }
656  }
657 
658  for (int recvRank=0; recvRank<m_nprocs; recvRank++){
659 
660  if (m_rank != recvRank){
661 
662  // Communication
663  MPI_Send(&vertexBufferSize, 1, MPI_LONG, recvRank, 100, m_communicator);
664  MPI_Send(vertexBuffer.data(), vertexBuffer.getSize(), MPI_CHAR, recvRank, 110, m_communicator);
665 
666  }
667  }
668 
669  //
670  // Send cell data
671  //
672  bitpit::OBinaryStream cellBuffer;
673  long cellBufferSize = 0;
674  long nCellsToCommunicate = 0;
675 
676  // Fill the buffer with cell data
677  cellBufferSize += sizeof(long);
678  for (const long cellId : dumptarget->getCellsIds()) {
679  if (dumptarget->getCells()[cellId].isInterior()){
680  cellBufferSize += sizeof(int) + sizeof(int) + dumptarget->getCells()[cellId].getBinarySize();
681  nCellsToCommunicate++;
682  }
683  }
684  cellBuffer.setSize(cellBufferSize);
685 
686  cellBuffer << nCellsToCommunicate;
687  for (const long cellId : dumptarget->getCellsIds()) {
688  if (dumptarget->getCells()[cellId].isInterior()){
689  const bitpit::Cell &cell = dumptarget->getCells()[cellId];
690  // Cell data
691  cellBuffer << cell;
692  }
693  }
694 
695  for (int recvRank=0; recvRank<m_nprocs; recvRank++){
696 
697  if (m_rank != recvRank){
698  // Communication
699  MPI_Send(&cellBufferSize, 1, MPI_LONG, recvRank, 200, m_communicator);
700  MPI_Send(cellBuffer.data(), cellBuffer.getSize(), MPI_CHAR, recvRank, 210, m_communicator);
701 
702  }
703  }
704 
705  }
706 
707  }// end external sendrank loop
708  }
709  else
710 #endif
711  {
712  //serial version
713  m_originalDumpingSurface = dumptarget->clone();
714  }

723 template<std::size_t NCOMP>
724 void
725 PropagateField<NCOMP>::computeDumpingFunction(){
726 
727  bitpit::PatchKernel * patch_ = getGeometry()->getPatch();
728  m_dumping.clear();
729  m_dumping.reserve(getGeometry()->getNCells());
730  m_dumping.setGeometry(getGeometry());
731  m_dumping.setDataLocation(MPVLocation::CELL);
732 
733  for (auto it = patch_->cellBegin(); it!=patch_->cellEnd(); ++it){
734  m_dumping.insert(it.getId(), 1.0);
735  }
736 
737  if(!m_originalDumpingSurface.get()){
738  //this unique_ptr structure is initialized by initializeDumpingSurface
739  // if dumping is not active is set to nullptr
740  return;
741  }
742 
743  const double maxd(m_radius);
744 
745  livector1D seedlist;
746  {
747  // Initialize seeds to avoid use of skdtree in narrowband computing
748  std::unordered_set<long> seedtemp;
749  for(auto it= m_dumping.begin(); it!=m_dumping.end(); ++it){
750  long id = it.getId();
751  bitpit::Cell & cell = patch_->getCell(id);
752  if(cell.isInterior()){
753  for (int iface=0; iface<cell.getFaceCount(); iface++){
754  if (cell.isFaceBorder(iface)){
755  seedtemp.insert(id);
756  }
757  }
758  }else{
759  //this cannot happen in serial, only in partitioned grids.
760  seedtemp.insert(id);
761  }
762  }
763  seedlist.insert(seedlist.end(), seedtemp.begin(), seedtemp.end());
764  }
765 
766  bitpit::PiercedVector<double> distFactor = getGeometry()->getCellsNarrowBandToExtSurfaceWDist(*(m_originalDumpingSurface.get()), maxd, &seedlist);
767
