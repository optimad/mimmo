243 template<typename mpv_t>
244 bool
245 MimmoPiercedVector<mpv_t>::checkDataSizeCoherence(){
246  if(getGeometry()==NULL) return false;
247  bool check = false;
248  switch(m_loc){
249  case MPVLocation::CELL:
250  check = (this->size()==m_geometry->getPatch()->getCells().size());
251  break;
252  case MPVLocation::INTERFACE:
253  {
254  size_t sizeInterfaces = m_geometry->getPatch()->getInterfaces().size();
255  check = (this->size()==sizeInterfaces);
256  if(sizeInterfaces == 0){
257  (*m_log)<<"Warning: Asked Data Size Coherence in MimmoPiercedVector for INTERFACES, but linked geometry may not have them built."<<std::endl;
258  }
259  }
260  break;
261  case MPVLocation::POINT:
262  check = (this->size()==m_geometry->getPatch()->getVertices().size());
263  break;
264  default:
265  check=false;
266  (*m_log)<<"NO suitable location data found to perform data size coherence check"<<std::endl;
267  break;
268  }
269  return check;
270 }

281 template<typename mpv_t>
282 bool
283 MimmoPiercedVector<mpv_t>::checkDataIdsCoherence(){
284  if(getGeometry()==NULL) return false;
285  auto ids = this->getIds();
286  bool check = !this->isEmpty();
287  switch(m_loc){
288  case MPVLocation::CELL:
289  {
290  auto vcell = m_geometry->getPatch()->getCells();
291  for(auto el : ids){
292  check =check && vcell.exists(el);
293  }
294  }
295  break;
296  case MPVLocation::INTERFACE:
297  {
298  size_t sizeInterfaces = m_geometry->getPatch()->getInterfaces().size();
299  if(sizeInterfaces == 0){
300  (*m_log)<<"Warning: Asked Data Ids Coherence in MimmoPiercedVector for INTERFACES, but linked geometry may not have them built."<<std::endl;
301  }
302  auto vint = m_geometry->getPatch()->getInterfaces();
303  for(auto el : ids){
304  check =check && vint.exists(el);
305  }
306  }
307  break;
308  case MPVLocation::POINT:
309  {
310  auto vvert = m_geometry->getPatch()->getVertices();
311  for(auto el : ids){
312  check =check && vvert.exists(el);
313  }
314  }
315  break;
316  default:
317  check = false;
318  (*m_log)<<"NO suitable location data found to perform ids coherence check"<<std::endl;
319  break;
320  }
321  return check;
322 }

408 template<typename mpv_t>
409 void
410 MimmoPiercedVector<mpv_t>::initialize(MimmoObject * geo, MPVLocation loc, const mpv_t & data){
411 
412  switch(loc){
413  case MPVLocation::POINT :
414  if(geo->getNVertex() == 0){
415  (*m_log)<<"MimmoPiercedVector warning: initialization failed"<<std::endl;
416  return;
417  }else{
418  this->clear();
419  this->reserve(geo->getNVertex());
420  m_geometry = geo;
421  m_loc = loc;
422  for(const auto & vertex: geo->getVertices()){
423  this->insert(vertex.getId(), data);
424  }
425  }
426  break;
427  case MPVLocation::CELL :
428  if(geo->getNCells() == 0){
429  (*m_log)<<"MimmoPiercedVector warning: initialization failed"<<std::endl;
430  return;
431  }else{
432  this->clear();
433  this->reserve(geo->getNCells());
434  m_geometry = geo;
435  m_loc = loc;
436  for(const auto & cell: geo->getCells()){
437  this->insert(cell.getId(), data);
438  }
439  }
440  break;
441  case MPVLocation::INTERFACE :
442  if(!geo->areInterfacesBuilt()){
443  (*m_log)<<"MimmoPiercedVector warning: initialization failed"<<std::endl;
444  return;
445  }else{
446  this->clear();
447  this->reserve(geo->getPatch()->getInterfaceCount());
448  m_geometry = geo;
449  m_loc = loc;
450  for(const auto & interf: geo->getInterfaces()){
451  this->insert(interf.getId(), data);
452  }
453  }
454  break;
455  default:
456  //do nothing
457  break;
458  }
459 }
