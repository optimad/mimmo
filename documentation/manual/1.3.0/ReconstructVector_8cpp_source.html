193 void
194 ReconstructVector::removeData(MimmoObject * patch){
195  //in m_subpatch remove progressively all pierced vector elements in last position
196  //which links towards a geometry of type patch.
197  while(m_subpatch.back().getGeometry() == patch){
198  m_subpatch.pop_back();
199  }
200 
201  //start searching from the begin on element linking patch
202  std::vector<dmpvecarr3E>::iterator it = m_subpatch.begin();
203 
204  while(it != m_subpatch.end()){
205  if (it->getGeometry() == patch){
206  *it = m_subpatch.back();
207  m_subpatch.pop_back();
208  //be always sure the last element is clean and does not link
209  // the patch
210  while(m_subpatch.back().getGeometry() == patch){
211  m_subpatch.pop_back();
212  }
213  }
214  ++it; //forward the iterator.
215  }

344 void
345 ReconstructVector::execute(){
346 
347  if(getGeometry() == NULL){
348  (*m_log)<<m_name + " : NULL pointer to linked geometry found"<<std::endl;
349  throw std::runtime_error("Error in "+m_name + " : NULL pointer to linked geometry found");
350  }
351 
352  if(getGeometry()->isEmpty()){
353  (*m_log)<<m_name + " : empty linked geometry found"<<std::endl;
354  }
355 
356  //Overlap fields
357  m_result.clear();
358  m_result.setGeometry(getGeometry());
359  m_result.setDataLocation(m_loc);
360 
361  m_subresults.clear();
362 
363  std::unordered_set<long> idsTarget;
364  {
365  livector1D ids = idsGeoDataLocation(getGeometry());
366  idsTarget.insert(ids.begin(), ids.end());
367  }
368 
369  bitpit::PiercedVector<int> counter;
370  for (int i=0; i<getNData(); i++){
371  dmpvecarr3E* pv = &m_subpatch[i];
372  livector1D ids = idsGeoDataLocation(pv->getGeometry());
373  for (auto ID: ids){
374  if(idsTarget.count(ID)==0) continue;
375 
376  if (!m_result.exists(ID)){
377  m_result.insert(ID, (*pv)[ID]);
378  counter.insert(ID, 1);
379  }
380  else{
381  overlapFields(ID, (*pv)[ID]);
382  counter[ID]++;
383  }
384  }
385  }
386 
387  if (m_result.isEmpty()){
388  (*m_log)<<"Warning in "<<m_name<<". Resulting reconstructed field is empty.This is could be caused by unrelated fields linked geometry and target geometry"<<std::endl;
389  }
390 
391  if (m_overlapCriterium == OverlapMethod::AVERAGE){
392  long int ID;
393  auto itend = m_result.end();
394  for (auto it=m_result.begin(); it!=itend; ++it){
395  ID = it.getId();
396  for (int j=0; j<3; j++)
397  (*it)[j] = (*it)[j] / double(counter[ID]);
398  }
399  }
400 
401  //Update field on whole geometry
402  darray3E zero = {{0.0,0.0,0.0}};
403 
404  if(!m_result.completeMissingData(zero)){
405  (*m_log)<<"Warning in "<<m_name<<". It seems a reconstruct field with values uncoherent with target geometry is generated."<<std::endl;
406  }
407 
408  //Create subresults
409  m_subresults.resize(getNData());
410  for (int i=0; i<getNData(); i++){
411  dmpvecarr3E* pv = &m_subpatch[i];
412  m_subresults[i].setGeometry(pv->getGeometry());
413  m_subresults[i].setDataLocation(m_loc);
414  livector1D ids = idsGeoDataLocation(pv->getGeometry());
415  for (auto ID : ids){
416  if(idsTarget.count(ID)>0){
417  m_subresults[i].insert(ID, m_result[ID]);
418  }
419  }
420  }
421 }

443 void
444 ReconstructVector::overlapFields(long int ID, darray3E & locField){
445 
446  switch(m_overlapCriterium){
447  case OverlapMethod::MAX :
448 
449  {
450  double actual = norm2(m_result[ID]);
451  double dummy = norm2(locField);
452 
453  if (actual < dummy)
454  m_result[ID] = locField;
455  }
456  break;
457 
458  case OverlapMethod::MIN :
459  {
460  double actual = norm2(m_result[ID]);
461  double dummy = norm2(locField);
462 
463  if (actual > dummy)
464  m_result[ID] = locField;
465  }
466  break;
467 
468  case OverlapMethod::AVERAGE :
469 
470  m_result[ID] += locField;
471  break;
472 
473  case OverlapMethod::SUM :
474 
475  m_result[ID] += locField;
476  break;
477 
478  default : //never been reached
479  break;
480  }

511 void ReconstructVector::absorbSectionXML(const bitpit::Config::Section & slotXML, std::string name){
512 
513  BITPIT_UNUSED(name);
514 
515  if(slotXML.hasOption("DataLocation")){
516  std::string input = slotXML.get("DataLocation");
517  input = bitpit::utils::string::trim(input);
518  int temp = -1;
519  if(!input.empty()){
520  std::stringstream ss(input);
521  ss>>temp;
522  }
523  if(int(m_loc) != temp){
524  (*m_log)<<"Error absorbing DataLocation in "<<m_name<<". Class and read locations mismatch"<<std::endl;
525  if (temp == 0) (*m_log)<<"XML DataLocation in "<<m_name<<" is set to 0-UNDEFINED"<<std::endl;
526  if (temp == 3) (*m_log)<<"XML DataLocation in "<<m_name<<" is set to 3-INTERFACE, not supported for now."<<std::endl;
527  throw std::runtime_error (m_name + " : xml absorbing failed.");
528  }
529  }
530 
531  //start absorbing
532  BaseManipulation::absorbSectionXML(slotXML, name);
533 
534  if(slotXML.hasOption("OverlapCriterium")){
535  std::string input = slotXML.get("OverlapCriterium");
536  input = bitpit::utils::string::trim(input);
537  int value = 4;
538  if(!input.empty()){
539  std::stringstream ss(input);
540  ss >> value;
541  value = std::min(std::max(1, value),4);
542  }
543  setOverlapCriterium(value);
544  }
