208 void
209 ProjPatchOnSurface::projection(){
210  std::unique_ptr<MimmoObject> dum = m_cobj->clone();
211  //...and projecting them onto target surface
212  if(!getGeometry()->isSkdTreeSync()) getGeometry()->buildSkdTree();
213  bitpit::PiercedVector<bitpit::Vertex> & verts = dum->getVertices();
214  darray3E point, proj;
215  for(auto it = verts.begin(); it != verts.end(); ++it){
216  point = it->getCoords();
217  proj = skdTreeUtils::projectPoint(&point, getGeometry()->getSkdTree());
218  it->setCoords(proj);
219  }
220  m_patch = std::move(dum);
221 };
