46 void example00002() {

48  /* Create IO_CGNS object to import input file. */

51  cgnsI->setDir("geodata");
52  cgnsI->setFilename("grid");
53 
54 #if MIMMO_ENABLE_MPI
55  /* Instantiation of a Partition object with default patition method space filling curve.

58  mimmo::Partition *partition = new mimmo::Partition();
59  partition->setPartitionMethod(mimmo::PartitionMethod::PARTGEOM);
60  partition->setPlotInExecution(true);
61 #endif
62 
63  mimmo::MeshChecker* checkmesh = new mimmo::MeshChecker();
64  checkmesh->setPlotInExecution(true);
65  checkmesh->setMinimumVolumeTolerance(1.0E-1);
66  checkmesh->setMaximumVolumeTolerance(1.0E9);
67  checkmesh->setMaximumSkewnessTolerance(80.0);
68  checkmesh->setMaximumBoundarySkewnessTolerance(90.0);
69  checkmesh->setMinimumFaceValidityTolerance(0.2);
70  checkmesh->setMinimumVolumeChangeTolerance(5.0E-2);
71 
72  /* Create PINs. */
73 #if MIMMO_ENABLE_MPI
74  mimmo::pin::addPin(cgnsI, partition, M_GEOM, M_GEOM) ;
75  mimmo::pin::addPin(cgnsI, partition, M_GEOM2, M_GEOM2) ;
76  mimmo::pin::addPin(partition, checkmesh, M_GEOM, M_GEOM);
77 #else
78  mimmo::pin::addPin(cgnsI, checkmesh, M_GEOM, M_GEOM) ;
79 #endif
80 
81  /* Create and execute chain. */

83  ch0.addObject(cgnsI);
84 #if MIMMO_ENABLE_MPI
85  ch0.addObject(partition);
86 #endif
87  ch0.addObject(checkmesh);
88 
89  ch0.exec(true);
90 
91  /* Destroy objects. */
92  delete cgnsI;
93 #if MIMMO_ENABLE_MPI
94  delete partition;
95 #endif
