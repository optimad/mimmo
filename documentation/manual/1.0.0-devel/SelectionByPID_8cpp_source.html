210 livector1D SelectionByPID::extractSelection(){
211  livector1D result;
212  std::set<long> extraction;
213 
214  syncPIDList();
215  shivector1D pids = getActivePID();
216 
217  for(auto && pid : pids){
218  livector1D local = getGeometry()->extractPIDCells(pid);
219  extraction.insert(local.begin(),local.end());
220  }
221 
222  //check if dual selection is triggered
223 
224  if(m_dual){
225  livector1D totID = getGeometry()->getMapCell();
226  result.resize(totID.size() - extraction.size());
227  if(result.size() == 0) return result;
228 
229  std::sort(totID.begin(), totID.end());
230  int counter = 0;
231  auto tot_it = totID.begin();
232  auto cand_it = extraction.begin();
233  while(tot_it != totID.end()){
234  long val = *tot_it;
235  if (cand_it == extraction.end() || val != *cand_it) {
236  result[counter] = val;
237  ++counter;
238  } else {
239  ++cand_it;
240  }
241  ++tot_it;
242  }
243  }else{
244  result.insert(result.end(), extraction.begin(), extraction.end());
245  }
246  return result;
247 };

254 void SelectionByPID::syncPIDList(){
255  if(getGeometry() == NULL) return;
256 
257  if(m_setPID.count(-1) == 0){
258 
259  std::unordered_set<short>::iterator itU;
260  shivector1D negative;
261  for(itU = m_setPID.begin(); itU != m_setPID.end(); ++itU){
262  short value = *itU;
263  if(m_activePID.count(value) > 0) m_activePID[value] = true;
264  else negative.push_back(value);
265  }
266 
267  for(auto &val: negative) m_setPID.erase(val);
268 
269 
270  }else{
271  m_setPID.clear();
272  for(auto &val: m_activePID ){
273  m_setPID.insert(val.first);
274  val.second = true;
275 
276  }
277  }
278 }

300  BITPIT_UNUSED(name);

317  std::stringstream ss(input);
318  ss >> value;

320  setDual(value);

322 
323  int nPID = 0;
324  if(slotXML.hasOption("nPID")){
325  std::string input = slotXML.get("nPID");
326  input = bitpit::utils::trim(input);
327  nPID = 0;
328  if(!input.empty()){
329  std::stringstream ss(input);

331  std::max(0, nPID);

333  }
334  shivector1D pidlist(nPID, -1);

337  std::string input = slotXML.get("PID");

339  if(!input.empty()){
340  std::stringstream ss(input);
341  for(int i=0; i<nPID; ++i){
342  ss>>pidlist[i];
343  }
344  }
345  }
346 
347  setPID(pidlist);
348 
349 
350  if(slotXML.hasOption("PlotInExecution")){
351  std::string input = slotXML.get("PlotInExecution");
352  input = bitpit::utils::trim(input);
353  bool value = false;
354  if(!input.empty()){
355  std::stringstream ss(input);
356  ss >> value;
357  }
358  setPlotInExecution(value);
359  }
360 
361  if(slotXML.hasOption("OutputPlot")){
362  std::string input = slotXML.get("OutputPlot");
363  input = bitpit::utils::trim(input);
