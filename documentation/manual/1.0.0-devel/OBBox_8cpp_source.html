463 dmatrix33E OBBox::eigenVectors( dmatrix33E & matrix, darray3E & eigenvalues){
464 
465  dmatrix33E result;
466  double * a = new double [9];
467  double * u = new double [9];
468  double * vt = new double [9];
469  double * s = new double[3];
470  double * superb = new double[3];
471 
472  int k=0;
473  for(int i=0; i<3; i++){
474  for(int j=0; j<3; j++){
475  //transpose
476  a[k] = matrix[j][i];
477  k++;
478  }
479  }
480 
481 
482  LAPACKE_dgesvd( LAPACK_COL_MAJOR, 'N','S', 3, 3, a, 3, s, u, 3, vt, 3, superb);
483 
484  //solution norm
485  for (int i=0; i<9; i++){
486  result[i/3][i%3] = vt[i];
487  }
488  for(int i=0; i<2; ++i) {
489  result[i] /= norm2(result[i]);
490  eigenvalues[i] = s[i];
491  }
492  result[2] = crossProduct(result[0],result[1]);
493  eigenvalues[2] = s[2];
494 
495  delete [] a; a = NULL;
496  delete [] u; u = NULL;
497  delete [] vt; vt = NULL;
498  delete [] s; s = NULL;
499  delete [] superb; superb = NULL;
500 
501  return result;
502 }

512 void OBBox::evaluateCovarianceMatrix( dmatrix33E & covariance, darray3E & eta){
513 
514  if(getGeometry()==NULL) {return;}
515 
516  eta.fill(0.0);
517  for(auto & val:covariance) val.fill(0.0);
518  int size, counter;
519  dvector1D area;
520  darray3E temp;
521  double areaTot = 0.0;
522  double maxVal= 1.e-18;
523 
524  if(getGeometry()->getType() == 3){
525  size = getGeometry()->getNVertex();
526  //evaluate eta;
527  for(auto & vert: getGeometry()->getVertices()){
528  eta += vert.getCoords();
529  }
530  eta /= (double)size;
531 
532  //evaluate covariance
533  for(auto & vert: getGeometry()->getVertices()){
534  temp = vert.getCoords() - eta;
535 
536  for(int j=0; j<3; ++j){
537  for(int k=j; k<3; ++k){
538  covariance[j][k] += temp[j]*temp[k];
539  }
540  }
541 
542  for(int j=0; j<3; ++j){
543  for(int k=j; k<3; ++k){
544  covariance[j][k] /= (double) size;
545  maxVal = std::fmax(maxVal, std::abs(covariance[j][k]));
546  }
547  }
548 
549  }
550 
551  }else{
552  size = getGeometry()->getNCells();
553 
554  bitpit::SurfUnstructured * tri = static_cast<bitpit::SurfUnstructured * >(getGeometry()->getPatch());
555  area.resize(size);
556  counter = 0;
557  for(auto &cell : tri->getCells()){
558  area[counter] = tri->evalCellArea(cell.getId());
559  areaTot +=area[counter];
560  ++counter;
561  }
562  for(auto & val: area) val /= areaTot;
563 
564  //evaluate eta;
565  counter = 0;
566  darray3E pp;
567  long vCount;
568  for(auto & cell: tri->getCells()){
569  vCount = cell.getVertexCount();
570  pp.fill(0.0);
571  for(int kk=0; kk<vCount; ++kk){
572  pp += tri->getVertexCoords(cell.getVertex(kk));
573  }
574 
575  eta += pp*area[counter];
576  ++counter;
577  }
578 
579  eta /= (3.0);
580 
581 
582  counter = 0;
583  dvecarr3E p2;
584  for(auto & cell: tri->getCells()){
585  vCount = cell.getVertexCount();
586  p2.resize(vCount);
587  for(int kk=0; kk<vCount; ++kk){
588  p2[kk] = tri->getVertexCoords(cell.getVertex(kk)) - eta;
589  }
590 
591  dmatrix33E tempM = evalCovTriangle(p2);
592  for(int j=0; j<3; ++j){
593  for(int k=j; k<3; ++k){
594  covariance[j][k] += area[counter]*tempM[j][k];
595  }
596  }
597 
598  ++counter;
599  }
600 
601 
602  for(int j=0; j<3; ++j){
603  for(int k=j; k<3; ++k){
604  covariance[j][k] /= (12.0*size);
605  maxVal = std::fmax(maxVal, std::abs(covariance[j][k]));
606  }
607  }
608  }
609 
610  for(int j=0; j<3; ++j){
611  for(int k=j; k<3; ++k){
612  covariance[j][k] /= maxVal;
613  }
614  }
615 
616  covariance[1][0] = covariance[0][1];
617  covariance[2][0] = covariance[0][2];
618  covariance[2][1] = covariance[1][2];
619 
620 };

676 void OBBox::adjustBasis(dmatrix33E & eigVec, darray3E & eigVal){
677 
678  darray3E diff;
679  double tol = 1.0E-3;
680 
681  diff[0] = std::abs(eigVal[1]-eigVal[0]);
682  diff[1] = std::abs(eigVal[2]-eigVal[1]);
683  diff[2] = std::abs(eigVal[0]-eigVal[2]);
684 
685  if(diff[0]>tol && diff[1]>tol && diff[2]>tol) return;
686  if(diff[0]<=tol && diff[1]<=tol && diff[2]<= tol){
687  int counter=0;
688  for(auto & val : eigVec){
689  val.fill(0.0);
690  val[counter] = 1.0;
691  ++counter;
692  }
693  return;
694  }
695 
696  int guess = 0, third =1, stable=2;
697 
698  if(diff[1] <= tol) {
699  guess = 1;
700  third = 2;
701  stable= 0;
702  }
703 
704  if(diff[2] <= tol) {
705  guess = 2;
706  third = 0;
707  stable= 1;
708  }
709 
710  dmatrix33E axes, trasp;
711  int counter=0;
712  for(auto & val: eigVec){
713  axes[counter] = val;
714  ++counter;
715  }
716 
717  darray3E guessVec = axes[guess];
718  darray3E refVec = axes[stable];
719 
720  int nstage = 15;
721  int niterations = 8;
722 
723  double distance = M_PI/(2.0*(nstage));
724  double volume, theta, val;
725  std::map<double, double> mapval;
726  darray3E pmin, pmax, temp;
727 
728  for(int i=0; i<nstage; ++i){
729 
730  theta = i*distance;
731 
732  axes[guess] = guessVec*std::cos(theta) + std::sin(theta)*crossProduct(refVec, guessVec);
733  axes[third] = crossProduct(axes[stable],axes[guess]);
734 
735  pmin.fill(1.e18);
736  pmax.fill(-1.e18);
737 
738  //trasp = bitpit::linearalgebra::transpose(axes);
739 
740  for(auto & vert: getGeometry()->getVertices()){
741  darray3E coord = vert.getCoords();
742  for(int i=0;i<3; ++i){
743  val = dotProduct(coord, axes[i]); //trasp[i]);
744  pmin[i] = std::fmin(pmin[i], val);
745  pmax[i] = std::fmax(pmax[i], val);
746  }
747  }
748 
749  temp = pmax - pmin;
750  volume = temp[0]*temp[1]*temp[2];
751 
752  mapval[volume] = theta;
753  }
754 
755  int it =0;
756  while (it < niterations){
757 
758 
759  distance /= 2.0;
760 
761  double thetadx = mapval.begin()->second + distance;
762  if(thetadx >= M_PI/2.0) thetadx += -1.0*M_PI/2.0;
763 
764  double thetasx = mapval.begin()->second - distance;
765  if(thetasx <= 0.0) thetasx += M_PI/2.0;
766 
767  //evaluate on the right
768  axes[guess] = guessVec*std::cos(thetadx) + std::sin(thetadx)*crossProduct(refVec, guessVec);
769  axes[third] = crossProduct(axes[stable],axes[guess]);
770 
771 
772  //trasp = bitpit::linearalgebra::transpose(axes);
773  pmin.fill(1.e18);
774  pmax.fill(-1.e18);
775  for(auto & vert: getGeometry()->getVertices()){
776  darray3E coord = vert.getCoords();
777  for(int i=0;i<3; ++i){
778  val = dotProduct(coord,axes[i]); // trasp[i]);
779  pmin[i] = std::fmin(pmin[i], val);
780  pmax[i] = std::fmax(pmax[i], val);
781  }
782  }
783 
784  temp = pmax - pmin;
785  volume = temp[0]*temp[1]*temp[2];
786 
787  mapval[volume] = thetadx;
788 
789  //evaluate on the left
790  axes[guess] = guessVec*std::cos(thetasx) + std::sin(thetasx)*crossProduct(refVec, guessVec);
791  axes[third] = crossProduct(axes[stable],axes[guess]);
792 
793  trasp = bitpit::linearalgebra::transpose(axes);
794  pmin.fill(1.e18);
795  pmax.fill(-1.e18);
796  for(auto & vert: getGeometry()->getVertices()){
797  darray3E coord = vert.getCoords();
798  for(int i=0;i<3; ++i){
799  val = dotProduct(coord,axes[i]); // trasp[i]);
800  pmin[i] = std::fmin(pmin[i], val);
801  pmax[i] = std::fmax(pmax[i], val);
802  }
803  }
804 
805  temp = pmax - pmin;
806  volume = temp[0]*temp[1]*temp[2];
807 
808  mapval[volume] = thetasx;
809  ++it;
810  }
811 
812  theta = mapval.begin()->second;
813  eigVec[guess] = guessVec*std::cos(theta) + std::sin(theta)*crossProduct(refVec, guessVec);
814  eigVec[third] = crossProduct(eigVec[stable],eigVec[guess]);
815
