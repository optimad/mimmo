353 dmatrix33E OBBox::eigenVectors( dmatrix33E & matrix, darray3E & eigenvalues){
354 
355  dmatrix33E result;
356  double * a = new double [9];
357  double * u = new double [9];
358  double * vt = new double [9];
359  double * s = new double[3];
360  double * superb = new double[3];
361  int info;
362 
363  int k=0;
364  for(int i=0; i<3; i++){
365  for(int j=0; j<3; j++){
366  //transpose
367  a[k] = matrix[j][i];
368  k++;
369  }
370  }
371 
372 
373  info = LAPACKE_dgesvd( LAPACK_COL_MAJOR, 'N','S', 3, 3, a, 3, s, u, 3, vt, 3, superb);
374 
375  //solution norm
376  for (int i=0; i<9; i++){
377  result[i/3][i%3] = vt[i];
378  }
379  for(int i=0; i<2; ++i) {
380  result[i] /= norm2(result[i]);
381  eigenvalues[i] = s[i];
382  }
383  result[2] = crossProduct(result[0],result[1]);
384  eigenvalues[2] = s[2];
385 
386  delete [] a; a = NULL;
387  delete [] u; u = NULL;
388  delete [] vt; vt = NULL;
389  delete [] s; s = NULL;
390  delete [] superb; superb = NULL;
391 
392  return result;
393 }

403 void OBBox::evaluateCovarianceMatrix( dmatrix33E & covariance, darray3E & eta){
404 
405  if(getGeometry()==NULL) {return;}
406 
407  eta.fill(0.0);
408  for(auto & val:covariance) val.fill(0.0);
409  int size, counter;
410  dvector1D area;
411  darray3E temp;
412  double areaTot = 0.0;
413  double maxVal= 1.e-18;
414 
415  if(getGeometry()->getType() == 3){
416  size = getGeometry()->getNVertex();
417  //evaluate eta;
418  for(auto & vert: getGeometry()->getVertices()){
419  eta += vert.getCoords();
420  }
421  eta /= (double)size;
422 
423  //evaluate covariance
424  for(auto & vert: getGeometry()->getVertices()){
425  temp = vert.getCoords() - eta;
426 
427  for(int j=0; j<3; ++j){
428  for(int k=j; k<3; ++k){
429  covariance[j][k] += temp[j]*temp[k];
430  }
431  }
432 
433  for(int j=0; j<3; ++j){
434  for(int k=j; k<3; ++k){
435  covariance[j][k] /= (double) size;
436  maxVal = std::fmax(maxVal, std::abs(covariance[j][k]));
437  }
438  }
439 
440  }
441 
442  }else{
443  size = getGeometry()->getNCells();
444 
445  bitpit::SurfUnstructured * tri = static_cast<bitpit::SurfUnstructured * >(getGeometry()->getPatch());
446  area.resize(size);
447  counter = 0;
448  for(auto &cell : tri->getCells()){
449  area[counter] = tri->evalCellArea(cell.getId());
450  areaTot +=area[counter];
451  ++counter;
452  }
453  for(auto & val: area) val /= areaTot;
454 
455  //evaluate eta;
456  counter = 0;
457  darray3E pp;
458  long vCount;
459  for(auto & cell: tri->getCells()){
460  vCount = cell.getVertexCount();
461  pp.fill(0.0);
462  for(int kk=0; kk<vCount; ++kk){
463  pp += tri->getVertexCoords(cell.getVertex(kk));
464  }
465 
466  eta += pp*area[counter];
467  ++counter;
468  }
469 
470  eta /= (3.0);
471 
472 
473  counter = 0;
474  dvecarr3E p2;
475  long id;
476  for(auto & cell: tri->getCells()){
477  vCount = cell.getVertexCount();
478  p2.resize(vCount);
479  for(int kk=0; kk<vCount; ++kk){
480  p2[kk] = tri->getVertexCoords(cell.getVertex(kk)) - eta;
481  }
482 
483  dmatrix33E tempM = evalCovTriangle(p2);
484  for(int j=0; j<3; ++j){
485  for(int k=j; k<3; ++k){
486  covariance[j][k] += area[counter]*tempM[j][k];
487  }
488  }
489 
490  ++counter;
491  }
492 
493 
494  for(int j=0; j<3; ++j){
495  for(int k=j; k<3; ++k){
496  covariance[j][k] /= (12.0*size);
497  maxVal = std::fmax(maxVal, std::abs(covariance[j][k]));
498  }
499  }
500  }
501 
502  for(int j=0; j<3; ++j){
503  for(int k=j; k<3; ++k){
504  covariance[j][k] /= maxVal;
505  }
506  }
507 
508  covariance[1][0] = covariance[0][1];
509  covariance[2][0] = covariance[0][2];
510  covariance[2][1] = covariance[1][2];
511 
512 };

568 void OBBox::adjustBasis(dmatrix33E & eigVec, darray3E & eigVal){
569 
570  darray3E diff;
571  double tol = 1.0E-3;
572 
573  diff[0] = std::abs(eigVal[1]-eigVal[0]);
574  diff[1] = std::abs(eigVal[2]-eigVal[1]);
575  diff[2] = std::abs(eigVal[0]-eigVal[2]);
576 
577  if(diff[0]>tol && diff[1]>tol && diff[2]>tol) return;
578  if(diff[0]<=tol && diff[1]<=tol && diff[2]<= tol){
579  int counter=0;
580  for(auto & val : eigVec){
581  val.fill(0.0);
582  val[counter] = 1.0;
583  ++counter;
584  }
585  return;
586  }
587 
588  int guess = 0, third =1, stable=2;
589 
590  if(diff[1] <= tol) {
591  guess = 1;
592  third = 2;
593  stable= 0;
594  }
595 
596  if(diff[2] <= tol) {
597  guess = 2;
598  third = 0;
599  stable= 1;
600  }
601 
602  dmatrix33E axes, trasp;
603  int counter=0;
604  for(auto & val: eigVec){
605  axes[counter] = val;
606  ++counter;
607  }
608 
609  darray3E guessVec = axes[guess];
610  darray3E refVec = axes[stable];
611 
612  int nstage = 15;
613  int niterations = 8;
614 
615  double distance = M_PI/(2.0*(nstage));
616  double volume, theta, val;
617  std::map<double, double> mapval;
618  darray3E pmin, pmax, temp;
619 
620  for(int i=0; i<nstage; ++i){
621 
622  theta = i*distance;
623 
624  axes[guess] = guessVec*std::cos(theta) + std::sin(theta)*crossProduct(refVec, guessVec);
625  axes[third] = crossProduct(axes[stable],axes[guess]);
626 
627  pmin.fill(1.e18);
628  pmax.fill(-1.e18);
629 
630  //trasp = bitpit::linearalgebra::transpose(axes);
631 
632  for(auto & vert: getGeometry()->getVertices()){
633  darray3E coord = vert.getCoords();
634  for(int i=0;i<3; ++i){
635  val = dotProduct(coord, axes[i]); //trasp[i]);
636  pmin[i] = std::fmin(pmin[i], val);
637  pmax[i] = std::fmax(pmax[i], val);
638  }
639  }
640 
641  temp = pmax - pmin;
642  volume = temp[0]*temp[1]*temp[2];
643 
644  mapval[volume] = theta;
645  }
646 
647  int it =0;
648  while (it < niterations){
649 
650 
651  distance /= 2.0;
652 
653  double thetadx = mapval.begin()->second + distance;
654  if(thetadx >= M_PI/2.0) thetadx += -1.0*M_PI/2.0;
655 
656  double thetasx = mapval.begin()->second - distance;
657  if(thetasx <= 0.0) thetasx += M_PI/2.0;
658 
659  //evaluate on the right
660  axes[guess] = guessVec*std::cos(thetadx) + std::sin(thetadx)*crossProduct(refVec, guessVec);
661  axes[third] = crossProduct(axes[stable],axes[guess]);
662 
663 
664  //trasp = bitpit::linearalgebra::transpose(axes);
665  pmin.fill(1.e18);
666  pmax.fill(-1.e18);
667  for(auto & vert: getGeometry()->getVertices()){
668  darray3E coord = vert.getCoords();
669  for(int i=0;i<3; ++i){
670  val = dotProduct(coord,axes[i]); // trasp[i]);
671  pmin[i] = std::fmin(pmin[i], val);
672  pmax[i] = std::fmax(pmax[i], val);
673  }
674  }
675 
676  temp = pmax - pmin;
677  volume = temp[0]*temp[1]*temp[2];
678 
679  mapval[volume] = thetadx;
680 
681  //evaluate on the left
682  axes[guess] = guessVec*std::cos(thetasx) + std::sin(thetasx)*crossProduct(refVec, guessVec);
683  axes[third] = crossProduct(axes[stable],axes[guess]);
684 
685  trasp = bitpit::linearalgebra::transpose(axes);
686  pmin.fill(1.e18);
687  pmax.fill(-1.e18);
688  for(auto & vert: getGeometry()->getVertices()){
689  darray3E coord = vert.getCoords();
690  for(int i=0;i<3; ++i){
691  val = dotProduct(coord,axes[i]); // trasp[i]);
692  pmin[i] = std::fmin(pmin[i], val);
693  pmax[i] = std::fmax(pmax[i], val);
694  }
695  }
696 
697  temp = pmax - pmin;
698  volume = temp[0]*temp[1]*temp[2];
699 
700  mapval[volume] = thetasx;
701  ++it;
702  }
703 
704  theta = mapval.begin()->second;
705  eigVec[guess] = guessVec*std::cos(theta) + std::sin(theta)*crossProduct(refVec, guessVec);
706  eigVec[third] = crossProduct(eigVec[stable],eigVec[guess]);
707 
708 
709 
710  return;
