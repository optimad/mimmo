280 bool MRBFStyleObj::removeNode(std::vector<int> & list){
281 
282  std::set<int> setList;

284 
285  int extracted = 0;

287  if(id>=0 && id <m_nodes){;
288  m_nodes--;
289  int index = id-extracted;
290  auto key = *(m_node.begin() + index);
291  m_node.erase(m_node.begin() + index);

293  m_bboxes.erase(key);
294  extracted++;
295  }
296  }
297  return(extracted == (int)(list.size()));

461 void MRBFStyleObj::execute(){
462 
463  MimmoObject * container = getGeometry();
464  if(container->isEmpty() ) return;
465 
466  int size, sizeF = getDataCount();
467  for (int i=0; i<sizeF; i++){
468 
469  if(m_solver == MRBFSol::NONE) size = m_weight[i].size();
470  else size = m_value[i].size();
471 
472  if(size != getTotalNodesCount()){
473  std::cout << "MiMMO : WARNING : " << getName() << " has displacements of " << i << " field with size (" << size << ") that does not fit number of RBF nodes ("<< getTotalNodesCount() << ")" << std::endl;
474  fitDataToNodes(i);
475  }
476  }
477 
478  double bboxDiag;
479  {
480  darray3E pmin, pmax;
481  container->getPatch()->getBoundingBox(pmin, pmax);
482  bboxDiag= norm2(pmax - pmin);
483  }
484 
485  //Checking supportRadius.
486  double distance = 0.0;
487  if(m_SRRatio <=0.0){ //get maximum weight/value displ and assign support radius a 3 times this value.
488 
489  for(int i=0; i<size; ++i){
490 
491  dvector1D data(sizeF);
492 
493  for(int j=0; j<sizeF; ++j){
494  if(m_solver == MRBFSol::NONE) data[j] = m_weight[j][i];
495  else data[j] = m_value[j][i];
496  }
497 
498  distance = std::max(distance, norm2(data));
499  }
500 
501  distance *=3.0;
502 
503  }else{
504  if (m_supRIsValue){
505  distance = m_SRRatio;
506  }
507  else{
508  distance = m_SRRatio * bboxDiag;
509  }
510  }
511 
512  //TODO remove it (I can't use a support radius 0....why?)
513  if(distance <=1.E-18){ //checkSupportRadius if too small, set it to the semidiagonal value of the geometry AABB
514  distance = 0.5*bboxDiag;
515  }
516 
517  const double radius = distance;
518  RBFKernel::setSupportRadius(radius);
519 
520  if (m_solver == MRBFSol::WHOLE) solve();
521  if (m_solver == MRBFSol::GREEDY) greedy(m_tol);
522 
523  int nv = container->getNVertex();
524  dvecarr3E vertex = container->getVertexCoords();
525 
526  m_displ.resize(nv, darray3E{0,0,0});
527  dvector1D displ;
528  for(int i=0; i<nv; ++i){
529  displ = evalRBF(vertex[i]);
530  for (int j=0; j<3; j++) m_displ[i][j] = displ[j];
531  }
532 
533  //if m_filter is active;
534  if(m_bfilter){
535  m_filter.resize(nv,1.0);
536  int counter = 0;
537  for (auto && vec : m_displ){
538  vec = vec * m_filter[counter];
539  ++counter;
540  }
541  }
542 
543  return;
544 };

645  double MRBFStyleObj::calcDist(int i, int j){
646 
647  double distMin = 1.0E+18;
648  if(!m_node[i]->isBvTreeBuilt()) m_node[i]->buildBvTree();
649  if(!m_node[j]->isBvTreeBuilt()) m_node[j]->buildBvTree();
650 
651  //find an indicative distance of search between geometries
652  double initRadius, workRadius;
653  double dist = 1.E+18;
654  long id;
655 
656  {
657  darray3E cA,cB;
658  cA = 0.5*(m_bboxes[m_node[i]][0]+m_bboxes[m_node[i]][1]);
659  cB = 0.5*(m_bboxes[m_node[j]][0]+m_bboxes[m_node[j]][1]);
660  initRadius = norm2(cB - cA);
661  }
662 
663  //search first set of points against second figure
664  for(auto & val : m_node[i]->getVertexCoords()){
665  workRadius= initRadius;
666  dist = bvTreeUtils::distance(&val, m_node[j]->getBvTree(), id, workRadius);
667  if(dist < distMin) distMin = dist;
668  }
669 
670  //search second set of points against first figure
671  for(auto & val : m_node[j]->getVertexCoords()){
672  workRadius= initRadius;
673  dist = bvTreeUtils::distance(&val, m_node[i]->getBvTree(), id, workRadius);
674  if(dist < distMin) distMin = dist;
675  }
676 
677  return distMin;
678 
679 }

684 double MRBFStyleObj::calcDist(const darray3E & point, int j){
685 
686  double dist = 1.0E+18;
687  double rate = 0.02;
688  int kmax = 1000;
689  int kiter = 0;
690  bool flag = true;
691  long id;
692  auto pp = point;
693 
694  if(!m_node[j]->isBvTreeBuilt()) m_node[j]->buildBvTree();
695  double initRadius = norm2(m_bboxes[m_node[j]][1] - m_bboxes[m_node[j]][0]);
696 
697  while(flag && kiter < kmax){
698  dist = bvTreeUtils::distance(&pp, m_node[j]->getBvTree(), id, initRadius);
699  flag = (dist == 1.0E+18);
700  if(flag) initRadius *= (1.0+ rate*((double)flag));
701  kiter++;
702  }
703 
704  return dist;
705 }
