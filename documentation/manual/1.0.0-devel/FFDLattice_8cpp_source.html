33 /*
34  * \date 09/feb/2016
35  * \authors Rocco Arpa
36  * \authors Edoardo Lombardi
37  *
38  * \brief Free Form Deformation of a 3D surface and point clouds, with structured lattice.
39  *
40  * Free Form deformation tool for 3D geometries (surface and point clouds). Basically, it builds an elemental 3D shape
41  * (box, sphere, cylinder or part of them) around the geometry and set a structured cartesian mesh of control
42  * points on it (lattice). Displacements of each control point is linked to the geometry inside
43  * the shape by means of a NURBS volumetric parameterization. Deformation will be applied only to
44  * those portion of geometry encased into the 3D shape.
45  *
46  */

664 darray3E FFDLattice::nurbsEvaluator(darray3E & pointOr){
665 
666  darray3E point = transfToLocal(pointOr);
667  darray3E scaling = getShape()->getScaling();
668 
669  ivector1D knotInterval(3,0);
670  dvector2D BSbasis(3);
671  dvector1D valH(4,0), temp1(4,0),temp2(4,0), zeros(4,0);
672  int uind, vind, wind, index;
673 
674  dvecarr3E* displ = getDisplacements();
675 
676  int i0 = m_mapdeg[0];
677  int i1 = m_mapdeg[1];
678  int i2 = m_mapdeg[2];
679 
680  iarray3E mappedIndex;
681 
682  for(int i=0; i<3; i++){
683  knotInterval[i] = getKnotInterval(point[i],i);
684  BSbasis[i] = basisITS0(knotInterval[i], i, point[i]);
685  }
686 
687  uind = knotInterval[i0] - m_deg[i0];
688  vind = knotInterval[i1] - m_deg[i1];
689  wind = knotInterval[i2] - m_deg[i2];
690 
691  for(int i=0; i<=m_deg[i0]; ++i){
692 
693  mappedIndex[0] = uind+i;
694  temp1 = zeros;
695 
696  for(int j=0; j<=m_deg[i1]; ++j){
697 
698  mappedIndex[1] = vind+j;
699  temp2 = zeros;
700 
701  for(int k=0; k<=m_deg[i2]; ++k){
702 
703  mappedIndex[2] = wind+k;
704  index = accessMapNodes(mappedIndex[0],mappedIndex[1],mappedIndex[2]);
705 
706  for(int intv=0; intv<3; ++intv){
707  temp2[intv] += BSbasis[i2][k]*m_weights[m_intMapDOF[index]]*(*displ)[m_intMapDOF[index]][intv];
708  }
709  temp2[3] += BSbasis[i2][k]*m_weights[index];
710  }
711  for(int intv=0; intv<4; ++intv){
712  temp1[intv] += BSbasis[i1][j]*temp2[intv];
713  }
714 
715  }
716  for(int intv=0; intv<4; ++intv){
717  valH[intv] += BSbasis[i0][i]*temp1[intv];
718  }
719  }
720 
721  darray3E outres;
722  if(isDisplGlobal()){
723  for(int i=0; i<3; ++i){
724  outres[i] = valH[i]/valH[3];
725  }
726 
727  }else{
728  //summing scaled displ in local ref frame;
729  for(int i=0; i<3; ++i){
730  point[i] += valH[i]/(valH[3]*scaling[i]);
731  }
732 
733  //get final displ in global ref frame:
734  outres = transfToGlobal(point) - pointOr;
735  }
736  return(outres);
737 
738 };
