33 /*
34  * \date 09/feb/2016
35  * \authors Rocco Arpa
36  * \authors Edoardo Lombardi
37  *
38  * \brief Free Form Deformation of a 3D surface and point clouds, with structured lattice.
39  *
40  * Free Form deformation tool for 3D geometries (surface and point clouds). Basically, it builds an elemental 3D shape
41  * (box, sphere, cylinder or part of them) around the geometry and set a structured cartesian mesh of control
42  * points on it (lattice). Displacements of each control point is linked to the geometry inside
43  * the shape by means of a NURBS volumetric parameterization. Deformation will be applied only to
44  * those portion of geometry encased into the 3D shape.
45  *
46  */

295 void FFDLattice::setLattice(darray3E &origin,darray3E & span, ShapeType type, dvector1D & spacing, iarray3E & degrees ){
296 
297  ivector1D dimLimit(3,2);
298  //create internal shape using unique_ptr member.
299  if(m_shape){m_shape.reset(nullptr);}
300 
301  switch(type){
302  case ShapeType::CYLINDER :
303  dimLimit[1] = 5;
304  break;
305  case ShapeType::SPHERE :
306  dimLimit[1] = 5; dimLimit[2] = 3;
307  break;

318

322  }else{
323  dim[i] = dimLimit[i];
324  }
325  }

509 void FFDLattice::execute(){
510 
511  if(!isBuilt()){build();}
512 
513  MimmoObject * container = getGeometry();
514  if(container == NULL ) return;
515 
516  //build trees
517  if(container->isBvTreeSupported() && !container->isBvTreeBuilt()) container->buildBvTree();
518  else if(!container->isKdTreeBuilt()) container->buildKdTree();
519 
520  livector1D map;
521  dvecarr3E localdef = apply(map);
522 
523  //reset displacement in a unique vector
524  int size = container->getNVertex();
525  m_gdispl.resize(size, darray3E{0,0,0});
526  for(int i=0; i<map.size(); ++i){
527  m_gdispl[container->getMapDataInv(map[i])] = localdef[i];
528  }

550 dvecarr3E FFDLattice::apply(livector1D & list){
551 
552  MimmoObject * container = getGeometry();
553  if(container == NULL || !isBuilt()) return dvecarr3E(0);
554 
555  list.clear();
556 
557  //check simplex included and extract their vertex in global IDs;
558  if(container->isBvTreeSupported()) list= container->getVertexFromCellList(getShape()->includeGeometry(container));
559  else list= getShape()->includeCloudPoints(container);
560  //return deformation
561  dvecarr3E result = nurbsEvaluator(list);
562  if(m_bfilter){
563 
564  m_filter.resize(container->getNVertex(),0.0);
565 
566  dvecarr3E::iterator itL= result.begin();
567  liimap & vMap = container->getMapDataInv();
568  for ( auto && vID : list){
569  *itL = (*itL) * m_filter[vMap[vID]];
570  ++itL;
571  }
572  }
573  return(result);
574 };

603 darray3E FFDLattice::convertDisplToXYZ(darray3E & target, int i){
604 
605  darray3E scaling = getScaling();
606  darray3E work;
607  for(int i=0; i<3; ++i){
608  work[i] = target[i]/scaling[i];

634 darray3E FFDLattice::nurbsEvaluator(darray3E & pointOr){
635 
636  darray3E point = transfToLocal(pointOr);
637  darray3E scaling = getShape()->getScaling();
638 
639  ivector1D knotInterval(3,0);
640  dvector2D BSbasis(3);
641  dvector1D valH(4,0), temp1(4,0),temp2(4,0), zeros(4,0);
642  int uind, vind, wind, index;
643 
644  dvecarr3E* displ = getDisplacements();
645 
646  int i0 = m_mapdeg[0];
647  int i1 = m_mapdeg[1];
648  int i2 = m_mapdeg[2];
649 
650  iarray3E mappedIndex;
651 
652  for(int i=0; i<3; i++){
653  knotInterval[i] = getKnotInterval(point[i],i);
654  BSbasis[i] = basisITS0(knotInterval[i], i, point[i]);
655  }
656 
657  uind = knotInterval[i0] - m_deg[i0];
658  vind = knotInterval[i1] - m_deg[i1];
659  wind = knotInterval[i2] - m_deg[i2];
660 
661  for(int i=0; i<=m_deg[i0]; ++i){
662 
663  mappedIndex[0] = uind+i;
664  temp1 = zeros;
665 
666  for(int j=0; j<=m_deg[i1]; ++j){
667 
668  mappedIndex[1] = vind+j;
669  temp2 = zeros;
670 
671  for(int k=0; k<=m_deg[i2]; ++k){
672 
673  mappedIndex[2] = wind+k;
674  index = accessMapNodes(mappedIndex[0],mappedIndex[1],mappedIndex[2]);
675 
676  for(int intv=0; intv<3; ++intv){
677  temp2[intv] += BSbasis[i2][k]*m_weights[m_intMapDOF[index]]*(*displ)[m_intMapDOF[index]][intv];
678  }
679  temp2[3] += BSbasis[i2][k]*m_weights[index];
680  }
681  for(int intv=0; intv<4; ++intv){
682  temp1[intv] += BSbasis[i1][j]*temp2[intv];
683  }
684 
685  }
686  for(int intv=0; intv<4; ++intv){
687  valH[intv] += BSbasis[i0][i]*temp1[intv];
688  }
689  }
690 
691  darray3E outres;
692  if(isDisplGlobal()){
693  for(int i=0; i<3; ++i){
694  outres[i] = valH[i]/valH[3];
695  }
696 
697  }else{
698  //summing scaled displ in local ref frame;
699  for(int i=0; i<3; ++i){
700  point[i] += valH[i]/(valH[3]*scaling[i]);
701  }
702 
703  //get final displ in global ref frame:
704  outres = transfToGlobal(point) - pointOr;
705  }
706  return(outres);
707 
708 };

715 dvecarr3E FFDLattice::nurbsEvaluator(livector1D & list){
716 
717  bitpit::PatchKernel * tri = getGeometry()->getPatch();
718  long int id;
719  int lsize = list.size();
720  livector1D::iterator it, itend = list.end();
721  darray3E target, point;
722  ivector1D knotInterval(3,0);
723  dvector1D BSbasisi0, BSbasisi1, BSbasisi2;
724  dvector1D valH(4,0), temp1(4,0),temp2(4,0);
725 
726  dvecarr3E displ = recoverFullGridDispl();
727  dvector1D weig = recoverFullNodeWeights();
728 
729  int uind, vind, wind, index;
730 
731  int intv, i, j, k;
732 
733  int i0 = m_mapdeg[0];
734  int i1 = m_mapdeg[1];
735  int i2 = m_mapdeg[2];
736  iarray3E mappedIndex;
737 
738  int md0 = m_deg[i0];
739  int md1 = m_deg[i1];
740  int md2 = m_deg[i2];
741 
742  dvecarr3E outres(lsize);
743  dvecarr3E::iterator itout = outres.begin();
744  darray3E scaling = getShape()->getScaling();
745 
746  double bbasisw2;
747  double bbasis1, bbasis0;
748 
749  for(it = list.begin(); it != itend; ++it){
750 
751  for(intv=0; intv<4; ++intv){
752  valH[intv] = 0.0;
753  }
754  id = *it;
755  target = tri->getVertex(id).getCoords();
756  point = transfToLocal(target);
757 
758  // get reference Interval int the knot matrix
759  for(i=0; i<3; i++){
760  knotInterval[i] = getKnotInterval(point[i],i);
761  }
762  BSbasisi0 = basisITS0(knotInterval[i0], i0, point[i0]);
763  BSbasisi1 = basisITS0(knotInterval[i1], i1, point[i1]);
764  BSbasisi2 = basisITS0(knotInterval[i2], i2, point[i2]);
765 
766  uind = knotInterval[i0] - md0;
767  vind = knotInterval[i1] - md1;
768  wind = knotInterval[i2] - md2;
769 
770 
771  for(i=0; i<=md0; ++i){
772 
773  mappedIndex[i0] = uind + i;
774 
775  for(int intv=0; intv<4; ++intv){
776  temp1[intv] = 0.0;
777  }
778 
779  for(j=0; j<=md1; ++j){
780 
781  mappedIndex[i1] = vind + j;
782 
783  for(intv=0; intv<4; ++intv){
784  temp2[intv] = 0.0;
785  }
786 
787  for(k=0; k<=md2; ++k){
788 
789  mappedIndex[i2] = wind + k;
790 
791  index = accessMapNodes(mappedIndex[0], mappedIndex[1], mappedIndex[2]);
792 
793  bbasisw2 = BSbasisi2[k]* weig[index];
794 
795  for(intv=0; intv<3; ++intv){
796  temp2[intv] += bbasisw2 * displ[index][intv];
797  }
798  temp2[3] += bbasisw2;
799 
800  }
801  bbasis1 = BSbasisi1[j];
802  for(intv=0; intv<4; ++intv){
803  temp1[intv] += bbasis1*temp2[intv];
804  }
805 
806  }
807  bbasis0 = BSbasisi0[i];
808  for(intv=0; intv<4; ++intv){
809  valH[intv] += bbasis0*temp1[intv];
810  }
811  }
812 
813 
814  if(isDisplGlobal()){
815 
816  //adding to local point displ rescaled
817  for(i=0; i<3; ++i){
818  (*itout)[i] = valH[i]/valH[3];
819  }
820 
821  }else{
822 
823  //adding to local point displ rescaled
