142 int
143 Chain::addObject(BaseManipulation* obj, int id_){
144  int id = id_;
145  if (id_ < 0){
146  id = m_objcounter;
147  m_objcounter++;
148  }
149  if (m_objects.size() == 0){
150  m_objects.push_back(obj);
151  m_idObjects.push_back(id);
152  return 0;
153  }
154  vector<BaseManipulation*>::iterator itchild = m_objects.end();
155  vector<BaseManipulation*>::iterator itparent;
156  int idxchild = m_objects.size();
157  ivector1D idxparent(obj->getNParent(), -1);
158  if (obj->getNChild()>0){
159  for (int i=0; i<obj->getNChild(); i++){
160  itchild = find(m_objects.begin(), m_objects.end(), obj->getChild(i));
161  if (itchild != m_objects.end()){
162  idxchild = min(idxchild, int(distance(m_objects.begin(), itchild)));
163  }
164  }
165  }
166  if (obj->getNParent()>0){
167  for (int i=0; i<obj->getNParent(); i++){
168  if (obj->getParent(i) != NULL){
169  itparent = find(m_objects.begin(), m_objects.end(), obj->getParent(i));
170  if (itparent != m_objects.end()){
171  idxparent[i] = distance(m_objects.begin(), itparent);
172  }
173  }
174  }
175  }
176  for (int i=0; i<obj->getNParent(); i++){
177  if (idxparent[i] == idxchild) return -1;
178  }
179 
180  bool parentsMinor = true;
181  for (int i=0; i<obj->getNParent(); i++){
182  if (idxparent[i] > idxchild){
183  parentsMinor = false;
184  }
185  }
186  if (!parentsMinor){
187  ivector1D idparent(obj->getNParent(), -1);
188  vector<BaseManipulation*> parent(obj->getNParent());
189  for (int i=0; i<obj->getNParent(); i++){
190  if (idxparent[i] > idxchild){
191  parent[i] = obj->getParent(i);
192  itparent = find(m_objects.begin(), m_objects.end(), parent[i]);
193  int idxtemp = distance(m_objects.begin(), itparent);
194  idparent[i] = m_idObjects[idxtemp];
195  m_objects.erase(itparent);
196  m_idObjects.erase(m_idObjects.begin()+idxtemp);
197  }
198  }
199  int idx = addObject(obj, id);
200  for (int i=0; i<obj->getNParent(); i++){
201  if (idxparent[i] > idxchild){
202  addObject(parent[i], idparent[i]);
203  parent[i] = NULL;
204  }
205  }
206  return idx+1;
207  }else{
208  m_objects.insert(m_objects.begin()+idxchild, obj);
209  m_idObjects.insert(m_idObjects.begin()+idxchild, id);
210  return idxchild;
211  }
212  return 0;
213 }
