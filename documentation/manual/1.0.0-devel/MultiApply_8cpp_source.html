118 void
119 MultiApply::execute(){
120 
121 
122  if (m_input.empty()) return;
123 
124 
125  for(auto val: m_input){
126 
127 
128 
129  dvecarr3E vertex = val.first->getVertexCoords();
130  long nv = val.first->getNVertex();
131  nv = long(std::min(int(nv), int(val.second->size())));
132 
133  livector1D & idmap = val.first->getMapData();
134  for (long i=0; i<nv; i++){
135 
136  vertex[i] += (*(val.second))[i];
137  val.first->modifyVertex(vertex[i], idmap[i]);
138  }
139 
140  if(m_force){
141  val.first->buildBvTree();
142  val.first->buildKdTree();
143  }
144  }
145  return;
146 };
