181 void
182 Mask::execute(){
183  int ndispl = m_displ.size();
184  ndispl = std::min(ndispl, int(m_coords.size()));
185  for (int i=0; i<ndispl; i++){
186  if (m_coords[i][0]>m_thres[0][0] && m_coords[i][0]<m_thres[0][1] &&
187  m_coords[i][1]>m_thres[1][0] && m_coords[i][1]<m_thres[1][1] &&
188  m_coords[i][2]>m_thres[2][0] && m_coords[i][2]<m_thres[2][1]){
189  for (int j=0; j<3; j++){
190  m_displ[i][j] = (1-m_inside[j])*m_displ[i][j];
191  }
192  }
193  else{
194  for (int j=0; j<3; j++){
195  m_displ[i][j] = (m_inside[j])*m_displ[i][j];
196  }
197  }
198  }
199  return;
200 };
