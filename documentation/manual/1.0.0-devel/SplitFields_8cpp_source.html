260 bitpit::VTKElementType
261 SplitField::desumeElement(ivector2D & conn){
262  bitpit::VTKElementType result = bitpit::VTKElementType::UNDEFINED;
263  if(conn.empty() && m_topo != 3) return result;
264 
265  switch(m_topo){
266  case 1:
267  if(conn[0].size() == 3) result = bitpit::VTKElementType::TRIANGLE;
268  if(conn[0].size() == 4) result = bitpit::VTKElementType::QUAD;
269  break;
270  case 2:
271  if(conn[0].size() == 4) result = bitpit::VTKElementType::TETRA;
272  if(conn[0].size() == 8) result = bitpit::VTKElementType::HEXAHEDRON;
273  break;
274  case 3:
275  result= bitpit::VTKElementType::VERTEX;
276  break;
277  default : //never been reached
278  break;
279  }
280 
281  return result;
282 };

365  if(isEmpty()) return;

367 
368  bitpit::VTKLocation loc = bitpit::VTKLocation::POINT;
369  if(m_mapVertDivision.empty()) loc = bitpit::VTKLocation::CELL;
370 
371  int counter=0;
372  for(auto & geo : m_originals){
373  dvecarr3E points = geo->getVertexCoords();
374 
375  if(getTopo() != 3){
376  ivector2D connectivity = geo->getCompactConnectivity();
377  bitpit::VTKElementType cellType = desumeElement(connectivity);

383  output.setCounter(counter);
384  output.setCodex(bitpit::VTKFormat::APPENDED);

386  }else{

391  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);

407 bool
408 SplitScalarField::split(){
409  if(m_mapCellDivision.empty() && m_mapVertDivision.empty()) return false;
410  if(isEmpty()) return false;
411 
412  //check original field;
413  bool loc = m_mapVertDivision.empty(); //true by cells, false by vert.
414 
415  int nGeo = m_originals.size();
416  m_result.resize(nGeo);
417 
418  std::unordered_map<long, std::pair<int, long>> * mapp;
419  livector1D * mapIdTarget;
420  std::vector< std::map<long, int> * > invIdLoc(nGeo);
421 
422  if(loc) {
423  m_field.resize(getGeometry()->getNCells(), 0.0);
424  mapp = &m_mapCellDivision;
425  mapIdTarget = &(getGeometry()->getMapCell());
426  for(int i=0; i<nGeo; ++i){
427  invIdLoc[i] = &(m_originals[i]->getMapCellInv());
428  }
429  }
430  else {
431  m_field.resize(getGeometry()->getNVertex(), 0.0);
432  mapp = &m_mapVertDivision;
433  mapIdTarget = &(getGeometry()->getMapData());
434  for(int i=0; i<nGeo; ++i){
435  invIdLoc[i] = &(m_originals[i]->getMapDataInv());
436  }
437  }
438 
439  if(m_field.size() != mapp->size()) return false;
440 
441  //allocate memory for results;
442  for(int i=0; i<nGeo; ++i){
443 
444  if(loc) m_result[i].resize(m_originals[i]->getNCells(),0.0);
445  else m_result[i].resize(m_originals[i]->getNVertex(),0.0);
446  }
447 
448  int counter = 0;
449  long IDtarget;
450  std::pair<int,long> douple;
451  int locali;
452  for(auto & val: m_field){
453 
454  IDtarget = (*mapIdTarget)[counter];
455  if(mapp->count(IDtarget)> 0 ){
456  douple = (*mapp)[IDtarget];
457  if(douple.first > nGeo) continue;
458  if((invIdLoc[douple.first])->count(douple.second) > 0){
459  locali = (*invIdLoc[douple.first])[douple.second];
460  m_result[douple.first][locali] = val;
461  }
462  }
463  ++counter;
464  }
465  return true;
466 }
