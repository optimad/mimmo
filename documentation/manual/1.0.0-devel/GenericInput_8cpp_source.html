164 void GenericInput::absorbSectionXML(const bitpit::Config::Section & slotXML, std::string name){
165 
166  BITPIT_UNUSED(name);
167 
168  std::string input;
169 
170  if(slotXML.hasOption("Priority")){
171  input = slotXML.get("Priority");
172  int value =0;
173  if(!input.empty()){
174  std::stringstream ss(bitpit::utils::trim(input));
175  ss>>value;
176  }
177  setPriority(value);
178  };
179 
180  if(slotXML.hasOption("ReadFromFile")){
181  std::string input = slotXML.get("ReadFromFile");
182  input = bitpit::utils::trim(input);
183  bool temp = false;
184  if(!input.empty()){
185  std::stringstream ss(input);
186  ss>>temp;
187  }
188  setReadFromFile(temp);
189  };
190 
191  if(slotXML.hasOption("Filename") && m_readFromFile){
192  std::string input = slotXML.get("Filename");
193  input = bitpit::utils::trim(input);
194  setFilename(input);
195  };
196 
197 }
