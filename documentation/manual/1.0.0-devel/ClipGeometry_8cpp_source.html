154 void
155 ClipGeometry::execute(){
156 
157  if(getGeometry() == NULL || getGeometry()->isEmpty()) return;
158 
159  m_patch.reset(nullptr);
160 
161  livector1D extracted = clipPlane();
162  if(extracted.empty()) return;
163 
164  //create your subpatch.
165  std::unique_ptr<MimmoObject> temp(new MimmoObject(getGeometry()->getType()));
166  bitpit::PatchKernel * tri = getGeometry()->getPatch();
167  bitpit::PiercedVector<bitpit::Vertex> & mapV = temp->getPatch()->getVertices();
168 
169  livector1D TT;
170 
171  long idV;
172  int sizeCC;
173  bitpit::ElementInfo::Type eltype;
174  int PID;
175 
176  if (getGeometry()->getType() != 3){
177  for(auto && idCell : extracted){
178 
179  bitpit::Cell & cell = tri->getCell(idCell);
180  eltype = cell.getType();
181  sizeCC = cell.getVertexCount();
182  PID = cell.getPID();
183  TT.resize(sizeCC);
184 
185  for(int i=0; i<sizeCC; ++i){
186  idV = cell.getVertex(i);
187  TT[i] = idV;
188 
189  if(!mapV.exists(idV)) temp->addVertex(tri->getVertexCoords(idV),idV);
190  }
191  temp->addConnectedCell(TT,eltype,short(PID), idCell);
192  TT.clear();
193  }
194  }
195  else{
196  for(auto && idV : extracted){
197  temp->addVertex(tri->getVertexCoords(idV),idV);
198  }
199  }
200 
201  m_patch = std::move(temp);
202  tri = NULL;
203 };

209 livector1D ClipGeometry::clipPlane(){
210 
211  livector1D result;
212  darray3E norm;
213  double offset;
214  int counter;
215  double sig = 1.0 - 2.0*(int)isInsideOut();
216  long iD;
217 
218 
219  for(int i=0; i<3; ++i) norm[i] = m_plane[i];
220  offset = m_plane[3];
221 
222  double normPlane = norm2(norm);
223  if(normPlane < 1.E-18) return result;
224  norm /= normPlane;
225 
226  bitpit::PatchKernel * tri = getGeometry()->getPatch();
227 
228  if(getGeometry()->getType() == 3){
229  counter = 0;
230  result.resize(tri->getVertexCount());
231  for(auto vert : tri->getVertices()){
232  iD = vert.getId();
233  if(sig*(dotProduct(norm, vert.getCoords()) + offset) >0) {
234  result[counter] = iD;
235  ++counter;
236  }
237  }
238 
239  }else{
240  counter = 0;
241  result.resize(tri->getCellCount());
242  for(auto cell : tri->getCells()){
243  iD = cell.getId();
244  if(sig*(dotProduct(norm, tri->evalCellCentroid(iD)) + offset) >0) {
245  result[counter] = iD;
246  ++counter;
247  }
248  }
249  }
250  result.resize(counter);
251  return result;
252 };

259 void ClipGeometry::plotOptionalResults(){
260  if(getClippedPatch() == NULL) return;
261  if(getClippedPatch()->isEmpty()) return;
262 
263  dvecarr3E points = getClippedPatch()->getVertexCoords();
264  ivector2D connectivity;
265  bitpit::VTKElementType cellType;
266 
267  std::string dir = m_outputPlot;
268  std::string name = m_name + "_Patch";
269 
270 
271  if (getClippedPatch()->getType() != 3){
272  connectivity = getClippedPatch()->getCompactConnectivity();
273  }
274  else{
275  int np = points.size();
276  connectivity.resize(np);
277  for (int i=0; i<np; i++){
278  connectivity[i].resize(1);
279  connectivity[i][0] = i;
280 
281  }
282  }
283  cellType = getClippedPatch()->desumeElement();
284 
285 
286  bitpit::VTKUnstructuredGrid output(dir,name,cellType);
287  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
288  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
289  output.setDimensions(connectivity.size(), points.size());
290 
291  auto pids = getClippedPatch()->getCompactPID();
292  if(pids.size() > 0) output.addData("PID", bitpit::VTKFieldType::SCALAR, bitpit::VTKLocation::CELL, pids);
293 
294  output.setCounter(getClassCounter());
295  output.setCodex(bitpit::VTKFormat::APPENDED);
296 
297  output.write();

408 void ClipGeometry::flushSectionXML(bitpit::Config::Section & slotXML, std::string name){
409 
410  BITPIT_UNUSED(name);
411 
412  slotXML.set("ClassName", m_name);
413  slotXML.set("Priority", std::to_string(getPriority()));
414 
415  int value = m_insideout;
416  slotXML.set("InsideOut", std::to_string(value));
417 
418  {
419  darray4E org = getClipPlane();
420  darray3E normal;
421  darray3E point = {{0.0,0.0,0.0}};
422  int imax = -1;
423  double dum = 0.0;
424  for(int i=0; i<3; ++i) {
425  normal[i] =org[i];
426  if(abs(normal[i]) > dum) {
427  imax = i;
428  dum = abs(normal[i]);
429  }
430  }
431  if(imax != -1) point[imax] = -1.0*org[3]/normal[imax];
432 
433  std::stringstream ss1, ss2;
434  ss1<<std::scientific<<point[0]<<'\t'<<point[1]<<'\t'<<point[2];
435  ss2<<std::scientific<<normal[0]<<'\t'<<normal[1]<<'\t'<<normal[2];
436 
437  bitpit::Config::Section & planeXML = slotXML.addSection("ClipPlane");
438  planeXML.set("Point",ss1.str());
439  planeXML.set("Normal",ss2.str());
440  }
441 
442  if(isPlotInExecution()){
443  slotXML.set("PlotInExecution", std::to_string(1));
444  }
445 
446  if(m_outputPlot != "."){
447  slotXML.set("OutputPlot", m_outputPlot);
448  }
449 
450  return;
