223 void ReconstructScalar::plotData(std::string dir, std::string name, bool flag, int counter){
224 
225  if(getGeometry() == NULL || getGeometry()->isEmpty()) return;
226  dvecarr3E points = getGeometry()->getVertexCoords();
227  ivector2D connectivity;
228  bitpit::VTKElementType cellType = getGeometry()->desumeElement();
229 
230  if (getGeometry()->getType() != 3){
231  connectivity = getGeometry()->getCompactConnectivity();
232  }
233  else{
234  int np = points.size();
235  connectivity.resize(np);
236  for (int i=0; i<np; i++){
237  connectivity[i].resize(1);
238  connectivity[i][0] = i;
239  }
240  }
241  bitpit::VTKUnstructuredGrid output(dir,name,cellType);
242  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
243  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
244  output.setDimensions(connectivity.size(), points.size());
245 
246  output.addData("field", bitpit::VTKFieldType::SCALAR, bitpit::VTKLocation::POINT, m_result);
247 
248  std::vector<long> ids(points.size());
249  long ID;
250  for (auto vertex : getGeometry()->getVertices()){
251  ID = vertex.getId();
252  ids[getGeometry()->getMapDataInv(ID)] = ID;
253  }
254 
255  output.addData("ID", bitpit::VTKFieldType::SCALAR, bitpit::VTKLocation::POINT, ids);
256 
257  output.setCounter(counter);
258  output.setCodex(bitpit::VTKFormat::APPENDED);
259  if(!flag) output.setCodex(bitpit::VTKFormat::ASCII);
260 
261  output.write();
262 };

268 void ReconstructScalar::execute(){
269 
270  if(getGeometry() == NULL) return;
271 
272  liimap & vMotherMap = getGeometry()->getMapDataInv();
273 
274  m_result.resize(getGeometry()->getPatch()->getVertexCount(), 0.0);
275 
276  if(m_subpatch.empty()) return;
277 
278  std::unordered_map<long, dvector1D> map = checkOverlapping();
279 
280  for(auto && obj : map){
281  m_result[vMotherMap[obj.first]] = overlapFields(obj.second);
282  obj.second.clear();
283  }
284 
285 }

305 double ReconstructScalar::overlapFields(dvector1D & locField){
306  int size = locField.size();
307  if(size < 1) return 0.0;
308 
309  double value = locField[0];
310  if(size ==1 )return value;
311 
312  switch(m_overlapCriterium){
313  case OverlapMethod::MAX :
314  for(auto && loc : locField){
315  value = std::fmax(loc,value);
316  }
317  break;
318  case OverlapMethod::MIN :
319  for(auto && loc : locField){
320  value = std::fmin(loc,value);
321  }
322  break;
323  case OverlapMethod::AVERAGE :
324  value = 0.0;
325  for(auto && loc : locField){
326  value += loc/double(size);
327  }
328  break;
329  case OverlapMethod::SUM :
330  value = 0.0;
331  for(auto && loc : locField){
332  value += loc;
333  }
334  break;
335  default : //never been reached
336  break;
337  }
338 
339  return value;
340 };

348 std::unordered_map<long, dvector1D> ReconstructScalar::checkOverlapping(){
349 
350  std::unordered_map<long, dvector1D> result;
351  int counter;
352 
353  for(auto && pairInd : m_subpatch){
354 
355  livector1D & vMap = pairInd.first->getMapData();
356  counter = 0;
357  dvector1D field = *(pairInd.second);
358  for(int i=0; i<(int)field.size(); ++i){
359  result[vMap[counter]].push_back(field[i]);
360  ++counter;
361  }
362  }
363 
364  return(result);
365 };

405 
406  BITPIT_UNUSED(name);

418 
419  if(slotXML.hasOption("OverlapCriterium")){
420  std::string input = slotXML.get("OverlapCriterium");
421  input = bitpit::utils::trim(input);
422  int value = 1;
423  if(!input.empty()){
424  std::stringstream ss(input);
425  ss >> value;
426  value = std::min(std::max(1, value),4);
427  }
428  setOverlapCriterium(value);
429  }
430 
431 
432  if(slotXML.hasOption("PlotInExecution")){
433  std::string input = slotXML.get("PlotInExecution");
434  input = bitpit::utils::trim(input);
435  bool value = false;
436  if(!input.empty()){
437  std::stringstream ss(input);
438  ss >> value;
439  }
440  setPlotInExecution(value);
441  }
442 
443  if(slotXML.hasOption("OutputPlot")){
444  std::string input = slotXML.get("OutputPlot");
