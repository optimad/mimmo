157 void
158 ControlDeformMaxDistance::execute(){
159 
160  MimmoObject * geo = getGeometry();
161  if(geo == NULL || geo->isEmpty() || !(geo->isBvTreeSupported())) return;
162 
163  m_defField.resize(getGeometry()->getNVertex(),darray3E{{0.0,0.0,0.0}});
164  m_violationField.resize(m_defField.size());
165 
166  if(!(geo->isBvTreeBuilt())) geo->buildBvTree();
167 
168  dvecarr3E points = geo->getVertexCoords();
169  points+= m_defField;
170  dvector1D normDef(m_defField.size());
171 
172  int count=0;
173  for(auto & def : m_defField){
174  normDef[count] = norm2(def);
175  ++count;
176  }
177 
178  double dist;
179  double radius ;
180  double rate = 0.05;
181  int kmax = 200;
182  int kiter;
183  bool flag;
184  long id;
185  count = 0;
186 
187  for(auto &p : points){
188  dist =1.0E+18;
189  kiter = 0;
190  flag = true;
191  radius = std::fmax(1.0E-8, normDef[count]);
192  while(flag && kiter < kmax){
193  dist = bvTreeUtils::distance(&p, geo->getBvTree(), id, radius);
194  flag = (dist == 1.0E+18);
195  if(flag) radius *= (1.0+ rate*((double)flag));
196  kiter++;
197  }
198  if(kiter == kmax) dist = m_maxDist - dist;
199  m_violationField[count] = (dist - m_maxDist);
200  count++;
201  }
202  return;
203 };
