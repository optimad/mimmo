134 SplitScalarField::plotOptionalResults(){
135  if(isEmpty()) return;
136  if(m_originals.size() != m_result.size()) return;
137 
138  bitpit::VTKLocation loc = bitpit::VTKLocation::POINT;
139  if(m_mapVertDivision.empty()) loc = bitpit::VTKLocation::CELL;
140 
141 
142  int counter=0;
143  for(auto & geo : m_originals){
144 
145  bitpit::VTKElementType cellType = geo->desumeElement();
146  dvecarr3E points = geo->getVertexCoords();
147 
148  if (cellType == bitpit::VTKElementType::UNDEFINED) continue;
149 
150  if(cellType != bitpit::VTKElementType::VERTEX){
151  ivector2D connectivity = geo->getCompactConnectivity();
152  bitpit::VTKUnstructuredGrid output(".",m_name+std::to_string(getClassCounter()),cellType);
153  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
154  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
155  output.setDimensions(connectivity.size(), points.size());
156  output.addData("field", bitpit::VTKFieldType::SCALAR, loc, m_result[counter]);
157  output.setCounter(counter);
158  output.setCodex(bitpit::VTKFormat::APPENDED);
159  output.write();
160  }else{
161  int size = points.size();
162  ivector1D connectivity(size);
163  for(int i=0; i<size; ++i) connectivity[i]=i;
164  bitpit::VTKUnstructuredGrid output(".",m_name+std::to_string(getClassCounter()), cellType);
165  output.setGeomData( bitpit::VTKUnstructuredField::POINTS, points);
166  output.setGeomData( bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
167  output.setDimensions(connectivity.size(), points.size());
168  output.addData("field", bitpit::VTKFieldType::SCALAR, loc, m_result[counter]);
169  output.setCounter(counter);
170  output.setCodex(bitpit::VTKFormat::APPENDED);
171  output.write();
172  }
173 
174  ++counter;
175  }

181 bool
182 SplitScalarField::split(){
183  if(m_mapCellDivision.empty() && m_mapVertDivision.empty()) return false;
184  if(isEmpty()) return false;
185 
186  //check original field;
187  bool loc = m_mapVertDivision.empty(); //true by cells, false by vert.
188 
189  int nGeo = m_originals.size();
190  m_result.resize(nGeo);
191 
192  std::unordered_map<long, std::pair<int, long>> * mapp;
193  livector1D * mapIdTarget;
194  std::vector< std::map<long, int> * > invIdLoc(nGeo);
195 
196  if(loc) {
197  m_field.resize(getGeometry()->getNCells(), 0.0);
198  mapp = &m_mapCellDivision;
199  mapIdTarget = &(getGeometry()->getMapCell());
200  for(int i=0; i<nGeo; ++i){
201  invIdLoc[i] = &(m_originals[i]->getMapCellInv());
202  }
203  }
204  else {
205  m_field.resize(getGeometry()->getNVertex(), 0.0);
206  mapp = &m_mapVertDivision;
207  mapIdTarget = &(getGeometry()->getMapData());
208  for(int i=0; i<nGeo; ++i){
209  invIdLoc[i] = &(m_originals[i]->getMapDataInv());
210  }
211  }
212 
213  if(m_field.size() != mapp->size()) return false;
214 
215  //allocate memory for results;
216  for(int i=0; i<nGeo; ++i){
217 
218  if(loc) m_result[i].resize(m_originals[i]->getNCells(),0.0);
219  else m_result[i].resize(m_originals[i]->getNVertex(),0.0);
220  }
221 
222  int counter = 0;
223  long IDtarget;
224  std::pair<int,long> douple;
225  int locali;
226  for(auto & val: m_field){
227 
228  IDtarget = (*mapIdTarget)[counter];
229  if(mapp->count(IDtarget)> 0 ){
230  douple = (*mapp)[IDtarget];
231  if(douple.first > nGeo) continue;
232  if((invIdLoc[douple.first])->count(douple.second) > 0){
233  locali = (*invIdLoc[douple.first])[douple.second];
234  m_result[douple.first][locali] = val;
235  }
236  }
237  ++counter;
238  }
239  return true;
240 }
