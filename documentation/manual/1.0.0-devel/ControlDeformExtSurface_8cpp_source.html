273 void
274 ControlDeformExtSurface::execute(){
275 
276  MimmoObject * geo = getGeometry();
277  if(geo == NULL || geo->isEmpty()) return;
278 
279  int nDFS = m_defField.size();
280  m_defField.resize(geo->getNVertex(),darray3E{{0.0,0.0,0.0}});
281  m_violationField.resize(nDFS,-1.E+18);
282 
283  dvector1D violationField;
284  violationField.resize(nDFS);
285 
286  dvecarr3E points = geo->getVertexCoords();
287  dvecarr3E pointsOR = points;
288 
289  //evaluate deformable geometry barycenter************************
290  darray3E geoBary = {{0.0,0.0,0.0}};
291  double distBary = 0.0;
292 
293  for(auto & p: points) geoBary += p;
294 
295  geoBary /= (double)geo->getNVertex();
296 
297  for(auto & p: points){
298  distBary= std::fmax(distBary,norm2(p-geoBary));
299  }
300  //***************************************************************
301 
302  //adding deformation to points **********************************
303  points+= m_defField;
304  //***************************************************************
305 
306  //read external surfaces*****************************************
307  std::vector<std::unique_ptr<MimmoGeometry> > extgeo;
308  dvector1D tols;
309  readGeometries(extgeo, tols);
310  //***************************************************************
311 
312  if(extgeo.size() < 1) return;
313 
314  //evaluating bounding box of deformation*************************
315  darray3E bbMinDef, bbMaxDef, bbMin, bbMax;
316  {
317  int count2 = 0;
318  for(auto & p : points){
319  if(count2 == 0) {
320  bbMinDef = p;
321  bbMaxDef = p;
322  }else{
323  for(int i=0; i<3; ++i ){
324  bbMinDef[i] = std::fmin(bbMinDef[i], p[i]);
325  bbMaxDef[i] = std::fmax(bbMaxDef[i], p[i]);
326  }
327  }
328  count2++;
329  }
330  count2=0;
331  for(auto & p : pointsOR){
332  for(int i=0; i<3; ++i ){
333  bbMinDef[i] = std::fmin(bbMinDef[i], p[i]);
334  bbMaxDef[i] = std::fmax(bbMaxDef[i], p[i]);
335  }
336  count2++;
337  }
338  }
339  //***************************************************************
340  int counterExtGeo =0;
341  // start examining all external geometries***********************
342  for(auto &gg : extgeo){
343 
344  //check constraints properties ******************************
345  MimmoObject * local = gg->getGeometry();
346  if(!(local->isBvTreeBuilt())) local->buildBvTree();
347  bool checkOpen = local->isClosedLoop();
348 
349  double dist;
350  double radius, radius_old;
351  long id;
352  darray3E normal;
353  int count;
354  //************************************************************
355 
356  //add local constraints to the bounding box compute **********
357  bbMin = bbMinDef;
358  bbMax = bbMaxDef;
359 
360  for(auto & p : local->getVertexCoords()){
361  for(int i=0; i<3; ++i ){
362  bbMin[i] = std::fmin(bbMin[i], p[i]);
363  bbMax[i] = std::fmax(bbMax[i], p[i]);
364  }
365  }
366  //*************************************************************
367 
368  //Evaluate background grid dimension for Level set ************
369  //calculate dh as a tot part of bb diagonal *******************
370 
371  darray3E span = bbMax - bbMin;
372  bbMin += -1.0*(0.05*span);
373  span *= 1.1;
374  double dh = norm2(span)/(double)m_cellBackground;
375  iarray3E dim;
376  for(int i=0; i<3; ++i){
377  dim[i] = (int)(span[i]/dh + 0.5);
378  dim[i] = std::max(2, dim[i]);
379  }
380  //*************************************************************
381 
382  //check if its more convenient evaluate distances with a direct BvTree
383  //evaluation of distance on target deformation points or using a background grid +
384  //interpolation.
385  double nReq = double(dim[0]+1)*double(dim[1]+1)*double(dim[2]+1);
386  double nAva = 0.8*nDFS;
387 
388  if( nReq > nAva){
389 
390  //going to use direct evaluation.
391 
392  radius = distBary;
393  radius_old = radius;
394  dvector1D refsigns(nDFS, 1.0);
395 
396  //get the actual sign of distance of the undeformed cloud w.r.t constraints
397  if(checkOpen){
398  count = 0;
399  for(auto &p : pointsOR){
400  dist = evaluateSignedDistance(pointsOR[count], local, id, normal, radius);
401  if(radius < 1.0E-12) radius = radius_old;
402  else radius_old = radius;
403  if(dist < 0.0) refsigns[count] = -1.0;
404  count++;
405  }
406  }else{
407  dist = evaluateSignedDistance(geoBary, local, id, normal, radius);
408 
409  if(dist < 0.0) refsigns *= -1.0;
410  if(radius <1.0E-12) radius = radius_old;
411  }
412 
413  //evaluate distance of the deformation cloud w.r.t. constraints
414  count = 0;
415  for(auto &p : points){
416  dist = evaluateSignedDistance(p, local, id, normal, radius);
417  if(radius < 1.0E-12) radius = radius_old;
418  else radius_old = radius;
419  violationField[count] = -1.0*refsigns[count]*dist;
420  count++;
421  }
422 
423  }else{
424 
425  //going to use background grid to evaluate distances
426 
427  //instantiate a VolCartesian;
428  bitpit::VolCartesian * mesh = new bitpit::VolCartesian(0,3,bbMin,span, dim);
429  mesh->update();
430 
431  //calculate distance on point of background grid.
432  dvecarr3E background_points(mesh->getVertexCount());
433  int count = 0;
434  for(auto & v: mesh->getVertices()){
435  background_points[count] = v.getCoords();
436  count++;
437  }
438  dvector1D background_LS(background_points.size());
439 
440  radius = 0.5*norm2(span);
441  radius_old = radius;
442  count = 0;
443  for(auto &p : background_points){
444  dist = evaluateSignedDistance(p, local, id, normal, radius);
445  if(radius < 1.0E-12) radius = radius_old;
446  else radius_old = radius;
447  background_LS[count] = dist;
448  count++;
449  }
450 
451  //interpolate on background to obtain distances
452 
453  //evaluate sign of distances of the undeformed cloud w.r.t to constraints.
454  dvector1D refsigns(nDFS, 1.0);
455  if(checkOpen){
456  count= 0;
457  for(auto & p : pointsOR){
458  ivector1D stencil;
459  dvector1D weights;
460  int res = mesh->linearVertexInterpolation(p, stencil,weights);
461 
462  dist = 0.0;
463  for(int j=0; j<res; ++j){
464  dist += background_LS[stencil[j]]*weights[j];
465  }
466 
467  if(dist < 0.0) refsigns[count] = -1.0;
468  ++count;
469  }
470  }else{
471  dist = evaluateSignedDistance(geoBary, local, id, normal, radius);
472  if(dist< 0) refsigns *= -1.0;
473  if(radius < 1.0E-12) radius = radius_old;
474  }
475 
476  //evaluate violation field
477  count = 0;
478  for(auto & p : points){
479  ivector1D stencil;
480  dvector1D weights;
481 
482  int res = mesh->linearVertexInterpolation(p, stencil,weights);
483 
484  violationField[count] = 0.0;
485  for(int j=0; j<res; ++j){
486  violationField[count] += background_LS[stencil[j]]*weights[j];
487  }
488 
489  violationField[count] *= -1.0*refsigns[count];
490  ++count;
491 
492  }
493  delete mesh;
494  mesh = NULL;
495  }//end if else
496 
497  for(int i=0; i< nDFS; ++i){
498  m_violationField[i] = std::fmax(m_violationField[i], (violationField[i] + tols[counterExtGeo]));
499  }
500  ++counterExtGeo;
501  }
502  return;
503 };

720 double ControlDeformExtSurface::evaluateSignedDistance(darray3E &point, mimmo::MimmoObject * geo, long & id, darray3E & normal, double &initRadius){
721 
722  double dist = 1.0E+18;
723  double rate = 0.02;
724  int kmax = 1000;
725  int kiter = 0;
726  bool flag = true;
727 
728  if(!geo->isBvTreeBuilt()) geo->buildBvTree();
729 
730  while(flag && kiter < kmax){
731  dist = bvTreeUtils::signedDistance(&point, geo->getBvTree(), id, normal, initRadius);
732  flag = (dist == 1.0E+18);
733  if(flag) initRadius *= (1.0+ rate*((double)flag));
734  kiter++;
735  }
736 
737  return dist;
