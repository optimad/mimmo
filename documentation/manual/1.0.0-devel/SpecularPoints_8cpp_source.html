208 void
209 SpecularPoints::execute(){
210 
211  darray3E norm;
212  double offset;
213  for(int i=0; i<3; ++i) norm[i] = m_plane[i];
214  offset = m_plane[3];
215  double normPlane = norm2(norm);
216  if(normPlane < 1.E-18 || m_points.empty()) return;
217  norm /= normPlane;
218  bool project =!(getGeometry() == NULL || getGeometry()->isEmpty());
219 
220 
221  //choosing margin for plane offset
222  double margin = 1.e-12;
223  if(project && getGeometry()->getType() == 1){
224  double aTot = 0.0;
225  int cellSize = getGeometry()->getNCells();
226  bitpit::SurfaceKernel * tri = static_cast<bitpit::SurfaceKernel * >(getGeometry()->getPatch());
227  for(auto &cell: tri->getCells()){
228  aTot += tri->evalCellArea(cell.getId());
229  }
230 
231  if(aTot > 0.0) {margin = 1.2*pow(aTot/((double)cellSize),0.5);}
232 
233  }else if(project && getGeometry()->getType() == 2){
234  double vTot = 0.0;
235  int cellSize = getGeometry()->getNCells();
236  bitpit::VolumeKernel * tetra = static_cast<bitpit::VolumeKernel * >(getGeometry()->getPatch());
237  for(auto &cell: tetra->getCells()){
238  vTot += tetra->evalCellVolume(cell.getId());
239  }
240 
241  if(vTot > 0.0) {margin = 1.2*pow(vTot/((double)cellSize),0.5);}
242  }
243 
244 
245  double sig = (1.0 - 2.0*((int)m_insideout));
246  double distance;
247 
248  //mirroring.
249  m_scalar.resize(m_points.size());
250  m_vector.resize(m_points.size());
251 
252  int counterProj = m_points.size();
253  int counterData = 0;
254 
255  m_proj = m_points;
256  m_scalarMirrored = m_scalar;
257  m_vectorMirrored = m_vector;
258 
259  m_proj.resize(2*counterProj);
260  m_scalarMirrored.resize(2*counterProj);
261  m_vectorMirrored.resize(2*counterProj);
262 
263  for(auto &val: m_points){
264  distance = sig*(dotProduct(norm, val) + offset);
265  if(distance > margin || m_force){
266  m_proj[counterProj] = val - 2.0*distance*sig*norm;
267  m_scalarMirrored[counterProj] = m_scalar[counterData];
268  m_vectorMirrored[counterProj] = m_vector[counterData] -2.0*dotProduct(m_vector[counterData], sig*norm)*sig*norm;
269  counterProj++;
270  }
271  counterData++;
272  }
273 
274  m_proj.resize(counterProj);
275  m_scalarMirrored.resize(counterProj);
276  m_vectorMirrored.resize(counterProj);
277 
278  if(project){
279  if(!getGeometry()->isBvTreeBuilt()) getGeometry()->buildBvTree();
280 
281  //project points on surface.
282  int counter = 0;
283  for(auto &val : m_proj){
284  m_proj[counter]= bvTreeUtils::projectPoint(&val, getGeometry()->getBvTree());
285  ++counter;
286  }
287  }
288  return;
289 };
