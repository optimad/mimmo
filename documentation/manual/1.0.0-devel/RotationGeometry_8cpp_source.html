129 void
130 RotationGeometry::execute(){
131 
132  m_displ.resize(m_geometry->getNVertex());
133 
134  //compute coefficients and constant vectors of rodriguez formula
135  double a = cos(m_alpha);
136  darray3E b = (1 - cos(m_alpha)) * m_direction;
137  double c = sin(m_alpha);
138 
139  darray3E point, rotated;
140  long ID;
141  int idx;
142  liimap mapID = m_geometry->getMapDataInv();
143 
144  for (auto vertex : m_geometry->getVertices()){
145  point = vertex.getCoords();
146  ID = vertex.getId();
147  idx = mapID[ID];
148 
149  point -= m_origin;
150  //rodrigues formula
151  rotated = a * point +
152  b * dotProduct(m_direction, point) +
153  c * crossProduct(m_direction, point);
154 
155  rotated += m_origin;
156  point += m_origin;
157 
158  m_displ[idx] = (rotated-point)*m_filter[idx];
159 
160  }
161 
162  return;
163 };
