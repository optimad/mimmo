209 void
210 StitchGeometry::execute(){
211  if(m_extgeo.empty()) return;
212 
213  std::unique_ptr<MimmoObject> dum(new MimmoObject(m_topo));
214  long nCells = 0;
215  long nVerts = 0;
216 
217  for(auto &obj: m_extgeo){
218  nCells += obj->getNCells();
219  nVerts += obj->getNVertex();
220  }
221 
222  //reserving memory
223  dum->getPatch()->reserveVertices(nVerts);
224  dum->getPatch()->reserveCells(nCells);
225 
226  //clean maps
227  m_mapCellDivision.clear();
228  m_mapVertDivision.clear();
229 
230  long cV = 0, cC = 0;
231  int objId = 0;
232 
233  //start filling your stitched object.
234  //divion maps will be filled coherently
235 
236  {
237  //optional vars;
238  long vId, cId;
239  short PID;
240  bitpit::ElementInfo::Type eltype;
241 
242  for(auto &obj : m_extgeo){
243 
244  //start extracting/reversing vertices of the current obj
245  std::unordered_map<long,long> mapVloc;
246 
247  for(auto & vv : obj->getVertices()){
248  vId = vv.getId();
249  dum->addVertex(obj->getVertexCoords(vId), cV);
250  //update map;
251  m_mapVertDivision[cV] = std::make_pair(objId, vId);
252  mapVloc[vId] = cV;
253  cV++;
254  }
255 
256  //start extracting/reversing cells of the current obj
257  for(auto & cc : obj->getCells()){
258  cId = cc.getId();
259  PID = cc.getPID();
260  eltype = cc.getType();
261  //get the local connectivity and update with new vertex numbering;
262  livector1D conn = obj->getCellConnectivity(cId);
263  for(auto && ee: conn) ee = mapVloc[ee];
264 
265  dum->addConnectedCell(conn, eltype, PID, cC);
266  //update map;
267  m_mapCellDivision[cC] = std::make_pair(objId, cId);
268  cC++;
269  }
270  objId++;
271  }
272  }//scope for optional vars;
273 
274  if(m_buildBvTree) dum->buildBvTree();
275  if(m_buildKdTree) dum->buildKdTree();
276 
277  m_patch = std::move(dum);
278 }
