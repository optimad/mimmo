226 void ReconstructVector::plotData(std::string dir, std::string name, bool flag, int counter){
227  if(getGeometry() == NULL || getGeometry()->isEmpty()) return;
228 
229  dvecarr3E points = getGeometry()->getVertexCoords();
230  ivector2D connectivity;
231  bitpit::VTKElementType cellType = getGeometry()->desumeElement();
232 
233  if (getGeometry()->getType() != 3){
234  connectivity = getGeometry()->getCompactConnectivity();
235  }
236  else{
237  int np = points.size();
238  connectivity.resize(np);
239  for (int i=0; i<np; i++){
240  connectivity[i].resize(1);
241  connectivity[i][0] = i;
242  }
243  }
244  bitpit::VTKUnstructuredGrid output(dir,name,cellType);
245  output.setGeomData(bitpit::VTKUnstructuredField::POINTS, points);
246  output.setGeomData(bitpit::VTKUnstructuredField::CONNECTIVITY, connectivity);
247  output.setDimensions(connectivity.size(), points.size());
248 
249  output.addData("vectorfield", bitpit::VTKFieldType::VECTOR, bitpit::VTKLocation::POINT, m_result);
250 
251  std::vector<long> ids(points.size());
252  long ID;
253  for (auto vertex : getGeometry()->getVertices()){
254  ID = vertex.getId();
255  ids[getGeometry()->getMapDataInv(ID)] = ID;
256  }
257 
258  output.addData("ID", bitpit::VTKFieldType::SCALAR, bitpit::VTKLocation::POINT, ids);
259 
260  output.setCounter(counter);
261  output.setCodex(bitpit::VTKFormat::APPENDED);
262  if(!flag) output.setCodex(bitpit::VTKFormat::ASCII);
263 
264 
265  output.write();
266 };

272 void ReconstructVector::execute(){
273  if(getGeometry() == NULL) return;
274 
275  liimap & vMotherMap = getGeometry()->getMapDataInv();
276 
277  m_result.resize(getGeometry()->getPatch()->getVertexCount(), {{0.0,0.0,0.0}});
278  if(m_subpatch.empty()) return;
279  std::unordered_map<long, dvecarr3E > map = checkOverlapping();
280 
281  for(auto && obj : map){
282  m_result[vMotherMap[obj.first]] = overlapFields(obj.second);
283  obj.second.clear();
284  }
285 
286 }

297 darray3E ReconstructVector::overlapFields(dvecarr3E & locField){
298  int size = locField.size();
299  if(size < 1) return {{0.0,0.0,0.0}};
300 
301  if(size ==1 )return locField[0];
302  darray3E value;
303  darray3E dir; dir.fill(0.0);
304  double match;
305 
306 
307 
308 
309  switch(m_overlapCriterium){
310  case OverlapMethod::MAX :
311  value = {{0.0,0.0,0.0}};
312  for(auto && loc : locField){
313  dir += loc;
314  }
315 
316  dir /= norm2(dir);
317 
318  match = 1.e-18;
319  for(auto && loc : locField){
320  double dummy = dotProduct(loc, dir);
321  match = std::fmax(match,dummy);
322  }
323 
324  value = match*dir;
325  break;
326 
327  case OverlapMethod::MIN :
328 
329  value = {{0.0,0.0,0.0}};
330  for(auto && loc : locField){
331  dir += loc;
332  }
333 
334  dir /= norm2(dir);
335 
336  match = 1.e18;
337  for(auto && loc : locField){
338  double dummy = dotProduct(loc, dir);
339  match = std::fmin(match,dummy);
340  }
341 
342  value = match*dir;
343  break;
344 
345  case OverlapMethod::AVERAGE :
346  value = {{0.0,0.0,0.0}};
347  for(auto && loc : locField){
348  value += loc/double(size);
349  }
350  break;
351 
352  case OverlapMethod::SUM :
353  value = {{0.0,0.0,0.0}};
354  for(auto && loc : locField){
355  value += loc;
356  }
357  break;
358  default : //never been reached
359  break;
360  }
361 
362  return value;
363 };

371 std::unordered_map<long, dvecarr3E> ReconstructVector::checkOverlapping(){
372 
373  std::unordered_map<long, dvecarr3E> result;

463

465  std::string input = slotXML.get("PlotInExecution");

476  std::string input = slotXML.get("OutputPlot");
