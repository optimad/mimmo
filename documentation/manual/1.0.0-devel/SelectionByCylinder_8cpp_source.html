318 
319  BITPIT_UNUSED(name);
320 
321  slotXML.set("ClassName", m_name);

324  int value = m_dual;

327 
328  {

334

336  darray3E span = getSpan();

339  slotXML.set("Span",ss.str());
340  }
341 
342  {
343  dmatrix33E axes = getRefSystem();
344  bitpit::Config::Section & axesXML = slotXML.addSection("RefSystem");
345 
346  for(int i=0; i<3; ++i){
347  std::string name = "axis"+std::to_string(i);
348  std::stringstream ss;
349  ss<<std::scientific<<axes[i][0]<<'\t'<<axes[i][1]<<'\t'<<axes[i][2];
350  axesXML.set(name, ss.str());
351  }
352  }
353 
354  {
355  darray3E inflim = getInfLimits();
356  std::stringstream ss;
357  ss<<std::scientific<<inflim[0]<<'\t'<<inflim[1]<<'\t'<<inflim[2];
358  slotXML.set("InfLimits",ss.str());
359  }
360 
361  if(isPlotInExecution()){
362  slotXML.set("PlotInExecution", std::to_string(1));
363  }
364 
365  if(m_outputPlot != "."){
366  slotXML.set("OutputPlot", m_outputPlot);
367  }
368 
369  return;
