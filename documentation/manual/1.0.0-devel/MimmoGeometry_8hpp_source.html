195  void writeValue (Type& value, std::ofstream& os){
196 
197  int offset = 5;
198  if (wformat == Long) offset = 13;
199 
200  std::stringstream manip;
201  manip << value;
202  std::string manips = manip.str();
203  std::string mantissa, expon;
204  int pos = manips.find("E");
205  if (pos < manips.size()){
206  mantissa = manips.substr(0,std::min(offset,pos));
207  expon = manips.substr(pos+1,manips.size());
208  manips = mantissa + expon;
209  }
210  pos = manips.find("e");
211  if (pos < manips.size()){
212  mantissa = manips.substr(0,std::min(offset,pos));
213  expon = manips.substr(pos+1,manips.size());
214  manips = mantissa + expon;
215  }
216  manips = manips.substr(0, offset+3);
217 
218  switch (wformat)
219  {
220  case Short:
221  {
222  os << std::setw(8) << manips;
223  break;
224  }
225  case Long:
226  {
227  os << std::setw(16) << manips;
228  break;
229  }
230  }
231  }
